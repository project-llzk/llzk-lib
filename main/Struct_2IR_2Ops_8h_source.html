69template <char const *FuncName, unsigned PrefixLen>
70mlir::LogicalResult verifyInStructFunctionNamed(
71 mlir::Operation *op, llvm::function_ref<llvm::SmallString<PrefixLen>()> prefix
72) {
73 return isInStructFunctionNamed(op, FuncName)
74 ? mlir::success()
75 : op->emitOpError(prefix())
76 << "only valid within a '" << getOperationName<function::FuncDefOp>()
77 << "' named \"@" << FuncName << "\" within a '"
78 << getOperationName<StructDefOp>() << "' definition";
79}

83template <char const *FuncName> struct InStructFunctionNamed {
84 template <typename TypeClass> class Impl : public mlir::OpTrait::TraitBase<TypeClass, Impl> {
85 public:
86 static mlir::LogicalResult verifyTrait(mlir::Operation *op) {
87 return verifyInStructFunctionNamed<FuncName, 0>(op, [] { return llvm::SmallString<0>(); });
88 }
89 };
