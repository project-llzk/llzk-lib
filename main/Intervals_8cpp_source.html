22Interval UnreducedInterval::reduce(const Field &field) const {
23 if (a > b) {
24 return Interval::Empty(field);
25 }
26 if (width() >= field.prime()) {
27 return Interval::Entire(field);
28 }
29 auto lhs = field.reduce(a), rhs = field.reduce(b);
30 if (rhs == lhs) {
31 return Interval::Degenerate(field, lhs);
32 }
33
34 const auto &half = field.half();
35 if (lhs <= rhs) {
36 if (lhs < half && rhs < half) {
37 return Interval::TypeA(field, lhs, rhs);
38 } else if (lhs < half) {
39 return Interval::TypeC(field, lhs, rhs);
40 } else {
41 return Interval::TypeB(field, lhs, rhs);
42 }
43 } else {
44 if (lhs >= half && rhs < half) {
45 return Interval::TypeF(field, lhs, rhs);
46 } else {
47 return Interval::Entire(field);
48 }
49 }
50}

108UnreducedInterval operator*(const UnreducedInterval &lhs, const UnreducedInterval &rhs) {
109 DynamicAPInt v1 = lhs.a * rhs.a;
110 DynamicAPInt v2 = lhs.a * rhs.b;
111 DynamicAPInt v3 = lhs.b * rhs.a;
112 DynamicAPInt v4 = lhs.b * rhs.b;
113
114 auto minVal = std::min({v1, v2, v3, v4});
115 auto maxVal = std::max({v1, v2, v3, v4});
116
117 return UnreducedInterval(minVal, maxVal);
118}

120bool UnreducedInterval::overlaps(const UnreducedInterval &rhs) const {
121 return isNotEmpty() && rhs.isNotEmpty() && (b >= rhs.a) && (a <= rhs.b);
122}

124std::strong_ordering operator<=>(const UnreducedInterval &lhs, const UnreducedInterval &rhs) {
125 if ((lhs.a < rhs.a) || ((lhs.a == rhs.a) && (lhs.b < rhs.b))) {
126 return std::strong_ordering::less;
127 }
128 if ((lhs.a > rhs.a) || ((lhs.a == rhs.a) && (lhs.b > rhs.b))) {
129 return std::strong_ordering::greater;
130 }
131 return std::strong_ordering::equal;
132}

134DynamicAPInt UnreducedInterval::width() const {
135 DynamicAPInt w;
136 if (a > b) {
137 // This would be reduced to an empty Interval, so the width is just zero.
138 w = 0;
139 } else {
140 // Since the range is inclusive, we add one to the difference to get the true width.
141 w = (b - a) + 1;
142 }
143 ensure(w >= 0, "cannot have negative width");
144 return w;
145}

158llvm::SmallVector<UnreducedInterval, 2> getUnsignedCanonicalParts(const Interval &iv) {
159 const Field &f = iv.getField();
160 llvm::SmallVector<UnreducedInterval, 2> parts;
161 if (iv.isEmpty()) {
162 return parts;
163 }
164 if (iv.isEntire()) {
165 parts.emplace_back(f.zero(), f.maxVal());
166 return parts;
167 }
168 if (iv.isTypeF()) {
169 parts.emplace_back(iv.lhs(), f.maxVal());
170 parts.emplace_back(f.zero(), iv.rhs());
171 return parts;
172 }
173
174 parts.emplace_back(iv.lhs(), iv.rhs());
175 return parts;
176}
177
178llvm::SmallVector<UnreducedInterval, 2> getSignedCanonicalParts(const Interval &iv) {
179 const Field &f = iv.getField();
180 llvm::SmallVector<UnreducedInterval, 2> parts;
181 if (iv.isEmpty()) {
182 return parts;
183 }
184 if (iv.isEntire()) {
185 parts.emplace_back(f.half() - f.prime(), f.half() - f.one());
186 return parts;
187 }
188 if (iv.isDegenerate()) {
189 DynamicAPInt v = iv.lhs();
190 if (v < f.half()) {
191 parts.emplace_back(v, v);
192 } else {
193 parts.emplace_back(v - f.prime(), v - f.prime());
194 }
195 return parts;
196 }
197 if (iv.isTypeA()) {
198 parts.emplace_back(iv.lhs(), iv.rhs());
199 return parts;
200 }
201 if (iv.isTypeB()) {
202 parts.emplace_back(iv.lhs() - f.prime(), iv.rhs() - f.prime());
203 return parts;
204 }
205 if (iv.isTypeC()) {
206 parts.emplace_back(f.half() - f.prime(), iv.rhs() - f.prime());
207 parts.emplace_back(iv.lhs(), f.half() - f.one());
208 return parts;
209 }
210
211 ensure(iv.isTypeF(), "expected TypeF interval");
212 parts.emplace_back(iv.lhs() - f.prime(), iv.rhs());
213 return parts;
214}

216bool containsZero(const UnreducedInterval &iv) { return iv.getLHS() <= 0 && iv.getRHS() >= 0; }

218Interval joinDivisionPiece(
219 const Field &f, Interval acc, const llvm::DynamicAPInt &q0, const llvm::DynamicAPInt &q1,
220 const llvm::DynamicAPInt &q2, const llvm::DynamicAPInt &q3
221) {
222 DynamicAPInt minQ = std::min({q0, q1, q2, q3});
223 DynamicAPInt maxQ = std::max({q0, q1, q2, q3});
224 Interval piece = UnreducedInterval(minQ, maxQ).reduce(f);
225 return acc.join(piece);
226}

230UnreducedInterval Interval::toUnreduced() const {
231 if (isEmpty()) {
232 // Since ranges are inclusive, empty is encoded as `[a, b]` where `a` > `b`.
233 // This matches the definition provided by UnreducedInterval::width().
234 return UnreducedInterval(field.get().one(), field.get().zero());
235 }
236 if (isEntire()) {
237 return UnreducedInterval(field.get().zero(), field.get().maxVal());
238 }
239 return UnreducedInterval(a, b);
240}

242UnreducedInterval Interval::firstUnreduced() const {
243 if (is<Type::TypeF>()) {
244 return UnreducedInterval(a - field.get().prime(), b);
245 }
246 return toUnreduced();

251 return UnreducedInterval(a - field.get().prime(), b - field.get().prime());

254Interval Interval::join(const Interval &rhs) const {
255 const auto &lhs = *this;
256 const Field &f = checkFields(lhs, rhs);
257
258 // Trivial cases
259 if (lhs.isEntire() || rhs.isEntire()) {
260 return Interval::Entire(f);
261 }
262 if (lhs.isEmpty()) {
263 return rhs;
264 }
265 if (rhs.isEmpty()) {
266 return lhs;
267 }
268 if (lhs.isDegenerate() || rhs.isDegenerate()) {
269 return lhs.toUnreduced().doUnion(rhs.toUnreduced()).reduce(f);
270 }
271
272 // More complex cases
273 if (areOneOf<
274 {Type::TypeA, Type::TypeA}, {Type::TypeB, Type::TypeB}, {Type::TypeC, Type::TypeC},
275 {Type::TypeA, Type::TypeC}, {Type::TypeB, Type::TypeC}>(lhs, rhs)) {
276 auto newLhs = std::min(lhs.a, rhs.a);
277 auto newRhs = std::max(lhs.b, rhs.b);
278 if (newLhs == newRhs) {
279 return Interval::Degenerate(f, newLhs);
280 }
281 return Interval(rhs.ty, f, newLhs, newRhs);
282 }
283 if (areOneOf<{Type::TypeA, Type::TypeB}>(lhs, rhs)) {
284 auto lhsUnred = lhs.firstUnreduced();
285 auto opt1 = rhs.firstUnreduced().doUnion(lhsUnred);
286 auto opt2 = rhs.secondUnreduced().doUnion(lhsUnred);
287 if (opt1.width() <= opt2.width()) {
288 return opt1.reduce(f);
289 }
290 return opt2.reduce(f);
291 }
292 if (areOneOf<{Type::TypeF, Type::TypeF}, {Type::TypeA, Type::TypeF}>(lhs, rhs)) {
293 return lhs.firstUnreduced().doUnion(rhs.firstUnreduced()).reduce(f);
294 }
295 if (areOneOf<{Type::TypeB, Type::TypeF}>(lhs, rhs)) {
296 return lhs.secondUnreduced().doUnion(rhs.firstUnreduced()).reduce(f);
297 }
298 if (areOneOf<{Type::TypeC, Type::TypeF}>(lhs, rhs)) {
299 return Interval::Entire(f);
300 }
301 if (areOneOf<
302 {Type::TypeB, Type::TypeA}, {Type::TypeC, Type::TypeA}, {Type::TypeC, Type::TypeB},
303 {Type::TypeF, Type::TypeA}, {Type::TypeF, Type::TypeB}, {Type::TypeF, Type::TypeC}>(
304 lhs, rhs
305 )) {
306 return rhs.join(lhs);
307 }
308 llvm::report_fatal_error("unhandled join case");
309 return Interval::Entire(f);
310}

312Interval Interval::intersect(const Interval &rhs) const {
313 const auto &lhs = *this;
314 const Field &f = checkFields(lhs, rhs);
315 // Trivial cases
316 if (lhs == rhs) {
317 return lhs;
318 }
319 if (lhs.isEmpty() || rhs.isEmpty()) {
320 return Interval::Empty(f);
321 }
322 if (lhs.isEntire()) {
323 return rhs;
324 }
325 if (rhs.isEntire()) {
326 return lhs;
327 }
328 if (lhs.isDegenerate() && rhs.isDegenerate()) {
329 // These must not be equal
330 return Interval::Empty(f);
331 }
332 if (lhs.isDegenerate()) {
333 return Interval::TypeA(f, lhs.a, lhs.a).intersect(rhs);
334 }
335 if (rhs.isDegenerate()) {
336 return Interval::TypeA(f, rhs.a, rhs.a).intersect(lhs);
337 }
338
339 // More complex cases
340 if (areOneOf<
341 {Type::TypeA, Type::TypeA}, {Type::TypeB, Type::TypeB}, {Type::TypeC, Type::TypeC},
342 {Type::TypeA, Type::TypeC}, {Type::TypeB, Type::TypeC}>(lhs, rhs)) {
343 auto maxA = std::max(lhs.a, rhs.a);
344 auto minB = std::min(lhs.b, rhs.b);
345 if (maxA < minB) {
346 return Interval(lhs.ty, f, maxA, minB);
347 } else if (maxA == minB) {
348 return Interval::Degenerate(f, maxA);
349 } else {
350 return Interval::Empty(f);
351 }
352 }
353 if (areOneOf<{Type::TypeA, Type::TypeB}>(lhs, rhs)) {
354 return Interval::Empty(f);
355 }
356 if (areOneOf<{Type::TypeF, Type::TypeF}, {Type::TypeA, Type::TypeF}>(lhs, rhs)) {
357 return lhs.firstUnreduced().intersect(rhs.firstUnreduced()).reduce(f);
358 }
359 if (areOneOf<{Type::TypeB, Type::TypeF}>(lhs, rhs)) {
360 return lhs.secondUnreduced().intersect(rhs.firstUnreduced()).reduce(f);
361 }
362 if (areOneOf<{Type::TypeC, Type::TypeF}>(lhs, rhs)) {
363 auto rhsUnred = rhs.firstUnreduced();
364 auto opt1 = lhs.firstUnreduced().intersect(rhsUnred).reduce(f);
365 auto opt2 = lhs.secondUnreduced().intersect(rhsUnred).reduce(f);
366 ensure(!opt1.isEntire() && !opt2.isEntire(), "impossible intersection");
367 if (opt1.isEmpty()) {
368 return opt2;
369 }
370 if (opt2.isEmpty()) {
371 return opt1;
372 }
373 return opt1.join(opt2);
374 }
375 if (areOneOf<
376 {Type::TypeB, Type::TypeA}, {Type::TypeC, Type::TypeA}, {Type::TypeC, Type::TypeB},
377 {Type::TypeF, Type::TypeA}, {Type::TypeF, Type::TypeB}, {Type::TypeF, Type::TypeC}>(
378 lhs, rhs
379 )) {
380 return rhs.intersect(lhs);
381 }
382 return Interval::Empty(f);
383}

385Interval Interval::difference(const Interval &other) const {
386 const Field &f = checkFields(*this, other);
387 // intersect checks that we're in the same field
388 Interval intersection = intersect(other);
389 if (intersection.isEmpty()) {
390 // There's nothing to remove, so just return this
391 return *this;
392 }
393
394 // Trivial cases with a non-empty intersection
395 if (isDegenerate() || other.isEntire()) {
396 return Interval::Empty(f);
397 }
398 if (isEntire()) {
399 // Since we don't support punching arbitrary holes in ranges, we only reduce
400 // entire ranges if other is [0, b] or [a, prime - 1]
401 if (other.a == f.zero()) {
402 return UnreducedInterval(other.b + f.one(), f.maxVal()).reduce(f);
403 }
404 if (other.b == f.maxVal()) {
405 return UnreducedInterval(f.zero(), other.a - f.one()).reduce(f);
406 }
407
408 return *this;
409 }
410
411 // Non-trivial cases
412 // - Internal+internal or external+external cases
413 if ((is<Type::TypeA, Type::TypeB, Type::TypeC>() &&
414 intersection.is<Type::TypeA, Type::TypeB, Type::TypeC>()) ||
415 areOneOf<{Type::TypeF, Type::TypeF}>(*this, intersection)) {
416 // The intersection needs to be at the end of the interval, otherwise we would
417 // split the interval in two, and we aren't set up to support multiple intervals
418 // per value.
419 if (a != intersection.a && b != intersection.b) {
420 return *this;
421 }
422 // Otherwise, remove the intersection and reduce
423 if (a == intersection.a) {
424 return UnreducedInterval(intersection.b + f.one(), b).reduce(f);
425 }
426 // else b == intersection.b
427 return UnreducedInterval(a, intersection.a - f.one()).reduce(f);
428 }
429 // - Mixed internal/external cases. We flip the comparison
430 if (isTypeF()) {
431 if (a != intersection.b && b != intersection.a) {
432 return *this;
433 }
434 // Otherwise, remove the intersection and reduce
435 if (a == intersection.b) {
436 return UnreducedInterval(intersection.a + f.one(), b).reduce(f);
437 }
438 // else b == intersection.a
439 return UnreducedInterval(a, intersection.b - f.one()).reduce(f);
440 }
441
442 // In cases we don't know how to handle, we over-approximate and return
443 // the original interval.
444 return *this;
445}

466Interval operator*(const Interval &lhs, const Interval &rhs) {
467 const Field &f = checkFields(lhs, rhs);
468 auto zeroInterval = Interval::Degenerate(f, f.zero());
469 if (lhs == zeroInterval || rhs == zeroInterval) {
470 return zeroInterval;
471 }
472 if (lhs.isEmpty() || rhs.isEmpty()) {
473 return Interval::Empty(f);
474 }
475 if (lhs.isEntire() || rhs.isEntire()) {
476 return Interval::Entire(f);
477 }
478
479 if (Interval::areOneOf<{Interval::Type::TypeB, Interval::Type::TypeB}>(lhs, rhs)) {
480 return (lhs.secondUnreduced() * rhs.secondUnreduced()).reduce(f);
481 }
482 return (lhs.firstUnreduced() * rhs.firstUnreduced()).reduce(f);
483}

485FailureOr<Interval> feltDiv(const Interval &lhs, const Interval &rhs) {
486 const Field &f = checkFields(lhs, rhs);
487 if (lhs.isEmpty() || rhs.isEmpty()) {
488 return success(Interval::Empty(f));
489 }
490 if (!rhs.isDegenerate() || rhs.lhs() == f.zero()) {
491 // Supporting arbitrary divisor intervals would require enumerating every
492 // possible divisor, inverting each value, and joining the products, which
493 // is too expensive. So, we return a failure in the non-degenerate case
494 // and in the divide-by-zero case.
495 return failure();
496 }
497 return success(lhs * Interval::Degenerate(f, f.inv(rhs.lhs())));
498}

500FailureOr<Interval> unsignedIntDiv(const Interval &lhs, const Interval &rhs) {
501 const Field &f = checkFields(lhs, rhs);
502 if (lhs.isEmpty() || rhs.isEmpty()) {
503 return success(Interval::Empty(f));
504 }
505
506 llvm::SmallVector<UnreducedInterval, 2> lhsParts = getUnsignedCanonicalParts(lhs);
507 llvm::SmallVector<UnreducedInterval, 2> rhsParts = getUnsignedCanonicalParts(rhs);
508 for (const UnreducedInterval &rhsPart : rhsParts) {
509 if (rhsPart.getLHS() == f.zero()) {
510 return failure();
511 }
512 }
513
514 Interval result = Interval::Empty(f);
515 for (const UnreducedInterval &lhsPart : lhsParts) {
516 for (const UnreducedInterval &rhsPart : rhsParts) {
517 result = joinDivisionPiece(
518 f, result, lhsPart.getLHS() / rhsPart.getRHS(), lhsPart.getLHS() / rhsPart.getLHS(),
519 lhsPart.getRHS() / rhsPart.getRHS(), lhsPart.getRHS() / rhsPart.getLHS()
520 );
521 }
522 }
523 return success(result);
524}

526FailureOr<Interval> signedIntDiv(const Interval &lhs, const Interval &rhs) {
527 const Field &f = checkFields(lhs, rhs);
528 if (lhs.isEmpty() || rhs.isEmpty()) {
529 return success(Interval::Empty(f));
530 }
531
532 llvm::SmallVector<UnreducedInterval, 2> lhsParts = getSignedCanonicalParts(lhs);
533 llvm::SmallVector<UnreducedInterval, 2> rhsParts = getSignedCanonicalParts(rhs);
534 for (const UnreducedInterval &rhsPart : rhsParts) {
535 if (containsZero(rhsPart)) {
536 return failure();
537 }
538 }
539
540 Interval result = Interval::Empty(f);
541 for (const UnreducedInterval &lhsPart : lhsParts) {
542 for (const UnreducedInterval &rhsPart : rhsParts) {
543 result = joinDivisionPiece(
544 f, result, lhsPart.getLHS() / rhsPart.getLHS(), lhsPart.getLHS() / rhsPart.getRHS(),
545 lhsPart.getRHS() / rhsPart.getLHS(), lhsPart.getRHS() / rhsPart.getRHS()
546 );
547 }
548 }
549 return success(result);
550}

557Interval operator&(const Interval &lhs, const Interval &rhs) {
558 const Field &f = checkFields(lhs, rhs);
559 if (lhs.isEmpty() || rhs.isEmpty()) {
560 return Interval::Empty(f);

563 return Interval::Degenerate(f, lhs.a & rhs.a);
564 } else if (lhs.isDegenerate()) {
565 return UnreducedInterval(f.zero(), lhs.a).reduce(f);
566 } else if (rhs.isDegenerate()) {
567 return UnreducedInterval(f.zero(), rhs.a).reduce(f);
568 }
569 return Interval::Entire(f);

572Interval operator|(const Interval &lhs, const Interval &rhs) {
573 const Field &f = checkFields(lhs, rhs);
574 if (lhs.isEmpty() || rhs.isEmpty()) {
575 return Interval::Empty(f);
576 }
577 auto zeroInterval = Interval::Degenerate(f, f.zero());
578 if (lhs == zeroInterval) {
579 return rhs;
580 }
581 if (rhs == zeroInterval) {
582 return lhs;
583 }
584 if (lhs.isDegenerate() && rhs.isDegenerate()) {
585 return Interval::Degenerate(f, f.reduce(lhs.a | rhs.a));
586 }
587 return Interval::Entire(f);

608Interval operator<<(const Interval &lhs, const Interval &rhs) {
609 const Field &f = checkFields(lhs, rhs);
610 if (lhs.isEmpty() || rhs.isEmpty()) {
611 return Interval::Empty(f);
612 }
613 if (lhs.isDegenerate() && rhs.isDegenerate()) {
614 if (rhs.a > f.bitWidth()) {
615 return Interval::Entire(f);
616 }
617
618 DynamicAPInt v = lhs.a << rhs.a;
619 return UnreducedInterval(v, v).reduce(f);
620 }
621 return Interval::Entire(f);
622}

624Interval operator>>(const Interval &lhs, const Interval &rhs) {
625 const Field &f = checkFields(lhs, rhs);
626 if (lhs.isEmpty() || rhs.isEmpty()) {
627 return Interval::Empty(f);
628 }
629 if (lhs.isDegenerate() && rhs.isDegenerate()) {
630 if (rhs.a > f.bitWidth()) {
631 return Interval::Degenerate(f, f.zero());
632 }
633
634 return Interval::Degenerate(f, lhs.a >> rhs.a);
635 }
636 return Interval::Entire(f);
637}

639DynamicAPInt Interval::width() const {
640 switch (ty) {
641 case Type::Empty:
642 return field.get().zero();
643 case Type::Degenerate:
644 return field.get().one();
645 case Type::Entire:
646 return field.get().prime();
647 default:
648 return field.get().reduce(toUnreduced().width());
649 }
650}

652Interval boolAnd(const Interval &lhs, const Interval &rhs) {
653 ensure(
654 lhs.getField() == rhs.getField(), "interval operations across differing fields is unsupported"
655 );
656 ensure(lhs.isBoolean() && rhs.isBoolean(), "operation only supported for boolean-type intervals");
657 const auto &field = rhs.getField();
658
659 if (lhs.isBoolFalse() || rhs.isBoolFalse()) {
660 return Interval::False(field);

700 return Interval::False(field);
701 }
702 if (lhs.isBoolTrue() || rhs.isBoolTrue()) {
703 return Interval::True(field);

706 return Interval::False(field);
707 }
708

718 }
719 if (iv.isBoolFalse()) {
