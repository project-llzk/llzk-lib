43class SourceRefIndex {
44 using IndexRange = std::pair<llvm::DynamicAPInt, llvm::DynamicAPInt>;
45
46public:
47 explicit SourceRefIndex(component::MemberDefOp f) : index(f) {}
48 explicit SourceRefIndex(SymbolLookupResult<component::MemberDefOp> f) : index(f) {}
49 explicit SourceRefIndex(mlir::StringAttr recordName) : index(recordName) {}
50 explicit SourceRefIndex(const llvm::DynamicAPInt &i) : index(i) {}
51 explicit SourceRefIndex(const llvm::APInt &i) : index(toDynamicAPInt(i)) {}
52 explicit SourceRefIndex(int64_t i) : index(llvm::DynamicAPInt(i)) {}
53 SourceRefIndex(const llvm::APInt &low, const llvm::APInt &high)
54 : index(IndexRange {toDynamicAPInt(low), toDynamicAPInt(high)}) {}
55 explicit SourceRefIndex(IndexRange r) : index(r) {}
56
57 bool isMember() const {
58 return std::holds_alternative<SymbolLookupResult<component::MemberDefOp>>(index) ||
59 std::holds_alternative<component::MemberDefOp>(index);
60 }
61 component::MemberDefOp getMember() const {
62 ensure(isMember(), "SourceRefIndex: member requested but not contained");
63 if (std::holds_alternative<component::MemberDefOp>(index)) {
64 return std::get<component::MemberDefOp>(index);
65 }
66 return std::get<SymbolLookupResult<component::MemberDefOp>>(index).get();
67 }
68
69 bool isPodRecord() const { return std::holds_alternative<mlir::StringAttr>(index); }
70 mlir::StringAttr getPodRecordNameAttr() const {
71 ensure(isPodRecord(), "SourceRefIndex: pod record requested but not contained");
72 return std::get<mlir::StringAttr>(index);
73 }
74 llvm::StringRef getPodRecordName() const { return getPodRecordNameAttr().getValue(); }
75
76 bool isIndex() const { return std::holds_alternative<llvm::DynamicAPInt>(index); }
77 llvm::DynamicAPInt getIndex() const {
78 ensure(isIndex(), "SourceRefIndex: index requested but not contained");
79 return std::get<llvm::DynamicAPInt>(index);
80 }
81
82 bool isIndexRange() const { return std::holds_alternative<IndexRange>(index); }
83 IndexRange getIndexRange() const {
84 ensure(isIndexRange(), "SourceRefIndex: index range requested but not contained");
85 return std::get<IndexRange>(index);
86 }
87
88 inline void dump() const { print(llvm::errs()); }
89 void print(mlir::raw_ostream &os) const;
90
91 inline bool operator==(const SourceRefIndex &rhs) const {
92 if (isMember() && rhs.isMember()) {
93 // We compare the underlying members, since the member could be in a symbol
94 // lookup or not.
95 return getMember() == rhs.getMember();
96 }
97 if (isIndex() && rhs.isIndex()) {
98 return getIndex() == rhs.getIndex();
99 }
100 return index == rhs.index;
101 }
102
103 std::strong_ordering operator<=>(const SourceRefIndex &rhs) const;
104
105 struct Hash {
106 size_t operator()(const SourceRefIndex &c) const;
107 };
108
109 inline size_t getHash() const { return Hash {}(*this); }
110
111private:
119 std::variant<
120 component::MemberDefOp, SymbolLookupResult<component::MemberDefOp>, mlir::StringAttr,
121 llvm::DynamicAPInt, IndexRange>
122 index;
123};

126 rhs.print(os);
127 return os;

143public:
144 using Path = std::vector<SourceRefIndex>;
145
146private:
147 // Sort in the following order:
148 // block arg < struct.new < nondet < other rooted result < template const < const index <
149 // const felt.
150 enum class SortCategory : std::uint8_t {
151 BlockArgument,
152 CreateStruct,
153 NonDet,
154 RootResult,
155 TemplateConstant,
156 ConstantIndex,
157 ConstantFelt,
158 };
159
160 template <typename OpT> static mlir::Value getSingleResultValue(OpT op) {
161 ensure(op, "SourceRef requires a non-null operation");
162 ensure(op->getNumResults() == 1, "SourceRef expects a single-result operation");
163 return op->getResult(0);
164 }
165
166 static mlir::Value getRootResultValue(mlir::OpResult result) {
167 ensure(
168 !llvm::isa<
169 felt::FeltConstantOp, mlir::arith::ConstantIndexOp, mlir::arith::ConstantIntOp,
170 polymorphic::ConstReadOp>(result.getOwner()),
171 "SourceRef rooted OpResult constructors must not be used for constant values"
172 );
173 return result;
174 }
175
176 template <typename OpT> mlir::FailureOr<OpT> getDefiningOp() const {
177 if (auto op = llvm::dyn_cast_if_present<OpT>(value.getDefiningOp())) {
178 return op;
179 }
180 return mlir::failure();
181 }
182
183 SourceRef(mlir::Value sourceValue, bool isConstantStorage, Path sourcePath = {})
184 : value(sourceValue), path(std::move(sourcePath)), constant(isConstantStorage) {
185 ensure(value != nullptr, "SourceRef requires a non-null value");
186 ensure(!constant || this->path.empty(), "constant SourceRef cannot have a path");
187 }
188
189 Path &getPathMut() { return path; }
190 const void *getAsOpaquePointer() const { return value.getAsOpaquePointer(); }
191 SortCategory getSortCategory() const;
192 llvm::StringRef getTemplateConstantName() const;
193 std::strong_ordering compareWithinCategory(const SourceRef &rhs, SortCategory category) const;
194
195public:
197 static std::vector<SourceRef>
198 getAllSourceRefs(mlir::SymbolTableCollection &tables, mlir::ModuleOp mod, const SourceRef &root);
199
201 static std::vector<SourceRef>
202 getAllSourceRefs(component::StructDefOp structDef, function::FuncDefOp fnOp);

206 static std::vector<SourceRef>
207 getAllSourceRefs(component::StructDefOp structDef, component::MemberDefOp memberDef);

209 /* Rooted constructors */
210
211 SourceRef(mlir::BlockArgument b, Path p = {})
212 : SourceRef(b, /*isConstantStorage=*/false, std::move(p)) {}
213 SourceRef(component::CreateStructOp createOp, Path p = {})
214 : SourceRef(getSingleResultValue(createOp), /*isConstantStorage=*/false, std::move(p)) {}
215 SourceRef(NonDetOp nondet, Path p = {})
216 : SourceRef(getSingleResultValue(nondet), /*isConstantStorage=*/false, std::move(p)) {}
217 SourceRef(mlir::OpResult rootResult, Path p = {})
218 : SourceRef(getRootResultValue(rootResult), /*isConstantStorage=*/false, std::move(p)) {}

220 /* Constant constructors */

223 : SourceRef(getSingleResultValue(c), /*isConstantStorage=*/true) {}
224 explicit SourceRef(mlir::arith::ConstantIndexOp c)
225 : SourceRef(getSingleResultValue(c), /*isConstantStorage=*/true) {}

227 : SourceRef(getSingleResultValue(c), /*isConstantStorage=*/true) {}

229 mlir::Type getType() const;

231 bool isConstantFelt() const {
232 return isConstant() && llvm::isa_and_present<felt::FeltConstantOp>(value.getDefiningOp());
233 }
234 bool isConstantIndex() const {
235 return isConstant() &&
236 llvm::isa_and_present<mlir::arith::ConstantIndexOp>(value.getDefiningOp());
237 }
238 bool isTemplateConstant() const {
239 return isConstant() && llvm::isa_and_present<polymorphic::ConstReadOp>(value.getDefiningOp());
240 }
241
242 bool isConstant() const { return constant; }
243 bool isConstantInt() const { return isConstantFelt() || isConstantIndex(); }
244
245 bool isFeltVal() const { return llvm::isa<felt::FeltType>(getType()); }
246 bool isIndexVal() const { return llvm::isa<mlir::IndexType>(getType()); }
247 bool isIntegerVal() const { return llvm::isa<mlir::IntegerType>(getType()); }
248 bool isTypeVarVal() const { return llvm::isa<polymorphic::TypeVarType>(getType()); }
249 bool isScalar() const {
250 return isConstant() || isFeltVal() || isIndexVal() || isIntegerVal() || isTypeVarVal();
251 }
252
253 bool isRooted() const { return !constant; }
254 bool isBlockArgument() const { return isRooted() && llvm::isa<mlir::BlockArgument>(value); }
255 mlir::FailureOr<mlir::Value> getRoot() const {
256 if (isRooted()) {
257 return value;

262 if (isConstant()) {

265 return mlir::failure();
266 }
267 mlir::FailureOr<mlir::BlockArgument> getBlockArgument() const {
268 if (auto blockArg = llvm::dyn_cast<mlir::BlockArgument>(value)) {
269 return blockArg;
270 }
271 return mlir::failure();
272 }
273 mlir::FailureOr<unsigned> getInputNum() const {
274 auto blockArg = getBlockArgument();
275 if (succeeded(blockArg)) {
276 return blockArg->getArgNumber();
277 }
278 return mlir::failure();
279 }
280
281 bool isCreateStructOp() const { return succeeded(getCreateStructOp()); }

283 return getDefiningOp<component::CreateStructOp>();

286 bool isNonDetOp() const { return succeeded(getNonDetOp()); }
287 mlir::FailureOr<NonDetOp> getNonDetOp() const { return getDefiningOp<NonDetOp>(); }
288
289 bool isCallResult() const { return succeeded(getCallOp()); }
290 mlir::FailureOr<function::CallOp> getCallOp() const { return getDefiningOp<function::CallOp>(); }
291
292 mlir::FailureOr<llvm::DynamicAPInt> getConstantFeltValue() const {
293 auto feltConst = getDefiningOp<felt::FeltConstantOp>();
294 if (succeeded(feltConst)) {
295 llvm::APInt i = feltConst->getValue();
296 return toDynamicAPInt(i);
297 }
298 return mlir::failure();
299 }
300 mlir::FailureOr<llvm::DynamicAPInt> getConstantIndexValue() const {
301 auto indexConst = getDefiningOp<mlir::arith::ConstantIndexOp>();
302 if (succeeded(indexConst)) {
303 return llvm::DynamicAPInt(indexConst->value());
304 }
305 return mlir::failure();
306 }
307 mlir::FailureOr<llvm::DynamicAPInt> getConstantValue() const {
308 auto feltVal = getConstantFeltValue();
309 if (succeeded(feltVal)) {
310 return *feltVal;
311 }
312 auto indexVal = getConstantIndexValue();
313 if (succeeded(indexVal)) {
314 return *indexVal;
315 }
316 return mlir::failure();
317 }

429 return llvm::hash_value(ref.getAsOpaquePointer());
430 }
431 return llzk::SourceRef::Hash {}(ref);
432 }
