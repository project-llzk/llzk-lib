46std::pair<SourceRefLatticeValue, mlir::ChangeResult>
47SourceRefLatticeValue::translate(const TranslationMap &translation) const {
48 auto newVal = *this;
49 auto res = mlir::ChangeResult::NoChange;
50 if (newVal.isScalar()) {
51 res = newVal.translateScalar(translation);
52 } else {
53 for (auto &elem : newVal.getArrayValue()) {
54 auto [newElem, elemRes] = elem->translate(translation);
55 (*elem) = newElem;
56 res |= elemRes;
57 }
58 }
59 return {newVal, res};
60}

69std::pair<SourceRefLatticeValue, mlir::ChangeResult>
70SourceRefLatticeValue::extract(const std::vector<SourceRefIndex> &indices) const {
71 if (isArray()) {
72 ensure(indices.size() <= getNumArrayDims(), "invalid extract array operands");
73
74 // First, compute what chunk(s) to index
75 std::vector<size_t> currIdxs {0};
76 for (unsigned i = 0; i < indices.size(); i++) {
77 const auto &idx = indices[i];
78 auto currDim = getArrayDim(i);
79
80 std::vector<size_t> newIdxs;
81 ensure(idx.isIndex() || idx.isIndexRange(), "wrong type of index for array");
82 if (idx.isIndex()) {
83 int64_t idxVal(idx.getIndex());
84 std::transform(
85 currIdxs.begin(), currIdxs.end(), std::back_inserter(newIdxs),
86 [&currDim, &idxVal](size_t j) { return j * currDim + idxVal; }
87 );
88 } else {
89 auto [low, high] = idx.getIndexRange();
90 int64_t lowInt(low), highInt(high);
91 for (int64_t idxVal = lowInt; idxVal < highInt; idxVal++) {
92 std::transform(
93 currIdxs.begin(), currIdxs.end(), std::back_inserter(newIdxs),
94 [&currDim, &idxVal](size_t j) { return j * currDim + idxVal; }
95 );
96 }
97 }
98
99 currIdxs = newIdxs;
100 }
101 std::vector<int64_t> newArrayDims;
102 size_t chunkSz = 1;
103 for (unsigned i = indices.size(); i < getNumArrayDims(); i++) {
104 auto dim = getArrayDim(i);
105 newArrayDims.push_back(dim);
106 chunkSz *= dim;
107 }
108 if (newArrayDims.empty()) {
109 // read case, where the return value is a scalar (single element)
110 SourceRefLatticeValue extractedVal;
111 for (auto idx : currIdxs) {
112 (void)extractedVal.update(getElemFlatIdx(idx));
113 }
114 return {extractedVal, mlir::ChangeResult::Change};
115 } else {
116 // extract case, where the return value is an array of fewer dimensions.
117 SourceRefLatticeValue extractedVal(newArrayDims);
118 for (auto chunkStart : currIdxs) {
119 for (size_t i = 0; i < chunkSz; i++) {
120 (void)extractedVal.getElemFlatIdx(i).update(getElemFlatIdx(chunkStart + i));
121 }
122 }
123 return {extractedVal, mlir::ChangeResult::Change};
124 }
125 } else {
126 auto currVal = *this;
127 auto res = mlir::ChangeResult::NoChange;
128 for (const auto &idx : indices) {
129 auto transform = [&idx](const SourceRef &r) -> SourceRef { return r.createChild(idx); };
130 auto [newVal, transformRes] = currVal.elementwiseTransform(transform);
131 currVal = newVal;
132 res |= transformRes;
133 }
134 return {currVal, res};
135 }
136}

138mlir::ChangeResult SourceRefLatticeValue::translateScalar(const TranslationMap &translation) {
139 auto res = mlir::ChangeResult::NoChange;
140 // copy the current value
141 auto currVal = getScalarValue();
142 // reset this value
143 getValue() = ScalarTy();
144 // For each current element, see if the translation map contains a valid prefix.
145 // If so, translate the current element with all replacement prefixes indicated
146 // by the translation value.
147 for (const SourceRef &currRef : currVal) {
148 for (const auto &[prefix, replacementVal] : translation) {
149 if (currRef.isValidPrefix(prefix)) {
150 for (const SourceRef &replacementPrefix : replacementVal.foldToScalar()) {
151 auto translatedRefRes = currRef.translate(prefix, replacementPrefix);
152 if (succeeded(translatedRefRes)) {
153 res |= insert(*translatedRefRes);
154 }
155 }
156 }
157 }
158 }
159 return res;
160}

162std::pair<SourceRefLatticeValue, mlir::ChangeResult> SourceRefLatticeValue::elementwiseTransform(
163 llvm::function_ref<SourceRef(const SourceRef &)> transform
164) const {
165 auto newVal = *this;
166 auto res = mlir::ChangeResult::NoChange;
167 if (newVal.isScalar()) {
168 ScalarTy indexed;
169 for (const auto &ref : newVal.getScalarValue()) {
170 auto [_, inserted] = indexed.insert(transform(ref));
171 if (inserted) {
172 res |= mlir::ChangeResult::Change;
173 }
174 }
175 newVal.getScalarValue() = indexed;
176 } else {
177 for (auto &elem : newVal.getArrayValue()) {
178 auto [newElem, elemRes] = elem->elementwiseTransform(transform);
179 (*elem) = newElem;
180 res |= elemRes;
181 }
182 }
183 return {newVal, res};
184}

193mlir::FailureOr<SourceRef> SourceRefLattice::getSourceRef(mlir::Value val) {
194 if (auto blockArg = llvm::dyn_cast<mlir::BlockArgument>(val)) {
195 return SourceRef(blockArg);
196 } else if (auto *defOp = val.getDefiningOp()) {
197 if (auto feltConst = llvm::dyn_cast<FeltConstantOp>(defOp)) {
198 return SourceRef(feltConst);
199 } else if (auto constIdx = llvm::dyn_cast<mlir::arith::ConstantIndexOp>(defOp)) {
200 return SourceRef(constIdx);
201 } else if (auto readConst = llvm::dyn_cast<ConstReadOp>(defOp)) {
202 return SourceRef(readConst);
203 } else if (auto structNew = llvm::dyn_cast<CreateStructOp>(defOp)) {
204 return SourceRef(structNew);
205 }
206 }
207 return mlir::failure();
208}

210void SourceRefLattice::print(mlir::raw_ostream &os) const {
211 os << "SourceRefLattice { ";
212 for (auto mit = valMap.begin(); mit != valMap.end();) {
213 const auto &[val, latticeVal] = *mit;
214 os << "\n (";
215 if (auto asVal = llvm::dyn_cast<Value>(val)) {
216 os << asVal;
217 } else if (auto *asOp = llvm::dyn_cast<Operation *>(val)) {
218 os << *asOp;
219 } else {
220 llvm_unreachable("unhandled ValueTy print case");
221 }
222 os << ") => " << latticeVal;
223 mit++;
224 if (mit != valMap.end()) {
225 os << ',';
226 } else {
227 os << '\n';
228 }
229 }
230 os << "}\n";
231}

253SourceRefLatticeValue SourceRefLattice::getOrDefault(SourceRefLattice::ValueTy v) const {
254 auto it = valMap.find(v);
255 if (it != valMap.end()) {
256 return it->second;
257 }
258
259 if (auto asVal = llvm::dyn_cast_if_present<Value>(v)) {
260 auto sourceRef = getSourceRef(asVal);
261 if (mlir::succeeded(sourceRef)) {
262 return SourceRefLatticeValue(*sourceRef);
263 }
264 }
265 return SourceRefLatticeValue();
266}

268SourceRefLatticeValue SourceRefLattice::getReturnValue(unsigned i) const {
269 ProgramPoint *pp = llvm::cast<ProgramPoint *>(this->getAnchor());
270 if (auto retOp = mlir::dyn_cast_if_present<function::ReturnOp>(pp->getPrevOp())) {
271 if (i >= retOp.getNumOperands()) {
272 llvm::report_fatal_error("return value requested is out of range");
273 }
274 return this->getOrDefault(retOp.getOperand(i));
275 }
276 return SourceRefLatticeValue();
277}
