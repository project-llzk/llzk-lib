47// Default to true if the check is not for that particular operation type.
48template <typename Check> inline bool runCheck(mlir::Operation *op, Check check) {
49 if (auto specificOp =
50 llvm::dyn_cast_if_present<typename llvm::function_traits<Check>::template arg_t<0>>(op)) {
51 return check(specificOp);
52 }
53 return true;
54}

94inline InstantiationLayout buildInstantiationLayout(
95 TemplateOp parentTemplate, mlir::ArrayAttr callParams,
96 const llvm::DenseMap<mlir::Attribute, mlir::Attribute> &paramNameToConcrete
97) {
98 mlir::SmallVector<mlir::Attribute> remainingNames;
99 mlir::SmallVector<mlir::Attribute> attrsForInstantiatedNameSuffix;
100 for (mlir::Attribute paramName : parentTemplate.getConstNames<TemplateParamOp>()) {
101 auto it = paramNameToConcrete.find(paramName);
102 if (it != paramNameToConcrete.end()) {
103 attrsForInstantiatedNameSuffix.push_back(it->second);
104 } else {
105 attrsForInstantiatedNameSuffix.push_back(nullptr);
106 remainingNames.push_back(paramName);
107 }
108 }
109
110 mlir::ArrayAttr rewrittenCallParams = nullptr;
111 if (!isNullOrEmpty(callParams) && !remainingNames.empty()) {
112 mlir::SmallVector<mlir::Attribute> remainingCallParams;
113 for (auto [paramOp, attr] :
114 llvm::zip_equal(parentTemplate.getConstOps<TemplateParamOp>(), callParams.getValue())) {
115 auto paramName = mlir::FlatSymbolRefAttr::get(paramOp.getSymNameAttr());
116 if (!paramNameToConcrete.contains(paramName)) {
117 remainingCallParams.push_back(attr);
118 }
119 }
120 rewrittenCallParams = mlir::ArrayAttr::get(parentTemplate.getContext(), remainingCallParams);
121 }
122
123 return {
124 std::move(remainingNames),
125 BuildShortTypeString::from(parentTemplate.getSymName().str(), attrsForInstantiatedNameSuffix),
126 rewrittenCallParams,
127 };
128}

165template <typename... AdditionalOpClasses, typename... AdditionalChecks>
166mlir::ConversionTarget newConverterDefinedTargetWithCallback(
167 mlir::TypeConverter &tyConv, mlir::MLIRContext *ctx, LegalityCheckCallback &cb,
168 AdditionalChecks &&...checks
169) {
170 mlir::ConversionTarget target = newBaseTarget(ctx);
171 auto inserter = [&](auto... opClasses) {
172 target.addDynamicallyLegalOp<decltype(opClasses)...>([&cb, &tyConv,
173 &checks...](mlir::Operation *op) {
174 LLVM_DEBUG(if (op) {
175 llvm::dbgs() << "[newConverterDefinedTarget] checking legality of ";
176 op->dump();
177 });
178 cb.checkStarted();
179 auto legality =
180 defaultLegalityCheck(tyConv, op) && (runCheck<AdditionalChecks>(op, checks) && ...);
181
182 cb.checkEnded(legality);
183 LLVM_DEBUG(if (legality) { llvm::dbgs() << "[newConverterDefinedTarget] is legal\n"; } else {
184 llvm::dbgs() << "[newConverterDefinedTarget] is not legal\n";
185 });
186 return legality;
187 });
188 };
189 std::apply(inserter, OpClassesWithStructTypes.NoGeneralBuilder);
190 std::apply(inserter, OpClassesWithStructTypes.WithGeneralBuilder);
191 applyToMoreTypes<decltype(inserter), AdditionalOpClasses...>(inserter);
192 return target;
193}
