111template <typename Context>
112concept ContextType = requires(const Context &a, const Context &b) {
113 { a == b } -> std::convertible_to<bool>;
114 { std::hash<Context> {}(a) } -> std::convertible_to<std::size_t>;
115};

123template <typename Analysis, typename Result, typename Context>
124concept StructAnalysisType = requires {
125 requires std::is_base_of<StructAnalysis<Result, Context>, Analysis>::value;
126 requires ContextType<Context>;

214 void constructChildAnalyses(mlir::AnalysisManager &am) {
215 auto init = dataflow::loadAndRunRequiredAnalyses(solver, modOp);
216 ensure(init.succeeded(), "solver failed to run on module!");
217
218 // The analysis is run at the module level so that lattices are computed
219 // for global functions as well.
220 initializeSolver();
221 auto res = solver.initializeAndRun(modOp);
222 ensure(res.succeeded(), "solver failed to run on module!");
223
224 const Context &ctx = getContext();
225 // Force construction of empty results here so `getCurrentResults()` on
226 // a module with no inner structs returns no results rather than an assertion
227 // failure.
228 results[ctx] = {};
229 modOp.walk([this, &am, &ctx](component::StructDefOp s) mutable {
230 auto &childAnalysis = am.getChildAnalysis<StructAnalysisTy>(s);
231 // Don't re-run the analysis if we already have the results.
232 // The analysis may have been run as part of a nested analysis.
233 if (!childAnalysis.constructed(ctx)) {
234 mlir::LogicalResult childAnalysisRes = childAnalysis.runAnalysis(solver, am, ctx);
235
236 if (mlir::failed(childAnalysisRes)) {
237 auto error_message = "StructAnalysis failed to run for " + s.getName();
238 s->emitError(error_message).report();
239 llvm::report_fatal_error(error_message);
240 }
241 }
242
243 auto [_, inserted] = results[ctx].insert(
244 std::make_pair(s, std::reference_wrapper(childAnalysis.getResult(ctx)))
245 );
246 ensure(inserted, "struct location conflict");
247 return mlir::WalkResult::skip();
248 });
249 }
250
251private:
252 mlir::ModuleOp modOp;
253 ResultMap results;
254
257 void ensureResultCreated(component::StructDefOp op) const {
258 ensure(hasResult(op), "Result does not exist for StructDefOp " + mlir::Twine(op.getName()));
259 }
260};
