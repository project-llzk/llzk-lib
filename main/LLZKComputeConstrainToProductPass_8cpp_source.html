47bool isValidRoot(StructDefOp root) {
48 FuncDefOp computeFunc = root.getComputeFuncOp();
49 FuncDefOp constrainFunc = root.getConstrainFuncOp();
50
51 if (!computeFunc || !constrainFunc) {
52 root->emitError()
53 .append("no ", FUNC_NAME_COMPUTE, "/", FUNC_NAME_CONSTRAIN, " to align")
54 .report();
55 return false;
56 }
57
58
60
61 return true;
62}

64LogicalResult alignStartingAt(
65 component::StructDefOp root, SymbolTableCollection &tables,
66 LightweightSignalEquivalenceAnalysis &equivalence
67) {
68 if (!isValidRoot(root)) {
69 return failure();
70 }
71
72 ProductAligner aligner {tables, equivalence};
73 if (!aligner.alignFuncs(root, root.getComputeFuncOp(), root.getConstrainFuncOp())) {
74 return failure();
75 }
76
77 return success();
78}

95 if (structDef.getSymName() == rootStruct) {

106FuncDefOp ProductAligner::alignFuncs(StructDefOp root, FuncDefOp compute, FuncDefOp constrain) {
107 OpBuilder funcBuilder(compute);
108
109 // Add compute/constrain attributes
110 compute.walk([&funcBuilder](Operation *op) {
111 op->setAttr(PRODUCT_SOURCE, funcBuilder.getStringAttr(FUNC_NAME_COMPUTE));
112 });
113
114 constrain.walk([&funcBuilder](Operation *op) {
115 op->setAttr(PRODUCT_SOURCE, funcBuilder.getStringAttr(FUNC_NAME_CONSTRAIN));
116 });
117
118 // Create an empty @product func...
119 FuncDefOp productFunc = funcBuilder.create<FuncDefOp>(
120 funcBuilder.getFusedLoc({compute.getLoc(), constrain.getLoc()}), FUNC_NAME_PRODUCT,
121 compute.getFunctionType()
122 );
123 productFunc->setAttr(DERIVED_ATTR_NAME, UnitAttr::get(funcBuilder.getContext()));
124 Block *entryBlock = productFunc.addEntryBlock();
125 funcBuilder.setInsertionPointToStart(entryBlock);
126
127 // ...with the right arguments
128 llvm::SmallVector<Value> args {productFunc.getArguments()};
129
130 // Add calls to @compute and @constrain...
131 CallOp computeCall = funcBuilder.create<CallOp>(funcBuilder.getUnknownLoc(), compute, args);
132 args.insert(args.begin(), computeCall->getResult(0));
133 CallOp constrainCall = funcBuilder.create<CallOp>(funcBuilder.getUnknownLoc(), constrain, args);
134 funcBuilder.create<ReturnOp>(funcBuilder.getUnknownLoc(), computeCall->getResult(0));
135
136 // ..and inline them
137 InlinerInterface inliner(productFunc.getContext());
138 if (failed(inlineCall(inliner, computeCall, compute, &compute.getBody(), true))) {
139 root->emitError().append("failed to inline ", FUNC_NAME_COMPUTE).report();
140 return nullptr;
141 }
142 if (failed(inlineCall(inliner, constrainCall, constrain, &constrain.getBody(), true))) {
143 root->emitError().append("failed to inline ", FUNC_NAME_CONSTRAIN).report();
144 return nullptr;
145 }
146 computeCall->erase();
147 constrainCall->erase();
148
149 // Mark the compute/constrain for deletion
150 alignedStructs.push_back(root);
151
152 // Make sure we can align sub-calls to @compute and @constrain
153 if (failed(alignCalls(productFunc))) {
154 return nullptr;
155 }
156 return productFunc;
157}

159LogicalResult ProductAligner::alignCalls(FuncDefOp product) {
160 // Gather up all the remaining calls to @compute and @constrain
161 llvm::SetVector<CallOp> computeCalls, constrainCalls;
162 product.walk([&](CallOp callOp) {
163 if (callOp.calleeIsStructCompute()) {
164 computeCalls.insert(callOp);
165 } else if (callOp.calleeIsStructConstrain()) {
166 constrainCalls.insert(callOp);
167 }
168 });
169
170 llvm::SetVector<std::pair<CallOp, CallOp>> alignedCalls;
171
172 // A @compute matches a @constrain if they belong to the same struct and all their input signals
173 // are pairwise equivalent
174 auto doCallsMatch = [&](CallOp compute, CallOp constrain) -> bool {
175 LLVM_DEBUG({
176 llvm::outs() << "Asking for equivalence between calls\n"
177 << compute << "\nand\n"
178 << constrain << "\n\n";
179 llvm::outs() << "In block:\n\n" << *compute->getBlock() << "\n";
180 });
181
182 auto computeStruct = getPrefixAsSymbolRefAttr(compute.getCallee());
183 auto constrainStruct = getPrefixAsSymbolRefAttr(constrain.getCallee());
184 if (computeStruct != constrainStruct) {
185 return false;
186 }
187 for (unsigned i = 0, e = compute->getNumOperands() - 1; i < e; i++) {
188 if (!equivalence.areSignalsEquivalent(compute->getOperand(i), constrain->getOperand(i + 1))) {
189 return false;
190 }
191 }
192
193 return true;
194 };
195
196 for (auto compute : computeCalls) {
197 // If there is exactly one @compute that matches a given @constrain, we can align them
198 auto matches = llvm::filter_to_vector(constrainCalls, [&](CallOp constrain) {
199 return doCallsMatch(compute, constrain);
200 });
201
202 if (matches.size() == 1) {
203 alignedCalls.insert({compute, matches[0]});
204 computeCalls.remove(compute);
205 constrainCalls.remove(matches[0]);
206 }
207 }
208
209 if (!computeCalls.empty() && constrainCalls.empty()) {
210 product.emitWarning()
211 .append("failed to align some @", FUNC_NAME_COMPUTE, " and @", FUNC_NAME_CONSTRAIN)
212 .report();
213 }
214
215 for (auto [compute, constrain] : alignedCalls) {
216 // If @A::@compute matches @A::@constrain, recursively align the functions in @A...
217 auto calleeTgt = compute.getCalleeTarget(tables);
218 if (failed(calleeTgt)) {
219 return failure();
220 }
221 auto newRoot = calleeTgt->get()->getParentOfType<StructDefOp>();
222 assert(newRoot);
223 FuncDefOp newProduct =
224 alignFuncs(newRoot, newRoot.getComputeFuncOp(), newRoot.getConstrainFuncOp());
225 if (!newProduct) {
226 return failure();
227 }
228
229 // ...and replace the two calls with a single call to @A::@product
230 OpBuilder callBuilder(compute);
231 CallOp newCall = callBuilder.create<CallOp>(
232 callBuilder.getFusedLoc({compute.getLoc(), constrain.getLoc()}), newProduct,
233 compute.getOperands()
234 );
235 compute->replaceAllUsesWith(newCall.getResults());
236 }
237
238 return success();
239}
