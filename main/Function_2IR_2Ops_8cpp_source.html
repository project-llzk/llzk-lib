77static LogicalResult verifyArgOrResNameAttrs(
78 ArrayAttr attrs, StringRef ownAttrName, StringRef crossAttrName, StringRef ownLabel,
79 StringRef crossLabel, EmitErrorFn emitFn
80) {
81 if (!attrs) {
82 return success();
83 }
84 llvm::DenseSet<StringAttr> seenNames;
85 for (auto [i, attr] : llvm::enumerate(attrs)) {
86 auto dictAttr = llvm::dyn_cast<DictionaryAttr>(attr);
87 if (!dictAttr) {
88 continue;
89 }
90 if (dictAttr.contains(crossAttrName)) {
91 return emitFn().append(
92 '\'', crossAttrName, "' is only valid on function ", crossLabel, "s but found on ",
93 ownLabel, ' ', i
94 );
95 }
96 Attribute nameAttr = dictAttr.get(ownAttrName);
97 if (!nameAttr) {
98 continue;
99 }
100 auto name = llvm::dyn_cast<StringAttr>(nameAttr);
101 if (!name) {
102 return emitFn().append(
103 '\'', ownAttrName, "' on ", ownLabel, ' ', i, " must be a string attribute"
104 );
105 }
106 if (!llvm::isa<NoneType>(name.getType())) {
107 return emitFn().append(
108 '\'', ownAttrName, "' on ", ownLabel, ' ', i, " must not have an explicit type"
109 );
110 }
111 if (name.getValue().empty()) {
112 return emitFn().append('\'', ownAttrName, "' on ", ownLabel, ' ', i, " must not be empty");
113 }
114 if (!seenNames.insert(name).second) {
115 return emitFn().append(
116 "duplicate '", ownAttrName, "' value \"", name.getValue(), "\" on ", ownLabel, ' ', i
117 );
118 }
119 }
120 return success();
121}

202void FuncDefOp::cloneInto(FuncDefOp dest, IRMapping &mapper) {
203 // Add the attributes of this function to dest.
204 llvm::MapVector<StringAttr, Attribute> newAttrMap;
205 for (const auto &attr : dest->getAttrs()) {
206 newAttrMap.insert({attr.getName(), attr.getValue()});
207 }
208 for (const auto &attr : (*this)->getAttrs()) {
209 newAttrMap.insert({attr.getName(), attr.getValue()});
210 }
211
212 auto newAttrs =
213 llvm::to_vector(llvm::map_range(newAttrMap, [](std::pair<StringAttr, Attribute> attrPair) {
214 return NamedAttribute(attrPair.first, attrPair.second);
215 }));
216 dest->setAttrs(DictionaryAttr::get(getContext(), newAttrs));
217
218 // Clone the body.
219 getBody().cloneInto(&dest.getBody(), mapper);
220}

227FuncDefOp FuncDefOp::clone(IRMapping &mapper) {
228 // Create the new function.
229 FuncDefOp newFunc = llvm::cast<FuncDefOp>(getOperation()->cloneWithoutRegions());
230
231 // If the function has a body, then the user might be deleting arguments to
232 // the function by specifying them in the mapper. If so, we don't add the
233 // argument to the input type vector.
234 if (!isExternal()) {
235 FunctionType oldType = getFunctionType();
236
237 unsigned oldNumArgs = oldType.getNumInputs();
238 SmallVector<Type, 4> newInputs;
239 newInputs.reserve(oldNumArgs);
240 for (unsigned i = 0; i != oldNumArgs; ++i) {
241 if (!mapper.contains(getArgument(i))) {
242 newInputs.push_back(oldType.getInput(i));
243 }
244 }
245
248 if (newInputs.size() != oldNumArgs) {
249 newFunc.setType(FunctionType::get(oldType.getContext(), newInputs, oldType.getResults()));
250
251 if (ArrayAttr argAttrs = getAllArgAttrs()) {
252 SmallVector<Attribute> newArgAttrs;
253 newArgAttrs.reserve(newInputs.size());
254 for (unsigned i = 0; i != oldNumArgs; ++i) {
255 if (!mapper.contains(getArgument(i))) {
256 newArgAttrs.push_back(argAttrs[i]);
257 }
258 }
259 newFunc.setAllArgAttrs(newArgAttrs);
260 }
261 }
262 }
263
265 cloneInto(newFunc, mapper);
266 return newFunc;
267}

338LogicalResult FuncDefOp::verify() {
339 OwningEmitErrorFn emitErrorFunc = getEmitOpErrFn(this);
340
341 if ((*this)->hasAttr(ARG_NAME_ATTR_NAME)) {
342 return emitErrorFunc() << '\'' << ARG_NAME_ATTR_NAME << "' is only valid on function arguments";
343 }
344 if ((*this)->hasAttr(RES_NAME_ATTR_NAME)) {
345 return emitErrorFunc() << '\'' << RES_NAME_ATTR_NAME << "' is only valid on function results";
346 }
347
348 if (failed(verifyArgOrResNameAttrs(
349 getAllResultAttrs(), RES_NAME_ATTR_NAME, ARG_NAME_ATTR_NAME, "result", "argument",
350 emitErrorFunc
351 ))) {
352 return failure();
353 }
354
355 if (failed(verifyArgOrResNameAttrs(
356 getAllArgAttrs(), ARG_NAME_ATTR_NAME, RES_NAME_ATTR_NAME, "argument", "result",
357 emitErrorFunc
358 ))) {
359 return failure();
360 }
361
362 // Ensure that only valid LLZK types are used for arguments and return. Additionally, the struct
363 // functions may not use AffineMapAttrs in their parameter types. If such a scenario seems to make
364 // sense when generating LLZK IR, it's likely better to introduce a struct parameter to use
365 // instead and instantiate the struct with that AffineMapAttr.
366 FunctionType type = getFunctionType();
367 for (Type t : type.getInputs()) {
368 if (llzk::checkValidType(emitErrorFunc, t).failed()) {
369 return failure();
370 }
371 if (isInStruct() && hasAffineMapAttr(t)) {
372 return emitErrorFunc().append(
373 "\"@", getName(), "\" parameters cannot contain affine map attributes but found ", t
374 );
375 }
376 }
377 for (Type t : type.getResults()) {
378 if (llzk::checkValidType(emitErrorFunc, t).failed()) {
379 return failure();
380 }
381 }
382 // Ensure that the function does not contain nested modules.
383 // Functions also cannot contain nested structs, but this check is handled
384 // via struct.def's requirement of having module as a parent.
385 WalkResult res = this->walk<WalkOrder::PreOrder>([this](ModuleOp nestedMod) {
386 getEmitOpErrFn(nestedMod)().append(
387 "cannot be nested within '", getOperation()->getName(), "' operations"
388 );
389 return WalkResult::interrupt();
390 });
391 if (res.wasInterrupted()) {
392 return failure();
393 }
394
395 return success();
396}

400LogicalResult
401verifyFuncTypeCompute(FuncDefOp &origin, SymbolTableCollection &tables, StructDefOp &parent) {
402 FunctionType funcType = origin.getFunctionType();
403 llvm::ArrayRef<Type> resTypes = funcType.getResults();
404 // Must return type of parent struct
405 if (resTypes.size() != 1) {
406 return origin.emitOpError().append(
407 "\"@", FUNC_NAME_COMPUTE, "\" must have exactly one return type"
408 );
409 }
410 if (failed(checkSelfType(tables, parent, resTypes.front(), origin, "return"))) {
411 return failure();
412 }
413
414 // After the more specific checks (to ensure more specific error messages would be produced if
415 // necessary), do the general check that all symbol references in the types are valid. The return
416 // types were already checked so just check the input types.
417 return llzk::verifyTypeResolution(tables, origin, funcType.getInputs());
418}
419
420LogicalResult
421verifyFuncTypeProduct(FuncDefOp &origin, SymbolTableCollection &tables, StructDefOp &parent) {
422 // The signature for @product is the same as the signature for @compute
423 return verifyFuncTypeCompute(origin, tables, parent);
424}
425
426LogicalResult
427verifyFuncTypeConstrain(FuncDefOp &origin, SymbolTableCollection &tables, StructDefOp &parent) {
428 FunctionType funcType = origin.getFunctionType();
429 // Must return '()' type, i.e., have no return types
430 if (funcType.getResults().size() != 0) {
431 return origin.emitOpError() << "\"@" << FUNC_NAME_CONSTRAIN << "\" must have no return type";
432 }
433
434 // Type of the first parameter must match the parent StructDefOp of the current operation.
435 llvm::ArrayRef<Type> inputTypes = funcType.getInputs();
436 if (inputTypes.size() < 1) {
437 return origin.emitOpError() << "\"@" << FUNC_NAME_CONSTRAIN
438 << "\" must have at least one input type";
439 }
440 if (failed(checkSelfType(tables, parent, inputTypes.front(), origin, "first input"))) {
441 return failure();
442 }
443
444 // After the more specific checks (to ensure more specific error messages would be produced if
445 // necessary), do the general check that all symbol references in the types are valid. There are
446 // no return types, just check the remaining input types (the first was already checked via
447 // the checkSelfType() call above).
448 return llzk::verifyTypeResolution(tables, origin, inputTypes.drop_front());
449}

453LogicalResult FuncDefOp::verifySymbolUses(SymbolTableCollection &tables) {
454 // Additional checks for the compute/constrain/product functions within a struct
455 if (StructDefOp parentStructOpt = getParentOfType<StructDefOp>(*this)) {
456 // Verify return type restrictions for functions within a StructDefOp
457 if (nameIsCompute()) {
458 return verifyFuncTypeCompute(*this, tables, parentStructOpt);
459 } else if (nameIsConstrain()) {
460 return verifyFuncTypeConstrain(*this, tables, parentStructOpt);
461 } else if (nameIsProduct()) {
462 return verifyFuncTypeProduct(*this, tables, parentStructOpt);
463 }
464 }
465 // In the general case, verify symbol resolution in all input and output types.
466 return verifyTypeResolution(tables, *this, getFunctionType());
467}

479Value FuncDefOp::getSelfValueFromCompute() {
480 assert(nameIsCompute()); // skip inStruct check to allow dangling functions
481 // Get the single block of the function body
482 Region &body = getBody();
483 assert(!body.empty() && "compute() function body is empty");
484 Block &block = body.back();
485
486 // The terminator should be the return op
487 Operation *terminator = block.getTerminator();
488 assert(terminator && "compute() function has no terminator");
489 auto retOp = llvm::dyn_cast<ReturnOp>(terminator);
490 if (!retOp) {
491 llvm::errs() << "Expected '" << ReturnOp::getOperationName() << "' but found '"
492 << terminator->getName() << "'\n";
493 llvm_unreachable("compute() function must end with ReturnOp");
494 }
495 return retOp.getOperands().front();
496}

512LogicalResult ReturnOp::verify() {
513 auto function = getParentOp<FuncDefOp>(); // parent is FuncDefOp per ODS
514
515 // The operand number and types must match the function signature.
516 const auto results = function.getFunctionType().getResults();
517 if (getNumOperands() != results.size()) {
518 return emitOpError("has ") << getNumOperands() << " operands, but enclosing function (@"
519 << function.getName() << ") returns " << results.size();
520 }
521
522 for (unsigned i = 0, e = results.size(); i != e; ++i) {
523 if (!typesUnify(getOperand(i).getType(), results[i])) {
524 return emitError() << "type of return operand " << i << " (" << getOperand(i).getType()
525 << ") doesn't match function result type (" << results[i] << ")"
526 << " in function @" << function.getName();
527 }
528 }
529
530 return success();
531}

537// Custom implementation to deserialize bytecode produced prior to version 2 which added optional
538// `OptionalAttr<ArrayAttr>:$templateParams`.
539LogicalResult CallOp::readProperties(DialectBytecodeReader &reader, OperationState &state) {
540 auto &prop = state.getOrAddProperties<Properties>();
541 if (failed(reader.readAttribute(prop.callee)) ||
542 failed(reader.readAttribute(prop.mapOpGroupSizes)) ||
543 failed(reader.readOptionalAttribute(prop.numDimsPerMap))) {
544 return failure();
545 }
546
547 if (reader.getBytecodeVersion() < /*kNativePropertiesODSSegmentSize=*/6) {
548 auto &propStorage = prop.operandSegmentSizes;
549 DenseI32ArrayAttr attr;
550 if (failed(reader.readAttribute(attr))) {
551 return failure();
552 }
553 if (attr.size() > static_cast<int64_t>(sizeof(propStorage) / sizeof(int32_t))) {
554 reader.emitError("size mismatch for operand/result_segment_size");
555 return failure();
556 }
557 llvm::copy(ArrayRef<int32_t>(attr), propStorage.begin());
558 }
559
560 // The `templateParams` is only available in version 2 or later.
561 auto versionOpt = reader.getDialectVersion<FunctionDialect>();
562 if (succeeded(versionOpt)) {
563 const auto &ver = static_cast<const LLZKDialectVersion &>(**versionOpt);
564 if (ver.majorVersion >= 2) {
565 if (failed(reader.readOptionalAttribute(prop.templateParams))) {
566 return failure();
567 }
568 }
569 }
570
571 if (reader.getBytecodeVersion() >= /*kNativePropertiesODSSegmentSize=*/6) {
572 return reader.readSparseArray(MutableArrayRef(prop.operandSegmentSizes));
573 };
574 return success();
575}

577// Same as tablegen would generate to serialize current version IR.
578void CallOp::writeProperties(DialectBytecodeWriter &writer) {
579 auto &prop = getProperties();
580 writer.writeAttribute(prop.callee);
581 writer.writeAttribute(prop.mapOpGroupSizes);
582 writer.writeOptionalAttribute(prop.numDimsPerMap);

592 if (writer.getBytecodeVersion() >= /*kNativePropertiesODSSegmentSize=*/6) {
593 writer.writeSparseArray(ArrayRef(propStorage));
594 }

625LogicalResult
626CallOp::verifyTemplateParamCompatibility(Attribute paramFromCallOp, TemplateParamOp targetParam) {
627 // A wildcard `?` (represented as kDynamic) defers inference to a later pass.
628 // It is only valid for parameters with a `!poly.tvar` type restriction.
629 if (auto intAttr = llvm::dyn_cast<IntegerAttr>(paramFromCallOp)) {
630 if (isDynamic(intAttr)) {
631 std::optional<Type> declaredType = targetParam.getTypeOpt();
632 if (!declaredType || !llvm::isa<TypeVarType>(*declaredType)) {
633 auto diag = this->emitOpError().append(
634 "wildcard `?` can only be used for template parameters with `!poly.tvar` "
635 "type restriction, but parameter \"@",
636 targetParam.getName(), "\" has "
637 );
638 if (declaredType) {
639 diag.append("type restriction ", *declaredType);
640 } else {
641 diag.append("no type restriction");
642 }
643 return diag;
644 }
645 return success();
646 }
647 }
648 if (std::optional<Type> declaredType = targetParam.getTypeOpt()) {
649 bool compatible = false;
650 if (auto sym = llvm::dyn_cast<SymbolRefAttr>(paramFromCallOp)) {
651 if (sym.getNestedReferences().empty()) {
652 SymbolTableCollection tables;
653 FailureOr<TemplateOp> parentTemplate = getConstResolutionTemplate(tables, *this);
654 if (failed(parentTemplate)) {
655 return failure();
656 }
657 if (TemplateOp p = *parentTemplate) {
658 auto binding = p.getConstNamed<TemplateSymbolBindingOpInterface>(sym.getRootReference());
659 if (binding) {
660 // Once we know it references a template symbol binding, assume it's compatible unless
661 // the optional type is present and doesn't unify with the declared type.
662 if (std::optional<Type> actualType = binding.getTypeOpt()) {
663 compatible = typesUnify(*actualType, *declaredType);
664 } else {
665 compatible = true;
666 }
667 }
668 }
669 }
670 } else if (llvm::isa<TypeVarType>(*declaredType)) {
671 compatible = llvm::isa<TypeAttr>(paramFromCallOp);
672 } else if (llvm::isa<FeltType>(*declaredType)) {
673 compatible = llvm::isa<FeltConstAttr, IntegerAttr>(paramFromCallOp) &&
674 isValidConstReadType(llvm::cast<TypedAttr>(paramFromCallOp).getType());
675 } else if (llvm::isa<IndexType, IntegerType>(*declaredType)) {
676 // Note: Just like struct type instantiation, there is no restriction on passing a
677 // larger value to an `i1`. The flattening pass will treat 0 as false and any other
678 // value as true (but give a warning if it's not 1).
679 compatible = llvm::isa<IntegerAttr>(paramFromCallOp) &&
680 isValidConstReadType(llvm::cast<TypedAttr>(paramFromCallOp).getType());
681 } else {
682 // Note: `declaredType` is restricted by `isValidConstReadType()`
683 llvm_unreachable("inconsistent with `isValidConstReadType()`");
684 }
685 if (!compatible) {
686 // Tested in call_with_template_params_fail.llzk
687 return this->emitOpError().append(
688 "instantiation value '", paramFromCallOp, "' is not compatible with parameter \"@",
689 targetParam.getName(), "\" type restriction ", *declaredType
690 );
691 }
692 }
693 return success();

827 KnownTargetVerifier(CallOp *c, SymbolLookupResult<FuncDefOp> &&tgtRes)
828 : CallOpVerifier(c, tgtRes.get().getSymName()), tgt(*tgtRes), tgtType(tgt.getFunctionType()),
829 includeSymNames(tgtRes.getNamespace()) {}
830
831 LogicalResult verifyTargetAttributes() override {
832 return CallOpVerifier::verifyTargetAttributesMatch(tgt);
833 }
834
835 LogicalResult verifyInputs() override {
836 return verifyTypesMatch(callOp->getArgOperands().getTypes(), tgtType.getInputs(), "operand");
837 }
838
839 LogicalResult verifyOutputs() override {
840 return verifyTypesMatch(callOp->getResultTypes(), tgtType.getResults(), "result");
841 }
842
843 LogicalResult verifyTemplateParams() override {
844 Operation *tgtOp = tgt.getOperation();
845 if (isInStruct(tgtOp)) {
846 // Struct function calls cannot contain template parameter instantiations.
847 return verifyNoTemplateInstantiations();
848 } else if (TemplateOp tgtOpParent = getParentOfType<TemplateOp>(tgtOp)) {
849 // When the target function is a free function within a TemplateOp, the CallOp may have
850 // template parameter instantiations that must be checked against the template parameters.
851 // - If the function type signature references all template parameters, then the parameter
852 // instantiation list on the CallOp is optional, otherwise it's required.
853 // - If present, the instantiation list must provide a value for every template parameter
854 // and the value must be type-compatible with the parameter's declared type (if any).
855 // - If present, the instantiation list must result in a function type signature that can
856 // be unified with the CallOp's operand and result types.
857 auto realParams = tgtOpParent.getConstOps<TemplateParamOp>();
858 ArrayAttr callParams = callOp->getTemplateParamsAttr();
859
860 // When there is no instantiation list, just ensure that it's not required.
861 if (isNullOrEmpty(callParams)) {
862 llvm::SmallDenseSet<SymbolRefAttr> referencedInSignature;
863 llzk::getSymbolsUsedIn(tgtType.getInputs(), referencedInSignature);
864 llzk::getSymbolsUsedIn(tgtType.getResults(), referencedInSignature);
865
866 bool allParamsReferenced = llvm::all_of(realParams, [&](TemplateParamOp p) {
867 return referencedInSignature.contains(FlatSymbolRefAttr::get(p.getNameAttr()));
868 });
869 if (allParamsReferenced) {
870 return success();
871 }
872 // Tested in call_with_template_params_fail.llzk
873 return callOp->emitOpError().append(
874 "must provide template instantiation parameters when calling \"@", tgt.getSymName(),
875 "\" because not all template parameters of \"@", tgtOpParent.getSymName(),
876 "\" appear in the function type signature"
877 );
878 }
879
880 // Ensure `forceIntAttrTypes()` was successful on the CallOp's template parameters.
881 if (failed(llzk::forceIntAttrTypes(callParams.getValue(), [this] {
882 return llzk::InFlightDiagnosticWrapper(this->callOp->emitOpError());
883 }))) {
884 return failure();
885 }
886
887 // The instantiation list is present. Check it has exactly one entry per template param.
888 size_t numTemplateParams = llvm::range_size(realParams);
889 if (callParams.size() != numTemplateParams) {
890 // Tested in call_with_template_params_fail.llzk
891 return callOp->emitOpError().append(
892 "template instantiation has ", callParams.size(), " parameter(s) but \"@",
893 tgtOpParent.getSymName(), "\" expects ", numTemplateParams, " template parameter(s)"
894 );
895 }
896
897 // Check type compatibility of each provided value with the declared parameter type (if any).
898 if (failed(callOp->verifyTemplateParamCompatibility(realParams))) {
899 return failure();
900 }
901
902 // Check that the provided instantiation values are consistent with what type unification
903 // of the target function types against the call's operand and result types would determine.
904 FailureOr<UnificationMap> unifyResult = callOp->unifyTypeSignature(tgtType);
905 assert(succeeded(unifyResult) && "already checked by `verifyInputs()` and `verifyOutputs()`");
906 return callOp->verifyTemplateParamsMatchInferred(realParams, unifyResult.value());
907 } else {
908 // Non-template functions cannot contain template parameter instantiations.
909 return verifyNoTemplateInstantiations();
910 }
911 }
912
913 LogicalResult verifyAffineMapParams() override {
914 if ((FunctionKind::StructCompute == tgtKind || FunctionKind::StructProduct == tgtKind) &&
915 isInStruct(tgt.getOperation())) {
916 // Return type should be a single StructType. If that is not the case here, just bail without
917 // producing an error. The combination of this KnownTargetVerifier resolving the callee to a
918 // specific FuncDefOp and verifyFuncTypeCompute() ensuring all FUNC_NAME_COMPUTE FuncOps have
919 // a single StructType return value will produce a more relevant error message in that case.
920 if (StructType retTy = callOp->getSingleResultTypeOfWitnessGen()) {
921 if (ArrayAttr params = retTy.getParams()) {
922 // Collect the struct parameters that are defined via AffineMapAttr
923 SmallVector<AffineMapAttr> mapAttrs;
924 for (Attribute a : params) {
925 if (AffineMapAttr m = dyn_cast<AffineMapAttr>(a)) {
926 mapAttrs.push_back(m);
927 }
928 }
929 return affineMapHelpers::verifyAffineMapInstantiations(
930 callOp->getMapOperands(), callOp->getNumDimsPerMap(), mapAttrs, *callOp
931 );
932 }
933 }
934 return success();
935 } else {
936 // Global functions and constrain functions cannot have affine map instantiations.
937 return verifyNoAffineMapInstantiations();
938 }

1008 // struct compute, constrain, or product function.

1076 );

1109LogicalResult CallOp::verifySymbolUses(SymbolTableCollection &tables) {
1110 // First, verify symbol resolution in all input and output types.
1111 if (failed(verifyTypeResolution(tables, *this, getTypeSignature()))) {

1118 return emitOpError("requires a 'callee' symbol reference attribute");

1122 // the subset of checks that can be done even though the target is unknown.
1123 if (calleeAttr.getNestedReferences().size() == 1) {
1124 if (TemplateOp parent = getParentOfType<TemplateOp>(*this)) {
