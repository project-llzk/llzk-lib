48FunctionKind fnNameToKind(mlir::StringRef name) {
49 if (FUNC_NAME_COMPUTE == name) {
50 return FunctionKind::StructCompute;
51 } else if (FUNC_NAME_CONSTRAIN == name) {
52 return FunctionKind::StructConstrain;
53 } else if (FUNC_NAME_PRODUCT == name) {
54 return FunctionKind::StructProduct;
55 } else {
56 return FunctionKind::Free;
57 }
58}

135void FuncDefOp::cloneInto(FuncDefOp dest, IRMapping &mapper) {
136 // Add the attributes of this function to dest.
137 llvm::MapVector<StringAttr, Attribute> newAttrMap;
138 for (const auto &attr : dest->getAttrs()) {
139 newAttrMap.insert({attr.getName(), attr.getValue()});
140 }
141 for (const auto &attr : (*this)->getAttrs()) {
142 newAttrMap.insert({attr.getName(), attr.getValue()});
143 }
144
145 auto newAttrs =
146 llvm::to_vector(llvm::map_range(newAttrMap, [](std::pair<StringAttr, Attribute> attrPair) {
147 return NamedAttribute(attrPair.first, attrPair.second);
148 }));
149 dest->setAttrs(DictionaryAttr::get(getContext(), newAttrs));
150
151 // Clone the body.
152 getBody().cloneInto(&dest.getBody(), mapper);
153}

160FuncDefOp FuncDefOp::clone(IRMapping &mapper) {
161 // Create the new function.
162 FuncDefOp newFunc = llvm::cast<FuncDefOp>(getOperation()->cloneWithoutRegions());
163
164 // If the function has a body, then the user might be deleting arguments to
165 // the function by specifying them in the mapper. If so, we don't add the
166 // argument to the input type vector.
167 if (!isExternal()) {
168 FunctionType oldType = getFunctionType();
169
170 unsigned oldNumArgs = oldType.getNumInputs();
171 SmallVector<Type, 4> newInputs;
172 newInputs.reserve(oldNumArgs);
173 for (unsigned i = 0; i != oldNumArgs; ++i) {
174 if (!mapper.contains(getArgument(i))) {
175 newInputs.push_back(oldType.getInput(i));
176 }
177 }
178
181 if (newInputs.size() != oldNumArgs) {
182 newFunc.setType(FunctionType::get(oldType.getContext(), newInputs, oldType.getResults()));
183
184 if (ArrayAttr argAttrs = getAllArgAttrs()) {
185 SmallVector<Attribute> newArgAttrs;
186 newArgAttrs.reserve(newInputs.size());
187 for (unsigned i = 0; i != oldNumArgs; ++i) {
188 if (!mapper.contains(getArgument(i))) {
189 newArgAttrs.push_back(argAttrs[i]);
190 }
191 }
192 newFunc.setAllArgAttrs(newArgAttrs);
193 }
194 }
195 }
196
198 cloneInto(newFunc, mapper);
199 return newFunc;
200}

202FuncDefOp FuncDefOp::clone() {

241LogicalResult FuncDefOp::verify() {
242 OwningEmitErrorFn emitErrorFunc = getEmitOpErrFn(this);
243 // Ensure that only valid LLZK types are used for arguments and return. Additionally, the struct
244 // functions may not use AffineMapAttrs in their parameter types. If such a scenario seems to make
245 // sense when generating LLZK IR, it's likely better to introduce a struct parameter to use
246 // instead and instantiate the struct with that AffineMapAttr.
247 FunctionType type = getFunctionType();
248 for (Type t : type.getInputs()) {
249 if (llzk::checkValidType(emitErrorFunc, t).failed()) {
250 return failure();
251 }
252 if (isInStruct() && hasAffineMapAttr(t)) {
253 return emitErrorFunc().append(
254 "\"@", getName(), "\" parameters cannot contain affine map attributes but found ", t
255 );
256 }
257 }
258 for (Type t : type.getResults()) {
259 if (llzk::checkValidType(emitErrorFunc, t).failed()) {
260 return failure();
261 }
262 }
263 // Ensure that the function does not contain nested modules.
264 // Functions also cannot contain nested structs, but this check is handled
265 // via struct.def's requirement of having module as a parent.
266 WalkResult res = this->walk<WalkOrder::PreOrder>([this](ModuleOp nestedMod) {
267 getEmitOpErrFn(nestedMod)().append(
268 "cannot be nested within '", getOperation()->getName(), "' operations"
269 );
270 return WalkResult::interrupt();
271 });
272 if (res.wasInterrupted()) {
273 return failure();
274 }
275
276 return success();
277}

281LogicalResult
282verifyFuncTypeCompute(FuncDefOp &origin, SymbolTableCollection &tables, StructDefOp &parent) {
283 FunctionType funcType = origin.getFunctionType();
284 llvm::ArrayRef<Type> resTypes = funcType.getResults();
285 // Must return type of parent struct
286 if (resTypes.size() != 1) {
287 return origin.emitOpError().append(
288 "\"@", FUNC_NAME_COMPUTE, "\" must have exactly one return type"
289 );
290 }
291 if (failed(checkSelfType(tables, parent, resTypes.front(), origin, "return"))) {
292 return failure();
293 }
294
295 // After the more specific checks (to ensure more specific error messages would be produced if
296 // necessary), do the general check that all symbol references in the types are valid. The return
297 // types were already checked so just check the input types.
298 return llzk::verifyTypeResolution(tables, origin, funcType.getInputs());
299}
300
301LogicalResult
302verifyFuncTypeProduct(FuncDefOp &origin, SymbolTableCollection &tables, StructDefOp &parent) {
303 // The signature for @product is the same as the signature for @compute
304 return verifyFuncTypeCompute(origin, tables, parent);
305}
306
307LogicalResult
308verifyFuncTypeConstrain(FuncDefOp &origin, SymbolTableCollection &tables, StructDefOp &parent) {
309 FunctionType funcType = origin.getFunctionType();
310 // Must return '()' type, i.e., have no return types
311 if (funcType.getResults().size() != 0) {
312 return origin.emitOpError() << "\"@" << FUNC_NAME_CONSTRAIN << "\" must have no return type";
313 }
314
315 // Type of the first parameter must match the parent StructDefOp of the current operation.
316 llvm::ArrayRef<Type> inputTypes = funcType.getInputs();
317 if (inputTypes.size() < 1) {
318 return origin.emitOpError() << "\"@" << FUNC_NAME_CONSTRAIN
319 << "\" must have at least one input type";
320 }
321 if (failed(checkSelfType(tables, parent, inputTypes.front(), origin, "first input"))) {
322 return failure();
323 }
324
325 // After the more specific checks (to ensure more specific error messages would be produced if
326 // necessary), do the general check that all symbol references in the types are valid. There are
327 // no return types, just check the remaining input types (the first was already checked via
328 // the checkSelfType() call above).
329 return llzk::verifyTypeResolution(tables, origin, inputTypes.drop_front());
330}

334LogicalResult FuncDefOp::verifySymbolUses(SymbolTableCollection &tables) {
335 // Additional checks for the compute/constrain/product functions within a struct
336 if (StructDefOp parentStructOpt = getParentOfType<StructDefOp>(*this)) {
337 // Verify return type restrictions for functions within a StructDefOp
338 if (nameIsCompute()) {
339 return verifyFuncTypeCompute(*this, tables, parentStructOpt);
340 } else if (nameIsConstrain()) {
341 return verifyFuncTypeConstrain(*this, tables, parentStructOpt);
342 } else if (nameIsProduct()) {
343 return verifyFuncTypeProduct(*this, tables, parentStructOpt);
344 }
345 }
346 // In the general case, verify symbol resolution in all input and output types.
347 return verifyTypeResolution(tables, *this, getFunctionType());
348}

360Value FuncDefOp::getSelfValueFromCompute() {
361 assert(nameIsCompute()); // skip inStruct check to allow dangling functions
362 // Get the single block of the function body
363 Region &body = getBody();
364 assert(!body.empty() && "compute() function body is empty");
365 Block &block = body.back();
366
367 // The terminator should be the return op
368 Operation *terminator = block.getTerminator();
369 assert(terminator && "compute() function has no terminator");
370 auto retOp = llvm::dyn_cast<ReturnOp>(terminator);
371 if (!retOp) {
372 llvm::errs() << "Expected '" << ReturnOp::getOperationName() << "' but found '"
373 << terminator->getName() << "'\n";
374 llvm_unreachable("compute() function must end with ReturnOp");
375 }
376 return retOp.getOperands().front();
377}
378
379Value FuncDefOp::getSelfValueFromConstrain() {
380 assert(nameIsConstrain()); // skip inStruct check to allow dangling functions
381 return getArguments().front();
382}

393LogicalResult ReturnOp::verify() {
394 auto function = getParentOp<FuncDefOp>(); // parent is FuncDefOp per ODS
395
396 // The operand number and types must match the function signature.
397 const auto results = function.getFunctionType().getResults();
398 if (getNumOperands() != results.size()) {
399 return emitOpError("has ") << getNumOperands() << " operands, but enclosing function (@"
400 << function.getName() << ") returns " << results.size();
401 }
402
403 for (unsigned i = 0, e = results.size(); i != e; ++i) {
404 if (!typesUnify(getOperand(i).getType(), results[i])) {
405 return emitError() << "type of return operand " << i << " (" << getOperand(i).getType()
406 << ") doesn't match function result type (" << results[i] << ")"
407 << " in function @" << function.getName();
408 }
409 }
410
411 return success();
412}

418static void addTemplateParams(
419 OpBuilder &odsBuilder, CallOp::Properties &props, ArrayRef<Attribute> templateParams
420) {
421 if (!templateParams.empty()) {
422 // Must attempt to convert attribute types but `build()` functions do not have a failure path or
423 // error reporting. That comes during validation of the constructed op so ignore errors here.
424 FailureOr<SmallVector<Attribute>> r = llzk::forceIntAttrTypes(templateParams, [&odsBuilder]() {
425 return InFlightDiagnosticWrapper::createSilent(odsBuilder.getContext());
426 });
427 ArrayRef<Attribute> converted = succeeded(r) ? r.value() : templateParams;
428 props.setTemplateParams(odsBuilder.getArrayAttr(converted));
429 }
430}

460LogicalResult
461CallOp::verifyTemplateParamCompatibility(Attribute paramFromCallOp, TemplateParamOp targetParam) {
462 if (std::optional<Type> declaredType = targetParam.getTypeOpt()) {
463 // Note: `declaredType` is restricted by `isValidConstReadType()`
464 bool compatible = false;
465 if (llvm::isa<TypeVarType>(*declaredType)) {
466 compatible = llvm::isa<TypeAttr>(paramFromCallOp);
467 } else if (llvm::isa<FeltType>(*declaredType)) {
468 compatible = llvm::isa<FeltConstAttr, IntegerAttr>(paramFromCallOp) &&
469 isValidConstReadType(llvm::cast<TypedAttr>(paramFromCallOp).getType());
470 } else if (llvm::isa<IndexType, IntegerType>(*declaredType)) {
471 // Note: Just like struct type instantiation, there is no restriction on passing a
472 // larger value to an `i1`. The flattening pass will treat 0 as false and any other
473 // value as true (but give a warning if it's not 1).
474 compatible = llvm::isa<IntegerAttr>(paramFromCallOp) &&
475 isValidConstReadType(llvm::cast<TypedAttr>(paramFromCallOp).getType());
476 } else {
477 llvm_unreachable("inconsistent with `isValidConstReadType()`");
478 }
479 if (!compatible) {
480 // Tested in call_with_template_params_fail.llzk
481 return this->emitOpError().append(
482 "instantiation value '", paramFromCallOp, "' is not compatible with parameter \"@",
483 targetParam.getName(), "\" type restriction ", *declaredType
484 );
485 }
486 }
487 return success();
488}

490LogicalResult CallOp::verifyTemplateParamCompatibility(
491 llvm::iterator_range<Region::op_iterator<TemplateParamOp>> targetParamDefs
492) {
493 ArrayAttr callParams = this->getTemplateParamsAttr();
494 assert(!isNullOrEmpty(callParams) && "pre-condition");
495 assert((callParams.size() == llvm::range_size(targetParamDefs)) && "pre-condition");
496
497 for (auto [paramOp, attr] : llvm::zip_equal(targetParamDefs, callParams.getValue())) {
498 if (failed(verifyTemplateParamCompatibility(attr, paramOp))) {
499 return failure();
500 }
501 }
502 return success();
503}
504
505LogicalResult CallOp::verifyTemplateParamsMatchInferred(
506 llvm::iterator_range<Region::op_iterator<TemplateParamOp>> targetParamDefs,
507 const UnificationMap &unifications
508) {
509 ArrayAttr callParams = this->getTemplateParamsAttr();
510 assert(!isNullOrEmpty(callParams) && "pre-condition");
511 assert((callParams.size() == llvm::range_size(targetParamDefs)) && "pre-condition");
512
513 for (auto [paramOp, attr] : llvm::zip_equal(targetParamDefs, callParams.getValue())) {
514 auto it = unifications.find({FlatSymbolRefAttr::get(paramOp.getNameAttr()), Side::RHS});
515 if (it != unifications.end() && !typeParamsUnify({attr}, {it->second})) {
516 // Tested in call_with_template_params_fail.llzk
517 return this->emitOpError().append(
518 "template instantiation value '", attr, "' for parameter \"@", paramOp.getName(),
519 "\" conflicts with value '", it->second, "' inferred from function type signature"
520 );
521 }
522 }
523 return success();
524}

528struct CallOpVerifier {
529 CallOpVerifier(CallOp *c, FunctionKind tgtFuncKind) : callOp(c), tgtKind(tgtFuncKind) {}
530 CallOpVerifier(CallOp *c, StringRef tgtName) : CallOpVerifier(c, fnNameToKind(tgtName)) {}
531 virtual ~CallOpVerifier() = default;
532
533 LogicalResult verify() {
534 // Rather than immediately returning on failure, we check all verifier steps and aggregate to
535 // provide as many errors are possible in a single verifier run.
536 LogicalResult aggregateResult = success();
537 if (failed(verifyTargetAttributes())) {
538 aggregateResult = failure();
539 }
540 if (failed(verifyInputs())) {
541 aggregateResult = failure();
542 }
543 if (failed(verifyOutputs())) {
544 aggregateResult = failure();
545 }
546 if (failed(verifyTemplateParams())) {
547 aggregateResult = failure();
548 }
549 if (failed(verifyAffineMapParams())) {
550 aggregateResult = failure();
551 }
552 return aggregateResult;
553 }
554
555protected:
556 CallOp *callOp;
557 FunctionKind tgtKind;
558
559 virtual LogicalResult verifyTargetAttributes() = 0;
560 virtual LogicalResult verifyInputs() = 0;
561 virtual LogicalResult verifyOutputs() = 0;
562 virtual LogicalResult verifyTemplateParams() = 0;
563 virtual LogicalResult verifyAffineMapParams() = 0;
564
566 LogicalResult verifyTargetAttributesMatch(FuncDefOp target) {
567 LogicalResult aggregateRes = success();
568 if (FuncDefOp caller = (*callOp)->getParentOfType<FuncDefOp>()) {
569 auto emitAttrErr = [&](StringLiteral attrName) {
570 aggregateRes = callOp->emitOpError()
571 << "target '@" << target.getName() << "' has '" << attrName
572 << "' attribute, which is not specified by the caller '@" << caller.getName()
573 << '\'';
574 };
575
576 if (target.hasAllowConstraintAttr() && !caller.hasAllowConstraintAttr()) {
577 emitAttrErr(AllowConstraintAttr::name);
578 }
579 if (target.hasAllowWitnessAttr() && !caller.hasAllowWitnessAttr()) {
580 emitAttrErr(AllowWitnessAttr::name);
581 }
582 if (target.hasAllowNonNativeFieldOpsAttr() && !caller.hasAllowNonNativeFieldOpsAttr()) {
583 emitAttrErr(AllowNonNativeFieldOpsAttr::name);
584 }
585 }
586 return aggregateRes;
587 }
588
589 LogicalResult verifyNoTemplateInstantiations() {
590 if (!isNullOrEmpty(callOp->getTemplateParamsAttr())) {
591 // Tested in call_with_template_params_fail.llzk
592 return callOp->emitOpError().append(
593 "can only have template instantiations when targeting a templated free function"
594 );
595 }
596 return success();
597 }
598
599 LogicalResult verifyNoAffineMapInstantiations() {
600 if (!isNullOrEmpty(callOp->getMapOpGroupSizesAttr())) {
601 // Tested in call_with_affinemap_fail.llzk
602 return callOp->emitOpError().append(
603 "can only have affine map instantiations when targeting a \"@", FUNC_NAME_COMPUTE,
604 "\" function"
605 );
606 }
607 // ASSERT: the check above is sufficient due to VerifySizesForMultiAffineOps trait.
608 assert(isNullOrEmpty(callOp->getNumDimsPerMapAttr()));
609 assert(callOp->getMapOperands().empty());
610 return success();
611 }
612};

614struct KnownTargetVerifier : public CallOpVerifier {
615 KnownTargetVerifier(CallOp *c, SymbolLookupResult<FuncDefOp> &&tgtRes)
616 : CallOpVerifier(c, tgtRes.get().getSymName()), tgt(*tgtRes), tgtType(tgt.getFunctionType()),
617 includeSymNames(tgtRes.getNamespace()) {}
618
619 LogicalResult verifyTargetAttributes() override {
620 return CallOpVerifier::verifyTargetAttributesMatch(tgt);
621 }
622
623 LogicalResult verifyInputs() override {
624 return verifyTypesMatch(callOp->getArgOperands().getTypes(), tgtType.getInputs(), "operand");
625 }
626
627 LogicalResult verifyOutputs() override {
628 return verifyTypesMatch(callOp->getResultTypes(), tgtType.getResults(), "result");
629 }
630
631 LogicalResult verifyTemplateParams() override {
632 auto tgtOp = tgt.getOperation();
633 if (isInStruct(tgtOp)) {
634 // Struct function calls cannot contain template parameter instantiations.
635 return verifyNoTemplateInstantiations();
636 } else if (TemplateOp tgtOpParent = getParentOfType<TemplateOp>(tgtOp)) {
637 // When the target function is a free function within a TemplateOp, the CallOp may have
638 // template parameter instantiations that must be checked against the template parameters.
639 // - If the function type signature references all template parameters, then the parameter
640 // instantiation list on the CallOp is optional, otherwise it's required.
641 // - If present, the instantiation list must provide a value for every template parameter
642 // and the value must be type-compatible with the parameter's declared type (if any).
643 // - If present, the instantiation list must result in a function type signature that can
644 // be unified with the CallOp's operand and result types.
645 auto realParams = tgtOpParent.getConstOps<TemplateParamOp>();
646 ArrayAttr callParams = callOp->getTemplateParamsAttr();
647
648 // When there is no instantiation list, just ensure that it's not required.
649 if (isNullOrEmpty(callParams)) {
650 llvm::SmallDenseSet<SymbolRefAttr> referencedInSignature;
651 llzk::getSymbolsUsedIn(tgtType.getInputs(), referencedInSignature);
652 llzk::getSymbolsUsedIn(tgtType.getResults(), referencedInSignature);
653
654 bool allParamsReferenced = llvm::all_of(realParams, [&](TemplateParamOp p) {
655 return referencedInSignature.contains(FlatSymbolRefAttr::get(p.getNameAttr()));
656 });
657 if (allParamsReferenced) {
658 return success();
659 }
660 // Tested in call_with_template_params_fail.llzk
661 return callOp->emitOpError().append(
662 "must provide template instantiation parameters when calling \"@", tgt.getSymName(),
663 "\" because not all template parameters of \"@", tgtOpParent.getSymName(),
664 "\" appear in the function type signature"
665 );
666 }
667
668 // Ensure `forceIntAttrTypes()` was successful on the CallOp's template parameters.
669 if (failed(llzk::forceIntAttrTypes(callParams.getValue(), [this] {
670 return llzk::InFlightDiagnosticWrapper(this->callOp->emitOpError());
671 }))) {
672 return failure();
673 }
674
675 // The instantiation list is present. Check it has exactly one entry per template param.
676 size_t numTemplateParams = llvm::range_size(realParams);
677 if (callParams.size() != numTemplateParams) {
678 // Tested in call_with_template_params_fail.llzk
679 return callOp->emitOpError().append(
680 "template instantiation has ", callParams.size(), " parameter(s) but \"@",
681 tgtOpParent.getSymName(), "\" expects ", numTemplateParams, " template parameter(s)"
682 );
683 }
684
685 // Check type compatibility of each provided value with the declared parameter type (if any).
686 if (failed(callOp->verifyTemplateParamCompatibility(realParams))) {
687 return failure();
688 }
689
690 // Check that the provided instantiation values are consistent with what type unification
691 // of the target function types against the call's operand and result types would determine.
692 FailureOr<UnificationMap> unifyResult = callOp->unifyTypeSignature(tgtType);
693 assert(succeeded(unifyResult) && "already checked by `verifyInputs()` and `verifyOutputs()`");
694 return callOp->verifyTemplateParamsMatchInferred(realParams, unifyResult.value());
695 } else {
696 // Non-template functions cannot contain template parameter instantiations.
697 return verifyNoTemplateInstantiations();
698 }
699 }
700
701 LogicalResult verifyAffineMapParams() override {
702 if ((FunctionKind::StructCompute == tgtKind || FunctionKind::StructProduct == tgtKind) &&
703 isInStruct(tgt.getOperation())) {
704 // Return type should be a single StructType. If that is not the case here, just bail without
705 // producing an error. The combination of this KnownTargetVerifier resolving the callee to a
706 // specific FuncDefOp and verifyFuncTypeCompute() ensuring all FUNC_NAME_COMPUTE FuncOps have
707 // a single StructType return value will produce a more relevant error message in that case.
708 if (StructType retTy = callOp->getSingleResultTypeOfWitnessGen()) {
709 if (ArrayAttr params = retTy.getParams()) {
710 // Collect the struct parameters that are defined via AffineMapAttr
711 SmallVector<AffineMapAttr> mapAttrs;
712 for (Attribute a : params) {
713 if (AffineMapAttr m = dyn_cast<AffineMapAttr>(a)) {
714 mapAttrs.push_back(m);
715 }
716 }
717 return affineMapHelpers::verifyAffineMapInstantiations(
718 callOp->getMapOperands(), callOp->getNumDimsPerMap(), mapAttrs, *callOp
719 );
720 }
721 }
722 return success();
723 } else {
724 // Global functions and constrain functions cannot have affine map instantiations.
725 return verifyNoAffineMapInstantiations();
726 }
727 }

730 template <typename T>
731 LogicalResult
732 verifyTypesMatch(ValueTypeRange<T> callOpTypes, ArrayRef<Type> tgtTypes, const char *aspect) {
733 if (tgtTypes.size() != callOpTypes.size()) {
734 return callOp->emitOpError()
735 .append("incorrect number of ", aspect, "s for callee, expected ", tgtTypes.size())
736 .attachNote(tgt.getLoc())
737 .append("callee defined here");
738 }
739 for (unsigned i = 0, e = tgtTypes.size(); i != e; ++i) {
740 if (!typesUnify(callOpTypes[i], tgtTypes[i], includeSymNames)) {
741 return callOp->emitOpError().append(
742 aspect, " type mismatch: expected type ", tgtTypes[i], ", but found ", callOpTypes[i],
743 " for ", aspect, " number ", i
744 );
745 }
746 }
747 return success();
748 }
749
750 FuncDefOp tgt;
751 FunctionType tgtType;
752 std::vector<llvm::StringRef> includeSymNames;
753};

757LogicalResult checkSelfTypeUnknownTarget(
758 StringAttr expectedParamName, Type actualType, CallOp *origin, const char *aspect
759) {
760 if (!llvm::isa<TypeVarType>(actualType) ||
761 llvm::cast<TypeVarType>(actualType).getRefName() != expectedParamName) {
762 // Tested in function_restrictions_fail.llzk:
763 // Non-tvar for constrain input via "call_target_constrain_without_self_non_struct"
764 // Non-tvar for compute output via "call_target_compute_wrong_type_ret"
765 // Wrong tvar for constrain input via "call_target_constrain_without_self_wrong_tvar_param"
766 // Wrong tvar for compute output via "call_target_compute_wrong_tvar_param_ret"
767 return origin->emitOpError().append(

769 " type '!", TypeVarType::name, "<@", expectedParamName.getValue(), ">' but found ",
770 actualType

785struct UnknownTargetVerifier : public CallOpVerifier {
786 UnknownTargetVerifier(CallOp *c, FunctionKind tgtFuncKind, SymbolRefAttr callee)
787 : CallOpVerifier(c, tgtFuncKind), calleeAttr(callee) {
788 assert(
789 tgtFuncKind == FunctionKind::StructCompute ||
790 tgtFuncKind == FunctionKind::StructConstrain || tgtFuncKind == FunctionKind::StructProduct
791 ); // pre-condition mentioned above
792 }
793
794 LogicalResult verifyTargetAttributes() override {
795 // Based on the precondition of this verifier, the target must be either a
796 // struct compute, constrain, or product function.
797 LogicalResult aggregateRes = success();
798 if (FuncDefOp caller = (*callOp)->getParentOfType<FuncDefOp>()) {
799 auto emitAttrErr = [&](StringLiteral attrName) {
800 aggregateRes = callOp->emitOpError()
801 << "target '" << calleeAttr << "' has '" << attrName
802 << "' attribute, which is not specified by the caller '@" << caller.getName()
803 << '\'';
804 };
805
806 switch (tgtKind) {
807 case FunctionKind::StructConstrain:
808 if (!caller.hasAllowConstraintAttr()) {
809 emitAttrErr(AllowConstraintAttr::name);
810 }
811 break;
812 case FunctionKind::StructCompute:
813 if (!caller.hasAllowWitnessAttr()) {
814 emitAttrErr(AllowWitnessAttr::name);
815 }
816 break;
817 case FunctionKind::StructProduct:
818 if (!caller.hasAllowWitnessAttr()) {
819 emitAttrErr(AllowWitnessAttr::name);
820 }
821 if (!caller.hasAllowConstraintAttr()) {
822 emitAttrErr(AllowConstraintAttr::name);
823 }
824 break;
825 default:
826 break;
827 }
828 }
829 return aggregateRes;
830 }
831
832 LogicalResult verifyInputs() override {
833 if (FunctionKind::StructCompute == tgtKind || FunctionKind::StructProduct == tgtKind) {
834 // Without known target, no additional checks can be done.
835 } else if (FunctionKind::StructConstrain == tgtKind) {
836 // Without known target, this can only check that the first input is VarType using the same
837 // struct parameter as the base of the callee (later replaced with the target struct's type).
838 Operation::operand_type_range inputTypes = callOp->getArgOperands().getTypes();
839 if (inputTypes.size() < 1) {
840 // Tested in function_restrictions_fail.llzk
841 return callOp->emitOpError()
842 << "target \"@" << FUNC_NAME_CONSTRAIN << "\" must have at least one input type";
843 }
844 return checkSelfTypeUnknownTarget(
845 calleeAttr.getRootReference(), inputTypes.front(), callOp, "first input"
846 );
847 }
848 return success();
849 }
850
851 LogicalResult verifyOutputs() override {
852 if (FunctionKind::StructCompute == tgtKind || FunctionKind::StructProduct == tgtKind) {
853 // Without known target, this can only check that the function returns VarType using the same
854 // struct parameter as the base of the callee (later replaced with the target struct's type).
855 Operation::result_type_range resTypes = callOp->getResultTypes();
856 if (resTypes.size() != 1) {
857 // Tested in function_restrictions_fail.llzk
858 return callOp->emitOpError().append(
859 "target \"@", FUNC_NAME_COMPUTE, "\" must have exactly one return type"
860 );
861 }
862 return checkSelfTypeUnknownTarget(
863 calleeAttr.getRootReference(), resTypes.front(), callOp, "return"
864 );
865 } else if (FunctionKind::StructConstrain == tgtKind) {
866 // Without known target, this can only check that the function has no return
867 if (callOp->getNumResults() != 0) {
868 // Tested in function_restrictions_fail.llzk
869 return callOp->emitOpError()
870 << "target \"@" << FUNC_NAME_CONSTRAIN << "\" must have no return type";
871 }
872 }
873 return success();
874 }
875
876 LogicalResult verifyTemplateParams() override {
877 // Struct function calls cannot contain template parameter instantiations.
878 return verifyNoTemplateInstantiations();
879 }
880
881 LogicalResult verifyAffineMapParams() override {
882 if (FunctionKind::StructCompute == tgtKind || FunctionKind::StructProduct == tgtKind) {
883 // Without known target, no additional checks can be done.
884 } else if (FunctionKind::StructConstrain == tgtKind) {
885 // Without known target, this can only check that there are no affine map instantiations.
886 return verifyNoAffineMapInstantiations();
887 }
888 return success();
889 }
890
891private:
892 SymbolRefAttr calleeAttr;
893};

897LogicalResult CallOp::verifySymbolUses(SymbolTableCollection &tables) {
898 // First, verify symbol resolution in all input and output types.
899 if (failed(verifyTypeResolution(tables, *this, getTypeSignature()))) {
900 return failure(); // verifyTypeResolution() already emits a sufficient error message
901 }
902
903 // Check that the callee attribute was specified.
904 SymbolRefAttr calleeAttr = getCalleeAttr();
905 if (!calleeAttr) {
906 return emitOpError("requires a 'callee' symbol reference attribute");
907 }
908
909 // If the callee references a parameter of the template where this call appears, perform
910 // the subset of checks that can be done even though the target is unknown.
911 if (calleeAttr.getNestedReferences().size() == 1) {
912 if (TemplateOp parent = getParentOfType<TemplateOp>(*this)) {
913 if (parent.hasConstNamed<TemplateParamOp>(calleeAttr.getRootReference())) {
914 FunctionKind tgtKind = fnNameToKind(calleeAttr.getLeafReference().getValue());
915 if (tgtKind != FunctionKind::Free) {
916 return UnknownTargetVerifier(this, tgtKind, calleeAttr).verify();
917 }
918 return this->emitError("expected parameterized callee to target a struct function")
919 .append(
920 " (i.e. \"@", FUNC_NAME_PRODUCT, "\", \"@", FUNC_NAME_COMPUTE, "\", or \"@",
921 FUNC_NAME_CONSTRAIN, "\")"
922 );
923 }
924 }
925 }
926
927 // Otherwise, callee must be specified via full path from the root module. Perform the full set of
928 // checks against the known target function.
929 auto tgtOpt = lookupTopLevelSymbol<FuncDefOp>(tables, calleeAttr, *this);
930 if (failed(tgtOpt)) {
931 return this->emitError() << "expected '" << FuncDefOp::getOperationName() << "' named \""
932 << calleeAttr << '"';
933 }
934 return KnownTargetVerifier(this, std::move(*tgtOpt)).verify();
935}

952bool calleeIsStructFunctionImpl(
953 const char *funcName, SymbolRefAttr callee, llvm::function_ref<StructType()> getType
954) {
955 if (callee.getLeafReference() == funcName) {
956 if (StructType t = getType()) {
957 // If the name ref within the StructType matches the `callee` prefix (i.e., sans the function
958 // name itself), then the `callee` target must be within a StructDefOp because validation
959 // checks elsewhere ensure that every StructType references a StructDefOp (i.e., the `callee`
960 // function is not simply a free function nested within a ModuleOp)
961 return t.getNameRef() == getPrefixAsSymbolRefAttr(callee);
962 }
963 }
964 return false;
965}
