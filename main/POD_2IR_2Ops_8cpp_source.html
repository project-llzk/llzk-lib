46 OpBuilder &builder, OperationState &state, PodType result, InitializedRecords initialValues
47) {
48 SmallVector<Value, 4> values;
49 SmallVector<StringRef, 4> names;
50

52 names.push_back(record.name);
53 values.push_back(record.value);

55
56 auto &props = state.getOrAddProperties<NewPodOp::Properties>();
57 state.addTypes(result);
58 state.addOperands(values);
59 props.setInitializedRecords(builder.getStrArrayAttr(names));

86static void collectMapAttrs(Type type, SmallVector<AffineMapAttr> &mapAttrs) {
87 llvm::TypeSwitch<Type, void>(type)
88 .Case([&mapAttrs](PodType t) {
89 for (auto record : t.getRecords()) {
90 collectMapAttrs(record.getType(), mapAttrs);
91 }
92 })
93 .Case([&mapAttrs](array::ArrayType t) {
94 for (auto a : t.getDimensionSizes()) {
95 if (auto m = llvm::dyn_cast<AffineMapAttr>(a)) {
96 mapAttrs.push_back(m);
97 }
98 }
99 })
100 .Case([&mapAttrs](component::StructType t) {
101 for (auto param : t.getParams()) {
102 if (auto m = llvm::dyn_cast<AffineMapAttr>(param)) {
103 mapAttrs.push_back(m);
104 }
105 }
106 }).Default([](Type) {});
107}

116static LogicalResult verifyInitialValues(
117 ValueRange values, ArrayRef<Attribute> names, PodType retTy,
118 llvm::function_ref<InFlightDiagnostic()> emitError
119) {
120 bool failed = false;
121 if (names.size() != values.size()) {
122 emitError() << "number of initialized records and initial values does not match ("
123 << names.size() << " != " << values.size() << ")";
124 failed = true;
125 }
126
127 llvm::StringMap<Type> records = retTy.getRecordMap();
128 llvm::StringSet<> seenNames;
129 for (auto [nameAttr, value] : llvm::zip_equal(names, values)) {
130 auto name = llvm::cast<StringAttr>(nameAttr).getValue(); // Per the ODS spec.
131 if (seenNames.contains(name)) {
132 emitError() << "found duplicated record name '" << name << '\'';
133 failed = true;
134 }
135 seenNames.insert(name);
136
137 if (!records.contains(name)) {
138 emitError() << "record '" << name << "' is not part of the struct";
139 failed = true;
140 continue;
141 }
142
143 auto valueTy = value.getType();
144 auto recordTy = records.at(name);
145 if (valueTy != recordTy) {
146 auto err = emitError();
147 err << "record '" << name << "' expected type " << recordTy << " but got " << valueTy;
148 if (typesUnify(valueTy, recordTy)) {
149 err.attachNote()
150 << "types " << valueTy << " and " << recordTy
151 << " can be unified. Perhaps you can add a 'poly.unifiable_cast' operation?";
152 }
153 failed = true;
154 }
155 }
156
157 return failure(failed);
158}

175LogicalResult NewPodOp::verify() {
176 auto retTy = llvm::dyn_cast<PodType>(getResult().getType());
177 assert(retTy); // per ODS spec of NewPodOp
178
179 bool failed = false;
180 check(
181 verifyInitialValues(getInitialValues(), getInitializedRecords().getValue(), retTy, [this]() {
182 return this->emitError();
183 })
184 );
185 check(verifyAffineMapOperands(this, retTy));
186
187 return failure(failed);
188}

206ParseResult NewPodOp::parse(OpAsmParser &parser, OperationState &result) {
207 /* Grammar
208 * op : record_init map_operands `:` type($result) attr-dict
209 * record_init : `{` record_inits `}`| `{` `}` | $
210 * map_operands : custom<MapOperands> | $
211 * record_inits : symbol `=` operand `,` record_inits | symbol `=` operand
212 */
213
214 auto &props = result.getOrAddProperties<NewPodOp::Properties>();
215
216 SmallVector<Attribute> initializedRecords;
217 // The map may not preserve the order of the operands so it needs to be iterated using
218 // `initializedRecords` that preserves the original order.
219 llvm::StringMap<UnresolvedOp> initialValuesOperands;
220 auto parseElementFn = [&parser, &initializedRecords, &initialValuesOperands] {
221 StringAttr name;
222 UnresolvedOp operand;
223 if (failed(parseRecordInitialization(parser, name, operand))) {
224 return failure();
225 }
226 initializedRecords.push_back(name);
227 initialValuesOperands.insert({name.getValue(), operand});
228 return success();
229 };
230 auto initialValuesLoc = parser.getCurrentLocation();
231 if (parser.parseCommaSeparatedList(AsmParser::Delimiter::OptionalBraces, parseElementFn)) {
232 return failure();
233 }
234 SmallVector<int32_t> mapOperandsGroupSizes;
235 SmallVector<UnresolvedOp> allMapOperands;
236 Type indexTy = parser.getBuilder().getIndexType();
237 bool colonAlreadyParsed = true;
238 auto mapOperandsLoc = parser.getCurrentLocation();
239 // Peek to see if we have affine map operands.
240 // If we don't then the next token must be `:`
241 if (failed(parser.parseOptionalColon())) {
242 colonAlreadyParsed = false;
243 SmallVector<SmallVector<UnresolvedOp>> mapOperands {};
244 if (parseMultiDimAndSymbolList(parser, mapOperands, props.numDimsPerMap)) {
245 return failure();
246 }
247
248 mapOperandsGroupSizes.reserve(mapOperands.size());
249 for (const auto &subRange : mapOperands) {
250 allMapOperands.append(subRange.begin(), subRange.end());
251 mapOperandsGroupSizes.push_back(llzk::checkedCast<int32_t>(subRange.size()));
252 }
253 }
254
255 if (!colonAlreadyParsed && parser.parseColon()) {
256 return failure();
257 }
258
259 PodType resultType;
260 if (parser.parseCustomTypeWithFallback(resultType)) {
261 return failure();
262 }
263 // Now that we have the struct type we can resolve the operands
264 // using the types of the struct.
265 for (auto attr : initializedRecords) {
266 auto name = llvm::cast<StringAttr>(attr); // Per ODS spec of RecordAttr
267 auto lookup = resultType.getRecord(name.getValue(), [&parser, initialValuesLoc] {
268 return parser.emitError(initialValuesLoc);
269 });
270 if (failed(lookup)) {
271 return failure();
272 }
273 const auto &operand = initialValuesOperands.at(name.getValue());
274 if (failed(parser.resolveOperands({operand}, *lookup, initialValuesLoc, result.operands))) {
275 return failure();
276 }
277 }
278 props.operandSegmentSizes = {
279 llzk::checkedCast<int32_t>(initializedRecords.size()),
280 llzk::checkedCast<int32_t>(allMapOperands.size())
281 };
282 props.mapOpGroupSizes = parser.getBuilder().getDenseI32ArrayAttr(mapOperandsGroupSizes);
283 props.initializedRecords = parser.getBuilder().getArrayAttr(initializedRecords);
284 result.addTypes({resultType});
285
286 if (failed(parser.resolveOperands(allMapOperands, indexTy, mapOperandsLoc, result.operands))) {
287 return failure();
288 }
289 {
290 auto loc = parser.getCurrentLocation();
291 if (parser.parseOptionalAttrDict(result.attributes)) {
292 return failure();
293 }
294 if (failed(verifyInherentAttrs(result.name, result.attributes, [&]() {
295 return parser.emitError(loc) << '\'' << result.name.getStringRef() << "' op ";
296 }))) {
297 return failure();
298 }
299 }
300
301 return success();
302}

304void NewPodOp::print(OpAsmPrinter &printer) {
305 auto &os = printer.getStream();
306 auto initializedRecords = getInitializedRecordValues();
307 if (!initializedRecords.empty()) {
308 os << " { ";
309 llvm::interleaveComma(initializedRecords, os, [&os, &printer](auto record) {
310 printer.printSymbolName(record.name);
311 os << " = ";
312 printer.printOperand(record.value);
313 });
314 os << " } ";
315 }
316 printMultiDimAndSymbolList(printer, getOperation(), getMapOperands(), getNumDimsPerMapAttr());
317
318 os << " : ";
319
320 auto type = getResult().getType();
321 if (auto validType = llvm::dyn_cast<PodType>(type)) {
322 printer.printStrippedAttrOrType(validType);
323 } else {
324 printer.printType(type);
325 }
326
327 printer.printOptionalAttrDict(
328 (*this)->getAttrs(),
329 {"initializedRecords", "mapOpGroupSizes", "numDimsPerMap", "operandSegmentSizes"}
330 );
331}

346LogicalResult ReadPodOp::verify() {
347 auto podTy = llvm::dyn_cast<PodType>(getPodRef().getType());
348 if (!podTy) {
349 return emitError() << "reference operand expected a plain-old-data struct but got "
350 << getPodRef().getType();
351 }
352
353 auto lookup = podTy.getRecord(getRecordName(), [this]() { return this->emitError(); });
354 if (failed(lookup)) {
355 return lookup;
356 }
357
358 if (getResult().getType() != *lookup) {
359 return emitError() << "operation result type and type of record do not match ("
360 << getResult().getType() << " != " << *lookup << ")";
361 }
362
363 return success();
364}
