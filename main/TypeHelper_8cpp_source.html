47template <typename Derived, typename ResultType> struct LLZKTypeSwitch {
48 inline ResultType match(Type type) {
49 return llvm::TypeSwitch<Type, ResultType>(type)
50 .template Case<IndexType>([this](auto t) {
51 return static_cast<Derived *>(this)->caseIndex(t);
52 })
53 .template Case<FeltType>([this](auto t) {
54 return static_cast<Derived *>(this)->caseFelt(t);
55 })
56 .template Case<StringType>([this](auto t) {
57 return static_cast<Derived *>(this)->caseString(t);
58 })
59 .template Case<TypeVarType>([this](auto t) {
60 return static_cast<Derived *>(this)->caseTypeVar(t);
61 })
62 .template Case<ArrayType>([this](auto t) {
63 return static_cast<Derived *>(this)->caseArray(t);
64 })
65 .template Case<PodType>([this](auto t) { return static_cast<Derived *>(this)->casePod(t); })
66 .template Case<StructType>([this](auto t) {
67 return static_cast<Derived *>(this)->caseStruct(t);
68 }).Default([this](Type t) {
69 if (t.isSignlessInteger(1)) {
70 return static_cast<Derived *>(this)->caseBool(cast<IntegerType>(t));
71 } else {
72 return static_cast<Derived *>(this)->caseInvalid(t);
73 }
74 });
75 }
76
77private:
78 friend Derived;
79 LLZKTypeSwitch() = default;
80};

98BuildShortTypeString &BuildShortTypeString::append(Type type) {
99 size_t position = ret.size();
100 (void)position; // tell compiler it's intentionally unused in builds without assertions
101
102 struct Impl : LLZKTypeSwitch<Impl, void> {
103 BuildShortTypeString &outer;
104 Impl(BuildShortTypeString &outerRef) : outer(outerRef) {}
105
106 void caseInvalid(Type) { outer.ss << "!INVALID"; }
107 void caseBool(IntegerType) { outer.ss << 'b'; }
108 void caseIndex(IndexType) { outer.ss << 'i'; }
109 void caseFelt(FeltType) { outer.ss << 'f'; }
110 void caseString(StringType) { outer.ss << 's'; }
111 void caseTypeVar(TypeVarType t) {
112 outer.ss << "!t<";
113 outer.appendSymName(llvm::cast<TypeVarType>(t).getRefName());
114 outer.ss << '>';
115 }
116 void caseArray(ArrayType t) {
117 outer.ss << "!a<";
118 outer.append(t.getElementType());
119 outer.ss << ':';
120 outer.append(t.getDimensionSizes());
121 outer.ss << '>';
122 }
123 void casePod(PodType t) {
124 outer.ss << "!r<";
125 for (auto record : t.getRecords()) {
126 outer.appendSymRef(record.getNameSym());
127 }
128 outer.ss << '>';
129 }
130 void caseStruct(StructType t) {
131 outer.ss << "!s<";
132 outer.appendSymRef(t.getNameRef());
133 if (ArrayAttr params = t.getParams()) {
134 outer.ss << '_';
135 outer.append(params.getValue());
136 }
137 outer.ss << '>';
138 }
139 };
140 Impl(*this).match(type);
141
142 assert(
143 ret.find(PLACEHOLDER, position) == std::string::npos &&
144 "formatting a Type should not produce the 'PLACEHOLDER' char"
145 );
146 return *this;
147}

149BuildShortTypeString &BuildShortTypeString::append(Attribute a) {
150 // Special case for inserting the `PLACEHOLDER`
151 if (a == nullptr) {
152 ss << PLACEHOLDER;
153 return *this;
154 }
155
156 size_t position = ret.size();
157 (void)position; // tell compiler it's intentionally unused in builds without assertions
158
159 // Adapted from AsmPrinter::Impl::printAttributeImpl()
160 if (auto ia = llvm::dyn_cast<IntegerAttr>(a)) {
161 Type ty = ia.getType();
162 bool isUnsigned = ty.isUnsignedInteger() || ty.isSignlessInteger(1);
163 ia.getValue().print(ss, !isUnsigned);
164 } else if (auto sra = llvm::dyn_cast<SymbolRefAttr>(a)) {
165 appendSymRef(sra);
166 } else if (auto ta = llvm::dyn_cast<TypeAttr>(a)) {
167 append(ta.getValue());
168 } else if (auto ama = llvm::dyn_cast<AffineMapAttr>(a)) {
169 ss << "!m<";
170 // Filter to remove spaces from the affine_map representation
171 filtered_raw_ostream fs(ss, [](char c) { return c == ' '; });
172 ama.getValue().print(fs);
173 fs.flush();
174 ss << '>';
175 } else if (auto aa = llvm::dyn_cast<ArrayAttr>(a)) {
176 append(aa.getValue());
177 } else {
178 // All valid/legal cases must be covered above
179 assertValidAttrForParamOfType(a);
180 }
181 assert(
182 ret.find(PLACEHOLDER, position) == std::string::npos &&
183 "formatting a non-null Attribute should not produce the 'PLACEHOLDER' char"
184 );
185 return *this;
186}

188BuildShortTypeString &BuildShortTypeString::append(ArrayRef<Attribute> attrs) {
189 llvm::interleave(attrs, ss, [this](Attribute a) { append(a); }, "_");
190 return *this;

193std::string BuildShortTypeString::from(const std::string &base, ArrayRef<Attribute> attrs) {
194 BuildShortTypeString bldr;
195
196 bldr.ret.reserve(base.size() + attrs.size()); // reserve minimum space required
197
198 // First handle replacements of PLACEHOLDER
199 const auto *END = attrs.end();
200 const auto *IT = attrs.begin();
201 {
202 size_t start = 0;
203 for (size_t pos; (pos = base.find(PLACEHOLDER, start)) != std::string::npos; start = pos + 1) {
204 // Append original up to the PLACEHOLDER
205 bldr.ret.append(base, start, pos - start);
206 // Append the formatted Attribute
207 assert(IT != END && "must have an Attribute for every 'PLACEHOLDER' char");
208 bldr.append(*IT++);
209 }
210 // Append remaining suffix of the original
211 bldr.ret.append(base, start, base.size() - start);
212 }
213
214 // Append any remaining Attributes
215 if (IT != END) {
216 bldr.ss << '_';
217 bldr.append(ArrayRef(IT, END));
218 }
219
220 return bldr.ret;
221}

228 template <typename StreamType> struct Appender {
229
230 // single
231 template <typename Ty> static inline void append(StreamType &stream) {
232 stream << '\'' << Ty::name << '\'';
233 }
234
235 // multiple
236 template <typename First, typename Second, typename... Rest>
237 static void append(StreamType &stream) {
238 append<First>(stream);
239 stream << ", ";
240 append<Second, Rest...>(stream);
241 }
242
243 // full list with wrapping brackets
244 static inline void append(StreamType &stream) {
245 stream << '[';
246 append<Types...>(stream);
247 stream << ']';
248 }
249 };

253 template <typename T> static inline bool matches(const T &value) {
254 return llvm::isa_and_present<Types...>(value);

390 bool areValidArrayDimSizes(ArrayRef<Attribute> dimensionSizes, EmitErrorFn emitError = nullptr) {
391 // In LLZK, the number of array dimensions must always be known, i.e., `hasRank()==true`
392 if (dimensionSizes.empty()) {
393 if (emitError) {
394 emitError().append("array must have at least one dimension").report();
395 }
396 return false;
397 }
398 // Rather than immediately returning on failure, we check all dimensions and aggregate to
399 // provide as many errors are possible in a single verifier run.
400 bool success = true;
401 for (Attribute a : dimensionSizes) {
402 if (!ArrayDimensionTypes::matches(a)) {
403 ArrayDimensionTypes::reportInvalid(emitError, a, "Array dimension");
404 success = false;
405 } else if (no_var && !llvm::isa_and_present<IntegerAttr>(a)) {
406 TypeList<IntegerAttr>::reportInvalid(emitError, a, "Concrete array dimension");
407 success = false;
408 } else if (failed(verifyAffineMapAttrType(emitError, a))) {
409 success = false;
410 } else if (failed(verifyIntAttrType(emitError, a))) {
411 success = false;
412 }
413 }
414 return success;
415 }

417 bool isValidArrayElemTypeImpl(Type type) {
418 // ArrayType element can be any valid type sans ArrayType itself.
419 return !llvm::isa<ArrayType>(type) && isValidTypeImpl(type);
420 }
421
422 bool isValidArrayTypeImpl(
423 Type elementType, ArrayRef<Attribute> dimensionSizes, EmitErrorFn emitError = nullptr
424 ) {
425 if (!areValidArrayDimSizes(dimensionSizes, emitError)) {
426 return false;
427 }
428
429 // Ensure array element type is valid
430 if (!isValidArrayElemTypeImpl(elementType)) {
431 if (emitError) {
432 // Print proper message if `elementType` is not a valid LLZK type or
433 // if it's simply not the right kind of type for an array element.
434 if (succeeded(checkValidType(emitError, elementType))) {
435 emitError()
436 .append(
437 '\'', ArrayType::name, "' element type cannot be '",
438 elementType.getAbstractType().getName(), '\''
439 )
440 .report();
441 }
442 }
443 return false;
444 }
445 return true;
446 }

448 bool isValidArrayTypeImpl(Type type) {
449 if (ArrayType arrTy = llvm::dyn_cast<ArrayType>(type)) {
450 return isValidArrayTypeImpl(arrTy.getElementType(), arrTy.getDimensionSizes());
451 }
452 return false;
453 }

455 // Note: The `no*` flags here refer to Types nested within a TypeAttr parameter (if any) except
456 // for the `no_struct_params` flag which requires that `params` is null or empty.
457 bool areValidStructTypeParams(ArrayAttr params, EmitErrorFn emitError = nullptr) {
458 if (isNullOrEmpty(params)) {
459 return true;
460 }
461 if (no_struct_params) {
462 return false;
463 }
464 bool success = true;
465 for (Attribute p : params) {
466 if (!StructParamTypes::matches(p)) {
467 StructParamTypes::reportInvalid(emitError, p, "Struct parameter");
468 success = false;
469 } else if (TypeAttr tyAttr = llvm::dyn_cast<TypeAttr>(p)) {
470 if (!isValidTypeImpl(tyAttr.getValue())) {
471 if (emitError) {
472 emitError().append("expected a valid LLZK type but found ", tyAttr.getValue()).report();
473 }
474 success = false;
475 }
476 } else if (no_var && !llvm::isa<IntegerAttr>(p)) {
477 TypeList<IntegerAttr>::reportInvalid(emitError, p, "Concrete struct parameter");
478 success = false;
479 } else if (failed(verifyAffineMapAttrType(emitError, p))) {
480 success = false;
481 } else if (failed(verifyIntAttrType(emitError, p))) {
482 success = false;
483 }
484 }
485
486 return success;
487 }

494bool AllowedTypes::isValidTypeImpl(Type type) {
495 assert(
496 !(no_int && no_felt && no_string && no_var && no_struct && no_array && no_pod) &&
497 "All types have been deactivated"
498 );
499 struct Impl : LLZKTypeSwitch<Impl, bool> {
500 AllowedTypes &outer;
501 Impl(AllowedTypes &outerRef) : outer(outerRef) {}
502
503 bool caseBool(IntegerType t) { return !outer.no_int && t.isSignlessInteger(1); }
504 bool caseIndex(IndexType) { return !outer.no_int; }
505 bool caseFelt(FeltType) { return !outer.no_felt; }
506 bool caseString(StringType) { return !outer.no_string; }
507 bool caseTypeVar(TypeVarType) { return !outer.no_var; }
508 bool caseArray(ArrayType t) {
509 return !outer.no_array &&
510 outer.isValidArrayTypeImpl(t.getElementType(), t.getDimensionSizes());
511 }
512 bool casePod(PodType t) { return !outer.no_pod && outer.areValidPodRecords(t.getRecords()); }
513 bool caseStruct(StructType t) {
514 // Note: The `no*` flags here refer to Types nested within a TypeAttr parameter.
515 if (outer.no_struct || !outer.validColumns(t)) {
516 return false;
517 }
518 return !outer.no_struct && outer.areValidStructTypeParams(t.getParams());
519 }
520 bool caseInvalid(Type) { return false; }
521 };
522 return Impl(*this).match(type);
523}

563uint64_t computeEmitEqCardinality(Type type) {
564 struct Impl : LLZKTypeSwitch<Impl, uint64_t> {
565 uint64_t caseBool(IntegerType) { return 1; }
566 uint64_t caseIndex(IndexType) { return 1; }
567 uint64_t caseFelt(FeltType) { return 1; }
568 uint64_t caseArray(ArrayType t) {
569 int64_t n = t.getNumElements();
570 return llzk::checkedCast<uint64_t>(n) * computeEmitEqCardinality(t.getElementType());
571 }
572 uint64_t caseStruct(StructType) { llvm_unreachable("not a valid EmitEq type"); }
573 uint64_t casePod(PodType t) {
574 return std::accumulate(
575 t.getRecords().begin(), t.getRecords().end(), 0,
576 [](const uint64_t &acc, const RecordAttr &record) {
577 return computeEmitEqCardinality(record.getType()) + acc;
578 }
579 );
580 }
581 uint64_t caseString(StringType) { llvm_unreachable("not a valid EmitEq type"); }
582 uint64_t caseTypeVar(TypeVarType) { llvm_unreachable("tvar has unknown cardinality"); }
583 uint64_t caseInvalid(Type) { llvm_unreachable("not a valid LLZK type"); }
584 };
585 return Impl().match(type);
586}

600struct UnifierImpl {
601 ArrayRef<StringRef> rhsRevPrefix;
602 UnificationMap *unifications;
603 AffineInstantiations *affineToIntTracker;
604 // This optional function can be used to provide an exception to the standard unification
605 // rules and return a true/success result when it otherwise may not.
606 llvm::function_ref<bool(Type oldTy, Type newTy)> overrideSuccess;
607
608 UnifierImpl(UnificationMap *unificationMap, ArrayRef<StringRef> rhsReversePrefix = {})
609 : rhsRevPrefix(rhsReversePrefix), unifications(unificationMap), affineToIntTracker(nullptr),
610 overrideSuccess(nullptr) {}
611
612 bool typeParamsUnify(
613 const ArrayRef<Attribute> &lhsParams, const ArrayRef<Attribute> &rhsParams,
614 bool unifyDynamicSize = false
615 ) {
616 auto pred = [this, unifyDynamicSize](auto lhsAttr, auto rhsAttr) {
617 return paramAttrUnify(lhsAttr, rhsAttr, unifyDynamicSize);
618 };
619 return (lhsParams.size() == rhsParams.size()) &&
620 std::equal(lhsParams.begin(), lhsParams.end(), rhsParams.begin(), pred);
621 }
622
623 UnifierImpl &trackAffineToInt(AffineInstantiations *tracker) {
624 this->affineToIntTracker = tracker;
625 return *this;
626 }
627
628 UnifierImpl &withOverrides(llvm::function_ref<bool(Type oldTy, Type newTy)> overrides) {
629 this->overrideSuccess = overrides;
630 return *this;
631 }

637 bool typeParamsUnify(
638 const ArrayAttr &lhsParams, const ArrayAttr &rhsParams, bool unifyDynamicSize = false
639 ) {
640 ArrayRef<Attribute> emptyParams;
641 return typeParamsUnify(
642 lhsParams ? lhsParams.getValue() : emptyParams,
643 rhsParams ? rhsParams.getValue() : emptyParams, unifyDynamicSize
644 );

647 bool arrayTypesUnify(ArrayType lhs, ArrayType rhs) {
648 // Check if the element types of the two arrays can unify
649 if (!typesUnify(lhs.getElementType(), rhs.getElementType())) {
650 return false;
651 }
652 // Check if the dimension size attributes unify between the LHS and RHS
653 return typeParamsUnify(
654 lhs.getDimensionSizes(), rhs.getDimensionSizes(), /*unifyDynamicSize=*/true
655 );
656 }
657
658 bool structTypesUnify(StructType lhs, StructType rhs) {
659 LLVM_DEBUG({
660 llvm::dbgs() << "[structTypesUnify] lhs = " << lhs << ", rhs = " << rhs << '\n';
661 });
662 // Check if it references the same StructDefOp, considering the additional RHS path prefix.
663 SmallVector<StringRef> rhsNames = getNames(rhs.getNameRef());
664 rhsNames.insert(rhsNames.begin(), rhsRevPrefix.rbegin(), rhsRevPrefix.rend());
665 auto lhsNames = getNames(lhs.getNameRef());
666 if (rhsNames != lhsNames) {
667 LLVM_DEBUG({
668 llvm::interleaveComma(
669 lhsNames, llvm::dbgs() << "[structTypesUnify] names do not match\n"
670 << " lhsNames = ["
671 );
672 llvm::interleaveComma(
673 rhsNames, llvm::dbgs() << "]\n"
674 << " rhsNames = ["
675 );
676 llvm::dbgs() << "]\n";
677 });
678 return false;
679 }
680 LLVM_DEBUG({ llvm::dbgs() << "[structTypesUnify] checking unification of parameters\n"; });
681 // Check if the parameters unify between the LHS and RHS
682 return typeParamsUnify(lhs.getParams(), rhs.getParams(), /*unifyDynamicSize=*/false);
683 }
684
685 bool podTypesUnify(PodType lhs, PodType rhs) {
686 // Same number of records, with the same names in the same order and record types unify.
687 auto lhsRecords = lhs.getRecords();
688 auto rhsRecords = rhs.getRecords();
689
690 return lhsRecords.size() == rhsRecords.size() &&
691 llvm::all_of(llvm::zip_equal(lhsRecords, rhsRecords), [this](auto &&records) {
692 auto &&[lhsRecord, rhsRecord] = records;
693 return lhsRecord.getName() == rhsRecord.getName() &&
694 typesUnify(lhsRecord.getType(), rhsRecord.getType());
695 });
696 }
697
698 bool typesUnify(Type lhs, Type rhs) {
699 if (lhs == rhs) {
700 return true;
701 }
702 if (overrideSuccess && overrideSuccess(lhs, rhs)) {
703 return true;
704 }
705 // A type variable can be any type, thus it unifies with anything.
706 if (TypeVarType lhsTvar = llvm::dyn_cast<TypeVarType>(lhs)) {
707 track(Side::LHS, lhsTvar.getNameRef(), rhs);
708 return true;
709 }
710 if (TypeVarType rhsTvar = llvm::dyn_cast<TypeVarType>(rhs)) {
711 track(Side::RHS, rhsTvar.getNameRef(), lhs);
712 return true;
713 }
714 if (llvm::isa<StructType>(lhs) && llvm::isa<StructType>(rhs)) {
715 return structTypesUnify(llvm::cast<StructType>(lhs), llvm::cast<StructType>(rhs));
716 }
717 if (llvm::isa<ArrayType>(lhs) && llvm::isa<ArrayType>(rhs)) {
718 return arrayTypesUnify(llvm::cast<ArrayType>(lhs), llvm::cast<ArrayType>(rhs));
719 }
720 if (llvm::isa<PodType>(lhs) && llvm::isa<PodType>(rhs)) {
721 return podTypesUnify(llvm::cast<PodType>(lhs), llvm::cast<PodType>(rhs));
722 }
723 return false;
724 }

727 template <typename Tracker, typename Key, typename Val>
728 inline void track(Tracker &tracker, Side side, Key keyHead, Val val) {
729 auto key = std::make_pair(keyHead, side);
730 auto it = tracker.find(key);
731 if (it == tracker.end()) {
732 tracker.try_emplace(key, val);
733 } else if (it->getSecond() != val) {
734 it->second = nullptr;
735 }
736 }

738 void track(Side side, SymbolRefAttr symRef, Type ty) {
739 if (unifications) {
740 Attribute attr;
741 if (TypeVarType tvar = dyn_cast<TypeVarType>(ty)) {
742 // If 'ty' is TypeVarType<@S>, just map to @S directly.
743 attr = tvar.getNameRef();
744 } else {
745 // Otherwise wrap as a TypeAttr.
746 attr = TypeAttr::get(ty);
747 }
748 assert(symRef);
749 assert(attr);
750 track(*unifications, side, symRef, attr);
751 }

754 void track(Side side, SymbolRefAttr symRef, Attribute attr) {
755 if (unifications) {
756 // If 'attr' is TypeAttr<TypeVarType<@S>>, just map to @S directly.
757 if (TypeAttr tyAttr = dyn_cast<TypeAttr>(attr)) {
758 if (TypeVarType tvar = dyn_cast<TypeVarType>(tyAttr.getValue())) {
759 attr = tvar.getNameRef();
760 }
761 }
762 assert(symRef);
763 assert(attr);
764 // If 'attr' is a SymbolRefAttr, map in both directions for the correctness of
765 // `isMoreConcreteUnification()` which relies on RHS check while other external
766 // checks on the UnificationMap may do LHS checks, and in the case of both being
767 // SymbolRefAttr, unification in either direction is possible.
768 if (SymbolRefAttr otherSymAttr = dyn_cast<SymbolRefAttr>(attr)) {
769 track(*unifications, reverse(side), otherSymAttr, symRef);
770 }
771 track(*unifications, side, symRef, attr);
772 }
773 }

776 if (affineToIntTracker) {
777 assert(affineAttr);
778 assert(intAttr);
779 assert(!isDynamic(intAttr));
780 track(*affineToIntTracker, side, affineAttr, intAttr);
781 }

784 bool paramAttrUnify(Attribute lhsAttr, Attribute rhsAttr, bool unifyDynamicSize = false) {
785 assertValidAttrForParamOfType(lhsAttr);
786 assertValidAttrForParamOfType(rhsAttr);
787 // Straightforward equality check.
788 if (lhsAttr == rhsAttr) {
789 return true;
790 }
791 // AffineMapAttr can unify with IntegerAttr (other than kDynamic) because struct parameter
792 // instantiation will result in conversion of AffineMapAttr to IntegerAttr.
793 if (AffineMapAttr lhsAffine = llvm::dyn_cast<AffineMapAttr>(lhsAttr)) {
794 if (IntegerAttr rhsInt = llvm::dyn_cast<IntegerAttr>(rhsAttr)) {
795 if (!isDynamic(rhsInt)) {
796 track(Side::LHS, lhsAffine, rhsInt);
797 return true;
798 }
799 }
800 }
801 if (AffineMapAttr rhsAffine = llvm::dyn_cast<AffineMapAttr>(rhsAttr)) {
802 if (IntegerAttr lhsInt = llvm::dyn_cast<IntegerAttr>(lhsAttr)) {
803 if (!isDynamic(lhsInt)) {
804 track(Side::RHS, rhsAffine, lhsInt);
805 return true;
806 }
807 }
808 }
809 // If either side is a SymbolRefAttr, assume they unify because either flattening or a pass with
810 // a more involved value analysis is required to check if they are actually the same value.
811 if (SymbolRefAttr lhsSymRef = llvm::dyn_cast<SymbolRefAttr>(lhsAttr)) {
812 track(Side::LHS, lhsSymRef, rhsAttr);
813 return true;
814 }
815 if (SymbolRefAttr rhsSymRef = llvm::dyn_cast<SymbolRefAttr>(rhsAttr)) {
816 track(Side::RHS, rhsSymRef, lhsAttr);
817 return true;
818 }
819 // If either side is ShapedType::kDynamic then, similarly to Symbols, assume they unify.
820 // NOTE: Dynamic array dimensions (i.e. '?') are allowed in LLZK but should generally be
821 // restricted to scenarios where it can be replaced with a concrete value during the flattening
822 // pass, such as a `unifiable_cast` where the other side of the cast has concrete dimensions or
823 // extern functions with varargs.
824 if (unifyDynamicSize) {
825 auto dyn_cast_if_dynamic = [](Attribute attr) -> IntegerAttr {
826 if (IntegerAttr intAttr = llvm::dyn_cast<IntegerAttr>(attr)) {
827 if (isDynamic(intAttr)) {
828 return intAttr;
829 }
830 }
831 return nullptr;
832 };
833 auto is_const_like = [](Attribute attr) {
834 return llvm::isa_and_present<IntegerAttr, SymbolRefAttr, AffineMapAttr>(attr);
835 };
836 if (IntegerAttr lhsIntAttr = dyn_cast_if_dynamic(lhsAttr)) {
837 if (is_const_like(rhsAttr)) {
838 return true;
839 }
840 }
841 if (IntegerAttr rhsIntAttr = dyn_cast_if_dynamic(rhsAttr)) {
842 if (is_const_like(lhsAttr)) {
843 return true;
844 }
845 }
846 }
847 // If both are type refs, check for unification of the types.
848 if (TypeAttr lhsTy = llvm::dyn_cast<TypeAttr>(lhsAttr)) {
849 if (TypeAttr rhsTy = llvm::dyn_cast<TypeAttr>(rhsAttr)) {
850 return typesUnify(lhsTy.getValue(), rhsTy.getValue());
851 }
852 }
853 // Otherwise, they do not unify.
854 return false;
855 }
856};

894bool isMoreConcreteUnification(
895 Type oldTy, Type newTy, llvm::function_ref<bool(Type oldTy, Type newTy)> knownOldToNew
896) {
897 UnificationMap unifications;
898 AffineInstantiations affineInstantiations;
899 // Run type unification with the addition that affine map can become integer in the new type.
900 if (!UnifierImpl(&unifications)
901 .trackAffineToInt(&affineInstantiations)
902 .withOverrides(knownOldToNew)
903 .typesUnify(oldTy, newTy)) {
904 return false;
905 }
906
907 // If either map contains RHS-keyed mappings then the old type is "more concrete" than the new.
908 // In the UnificationMap, a RHS key would indicate that the new type contains a SymbolRef (i.e.
909 // the "least concrete" attribute kind) where the old type contained any other attribute. In the
910 // AffineInstantiations map, a RHS key would indicate that the new type contains an AffineMapAttr
911 // where the old type contains an IntegerAttr.
912 auto entryIsRHS = [](const auto &entry) { return entry.first.second == Side::RHS; };
913 return !llvm::any_of(unifications, entryIsRHS) && !llvm::any_of(affineInstantiations, entryIsRHS);
914}

916FailureOr<IntegerAttr> forceIntType(IntegerAttr attr, EmitErrorFn emitError) {
917 if (llvm::isa<IndexType>(attr.getType())) {
918 return attr;
919 }
920 // Ensure the APInt is the right bitwidth for IndexType or else
921 // IntegerAttr::verify(..) will report an error.
922 APInt value = attr.getValue();
923 auto compare = value.getBitWidth() <=> IndexType::kInternalStorageBitWidth;
924 if (compare < 0) {
925 value = value.zext(IndexType::kInternalStorageBitWidth);
926 } else if (compare > 0) {
927 return emitError().append("value is too large for `index` type: ", debug::toStringOne(value));
928 }
929 return IntegerAttr::get(IndexType::get(attr.getContext()), value);
930}

1009LogicalResult
1010verifySubArrayType(EmitErrorFn emitError, ArrayType arrayType, ArrayType subArrayType) {
1011 ArrayRef<Attribute> dimsFromArr = arrayType.getDimensionSizes();
1012 size_t numArrDims = dimsFromArr.size();
1013 ArrayRef<Attribute> dimsFromSubArr = subArrayType.getDimensionSizes();
1014 size_t numSubArrDims = dimsFromSubArr.size();
1015
1016 if (numArrDims < numSubArrDims) {
1017 return emitError().append(
1018 "subarray type ", subArrayType, " has more dimensions than array type ", arrayType
1019 );
1020 }
1021
1022 size_t toDrop = numArrDims - numSubArrDims;
1023 ArrayRef<Attribute> dimsFromArrReduced = dimsFromArr.drop_front(toDrop);
1024
1025 // Ensure dimension sizes are compatible (ignoring the indexed dimensions)
1026 if (!typeParamsUnify(dimsFromArrReduced, dimsFromSubArr)) {
1027 std::string message;
1028 llvm::raw_string_ostream ss(message);
1029 auto appendOne = [&ss](Attribute a) { appendWithoutType(ss, a); };
1030 ss << "cannot unify array dimensions [";
1031 llvm::interleaveComma(dimsFromArrReduced, ss, appendOne);
1032 ss << "] with [";
1033 llvm::interleaveComma(dimsFromSubArr, ss, appendOne);
1034 ss << "]";
1035 return emitError().append(message);
1036 }
1037
1038 // Ensure element types of the arrays are compatible
1039 if (!typesUnify(arrayType.getElementType(), subArrayType.getElementType())) {
1040 return emitError().append(
1041 "incorrect array element type; expected: ", arrayType.getElementType(),
1042 ", found: ", subArrayType.getElementType()
1043 );
1044 }
1045
1046 return success();
1047}
