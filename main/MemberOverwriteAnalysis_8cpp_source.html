31ChangeResult MemberOverwriteLattice::record(MemberWriteOp write) {
32 auto name = write.getMemberName();
33
34 bool changed = false;
35
36 if (auto it = mayWrites.find(name); it != mayWrites.end() && it->second != write) {
37 // .insert(...) returns true if an insertion was performed (i.e., it wasn't present before),
38 // meaning there was a change
39 changed |= overwrites.insert({mayWrites.at(name), write});
40 } else {
41 mayWrites.insert({name, write});
42 changed = true;
43 }
44
45 changed |= mustWrites.insert(name);
46 return ChangeResult {changed};
47}

49ChangeResult MemberOverwriteLattice::join(const AbstractDenseLattice &other) {
50 const auto *rhs = dynamic_cast<const MemberOverwriteLattice *>(&other);
51 ensure(rhs, "cannot join incomparable lattices");
52
53 LLVM_DEBUG(
54 llvm::dbgs() << "Joining " << *dyn_cast<ProgramPoint *>(getAnchor()) << "(" << *this
55 << ") with " << *dyn_cast<ProgramPoint *>(rhs->getAnchor()) << "(" << *rhs
56 << ")\n"
57 );
58 bool changed = false;
59
60 // Union the mayWrites
61 for (auto [name, write] : rhs->mayWrites) {
62 auto it = mayWrites.find(name);
63 changed |= it == mayWrites.end() || it->second != write;
64 mayWrites[name] = write;
65 }
66 changed |= overwrites.set_union(rhs->overwrites);
67
68 // "Intersect" the mustWrites
69 changed |= mustWrites.intersect(rhs->mustWrites);
70
71 return ChangeResult {changed};
72}

84LogicalResult MemberOverwriteAnalysis::visitOperation(
85 Operation *op, const MemberOverwriteLattice &before, MemberOverwriteLattice *after
86) {
87 ChangeResult result = after->join(before);
88
89 LLVM_DEBUG(llvm::dbgs() << "Visiting operation: " << *op << ": " << before << "\n");
90
91 if (auto write = dyn_cast<MemberWriteOp>(op)) {
92 result |= after->record(write);
93 }
94
95 propagateIfChanged(after, result);
96 return success();
97}

99llvm::FailureOr<std::pair<llvm::SetVector<Overwrite>, FuzzySet>>
100analyzeStruct(component::StructDefOp structDef) {
101 function::FuncDefOp computeOrProductFunc = structDef.getComputeFuncOp();
102 if (!computeOrProductFunc) {
103 computeOrProductFunc = structDef.getProductFuncOp();
104 }
105
106 DataFlowSolver solver {DataFlowConfig {}.setInterprocedural(false)};
107 llzk::dataflow::loadRequiredAnalyses(solver);
108 solver.load<MemberOverwriteAnalysis>();
109 if (failed(solver.initializeAndRun(computeOrProductFunc))) {
110 return llvm::failure();
111 }
112
113 auto &funcBody = computeOrProductFunc.getBody();
114 if (funcBody.empty()) {
115 // If there's nothing, just build a default lattice element (no overwrites, everything is
116 // unwritten)
117 return {{{}, {}}};
118 }
119
120 auto *returnOp = funcBody.back().getTerminator();
121 const auto *lattice =
122 solver.lookupState<MemberOverwriteLattice>(solver.getProgramPointAfter(returnOp));
123 return {{lattice->getOverwrites(), lattice->mustWrites}};
124}
