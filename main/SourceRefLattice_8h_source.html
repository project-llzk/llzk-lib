90class SourceRefLattice : public mlir::dataflow::AbstractDenseLattice {
91public:
92 // mlir::Value is used for read-like operations that create references in their results,
93 // mlir::Operation* is used for write-like operations that reference values as their destinations
94 using ValueTy = llvm::PointerUnion<mlir::Value, mlir::Operation *>;
95 using ValueMap = mlir::DenseMap<ValueTy, SourceRefLatticeValue>;
96 // Used to lookup MLIR values/operations from a given SourceRef (all values that a ref is
97 // referenced by)
98 using ValueSet = mlir::DenseSet<ValueTy>;
99 using Ref2Val = mlir::DenseMap<SourceRef, mlir::DenseSet<ValueTy>>;
100 using AbstractDenseLattice::AbstractDenseLattice;
101
102 /* Static utilities */
103
108 static mlir::FailureOr<SourceRef> getSourceRef(mlir::Value val);
109
110 /* Required methods */
111
113 mlir::ChangeResult join(const AbstractDenseLattice &rhs) override {
114 return setValues(static_cast<const SourceRefLattice &>(rhs).valMap);
115 }
116
118 mlir::ChangeResult meet(const AbstractDenseLattice & /*rhs*/) override {
119 llvm::report_fatal_error("meet operation is not supported for SourceRefLattice");
120 return mlir::ChangeResult::NoChange;
121 }
122
123 void print(mlir::raw_ostream &os) const override;
124
125 /* Update utility methods */
126
127 mlir::ChangeResult setValues(const ValueMap &rhs);
128
129 mlir::ChangeResult setValue(ValueTy v, const SourceRefLatticeValue &rhs);
130
131 mlir::ChangeResult setValue(ValueTy v, const SourceRef &ref);
132
133 SourceRefLatticeValue getOrDefault(ValueTy v) const;
134
135 SourceRefLatticeValue getReturnValue(unsigned i) const;
136
137 ValueSet lookupValues(const SourceRef &r) const;
138
139 size_t size() const { return valMap.size(); }
140
141 const ValueMap &getMap() const { return valMap; }
142
143 const Ref2Val &getRef2Val() const { return refMap; }
144
145 ValueMap::iterator begin() { return valMap.begin(); }
146 ValueMap::iterator end() { return valMap.end(); }
147 ValueMap::const_iterator begin() const { return valMap.begin(); }
148 ValueMap::const_iterator end() const { return valMap.end(); }
149
150 friend mlir::raw_ostream &operator<<(mlir::raw_ostream &os, const SourceRefLattice &v);
151
152private:
153 ValueMap valMap;
154 Ref2Val refMap;
155};
