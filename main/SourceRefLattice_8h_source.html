92class SourceRefLattice : public mlir::dataflow::AbstractDenseLattice {
93public:
94 // mlir::Value is used for read-like operations that create references in their results,
95 // mlir::Operation* is used for write-like operations that reference values as their destinations
96 using ValueTy = llvm::PointerUnion<mlir::Value, mlir::Operation *>;
97 using ValueMap = mlir::DenseMap<ValueTy, SourceRefLatticeValue>;
98 // Used to lookup MLIR values/operations from a given SourceRef (all values that a ref is
99 // referenced by)
100 using ValueSet = mlir::DenseSet<ValueTy>;
101 using Ref2Val = mlir::DenseMap<SourceRef, mlir::DenseSet<ValueTy>>;
102 using AbstractDenseLattice::AbstractDenseLattice;
103
104 /* Static utilities */
105
110 static mlir::FailureOr<SourceRef> getSourceRef(mlir::Value val);
111
112 /* Required methods */
113
115 mlir::ChangeResult join(const AbstractDenseLattice &rhs) override {
116 return setValues(static_cast<const SourceRefLattice &>(rhs).valMap);
117 }
118
120 mlir::ChangeResult meet(const AbstractDenseLattice & /*rhs*/) override {
121 llvm::report_fatal_error("meet operation is not supported for SourceRefLattice");
122 return mlir::ChangeResult::NoChange;
123 }

125 void print(mlir::raw_ostream &os) const override;
126
127 /* Update utility methods */
128

134
135 SourceRefLatticeValue getOrDefault(ValueTy v) const;
136
137 SourceRefLatticeValue getReturnValue(unsigned i) const;
138
139 ValueSet lookupValues(const SourceRef &r) const;
140
141 size_t size() const { return valMap.size(); }
142
143 const ValueMap &getMap() const { return valMap; }
144
145 const Ref2Val &getRef2Val() const { return refMap; }
146
147 ValueMap::iterator begin() { return valMap.begin(); }
148 ValueMap::iterator end() { return valMap.end(); }
149 ValueMap::const_iterator begin() const { return valMap.begin(); }
150 ValueMap::const_iterator end() const { return valMap.end(); }
151
152 friend mlir::raw_ostream &operator<<(mlir::raw_ostream &os, const SourceRefLattice &v);
153
154private:
155 ValueMap valMap;
156 Ref2Val refMap;
157};
