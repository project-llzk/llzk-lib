24bool IntToFeltOp::isCompatibleReturnTypes(::mlir::TypeRange lhs, ::mlir::TypeRange rhs) {
25 return lhs.size() == rhs.size() && llvm::all_of(llvm::zip_equal(lhs, rhs), [](auto pair) {
26 auto [lhsType, rhsType] = pair;
27 auto lhsFeltType = mlir::dyn_cast<llzk::felt::FeltType>(lhsType);
28 auto rhsFeltType = mlir::dyn_cast<llzk::felt::FeltType>(rhsType);
29
30 // If both types are felts but NOT structurally equal then check if the types are valid
31 // with the additional consideration that lhs is allowed to NOT have
32 // a declared field.
33 if (lhsFeltType && rhsFeltType && lhsFeltType != rhsFeltType) {
34 // If we reached this point we know that the felts are not equal and that only the lhs is
35 // allowed to not have a declared field. Thus, rhs must have a declared field. If lhs has a
36 // declared field, then, since they are not structurally equal, it must be a different field
37 // than rhs. With all that, the types are compatible if lhs does not have a field, so we can
38 // simply return that.
39 return !lhsFeltType.hasField();
40 }
41
42 // Any other case gets handled by standard equality.
43 return lhsType == rhsType;
44 });
45}
46} // namespace llzk::cast
