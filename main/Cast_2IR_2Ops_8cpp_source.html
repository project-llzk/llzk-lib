27bool IntToFeltOp::isCompatibleReturnTypes(::mlir::TypeRange lhs, ::mlir::TypeRange rhs) {
28 return lhs.size() == rhs.size() && llvm::all_of(llvm::zip_equal(lhs, rhs), [](auto pair) {
29 auto [lhsType, rhsType] = pair;
30 auto lhsFeltType = mlir::dyn_cast<llzk::felt::FeltType>(lhsType);
31 auto rhsFeltType = mlir::dyn_cast<llzk::felt::FeltType>(rhsType);
32
33 // If both types are felts but NOT structurally equal then check if the types are valid
34 // with the additional consideration that lhs is allowed to NOT have
35 // a declared field.
36 if (lhsFeltType && rhsFeltType && lhsFeltType != rhsFeltType) {
37 // If we reached this point we know that the felts are not equal and that only the lhs is
38 // allowed to not have a declared field. Thus, rhs must have a declared field. If lhs has a
39 // declared field, then, since they are not structurally equal, it must be a different field
40 // than rhs. With all that, the types are compatible if lhs does not have a field, so we can
41 // simply return that.
42 return !lhsFeltType.hasField();
43 }
44
45 // Any other case gets handled by standard equality.
46 return lhsType == rhsType;
47 });
48}

50LogicalResult FeltToIndexOp::canonicalize(FeltToIndexOp op, ::mlir::PatternRewriter &rewriter) {
51 if (auto constOp = op.getValue().getDefiningOp<felt::FeltConstantOp>()) {
52 auto value = constOp.getValue().getValue();
53 if (value.getBitWidth() <= 64) {
54 rewriter.replaceOpWithNewOp<arith::ConstantIndexOp>(op, value.getSExtValue());
55 return success();
56 }
57 }
58 return failure();
59}
