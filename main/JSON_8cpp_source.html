31static llvm::Expected<int64_t> jsonToInt(const llvm::json::Value *json) {
32 if (std::optional<int64_t> integer = json->getAsInteger()) {
33 return *integer;
34 }
35 if (std::optional<llvm::StringRef> str = json->getAsString()) {
36 int64_t value = 0;
37 if (!str->getAsInteger(10, value)) {
38 return value;
39 }
40 }
41 return makeError("expected integer-compatible JSON value");
42}

45static llvm::Expected<llvm::DynamicAPInt>
46jsonToFelt(const llvm::json::Value *json, const Field &field) {
47 if (std::optional<llvm::StringRef> str = json->getAsString()) {
48 return field.reduce(toDynamicAPInt(*str));
49 }
50 if (std::optional<int64_t> integer = json->getAsInteger()) {
51 return field.reduce(*integer);
52 }
53 return makeError("expected felt value as JSON integer or decimal string");
54}

57static llvm::Expected<WitnessVal> parseJSONArray(
58 const llvm::json::Value *json, array::ArrayType type, const Field &field, Operation *origin,
59 size_t dimIndex = 0
60) {
61 const auto *jsonArray = json->getAsArray();
62 if (!jsonArray) {
63 return makeError("expected JSON array");
64 }
65
66 llvm::ArrayRef<int64_t> shape = type.getShape();
67 if (dimIndex >= shape.size()) {
68 return makeError("invalid array shape");
69 }
70 auto expectedSize = checkedShapeDimToSize(shape[dimIndex], "JSON array input");
71 if (!expectedSize) {
72 return expectedSize.takeError();
73 }
74 if (jsonArray->size() != *expectedSize) {
75 return makeError("JSON array length does not match LLZK array dimension");
76 }
77
78 auto arrayValue = std::make_shared<ArrayValue>();
79 arrayValue->type = type;
80 if (dimIndex == shape.size() - 1) {
81 arrayValue->elements.reserve(jsonArray->size());
82 for (const llvm::json::Value &elem : *jsonArray) {
83 auto parsed = parseJSONValue(&elem, type.getElementType(), field, origin);
84 if (!parsed) {
85 return parsed.takeError();
86 }
87 arrayValue->elements.push_back(*parsed);
88 }
89 return arrayValue;
90 }
91
92 arrayValue->elements.reserve(jsonArray->size());
93 for (const llvm::json::Value &elem : *jsonArray) {
94 auto parsed = parseJSONArray(&elem, type, field, origin, dimIndex + 1);
95 if (!parsed) {
96 return parsed.takeError();
97 }
98 auto subArray = asArray(*parsed);
99 if (!subArray) {
100 return subArray.takeError();
101 }
102 for (const WitnessVal &subElem : (*subArray)->elements) {
103 arrayValue->elements.push_back(subElem);
104 }
105 }
106 return arrayValue;
107}

118static llvm::Expected<llvm::json::Value> serializeJSONArray(
119 const ArrayValueRef &arrayValue, array::ArrayType type, SymbolTableCollection &tables,
120 Operation *origin, SerializationMode mode, size_t dimIndex = 0, size_t flatOffset = 0
121) {
122 llvm::json::Array jsonArray;
123 llvm::ArrayRef<int64_t> shape = type.getShape();
124 auto dimSize = checkedShapeDimToSize(shape[dimIndex], "JSON array output");
125 if (!dimSize) {
126 return dimSize.takeError();
127 }
128 if (dimIndex == shape.size() - 1) {
129 for (size_t i = 0; i < *dimSize; ++i) {
130 bool overflow = false;
131 size_t elementOffset = llvm::SaturatingAdd(flatOffset, i, &overflow);
132 if (overflow) {
133 return makeError("JSON array output flat index would overflow size_t");
134 }
135 auto elem = serializeJSONValue(
136 arrayValue->elements[elementOffset], type.getElementType(), tables, origin, mode
137 );
138 if (!elem) {
139 return elem.takeError();
140 }
141 jsonArray.push_back(*elem);
142 }
143 return llvm::json::Value(std::move(jsonArray));
144 }
145
146 size_t subArraySize = 1;
147 for (size_t i = dimIndex + 1; i < shape.size(); ++i) {
148 auto nextDimSize = checkedShapeDimToSize(shape[i], "JSON array output");
149 if (!nextDimSize) {
150 return nextDimSize.takeError();
151 }
152 bool overflow = false;
153 subArraySize = llvm::SaturatingMultiply(subArraySize, *nextDimSize, &overflow);
154 if (overflow) {
155 return makeError("JSON array output sub-array size would overflow size_t");
156 }
157 }
158
159 for (size_t i = 0; i < *dimSize; ++i) {
160 bool overflow = false;
161 size_t nextOffset = llvm::SaturatingMultiplyAdd(i, subArraySize, flatOffset, &overflow);
162 if (overflow) {
163 return makeError("JSON array output flat offset would overflow size_t");
164 }
165 auto subArray =
166 serializeJSONArray(arrayValue, type, tables, origin, mode, dimIndex + 1, nextOffset);
167 if (!subArray) {
168 return subArray.takeError();
169 }
170 jsonArray.push_back(*subArray);
171 }
172 return llvm::json::Value(std::move(jsonArray));
173}

176llvm::Expected<WitnessVal>
177parseJSONValue(const llvm::json::Value *json, Type type, const Field &field, Operation *origin) {
178 return llvm::TypeSwitch<Type, llvm::Expected<WitnessVal>>(type)
179 .Case([&](felt::FeltType) -> llvm::Expected<WitnessVal> { return jsonToFelt(json, field); })
180 .Case([&](array::ArrayType arrayType) -> llvm::Expected<WitnessVal> {
181 return parseJSONArray(json, arrayType, field, origin);
182 })
183 .Case([&](pod::PodType) -> llvm::Expected<WitnessVal> {
184 return makeError("pod JSON inputs are not supported in llzk-witgen v1");
185 })
186 .Case([&](component::StructType) -> llvm::Expected<WitnessVal> {
187 return makeError("struct JSON inputs are not supported in llzk-witgen v1");
188 })
189 .Case([&](IndexType) -> llvm::Expected<WitnessVal> {
190 auto integer = jsonToInt(json);
191 if (!integer) {
192 return integer.takeError();
193 }
194 return *integer;
195 })
196 .Case([&](IntegerType intType) -> llvm::Expected<WitnessVal> {
197 if (intType.getWidth() == 1) {
198 if (std::optional<bool> boolValue = json->getAsBoolean()) {
199 return *boolValue;
200 }
201 auto integer = jsonToInt(json);
202 if (!integer) {
203 return integer.takeError();
204 }
205 return *integer != 0;
206 }
207 return makeError("only i1 integer JSON inputs are supported");
208 }).Default([&](Type) -> llvm::Expected<WitnessVal> {
209 return makeError("unsupported input type in llzk-witgen");
210 });
211}

214llvm::Expected<llvm::json::Value> serializeJSONValue(
215 const WitnessVal &value, Type type, SymbolTableCollection &tables, Operation *origin,
216 SerializationMode mode
217) {
218 return llvm::TypeSwitch<Type, llvm::Expected<llvm::json::Value>>(type)
219 .Case([&](felt::FeltType) -> llvm::Expected<llvm::json::Value> {
220 auto feltValue = asFelt(value);
221 if (!feltValue) {
222 return feltValue.takeError();
223 }
224 return feltToJSON(*feltValue);
225 })
226 .Case([&](array::ArrayType arrayType) -> llvm::Expected<llvm::json::Value> {
227 auto arrayValue = asArray(value);
228 if (!arrayValue) {
229 return arrayValue.takeError();
230 }
231 return serializeJSONArray(*arrayValue, arrayType, tables, origin, mode);
232 })
233 .Case([&](pod::PodType podType) -> llvm::Expected<llvm::json::Value> {
234 auto podValue = asPod(value);
235 if (!podValue) {
236 return podValue.takeError();
237 }
238 llvm::json::Object result;
239 for (pod::RecordAttr record : podType.getRecords()) {
240 auto it = (*podValue)->records.find(record.getName().getValue());
241 if (it == (*podValue)->records.end()) {
242 return makeError("missing POD record during JSON serialization");
243 }
244 auto serialized = serializeJSONValue(it->second, record.getType(), tables, origin, mode);
245 if (!serialized) {
246 return serialized.takeError();
247 }
248 result[record.getName().getValue()] = *serialized;
249 }
250 return llvm::json::Value(std::move(result));
251 })
252 .Case([&](component::StructType structType) -> llvm::Expected<llvm::json::Value> {
253 auto structValue = asStruct(value);
254 if (!structValue) {
255 return structValue.takeError();
256 }
257 auto defLookup = structType.getDefinition(tables, origin);
258 if (failed(defLookup)) {
259 return makeError("could not resolve struct type during JSON serialization");
260 }
261 llvm::json::Object result;
262 for (component::MemberDefOp member : defLookup->get().getMemberDefs()) {
263 auto it = (*structValue)->members.find(member.getSymName());
264 if (it == (*structValue)->members.end()) {
265 return makeError("missing struct member during JSON serialization");
266 }
267
268 if (mode == SerializationMode::PublicOutputsOnly) {
269 if (!member.hasPublicAttr()) {
270 continue;
271 }
272 } else {
273 if (!memberIsSignal(defLookup->get(), member) &&
274 !isa<component::StructType>(member.getType())) {
275 continue;
276 }
277 }
278
279 auto serialized = serializeJSONValue(it->second, member.getType(), tables, origin, mode);
280 if (!serialized) {
281 return serialized.takeError();
282 }
283 if (mode == SerializationMode::AllSignals && isa<component::StructType>(member.getType())) {
284 auto *object = serialized->getAsObject();
285 if (!object || object->empty()) {
286 continue;
287 }
288 }
289 result[member.getSymName()] = *serialized;
290 }
291 return llvm::json::Value(std::move(result));
292 })
293 .Case([&](IndexType) -> llvm::Expected<llvm::json::Value> {
294 auto indexValue = asIndex(value);
295 if (!indexValue) {
296 return indexValue.takeError();
297 }
298 return llvm::json::Value(*indexValue);
299 })
300 .Case([&](IntegerType intType) -> llvm::Expected<llvm::json::Value> {
301 if (intType.getWidth() != 1) {
302 return makeError("only i1 integer JSON serialization is supported");
303 }
304 auto boolValue = asBool(value);
305 if (!boolValue) {
306 return boolValue.takeError();
307 }
308 return llvm::json::Value(*boolValue);
309 }).Default([&](Type) -> llvm::Expected<llvm::json::Value> {
310 return makeError("unsupported output type in llzk-witgen");
311 });
312}

315llvm::Expected<llvm::json::Object> buildInputsJSONObject(
316 ArrayRef<InputBinding> bindings, ArrayRef<WitnessVal> values, SymbolTableCollection &tables,
317 Operation *origin
318) {
319 if (bindings.size() != values.size()) {
320 return makeError("input binding count mismatch during witness JSON assembly");
321 }
322
323 llvm::json::Object result;
324 for (auto [binding, value] : llvm::zip(bindings, values)) {
325 auto serialized =
326 serializeJSONValue(value, binding.type, tables, origin, SerializationMode::AllSignals);
327 if (!serialized) {
328 return serialized.takeError();
329 }
330 result[binding.name] = *serialized;
331 }
332 return result;
333}

336llvm::Expected<WitnessVal> extractValueAtPath(
337 const WitnessVal &root, Type rootType, ArrayRef<std::string> path,
338 SymbolTableCollection &tables, Operation *origin
339) {
340 if (path.empty()) {
341 return root;
342 }
343
344 if (auto structType = dyn_cast<component::StructType>(rootType)) {
345 auto structValue = asStruct(root);
346 if (!structValue) {
347 return structValue.takeError();
348 }
349 auto defLookup = structType.getDefinition(tables, origin);
350 if (failed(defLookup)) {
351 return makeError("could not resolve struct type while extracting witness value");
352 }
353 for (component::MemberDefOp member : defLookup->get().getMemberDefs()) {
354 if (member.getSymName() != path.front()) {
355 continue;
356 }
357 auto it = (*structValue)->members.find(member.getSymName());
358 if (it == (*structValue)->members.end()) {
359 return makeError("missing struct member while extracting witness value");
360 }
361 return extractValueAtPath(it->second, member.getType(), path.drop_front(), tables, origin);
362 }
363 return makeError("unknown struct member while extracting witness value");
364 }
365
366 if (auto podType = dyn_cast<pod::PodType>(rootType)) {
367 auto podValue = asPod(root);
368 if (!podValue) {
369 return podValue.takeError();
370 }
371 for (pod::RecordAttr record : podType.getRecords()) {
372 if (record.getName().getValue() != path.front()) {
373 continue;
374 }
375 auto it = (*podValue)->records.find(record.getName().getValue());
376 if (it == (*podValue)->records.end()) {
377 return makeError("missing POD record while extracting witness value");
378 }
379 return extractValueAtPath(it->second, record.getType(), path.drop_front(), tables, origin);
380 }
381 return makeError("unknown POD record while extracting witness value");
382 }
383
384 return makeError("extra witness path components for non-aggregate value");
385}
