61template <int N = DEFAULT_ELTS> class MapOperandsHelper {
62public:
63 MapOperandsHelper(intptr_t nMapOperands, MlirValueRange const *mapOperands) {
64 // resize allocates elements (needed for operator[] access on line 44)
65 storage.resize(nMapOperands);
66 ranges.reserve(nMapOperands);
67
68 for (intptr_t i = 0; i < nMapOperands; i++) {
69 mlir::SmallVector<mlir::Value, N> &sto = storage[i];
70 MlirValueRange ops = mapOperands[i];
71 ranges.push_back(mlir::ValueRange(unwrapList(ops.size, ops.values, sto)));
72 }
73 }
74
75 mlir::ArrayRef<mlir::ValueRange> operator*() const { return ranges; }
76
77private:
78 mlir::SmallVector<mlir::SmallVector<mlir::Value, N>, N> storage;
79 mlir::SmallVector<mlir::ValueRange, N> ranges;
80};
