19/// record and remembers how each original member was split.
20///
21/// 2. Run a dialect conversion that does the following:
22///
23/// - Replace `MemberReadOp` and `MemberWriteOp` targeting the members that were split in step 1
24/// so they instead perform scalar reads and writes from the new members. The transformation is
25/// local to the current op. Therefore, when replacing the `MemberReadOp` a new pod is
26/// created locally and all uses of the `MemberReadOp` are replaced with the new pod Value,
27/// then each scalar member read is followed by scalar write into the new pod. Similarly,
28/// when replacing a `MemberWriteOp`, each element in the pod operand needs a scalar read
29/// from the pod followed by a scalar write to the new member. Making only local changes
30/// keeps this step simple and later steps will optimize.
31///
32/// - Remove optional initialization from `NewPodOp` and instead insert a list of `WritePodOp`
33/// immediately following.
34///
35/// - Split pods to scalars in `FuncDefOp`, `CallOp`, and `ReturnOp` and insert the necessary
36/// create/read/write ops so the changes are as local as possible (just as described for
37/// `MemberReadOp` and `MemberWriteOp`)
38///
39/// 3. Promote pod reads and writes out of `scf.if`, `scf.for`, and `scf.while` regions when the
40/// access can be modeled as an SSA value flowing through the region boundary. This puts the
41/// pod accesses that mem2reg must eliminate into a parent block or loop-carried value.
42///
43/// 4. Run MLIR "sroa" pass to split each pod with `N` records into `N` pods with 1 record each
44/// (to prepare for the "mem2reg" pass because its API cannot split memory by itself).
45///
46/// 5. Run MLIR "mem2reg" pass to convert all single-record pod allocations and accesses into SSA
47/// values.
48///
49/// 6. Remove pod allocations that become unread after memory promotion, then remove SSA values
50/// made dead by that cleanup.
51///
52/// ** Steps 4-6 are rerun while nested POD types are still being exposed, until a fixpoint.
53///
54/// Note: This transformation imposes a "last write wins" semantics on pod records. If
55/// different/configurable semantics are added in the future, some additional transformation would
56/// be necessary before/during this pass so that multiple writes to the same record can be handled
57/// properly while they still exist.
58///
59/// Note: This transformation will introduce a `nondet` op when there exists a read from a pod
60/// record that was not earlier written to.
61///

339static void flattenPodMemberIntoLeaves(
340 MemberDefOp originalMember, PodType podTy, SmallVectorImpl<StringAttr> &recordChain,
341 LocalMemberReplacementMap &localRepMapRef, SymbolTable &structSymbolTable,
342 ConversionPatternRewriter &rewriter
343) {
344 for (RecordAttr record : podTy.getRecords()) {
345 recordChain.push_back(record.getName());
346 if (PodType nestedPodTy = dyn_cast<PodType>(record.getType())) {
347 flattenPodMemberIntoLeaves(
348 originalMember, nestedPodTy, recordChain, localRepMapRef, structSymbolTable, rewriter
349 );
350 recordChain.pop_back();
351 continue;
352 }
353
354 StringAttr name = getFlattenedMemberName(
355 originalMember.getContext(), originalMember.getSymNameAttr(), recordChain
356 );
357 Type ty = record.getType();
358 MemberDefOp newMember = rewriter.create<MemberDefOp>(
359 originalMember.getLoc(), name, ty, originalMember.getSignal(), originalMember.getColumn()
360 );
361 newMember.setPublicAttr(originalMember.hasPublicAttr());
362 localRepMapRef[RecordChain(recordChain)] =
363 std::make_pair(structSymbolTable.insert(newMember), ty);
364 recordChain.pop_back();
365 }

462 void rewrite(FuncDefOp op, OpAdaptor, ConversionPatternRewriter &rewriter) const override {
463 // Update in/out types of the function to replace pods with scalars
464 class Impl : public FunctionTypeConverter {
465 SmallVector<size_t> originalInputIdxToSize, originalResultIdxToSize;
466 SplitFunctionNameInfo inputNameInfo;
467 SplitFunctionNameInfo resultNameInfo;
468
469 protected:
470 SmallVector<Type> convertInputs(ArrayRef<Type> origTypes) override {
471 return splitPodType(origTypes, &originalInputIdxToSize);
472 }
473 SmallVector<Type> convertResults(ArrayRef<Type> origTypes) override {
474 return splitPodType(origTypes, &originalResultIdxToSize);
475 }
476 ArrayAttr convertInputAttrs(ArrayAttr origAttrs, SmallVector<Type> newTypes) override {
477 return replicateFunctionNameAttrsAsNeeded(
478 origAttrs, originalInputIdxToSize, newTypes, ARG_NAME_ATTR_NAME,
479 inputNameInfo.originalNames, inputNameInfo.existingNames,
480 inputNameInfo.splitNameSuffixes
481 );
482 }
483 ArrayAttr convertResultAttrs(ArrayAttr origAttrs, SmallVector<Type> newTypes) override {
484 return replicateFunctionNameAttrsAsNeeded(
485 origAttrs, originalResultIdxToSize, newTypes, RES_NAME_ATTR_NAME,
486 resultNameInfo.originalNames, resultNameInfo.existingNames,
487 resultNameInfo.splitNameSuffixes
488 );
489 }
490
495 void processBlockArgs(Block &entryBlock, RewriterBase &rewriter) override {
496 OpBuilder::InsertionGuard guard(rewriter);
497 rewriter.setInsertionPointToStart(&entryBlock);
498
499 for (unsigned i = 0; i < entryBlock.getNumArguments();) {
500 Value oldV = entryBlock.getArgument(i);
501 if (PodType pt = splittablePod(oldV.getType())) {
502 Location loc = oldV.getLoc();
503 // Generate `NewPodOp` and replace uses of the argument with it.
504 auto newPod = rewriter.create<NewPodOp>(loc, pt);
505 rewriter.replaceAllUsesWith(oldV, newPod);
506 // Remove the argument from the block
507 entryBlock.eraseArgument(i);
508 // For all indices in the PodType (i.e., the element count), generate a new
509 // block argument and a write of that argument to the new pod.
510 for (RecordAttr record : pt.getRecords()) {
511 BlockArgument newArg = entryBlock.insertArgument(i, record.getType(), loc);
512 genWrite(loc, newPod, record.getName(), newArg, rewriter);
513 ++i;
514 }
515 } else {
516 ++i;
517 }
518 }
519 }
520
521 public:
522 Impl(FuncDefOp op) {
523 inputNameInfo = collectSplitFunctionNameInfo(op.getArgumentTypes(), [&op](unsigned i) {
524 return op.getArgNameAttr(i);
525 }, getSplitRecordNameSuffixes);
526 resultNameInfo = collectSplitFunctionNameInfo(
527 op.getResultTypes(), [resultAttrs = op.getAllResultAttrs()](unsigned i) {
528 return getAttrAtIndexWithName(resultAttrs, i, RES_NAME_ATTR_NAME);
529 }, getSplitRecordNameSuffixes
530 );
531 }
532 };
533 Impl(op).convert(op, rewriter);
534 }

558static CallOp newCallOpWithSplitResults(
559 CallOp oldCall, CallOp::Adaptor adaptor, ConversionPatternRewriter &rewriter
560) {
561 OpBuilder::InsertionGuard guard(rewriter);
562 rewriter.setInsertionPointAfter(oldCall);
563
564 Operation::result_range oldResults = oldCall.getResults();
565 CallOp newCall = createCallPreservingInstantiationOperands(
566 oldCall.getLoc(), splitPodType(oldResults.getTypes()), oldCall, adaptor.getMapOperands(),
567 adaptor.getArgOperands(), rewriter
568 );
569
570 auto newResults = newCall.getResults().begin();
571 for (Value oldVal : oldResults) {
572 if (PodType pt = splittablePod(oldVal.getType())) {
573 Location loc = oldVal.getLoc();
574 // Generate `NewPodOp` and replace uses of the result with it.
575 auto newPod = rewriter.create<NewPodOp>(loc, pt);
576 rewriter.replaceAllUsesWith(oldVal, newPod);
577
578 // For each record in the PodType, write the next result from the new CallOp to the new pod.
579 for (RecordAttr record : pt.getRecords()) {
580 genWrite(loc, newPod, record.getName(), *newResults, rewriter);
581 newResults++;
582 }
583 } else {
584 rewriter.replaceAllUsesWith(oldVal, *newResults);
585 newResults++;
586 }
587 }
588 // erase the original CallOp
589 rewriter.eraseOp(oldCall);
590
591 return newCall;
592}

637static Value rebuildFlattenedPodRecord(
638 Location loc, Type recordType, SmallVectorImpl<StringAttr> &recordChain,
639 const DenseMap<RecordChain, Value> &leafValues, ConversionPatternRewriter &rewriter
640) {
641 if (PodType nestedPodTy = dyn_cast<PodType>(recordType)) {
642 NewPodOp nestedPod = rewriter.create<NewPodOp>(loc, nestedPodTy);
643 for (RecordAttr record : nestedPodTy.getRecords()) {
644 recordChain.push_back(record.getName());
645 Value recordValue =
646 rebuildFlattenedPodRecord(loc, record.getType(), recordChain, leafValues, rewriter);
647 genWrite(loc, nestedPod, record.getName(), recordValue, rewriter);
648 recordChain.pop_back();
649 }
650 return nestedPod;
651 }
652
653 auto it = leafValues.find(RecordChain(recordChain));
654 assert(it != leafValues.end() && "missing flattened POD leaf value");
655 return it->second;

1430 for (const LoopPodSlot &slot : slots) {
1431 newInits.push_back(genRead(loc, slot.podRef, slot.recordName, rewriter).getResult());
1432 newResultTypes.push_back(slot.type);
1433 }
1434
1435 auto newWhile = rewriter.create<scf::WhileOp>(loc, newResultTypes, newInits, nullptr, nullptr);
1436 newWhile->setAttrs(whileOp->getAttrs());
1437
1438 Block &newBeforeBody = *newWhile.getBeforeBody();
1439 Block &newAfterBody = *newWhile.getAfterBody();
1440 dropTerminatorIfPresent(newBeforeBody);
1441 dropTerminatorIfPresent(newAfterBody);
1442
1443 IRMapping beforeMapping;
1444 for (auto [oldArg, newArg] : llvm::zip_equal(
1445 whileOp.getBeforeArguments(),
1446 newWhile.getBeforeArguments().take_front(whileOp.getBeforeArguments().size())
1447 )) {
1448 beforeMapping.map(oldArg, newArg);
1449 }
1450
1451 SmallVector<Value> beforeSlotValues = llvm::map_to_vector(
1452 llvm::seq<size_t>(0, slots.size()),
1453 [base = whileOp.getBeforeArguments().size(), &newWhile](size_t idx) -> Value {
1454 return newWhile.getBeforeArguments()[llzk::checkedCast<unsigned>(base + idx)];
1455 }
1456 );
1457
1458 rewriter.setInsertionPointToEnd(&newBeforeBody);
1459 for (Operation &op : beforeBody) {
1460 if (auto conditionOp = dyn_cast<scf::ConditionOp>(&op)) {
1461 SmallVector<Value> conditionArgs =
1462 llvm::map_to_vector(conditionOp.getArgs(), [&beforeMapping](Value a) {
1463 return beforeMapping.lookupOrDefault(a);
1464 });
1465 llvm::append_range(conditionArgs, beforeSlotValues);
1466 rewriter.create<scf::ConditionOp>(
1467 conditionOp.getLoc(), beforeMapping.lookupOrDefault(conditionOp.getCondition()),
1468 conditionArgs
1469 );
1470 continue;
1471 }
1472
1473 if (auto readOp = dyn_cast<ReadPodOp>(&op)) {
1474 if (std::optional<size_t> slotIdx =
1475 findLoopSlotIndex(slots, readOp.getPodRef(), getRecordNameAsStringAttr(readOp))) {
1476 beforeMapping.map(readOp.getResult(), beforeSlotValues[*slotIdx]);
1477 continue;
1478 }
1479 }
1480
1481 if (auto writeOp = dyn_cast<WritePodOp>(&op)) {
1482 if (std::optional<size_t> slotIdx =
1483 findLoopSlotIndex(slots, writeOp.getPodRef(), getRecordNameAsStringAttr(writeOp))) {
1484 beforeSlotValues[*slotIdx] = beforeMapping.lookupOrDefault(writeOp.getValue());
1485 continue;
1486 }
1487 }
1488
1489 rewriter.clone(op, beforeMapping);
1490 }
1491
1492 IRMapping afterMapping;
1493 for (auto [oldArg, newArg] : llvm::zip_equal(
1494 whileOp.getAfterArguments(),
1495 newWhile.getAfterArguments().take_front(whileOp.getAfterArguments().size())
1496 )) {
1497 afterMapping.map(oldArg, newArg);
1498 }
1499
1500 SmallVector<Value> afterSlotValues = llvm::map_to_vector(
1501 llvm::seq<size_t>(0, slots.size()),
1502 [base = whileOp.getAfterArguments().size(), &newWhile](size_t idx) -> Value {
1503 return newWhile.getAfterArguments()[llzk::checkedCast<unsigned>(base + idx)];
1504 }
1505 );
1506
1507 rewriter.setInsertionPointToEnd(&newAfterBody);
1508 for (Operation &op : afterBody) {
1509 if (auto yieldOp = dyn_cast<scf::YieldOp>(&op)) {
1510 SmallVector<Value> yieldValues =
1511 llvm::map_to_vector(yieldOp.getOperands(), [&afterMapping](Value v) {
1512 return afterMapping.lookupOrDefault(v);
1513 });
1514 llvm::append_range(yieldValues, afterSlotValues);
1515 rewriter.create<scf::YieldOp>(yieldOp.getLoc(), yieldValues);
1516 continue;
1517 }
1518
1519 if (auto readOp = dyn_cast<ReadPodOp>(&op)) {
1520 if (std::optional<size_t> slotIdx =
1521 findLoopSlotIndex(slots, readOp.getPodRef(), getRecordNameAsStringAttr(readOp))) {
1522 afterMapping.map(readOp.getResult(), afterSlotValues[*slotIdx]);
1523 continue;
1524 }
1525 }
1526
1527 if (auto writeOp = dyn_cast<WritePodOp>(&op)) {
1528 if (std::optional<size_t> slotIdx =
1529 findLoopSlotIndex(slots, writeOp.getPodRef(), getRecordNameAsStringAttr(writeOp))) {
1530 afterSlotValues[*slotIdx] = afterMapping.lookupOrDefault(writeOp.getValue());
1531 continue;
1532 }
1533 }
1534
1535 rewriter.clone(op, afterMapping);
1536 }
1537
1538 rewriter.setInsertionPointAfter(newWhile);
1539 for (auto [idx, slot] : llvm::enumerate(slots)) {
1540 genWrite(
1541 loc, slot.podRef, slot.recordName, newWhile.getResult(whileOp.getNumResults() + idx),
1542 rewriter
1543 );
1544 }
1545
1546 rewriter.replaceOp(whileOp, newWhile.getResults().take_front(whileOp.getNumResults()));
1547 return success();
1548 }

1618 void runOnOperation() override {
1619 ModuleOp module = getOperation();
1620
1621 if (failed(step0(module))) {
1622 return signalPassFailure();
1623 }
1624 LLVM_DEBUG({
1625 llvm::dbgs() << "After step 0:\n";
1626 module.dump();
1627 });
1628
1629 {
1630 // This is divided into 2 steps to simplify the implementation for member-related ops. The
1631 // issue is that the conversions for member read/write expect the mapping of record name to
1632 // member name+type to already be populated for the referenced member (although this could be
1633 // computed on demand if desired but it complicates the implementation a bit).
1634 SymbolTableCollection symTables;
1635 MemberReplacementMap memberRepMap;
1636 if (failed(step1(module, symTables, memberRepMap))) {
1637 return signalPassFailure();
1638 }
1639 LLVM_DEBUG({
1640 llvm::dbgs() << "After step 1:\n";
1641 module.dump();
1642 });
1643
1644 if (failed(step2(module, symTables, memberRepMap))) {
1645 return signalPassFailure();
1646 }
1647 LLVM_DEBUG({
1648 llvm::dbgs() << "After step 2:\n";
1649 module.dump();
1650 });
1651 }
1652
1653 if (failed(step3(module))) {
1654 return signalPassFailure();
1655 }
1656 LLVM_DEBUG({
1657 llvm::dbgs() << "After step 3:\n";
1658 module.dump();
1659 });
1660
1661 // 1. Use SROA (Destructurable* interfaces) to split each pod with `N` records into `N` pods
1662 // with 1 record each. This is necessary because the mem2reg pass cannot deal with splitting
1663 // up memory, i.e., it can only convert scalar memory access into SSA values.
1664 // 2. The mem2reg pass converts the size 1 pod allocations and accesses into SSA values.
1665 OpPassManager scalarizePM(ModuleOp::getOperationName());
1666 scalarizePM.addPass(createSpecializedSROAPass<NewPodOp>());
1667 scalarizePM.addPass(createSpecializedMem2RegPass<NewPodOp>());
1668
1669 // Cleanup allocations made dead by memory promotion and other dead SSA values.
1670 OpPassManager cleanupPM(ModuleOp::getOperationName());
1671 cleanupPM.addPass(createRemoveUnusedDiscardableAllocationsPass(
1672 RemoveUnusedDiscardableAllocationsPassOptions {
1673 .allocatorOpName = NewPodOp::getOperationName().str()
1674 }
1675 ));
1676 cleanupPM.addPass(createRemoveDeadValuesPass());
1677
1678 size_t podAllocWeight = podAllocScalarizationWeight(module);
1679 while (podAllocWeight != 0) {
1680 if (failed(runPipeline(scalarizePM, module))) {
1681 signalPassFailure();
1682 return;
1683 }
1684
1685 // SROA+mem2reg can expose `scf.if`-carried POD values that become redundant after a
1686 // same-record write from another `scf.if` result. Fold those reads and clean up before
1687 // checking convergence.
1688 bool foldedIfCarriedRead = applyIfCarriedPodReadAfterWritePatterns(module);
1689 if (failed(runPipeline(cleanupPM, module))) {
1690 signalPassFailure();
1691 return;
1692 }
1693
1694 // Nested PODs can become visible only after an outer single-record POD has been promoted,
1695 // and SROA can transiently increase allocation count while splitting aggregates. Keep
1696 // iterating until the allocation-weight heuristic reaches a fixed point.
1697 size_t nextPodAllocWeight = podAllocScalarizationWeight(module);
1698 if (!foldedIfCarriedRead && nextPodAllocWeight == podAllocWeight) {
1699 break;
1700 }
1701 podAllocWeight = nextPodAllocWeight;
1702 }
1703 LLVM_DEBUG({
1704 llvm::dbgs() << "After SROA+Mem2Reg pipeline:\n";
1705 module.dump();
1706 });
1707 }
