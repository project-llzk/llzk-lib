101LogicalResult CreateArrayOp::verify() {
102 Type retTy = getResult().getType();
103 assert(llvm::isa<ArrayType>(retTy)); // per ODS spec of CreateArrayOp
104
105 // Collect the array dimensions that are defined via AffineMapAttr
106 SmallVector<AffineMapAttr> mapAttrs;
107 // Extend the lifetime of the temporary to suppress warnings.
108 ArrayType arrTy = llvm::cast<ArrayType>(retTy);
109 for (Attribute a : arrTy.getDimensionSizes()) {
110 if (AffineMapAttr m = dyn_cast<AffineMapAttr>(a)) {
111 mapAttrs.push_back(m);
112 }
113 }
114 return affineMapHelpers::verifyAffineMapInstantiations(
115 getMapOperands(), getNumDimsPerMap(), mapAttrs, *this
116 );
117}

120SmallVector<DestructurableMemorySlot> CreateArrayOp::getDestructurableSlots() {
121 assert(getElements().empty() && "must run after initialization is split from allocation");
122 ArrayType arrType = getType();
123 if (!arrType.hasStaticShape() || arrType.getNumElements() == 1) {
124 return {};
125 }
126 if (auto destructured = arrType.getSubelementIndexMap()) {
127 return {DestructurableMemorySlot {{getResult(), arrType}, std::move(*destructured)}};
128 }
129 return {};
130}

133DenseMap<Attribute, MemorySlot> CreateArrayOp::destructure(
134 const DestructurableMemorySlot &slot, const SmallPtrSetImpl<Attribute> &usedIndices,
135 OpBuilder &builder, SmallVectorImpl<DestructurableAllocationOpInterface> &newAllocators
136) {
137 assert(slot.ptr == getResult());
138 assert(slot.elemType == getType());
139
140 builder.setInsertionPointAfter(*this);
141
142 DenseMap<Attribute, MemorySlot> slotMap; // result
143 for (Attribute index : usedIndices) {
144 // This is an ArrayAttr since indexing is multi-dimensional
145 ArrayAttr indexAsArray = llvm::dyn_cast<ArrayAttr>(index);
146 assert(indexAsArray && "expected ArrayAttr");
147
148 Type destructAs = getType().getTypeAtIndex(indexAsArray);
149 assert(destructAs == slot.subelementTypes.lookup(indexAsArray));
150
151 ArrayType destructAsArrayTy = llvm::dyn_cast<ArrayType>(destructAs);
152 assert(destructAsArrayTy && "expected ArrayType");
153
154 auto subCreate = builder.create<CreateArrayOp>(getLoc(), destructAsArrayTy);
155 newAllocators.push_back(subCreate);
156 slotMap.try_emplace<MemorySlot>(index, {subCreate.getResult(), destructAs});
157 }
158
159 return slotMap;
160}

173 ArrayType arrType = getType();

223bool ArrayAccessOpInterface::canRewire(
224 const DestructurableMemorySlot &slot, SmallPtrSetImpl<Attribute> &usedIndices,
225 SmallVectorImpl<MemorySlot> & /*mustBeSafelyUsed*/, const DataLayout & /*dataLayout*/
226) {
227 if (slot.ptr != getArrRef()) {
228 return false;
229 }
230
231 ArrayAttr indexAsAttr = indexOperandsToAttributeArray();
232 if (!indexAsAttr) {
233 return false;
234 }
235
236 // Scalar read/write case has 0 dimensions in the read/write value.
237 if (!getValueOperandDims().empty()) {
238 return false;
239 }
240
241 // Just insert the index.
242 usedIndices.insert(indexAsAttr);
243 return true;
244}

247DeletionKind ArrayAccessOpInterface::rewire(
248 const DestructurableMemorySlot &slot, DenseMap<Attribute, MemorySlot> &subslots,
249 OpBuilder &builder, const DataLayout & /*dataLayout*/
250) {
251 assert(slot.ptr == getArrRef());
252 assert(slot.elemType == getArrRefType());
253 // ASSERT: non-scalar read/write should have been desugared earlier
254 assert(getValueOperandDims().empty() && "only scalar read/write supported");
255
256 ArrayAttr indexAsAttr = indexOperandsToAttributeArray();
257 assert(indexAsAttr && "canRewire() should have returned false");
258 const MemorySlot &memorySlot = subslots.at(indexAsAttr);
259
260 // Temporarily set insertion point before the current op for what's built below
261 OpBuilder::InsertionGuard guard(builder);
262 builder.setInsertionPoint(this->getOperation());
263
264 // Write to the sub-slot created for the index of `this`, using index 0
265 getArrRefMutable().set(memorySlot.ptr);
266 getIndicesMutable().clear();
267 getIndicesMutable().assign(builder.create<arith::ConstantIndexOp>(getLoc(), 0));
268
269 return DeletionKind::Keep;
270}

278LogicalResult
279ensureNumIndicesMatchDims(ArrayType ty, size_t numIndices, const OwningEmitErrorFn &errFn) {
280 ArrayRef<Attribute> dims = ty.getDimensionSizes();
281 // Ensure the number of provided indices matches the array dimensions
282 auto compare = numIndices <=> dims.size();
283 if (compare != 0) {
284 return errFn().append(
285 "has ", (compare < 0 ? "insufficient" : "too many"), " indexed dimensions: expected ",
286 dims.size(), " but found ", numIndices
287 );
288 }
289 return success();
290}

387LogicalResult ExtractArrayOp::inferReturnTypes(
388 MLIRContext * /*context*/, std::optional<Location> location, ExtractArrayOpAdaptor adaptor,
389 llvm::SmallVectorImpl<Type> &inferredReturnTypes
390) {
391 size_t numToSkip = adaptor.getIndices().size();
392 Type arrRefType = adaptor.getArrRef().getType();
393 assert(llvm::isa<ArrayType>(arrRefType)); // per ODS spec of ExtractArrayOp
394 ArrayType arrRefArrType = llvm::cast<ArrayType>(arrRefType);
395 ArrayRef<Attribute> arrRefDimSizes = arrRefArrType.getDimensionSizes();
396
397 // Check for invalid cases
398 auto compare = numToSkip <=> arrRefDimSizes.size();
399 if (compare == 0) {
400 return mlir::emitOptionalError(
401 location, '\'', ExtractArrayOp::getOperationName(),
402 "' op cannot select all dimensions of an array. Use '", ReadArrayOp::getOperationName(),
403 "' instead."
404 );
405 } else if (compare > 0) {
406 return mlir::emitOptionalError(
407 location, '\'', ExtractArrayOp::getOperationName(),
408 "' op cannot select more dimensions than exist in the source array"
409 );
410 }
411
412 // Generate and store reduced array type
413 inferredReturnTypes.resize(1);
414 inferredReturnTypes[0] =
415 ArrayType::get(arrRefArrType.getElementType(), arrRefDimSizes.drop_front(numToSkip));
416 return success();
417}

434LogicalResult InsertArrayOp::verify() {
435 ArrayType baseArrRefArrType = getArrRefType();
436 Type rValueType = getRvalue().getType();
437 assert(llvm::isa<ArrayType>(rValueType)); // per ODS spec of InsertArrayOp
438 ArrayType rValueArrType = llvm::cast<ArrayType>(rValueType);
439
440 // size of lhs dimensions == numIndices + size of rhs dimensions
441 size_t lhsDims = baseArrRefArrType.getDimensionSizes().size();
442 size_t numIndices = getIndices().size();
443 size_t rhsDims = rValueArrType.getDimensionSizes().size();
444
445 // Ensure the number of indices specified does not exceed base dimension count.
446 if (numIndices > lhsDims) {
447 return emitOpError("cannot select more dimensions than exist in the source array");
448 }
449
450 // Ensure the rValue dimension count equals the base reduced dimension count
451 auto compare = (numIndices + rhsDims) <=> lhsDims;
452 if (compare != 0) {
453 return emitOpError().append(
454 "has ", (compare < 0 ? "insufficient" : "too many"), " indexed dimensions: expected ",
455 (lhsDims - rhsDims), " but found ", numIndices
456 );
457 }
458
459 // Having verified the indices are of appropriate size, we verify the subarray type.
460 // This will verify the dimensions of the subarray, which is why we only check the
461 // size of the indices above.
462 return verifySubArrayType(getEmitOpErrFn(this), baseArrRefArrType, rValueArrType);
463}
