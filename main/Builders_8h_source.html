313 TemplateBuilder(polymorphic::TemplateOp t) : ModuleLikeBuilder(t.getContext()), myTemplate(t) {}
314
315 /* Getter methods */
316
317 mlir::Region &getBodyRegion() { return myTemplate.getBodyRegion(); }
318
320 polymorphic::TemplateOp &getTemplate() { return myTemplate; }
321
322 mlir::FailureOr<polymorphic::TemplateParamOp> getParam(std::string_view name) {
323 auto op = myTemplate.getConstNamed<polymorphic::TemplateParamOp>(name);
324 if (op) {
325 return op;
326 }
327 return mlir::failure();
328 }
329
330 mlir::FailureOr<polymorphic::TemplateExprOp> getExpr(std::string_view name) {
331 auto op = myTemplate.getConstNamed<polymorphic::TemplateExprOp>(name);
332 if (op) {
333 return op;
334 }
335 return mlir::failure();
336 }
337
338 TemplateBuilder &
339 insertParam(std::string_view name, mlir::Location loc, mlir::TypeAttr type = {}) {
340 if (succeeded(getParam(name))) {
341 llvm::report_fatal_error("Duplicate TemplateParamOp insertion attempted");
342 }
343
344 mlir::OpBuilder builder(context);
345
346 auto &region = getBodyRegion();
347 if (region.empty()) {
348 region.emplaceBlock();
349 }
350
351 builder.setInsertionPointToEnd(&region.front());
352
353 auto nameAttr = builder.getStringAttr(name);
354
355 builder.create<polymorphic::TemplateParamOp>(loc, nameAttr, type);
356
357 return *this;
358 }
359
360 inline TemplateBuilder &insertParam(std::string_view name) {
361 return insertParam(name, getUnknownLoc());
362 }
363
364 TemplateBuilder &insertExpr(std::string_view name, mlir::Location loc) {
365 if (succeeded(getExpr(name))) {
366 llvm::report_fatal_error("Duplicate TemplateExprOp insertion attempted");
367 }
368
369 mlir::OpBuilder builder(context);
370
371 auto &region = getBodyRegion();
372 if (region.empty()) {
373 region.emplaceBlock();
374 }
375
376 builder.setInsertionPointToEnd(&region.front());
377
378 auto nameAttr = builder.getStringAttr(name);
379
380 builder.create<polymorphic::TemplateExprOp>(loc, nameAttr);
381
382 return *this;
383 }
384
385 inline TemplateBuilder &insertExpr(std::string_view name) {
386 return insertExpr(name, getUnknownLoc());
