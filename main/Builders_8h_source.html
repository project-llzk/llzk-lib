232 inline mlir::FailureOr<function::FuncDefOp>
233 getFunc(function::FunctionKind kind, std::string_view name) const {
234 switch (kind) {
235 case function::FunctionKind::StructCompute:
236 return getComputeFn(name);
237 case function::FunctionKind::StructConstrain:
238 return getConstrainFn(name);
239 case function::FunctionKind::StructProduct:
240 return getProductFn(name);
241 case function::FunctionKind::Free:
242 return getFreeFunc(name);
243 }
244 return mlir::failure();
245 }
