345 void
346 updateReachability(Def2NodeMap &m, component::StructDefOp caller, component::StructDefOp callee) {
347 auto &callerNode = m[caller];
348 auto &calleeNode = m[callee];
349 callerNode.callees[callee] = &calleeNode;
350 }
351
352 bool isReachable(Def2NodeMap &m, component::StructDefOp caller, component::StructDefOp callee) {
353 StructDefSet visited;
354 std::deque<component::StructDefOp> frontier;
355 frontier.push_back(caller);
356
357 while (!frontier.empty()) {
358 auto s = frontier.front();
359 frontier.pop_front();
360 if (!visited.insert(s).second) {
361 continue;
362 }
363
364 if (s == callee) {
365 return true;
366 }
367 for (auto &[calleeStruct, _] : m[s].callees) {
368 frontier.push_back(calleeStruct);
369 }
370 }
371 return false;
372 }
