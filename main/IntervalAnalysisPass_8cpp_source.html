34class IntervalAnalysisPrinterPass
35 : public impl::IntervalAnalysisPrinterPassBase<IntervalAnalysisPrinterPass> {

38public:
39 explicit IntervalAnalysisPrinterPass(llvm::raw_ostream &ostream)
40 : impl::IntervalAnalysisPrinterPassBase<IntervalAnalysisPrinterPass>(), os(ostream) {}
41
42protected:
43 void runOnOperation() override {
44 markAllAnalysesPreserved();
45
46 // Suppress false positive from `clang-tidy`
47 // NOLINTNEXTLINE(clang-analyzer-core.NonNullParamChecker)
48 auto modOp = llvm::dyn_cast<mlir::ModuleOp>(getOperation());
49 if (!modOp) {
50 constexpr const char *msg = "IntervalAnalysisPrinterPass error: should be run on ModuleOp!";
51 getOperation()->emitError(msg).report();
52 return;
53 }
54
55 auto fieldLookupRes = Field::tryGetField(fieldName.c_str());
56
57 if (mlir::failed(fieldLookupRes)) {
58 modOp->emitError()
59 .append("IntervalAnalysisPrinterPass error: unknown field \"", fieldName, "\" specified")
60 .report();
61 return;
62 }
63
64 auto &mia = getAnalysis<ModuleIntervalAnalysis>();
65 mia.setField(fieldLookupRes.value());
66 mia.setPropagateInputConstraints(propagateInputConstraints);
67 auto am = getAnalysisManager();
68 mia.ensureAnalysisRun(am);
69
70 for (const auto &[s, si] : mia.getCurrentResults()) {
71 auto &structDef = const_cast<StructDefOp &>(s);
72 auto fullName = getPathFromTopRoot(structDef);
73 ensure(
74 mlir::succeeded(fullName),
75 "could not resolve fully qualified name of struct " + mlir::Twine(structDef.getName())
76 );
77 os << fullName.value() << ' ';
78 si.get().print(os, printSolverConstraints, printComputeIntervals);
79 }
80 }
81};
