121class ConstraintDependencyGraph {
122public:
133 static mlir::FailureOr<ConstraintDependencyGraph> compute(
134 mlir::ModuleOp mod, component::StructDefOp s, mlir::DataFlowSolver &solver,
135 mlir::AnalysisManager &am, const CDGAnalysisContext &ctx
136 );
137
139 void dump() const;
142 void print(mlir::raw_ostream &os) const;
143
151 ConstraintDependencyGraph translate(SourceRefRemappings translation) const;
152
161 SourceRefSet getConstrainingValues(const SourceRef &ref) const;
162
163 const SourceRefLattice::Ref2Val &getRef2Val() const { return ref2Val; }
164
165 /*
166 Rule of three, needed for the mlir::SymbolTableCollection, which has no copy constructor.
167 Since the mlir::SymbolTableCollection is a caching mechanism, we simply allow default, empty
168 construction for copies.
169 */
170
171 ConstraintDependencyGraph(const ConstraintDependencyGraph &other)
172 : mod(other.mod), structDef(other.structDef), ctx(other.ctx), signalSets(other.signalSets),
173 constantSets(other.constantSets), ref2Val(other.ref2Val), tables() {}
174
175 ConstraintDependencyGraph &operator=(const ConstraintDependencyGraph &other) {
176 mod = other.mod;
177 structDef = other.structDef;
178 ctx = other.ctx;
179 signalSets = other.signalSets;
180 constantSets = other.constantSets;
181 ref2Val = other.ref2Val;
182 return *this;
183 }
184 virtual ~ConstraintDependencyGraph() = default;
185
186private:
187 mlir::ModuleOp mod;
188 // Using mutable because many operations are not const by default, even for "const"-like
189 // operations, like "getName()", and this reduces const_casts.
190 mutable component::StructDefOp structDef;
191 CDGAnalysisContext ctx;
192
193 // Transitive closure only over signals.
194 llvm::EquivalenceClasses<SourceRef> signalSets;
195 // A simple set mapping of constants, as we do not want to compute a transitive closure over
196 // constants.
197 std::unordered_map<SourceRef, SourceRefSet, SourceRef::Hash> constantSets;
198
199 // Maps references to the values where they are found.
200 SourceRefLattice::Ref2Val ref2Val;
201
202 // Also mutable for caching within otherwise const lookup operations.
203 mutable mlir::SymbolTableCollection tables;
204
210 ConstraintDependencyGraph(mlir::ModuleOp m, component::StructDefOp s, const CDGAnalysisContext &c)
211 : mod(m), structDef(s), ctx(c) {}
212
218 mlir::LogicalResult computeConstraints(mlir::DataFlowSolver &solver, mlir::AnalysisManager &am);
219
226 void walkConstrainOp(mlir::DataFlowSolver &solver, mlir::Operation *emitOp);
227};
