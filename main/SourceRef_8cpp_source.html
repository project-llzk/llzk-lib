49bool SourceRefIndex::operator<(const SourceRefIndex &rhs) const {
50 if (isMember() && rhs.isMember()) {
51 return NamedOpLocationLess<MemberDefOp> {}(getMember(), rhs.getMember());
52 }
53 if (isIndex() && rhs.isIndex()) {
54 return getIndex() < rhs.getIndex();
55 }
56 if (isIndexRange() && rhs.isIndexRange()) {
57 auto [ll, lu] = getIndexRange();
58 auto [rl, ru] = rhs.getIndexRange();
59 return ll < rl || (ll == rl && lu < ru);
60 }
61
62 if (isMember()) {
63 return true;
64 }
65 if (isIndex() && !rhs.isMember()) {
66 return true;
67 }
68
69 return false;
70}

72size_t SourceRefIndex::Hash::operator()(const SourceRefIndex &c) const {
73 if (c.isIndex()) {
74 // We don't hash the index directly, because the built-in LLVM hash includes
75 // the bitwidth of the APInt in the hash, which is undesirable for this application.
76 // i.e., We want a N-bit version of x to hash to the same value as an M-bit version of X,
77 // because our equality checks would consider them equal regardless of bitwidth.
78 APSInt idx = toAPSInt(c.getIndex());
79 unsigned requiredBits = idx.getSignificantBits();
80 auto hash = llvm::hash_value(idx.trunc(requiredBits));
81 return hash;
82 } else if (c.isIndexRange()) {
83 auto r = c.getIndexRange();
84 return llvm::hash_value(std::get<0>(r)) ^ llvm::hash_value(std::get<1>(r));
85 } else {
86 return OpHash<component::MemberDefOp> {}(c.getMember());
87 }
88}

120std::vector<SourceRef> SourceRef::getAllSourceRefs(StructDefOp structDef, FuncDefOp fnOp) {
121 std::vector<SourceRef> res;
122
123 ensure(
124 structDef == fnOp->getParentOfType<StructDefOp>(), "function must be within the given struct"
125 );
126
127 FailureOr<ModuleOp> modOp = getRootModule(structDef);
128 ensure(succeeded(modOp), "could not lookup module from struct " + Twine(structDef.getName()));
129
130 SymbolTableCollection tables;
131 for (auto a : fnOp.getArguments()) {
132 auto argRes = getAllSourceRefs(tables, modOp.value(), SourceRef(a));
133 res.insert(res.end(), argRes.begin(), argRes.end());
134 }
135
136 // For compute functions, the "self" member is not arg0 like for constrain, but
137 // rather the struct value returned from the function.
138 if (fnOp.isStructCompute()) {
139 Value selfVal = fnOp.getSelfValueFromCompute();
140 auto createOp = dyn_cast_if_present<CreateStructOp>(selfVal.getDefiningOp());
141 ensure(createOp, "self value should originate from struct.new operation");
142 auto selfRes = getAllSourceRefs(tables, modOp.value(), SourceRef(createOp));
143 res.insert(res.end(), selfRes.begin(), selfRes.end());
144 }
145
146 return res;
147}

149std::vector<SourceRef> SourceRef::getAllSourceRefs(StructDefOp structDef, MemberDefOp memberDef) {
150 std::vector<SourceRef> res;
151 FuncDefOp constrainFnOp = structDef.getConstrainFuncOp();
152 ensure(
153 memberDef->getParentOfType<StructDefOp>() == structDef,
154 "Member " + Twine(memberDef.getName()) + " is not a member of struct " +
155 Twine(structDef.getName())
156 );
157 FailureOr<ModuleOp> modOp = getRootModule(structDef);
158 ensure(succeeded(modOp), "could not lookup module from struct " + Twine(structDef.getName()));
159
160 // Get the self argument (like `FuncDefOp::getSelfValueFromConstrain()`)
161 BlockArgument self = constrainFnOp.getArguments().front();
162 SourceRef memberRef = SourceRef(self, {SourceRefIndex(memberDef)});
163
164 SymbolTableCollection tables;
165 return getAllSourceRefs(tables, modOp.value(), memberRef);
166}

168Type SourceRef::getType() const {
169 if (isConstantFelt()) {
170 return std::get<FeltConstantOp>(*constantVal).getType();
171 } else if (isConstantIndex()) {
172 return std::get<arith::ConstantIndexOp>(*constantVal).getType();
173 } else if (isTemplateConstant()) {
174 return std::get<ConstReadOp>(*constantVal).getType();
175 } else {
176 int array_derefs = 0;
177 int idx = llzk::checkedCast<int>(memberRefs.size()) - 1;
178 while (idx >= 0 && memberRefs[idx].isIndex()) {
179 array_derefs++;
180 idx--;
181 }
182
183 Type currTy = nullptr;
184 if (idx >= 0) {
185 currTy = memberRefs[idx].getMember().getType();
186 } else {
187 currTy = isBlockArgument() ? getBlockArgument().getType() : getCreateStructOp().getType();
188 }
189
190 while (array_derefs > 0) {
191 currTy = dyn_cast<ArrayType>(currTy).getElementType();
192 array_derefs--;
193 }
194 return currTy;
195 }
196}

198bool SourceRef::isValidPrefix(const SourceRef &prefix) const {
199 if (isConstant()) {
200 return false;
201 }
202
203 if (root != prefix.root || memberRefs.size() < prefix.memberRefs.size()) {
204 return false;
205 }
206 for (size_t i = 0; i < prefix.memberRefs.size(); i++) {
207 if (memberRefs[i] != prefix.memberRefs[i]) {
208 return false;
209 }
210 }
211 return true;
212}

214FailureOr<std::vector<SourceRefIndex>> SourceRef::getSuffix(const SourceRef &prefix) const {
215 if (!isValidPrefix(prefix)) {
216 return failure();
217 }
218 std::vector<SourceRefIndex> suffix;
219 for (size_t i = prefix.memberRefs.size(); i < memberRefs.size(); i++) {
220 suffix.push_back(memberRefs[i]);
221 }
222 return suffix;
223}

225FailureOr<SourceRef> SourceRef::translate(const SourceRef &prefix, const SourceRef &other) const {
226 if (isConstant()) {
227 return *this;
228 }
229 auto suffix = getSuffix(prefix);
230 if (failed(suffix)) {
231 return failure();
232 }
233
234 auto newSignalUsage = other;
235 newSignalUsage.memberRefs.insert(newSignalUsage.memberRefs.end(), suffix->begin(), suffix->end());
236 return newSignalUsage;
237}

239std::vector<SourceRef> getAllChildren(
240 SymbolTableCollection & /*tables*/, ModuleOp /*mod*/, ArrayType arrayTy, const SourceRef &root
241) {
242 std::vector<SourceRef> res;
243 // Recurse into arrays by iterating over their elements
244 for (int64_t i = 0; i < arrayTy.getDimSize(0); i++) {
245 SourceRef childRef = root.createChild(SourceRefIndex(i));
246 res.push_back(childRef);
247 }
248
249 return res;
250}

252std::vector<SourceRef> getAllChildren(
253 SymbolTableCollection &tables, ModuleOp mod, SymbolLookupResult<StructDefOp> structDefRes,
254 const SourceRef &root
255) {
256 std::vector<SourceRef> res;
257 // Recurse into struct types by iterating over all their member definitions
258 for (auto f : structDefRes.get().getOps<MemberDefOp>()) {
259 // We want to store the MemberDefOp, but without the possibility of accidentally dropping the
260 // reference, so we need to re-lookup the symbol to create a SymbolLookupResult, which will
261 // manage the external module containing the member defs, if needed.
262 // TODO: It would be nice if we could manage module op references differently
263 // so we don't have to do this.
264 auto structDefCopy = structDefRes;
265 auto memberLookup = lookupSymbolIn<MemberDefOp>(
266 tables, SymbolRefAttr::get(f.getContext(), f.getSymNameAttr()), std::move(structDefCopy),
267 mod.getOperation()
268 );
269 ensure(succeeded(memberLookup), "could not get SymbolLookupResult of existing MemberDefOp");
270 SourceRef childRef = root.createChild(SourceRefIndex(memberLookup.value()));
271 // Make a reference to the current member, regardless of if it is a composite
272 // type or not.
273 res.push_back(childRef);
274 }
275 return res;
276}

278std::vector<SourceRef>
279SourceRef::getAllChildren(SymbolTableCollection &tables, ModuleOp mod) const {
280 auto ty = getType();
281 if (auto structTy = dyn_cast<StructType>(ty)) {
282 return llzk::getAllChildren(tables, mod, getStructDef(tables, mod, structTy), *this);
283 } else if (auto arrayType = dyn_cast<ArrayType>(ty)) {
284 return llzk::getAllChildren(tables, mod, arrayType, *this);
285 }
286 // Scalar type, no children
287 return {};

290void SourceRef::print(raw_ostream &os) const {
291 if (isConstantFelt()) {
292 os << "<felt.const: " << getConstantFeltValue() << '>';
293 } else if (isConstantIndex()) {
294 os << "<index: " << getConstantIndexValue() << '>';
295 } else if (isTemplateConstant()) {
296 auto constRead = std::get<ConstReadOp>(*constantVal);
297 auto structDefOp = constRead->getParentOfType<StructDefOp>();
298 ensure(structDefOp, "struct template should have a struct parent");
299 os << '@' << structDefOp.getName() << "<[@" << constRead.getConstName() << "]>";
300 } else {
301 if (isCreateStructOp()) {
302 os << "%self";
303 } else {
304 ensure(isBlockArgument(), "unhandled print case");
305 os << "%arg" << getInputNum();
306 }
307
308 for (const auto &f : memberRefs) {
309 os << "[" << f << "]";
310 }
311 }

314bool SourceRef::operator==(const SourceRef &rhs) const {
315 // This way two felt constants can be equal even if the declared in separate ops.
316 if (isConstantInt() && rhs.isConstantInt()) {
317 DynamicAPInt lhsVal = getConstantValue(), rhsVal = rhs.getConstantValue();
318 return getType() == rhs.getType() && lhsVal == rhsVal;
319 }
320 return (root == rhs.root) && (memberRefs == rhs.memberRefs) && (constantVal == rhs.constantVal);
321}

323// required for EquivalenceClasses usage
324bool SourceRef::operator<(const SourceRef &rhs) const {
325 if (isConstantFelt() && !rhs.isConstantFelt()) {
326 // Put all constants at the end
327 return false;
328 } else if (!isConstantFelt() && rhs.isConstantFelt()) {
329 return true;
330 } else if (isConstantFelt() && rhs.isConstantFelt()) {
331 DynamicAPInt lhsInt = getConstantFeltValue(), rhsInt = rhs.getConstantFeltValue();
332 return lhsInt < rhsInt;
333 }
334
335 if (isConstantIndex() && !rhs.isConstantIndex()) {
336 // Put all constant indices next at the end
337 return false;
338 } else if (!isConstantIndex() && rhs.isConstantIndex()) {
339 return true;
340 } else if (isConstantIndex() && rhs.isConstantIndex()) {
341 DynamicAPInt lhsVal = getConstantIndexValue(), rhsVal = rhs.getConstantIndexValue();
342 return lhsVal < rhsVal;
343 }
344
345 if (isTemplateConstant() && !rhs.isTemplateConstant()) {
346 // Put all template constants next at the end
347 return false;
348 } else if (!isTemplateConstant() && rhs.isTemplateConstant()) {
349 return true;
350 } else if (isTemplateConstant() && rhs.isTemplateConstant()) {
351 StringRef lhsName = std::get<ConstReadOp>(*constantVal).getConstName();
352 StringRef rhsName = std::get<ConstReadOp>(*rhs.constantVal).getConstName();
353 return lhsName.compare(rhsName) < 0;
354 }
355
356 // Sort out the block argument vs struct.new cases
357 if (isBlockArgument() && rhs.isCreateStructOp()) {
358 return true;
359 } else if (isCreateStructOp() && rhs.isBlockArgument()) {
360 return false;
361 } else if (isBlockArgument() && rhs.isBlockArgument()) {
362 if (getInputNum() < rhs.getInputNum()) {
363 return true;
364 } else if (getInputNum() > rhs.getInputNum()) {
365 return false;
366 }
367 } else if (isCreateStructOp() && rhs.isCreateStructOp()) {
368 CreateStructOp lhsOp = getCreateStructOp(), rhsOp = rhs.getCreateStructOp();
369 if (lhsOp < rhsOp) {
370 return true;
371 } else if (lhsOp > rhsOp) {
372 return false;
373 }
374 } else {
375 llvm_unreachable("unhandled operator< case");
376 }
377
378 for (size_t i = 0; i < memberRefs.size() && i < rhs.memberRefs.size(); i++) {
379 if (memberRefs[i] < rhs.memberRefs[i]) {
380 return true;
381 } else if (memberRefs[i] > rhs.memberRefs[i]) {
382 return false;
383 }
384 }
385 return memberRefs.size() < rhs.memberRefs.size();
386}

388size_t SourceRef::Hash::operator()(const SourceRef &val) const {
389 if (val.isConstantInt()) {
390 return llvm::hash_value(val.getConstantValue());
391 } else if (val.isTemplateConstant()) {
392 return OpHash<ConstReadOp> {}(std::get<ConstReadOp>(*val.constantVal));
393 } else {
394 ensure(val.isBlockArgument() || val.isCreateStructOp(), "unhandled SourceRef hash case");
395
396 size_t hash = val.isBlockArgument() ? std::hash<unsigned> {}(val.getInputNum())
397 : OpHash<CreateStructOp> {}(val.getCreateStructOp());
398 for (const auto &f : val.memberRefs) {
399 hash ^= f.getHash();
400 }
401 return hash;
402 }
403}
