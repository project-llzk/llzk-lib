29static llvm::Expected<size_t>
30dynamicAPIntToSize(const llvm::DynamicAPInt &value, llvm::Twine context) {
31 if (value < 0) {
32 return makeError(context + " would underflow size_t");
33 }
34 llvm::APSInt as = llzk::toAPSInt(value);
35 if (as.getActiveBits() > std::numeric_limits<size_t>::digits) {
36 return makeError(context + " would overflow size_t");
37 }
38 return llzk::checkedCast<size_t>(as.getZExtValue());
39}

50llvm::DynamicAPInt randomFieldElement(std::mt19937_64 &rng, const Field &field) {
51 const uint64_t prime = toAPSInt(field.prime()).getZExtValue();
52 if (prime == 0) {
53 return field.zero();
54 }
55 uint64_t candidate = std::uniform_int_distribution<uint64_t>(0, prime - 1)(rng);
56 return field.reduce(candidate);
57}

68llvm::Expected<size_t> checkedShapeDimToSize(int64_t dim, llvm::StringRef context) {
69 if (ShapedType::isDynamic(dim)) {
70 return makeError(llvm::Twine(context) + " requires a static shape");
71 }
72 if (dim < 0) {
73 return makeError(llvm::Twine(context) + " has a negative dimension");
74 }
75 llvm::DynamicAPInt value(dim);
76 return dynamicAPIntToSize(value, llvm::Twine(context) + " dimension");
77}

84llvm::Expected<size_t>
85getStaticShapeElementCount(llvm::ArrayRef<int64_t> shape, llvm::StringRef context) {
86 llvm::DynamicAPInt count(1);
87 for (int64_t dim : shape) {
88 auto dimSize = checkedShapeDimToSize(dim, context);
89 if (!dimSize) {
90 return dimSize.takeError();
91 }
92 count *= llvm::DynamicAPInt(*dimSize);
93 }
94 return dynamicAPIntToSize(count, context);
95}

97llvm::Expected<size_t> getStaticElementCount(ShapedType type, llvm::StringRef context) {
98 if (!type.hasStaticShape()) {
99 return makeError(llvm::Twine(context) + " requires a static shape");
100 }
101 int64_t count = type.getNumElements();
102 if (count < 0) {
103 return makeError(llvm::Twine(context) + " has an invalid negative element count");
104 }
105 return dynamicAPIntToSize(llvm::DynamicAPInt(count), context);
106}
