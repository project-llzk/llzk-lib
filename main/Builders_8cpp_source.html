22OwningOpRef<ModuleOp> createLLZKModule(MLIRContext * /*context*/, Location loc) {

28void addLangAttrForLLZKDialect(ModuleOp mod) {
29 MLIRContext *ctx = mod.getContext();
30 if (auto *dialect = ctx->getOrLoadDialect<LLZKDialect>()) {
31 mod->setAttr(LANG_ATTR_NAME, StringAttr::get(ctx, dialect->getNamespace()));
32 } else {
33 llvm::report_fatal_error("Could not load LLZK dialect!");
34 }
35}

99ModuleBuilder &
100ModuleBuilder::insertEmptyStruct(std::string_view structName, Location loc, int numStructParams) {
101 ensureNoSuchStruct(structName);
102
103 OpBuilder opBuilder(rootModule.getBody(), rootModule.getBody()->begin());
104 auto structNameAttr = StringAttr::get(context, structName);
105 ArrayAttr structParams = nullptr;
106 if (numStructParams >= 0) {
107 SmallVector<Attribute> paramNames;
108 for (int i = 0; i < numStructParams; ++i) {
109 paramNames.push_back(FlatSymbolRefAttr::get(context, "T" + std::to_string(i)));
110 }
111 structParams = opBuilder.getArrayAttr(paramNames);
112 }
113 auto structDef = opBuilder.create<StructDefOp>(loc, structNameAttr, structParams);
114 // populate the initial region
115 auto &region = structDef.getRegion();
116 (void)region.emplaceBlock();
117 structMap[structName] = structDef;
118
119 return *this;
120}

198 auto calleeFn = computeFnMap.at(callee.getName());

202 updateComputeReachability(caller, callee);

214ModuleBuilder &ModuleBuilder::insertConstrainCall(
215 StructDefOp caller, StructDefOp callee, Location callLoc, Location memberDefLoc
216) {
217 ensureConstrainFnExists(caller.getName());
218 ensureConstrainFnExists(callee.getName());
219
220 FuncDefOp callerFn = constrainFnMap.at(caller.getName());
221 FuncDefOp calleeFn = constrainFnMap.at(callee.getName());
222 StructType calleeTy = callee.getType();
223
224 size_t numOps = caller.getBody()->getOperations().size();
225 auto memberName = StringAttr::get(context, callee.getName().str() + std::to_string(numOps));
226
227 // Insert the member declaration op
228 {
229 OpBuilder builder(caller.getBodyRegion());
230 builder.create<MemberDefOp>(memberDefLoc, memberName, calleeTy);
231 }
232
233 // Insert the constrain function ops
234 {
235 OpBuilder builder(callerFn.getBody());
236
237 auto member = builder.create<MemberReadOp>(
238 callLoc, calleeTy, callerFn.getSelfValueFromConstrain(), memberName
239 );
240 builder.create<CallOp>(
241 callLoc, TypeRange {}, calleeFn.getFullyQualifiedName(), ValueRange {member}
242 );
243 }
244 updateConstrainReachability(caller, callee);
245 return *this;
246}
