32struct NestedTextualOptions {
34 std::string str;
35
41 template <typename ValueT, typename CreateFnT, typename InitializeFnT>
42 std::unique_ptr<ValueT> createValidatedValue(
43 CreateFnT &&createValue, llvm::StringRef kind, InitializeFnT &&initializeValue
44 ) const {
45 auto value = createValue();
46 if (str.empty()) {
47 return value;
48 }
49
50 std::string error;
51 if (failed(initializeValue(*value, str, error))) {
52 llvm::report_fatal_error(
53 llvm::Twine("failed to initialize previously-validated nested ") + kind +
54 " options: " + error
55 );
56 }
57 return value;
58 }
59};

65template <auto CreatePass> struct NestedPassOptions : detail::NestedTextualOptions {
67 std::unique_ptr<mlir::Pass> createPass() const {
68 return this->createValidatedValue<mlir::Pass>(CreatePass, "pass", initializePass);
69 }
70
71 static mlir::LogicalResult
72 initializePass(mlir::Pass &pass, llvm::StringRef options, std::string &error) {
73 return pass.initializeOptions(options, [&error](const llvm::Twine &message) {
74 error = message.str();
75 return mlir::failure();
76 });
77 }
78};

82template <typename PipelineOptionsT> struct NestedPipelineOptions : detail::NestedTextualOptions {
84 std::unique_ptr<PipelineOptionsT> createOptions() const {
85 return this->createValidatedValue<PipelineOptionsT>(

91 initializeOptions(PipelineOptionsT &options, llvm::StringRef value, std::string &error) {
92 llvm::raw_string_ostream errorStream(error);

103template <typename OptionsT> class NestedOptionsParserBase : public basic_parser<OptionsT> {
104public:
105 NestedOptionsParserBase(Option &O) : basic_parser<OptionsT>(O) {}
106
107protected:
110 bool parseNestedOptions(Option &O, StringRef Arg, StringRef kind, StringRef &options) const {
111 options = Arg;
112 if (options.consume_front("{") && !options.consume_back("}")) {
113 return O.error(llvm::Twine("expected nested ") + kind + " options to end with '}'");
114 }
115 return false;
116 }
117
118public:
119 static void print(llvm::raw_ostream &OS, const OptionsT &Val) { OS << '{' << Val.str << '}'; }
120
121 void printOptionDiff(
122 const Option &O, const OptionsT &V, const OptionValue<OptionsT> &Default, size_t GlobalWidth
123 ) const {
124 this->printOptionName(O, GlobalWidth);
125 print(llvm::outs(), V);
126 llvm::outs() << " (default: ";
127 if (Default.hasValue()) {
128 print(llvm::outs(), Default.getValue());
129 } else {
130 llvm::outs() << "<unspecified>";
131 }
132 llvm::outs() << ")\n";
133 }
134};

137template <auto CreatePass>
138class parser<llzk::NestedPassOptions<CreatePass>>
139 : public NestedOptionsParserBase<llzk::NestedPassOptions<CreatePass>> {
140public:
141 using OptionsT = llzk::NestedPassOptions<CreatePass>;
142 using Base = NestedOptionsParserBase<OptionsT>;
143
144 parser(Option &O) : Base(O) {}
145
146 bool parse(Option &O, StringRef, StringRef Arg, OptionsT &Val) {
147 StringRef options;
148 if (this->parseNestedOptions(O, Arg, "pass", options)) {
149 return true;
150 }
151
152 auto pass = CreatePass();
153 std::string error;
154 if (failed(OptionsT::initializePass(*pass, options, error))) {
155 return O.error(error);
156 }
157
158 Val.str = options.str();
159 return false;
160 }
161};

191// Parser for APInt
192template <> class parser<APInt> : public basic_parser<APInt> {
193public:
194 parser(Option &O) : basic_parser(O) {}
195
196 bool parse(Option &O, StringRef, StringRef Arg, APInt &Val) {
197 if (Arg.empty()) {
198 return O.error("empty integer literal");
199 }
200 if (!all_of(Arg, [](char c) { return isDigit(c); })) {
201 return O.error("arg must be in base 10 (digits).");
202 }
203 // Decimal-only: allocate a safe width then shrink.
204 unsigned bits = std::max(1u, 4u * llzk::checkedCast<unsigned>(Arg.size()));
205 APInt tmp(bits, Arg, 10);
206 unsigned active = tmp.getActiveBits();
207 if (active == 0) {
208 active = 1;
209 }
210 Val = tmp.zextOrTrunc(active);
211 return false;
212 }

214 // Prints how the passed option differs from the default one specified in the pass
215 // For example, if V = 17 and Default = 11 then it should print
216 // [OptionName] 17 (default: 11)
217 void printOptionDiff(
218 const Option &O, const APInt &V, const OptionValue<APInt> &Default, size_t GlobalWidth
219 ) const {
220 std::string Cur = llvm::toString(V, 10, false);
221
222 std::string Def = "<unspecified>";
223 if (Default.hasValue()) {
224 const APInt &D = Default.getValue();
225 Def = llvm::toString(D, 10, false);
226 }
227
228 printOptionName(O, GlobalWidth);
229 llvm::outs() << Cur << " (default: " << Def << ")\n";
230 }
231};
