34inline std::string
35reserveUniqueFunctionArgName(llvm::StringSet<> &usedNames, llvm::StringRef desiredName) {
36 if (!usedNames.contains(desiredName)) {
37 usedNames.insert(desiredName);
38 return desiredName.str();
39 }
40
41 for (unsigned suffix = 1;; ++suffix) {
42 std::string candidate = (desiredName + "#" + llvm::Twine(suffix)).str();
43 if (!usedNames.contains(candidate)) {
44 usedNames.insert(candidate);
45 return candidate;
46 }
47 }

68 void convert(function::FuncDefOp op, mlir::RewriterBase &rewriter) {
69 // Update in/out types of the function
70 mlir::FunctionType oldTy = op.getFunctionType();
71 llvm::SmallVector<mlir::Type> newInputs = convertInputs(oldTy.getInputs());
72 llvm::SmallVector<mlir::Type> newResults = convertResults(oldTy.getResults());
73 mlir::FunctionType newTy = mlir::FunctionType::get(
74 oldTy.getContext(), mlir::TypeRange(newInputs), mlir::TypeRange(newResults)
75 );
76 if (newTy == oldTy) {
77 return; // nothing to change
78 }
79
80 // Pre-condition: arg/result count equals corresponding attribute count
81 assert(!op.getResAttrsAttr() || op.getResAttrsAttr().size() == op.getNumResults());
82 assert(!op.getArgAttrsAttr() || op.getArgAttrsAttr().size() == op.getNumArguments());
83 rewriter.modifyOpInPlace(op, [&]() {
84 op.setFunctionType(newTy);
85
86 // If any input or result types were added, ensure the attributes are updated too.
87 if (mlir::ArrayAttr newArgAttrs = convertInputAttrs(op.getArgAttrsAttr(), newInputs)) {
88 op.setArgAttrsAttr(newArgAttrs);
89 }
90 if (mlir::ArrayAttr newResAttrs = convertResultAttrs(op.getResAttrsAttr(), newResults)) {
91 op.setResAttrsAttr(newResAttrs);
92 }
93 });
94 // Post-condition: arg/result count equals corresponding attribute count
95 assert(!op.getResAttrsAttr() || op.getResAttrsAttr().size() == op.getNumResults());
96 assert(!op.getArgAttrsAttr() || op.getArgAttrsAttr().size() == op.getNumArguments());
97
98 // If the function has a body, ensure the entry block arguments match the function inputs.
99 if (mlir::Region *body = op.getCallableRegion()) {
100 mlir::Block &entryBlock = body->front();
101 if (!std::cmp_equal(entryBlock.getNumArguments(), newInputs.size())) {
102 processBlockArgs(entryBlock, rewriter);
103 // Post-condition: block args must match function inputs
104 assert(std::cmp_equal(entryBlock.getNumArguments(), newInputs.size()));
105 }
106 }
107 }
108};
