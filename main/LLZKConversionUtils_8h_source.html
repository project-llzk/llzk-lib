39inline std::string
40reserveUniqueFunctionArgName(llvm::StringSet<> &usedNames, llvm::StringRef desiredName) {
41 if (!usedNames.contains(desiredName)) {
42 usedNames.insert(desiredName);
43 return desiredName.str();
44 }
45
46 for (unsigned suffix = 1;; ++suffix) {
47 std::string candidate = (desiredName + "#" + llvm::Twine(suffix)).str();
48 if (!usedNames.contains(candidate)) {
49 usedNames.insert(candidate);
50 return candidate;
51 }
52 }

73 void convert(function::FuncDefOp op, mlir::RewriterBase &rewriter) {
74 // Update in/out types of the function
75 mlir::FunctionType oldTy = op.getFunctionType();
76 llvm::SmallVector<mlir::Type> newInputs = convertInputs(oldTy.getInputs());
77 llvm::SmallVector<mlir::Type> newResults = convertResults(oldTy.getResults());
78 mlir::FunctionType newTy = mlir::FunctionType::get(
79 oldTy.getContext(), mlir::TypeRange(newInputs), mlir::TypeRange(newResults)
80 );
81 if (newTy == oldTy) {
82 return; // nothing to change
83 }
84
85 // Pre-condition: arg/result count equals corresponding attribute count
86 assert(!op.getResAttrsAttr() || op.getResAttrsAttr().size() == op.getNumResults());
87 assert(!op.getArgAttrsAttr() || op.getArgAttrsAttr().size() == op.getNumArguments());
88 rewriter.modifyOpInPlace(op, [&]() {
89 op.setFunctionType(newTy);
90
91 // If any input or result types were added, ensure the attributes are updated too.
92 if (mlir::ArrayAttr newArgAttrs = convertInputAttrs(op.getArgAttrsAttr(), newInputs)) {
93 op.setArgAttrsAttr(newArgAttrs);
94 }
95 if (mlir::ArrayAttr newResAttrs = convertResultAttrs(op.getResAttrsAttr(), newResults)) {
96 op.setResAttrsAttr(newResAttrs);
97 }
98 });
99 // Post-condition: arg/result count equals corresponding attribute count
100 assert(!op.getResAttrsAttr() || op.getResAttrsAttr().size() == op.getNumResults());
101 assert(!op.getArgAttrsAttr() || op.getArgAttrsAttr().size() == op.getNumArguments());
102
103 // If the function has a body, ensure the entry block arguments match the function inputs.
104 if (mlir::Region *body = op.getCallableRegion()) {
105 mlir::Block &entryBlock = body->front();
106 if (!std::cmp_equal(entryBlock.getNumArguments(), newInputs.size())) {
107 processBlockArgs(entryBlock, rewriter);
108 // Post-condition: block args must match function inputs
109 assert(std::cmp_equal(entryBlock.getNumArguments(), newInputs.size()));
110 }
111 }
112 }
113};

122template <
123 typename ImplClass, HasInterface<component::MemberRefOpInterface> MemberRefOpClass,
124 typename GenHeaderType, typename IdType>
125class SplitAggregateInMemberRefOp : public mlir::OpConversionPattern<MemberRefOpClass> {
126public:
128 using MemberInfo = std::pair<mlir::StringAttr, mlir::Type>;
130 using LocalMemberReplacementMap = llvm::DenseMap<IdType, MemberInfo>;
132 using MemberReplacementMap = llvm::DenseMap<
133 component::StructDefOp, llvm::DenseMap<mlir::StringAttr, LocalMemberReplacementMap>>;
134
135private:
136 mlir::SymbolTableCollection &tables;
137 const MemberReplacementMap &repMapRef;
138
139 // Static check to ensure the methods are implemented in all subclasses.
140 inline static void ensureImplementedAtCompile() {
141 static_assert(
142 sizeof(MemberRefOpClass) == 0,
143 "SplitAggregateInMemberRefOp not implemented for requested type."
144 );
145 }
146
147protected:
148 using OpAdaptor = typename MemberRefOpClass::Adaptor;
149
152 static GenHeaderType genHeader(MemberRefOpClass, mlir::ConversionPatternRewriter &) {
153 ensureImplementedAtCompile();
154 llvm_unreachable("must have concrete instantiation");
155 }
156
159 static void forId(
160 mlir::Location, GenHeaderType, IdType, MemberInfo, OpAdaptor,
161 mlir::ConversionPatternRewriter &
162 ) {
163 ensureImplementedAtCompile();
164 llvm_unreachable("must have concrete instantiation");
165 }
166
167public:
168 // Suppress false positive from `clang-tidy`
169 // NOLINTNEXTLINE(bugprone-crtp-constructor-accessibility)
170 SplitAggregateInMemberRefOp(
171 mlir::MLIRContext *ctx, mlir::SymbolTableCollection &symTables,
172 const MemberReplacementMap &memberRepMap
173 )
174 : mlir::OpConversionPattern<MemberRefOpClass>(ctx), tables(symTables),
175 repMapRef(memberRepMap) {}
176
177 static bool legal(MemberRefOpClass) {
178 ensureImplementedAtCompile();
179 llvm_unreachable("must have concrete instantiation");
180 return false;
181 }
182
183 mlir::LogicalResult match(MemberRefOpClass op) const override {
184 return mlir::failure(ImplClass::legal(op));
185 }
186
187 void rewrite(
188 MemberRefOpClass op, OpAdaptor adaptor, mlir::ConversionPatternRewriter &rewriter
189 ) const override {
190 component::StructType tgtStructTy =
191 llvm::cast<component::MemberRefOpInterface>(op.getOperation()).getStructType();
192 assert(tgtStructTy);
193 auto tgtStructDef = tgtStructTy.getDefinition(tables, op);
194 assert(mlir::succeeded(tgtStructDef));
195
196 GenHeaderType prefixResult = ImplClass::genHeader(op, rewriter);
197
198 const LocalMemberReplacementMap &idToName =
199 repMapRef.at(tgtStructDef->get()).at(op.getMemberNameAttr().getAttr());
200 // Split the aggregate member into a series of scalar member ops.
201 for (auto [id, newMember] : idToName) {
202 ImplClass::forId(op.getLoc(), prefixResult, id, newMember, adaptor, rewriter);
203 }
204 rewriter.eraseOp(op);
205 }
206};
