26template <typename TypeClass>
27// Suppress false positive from `clang-tidy`
28// NOLINTNEXTLINE(bugprone-crtp-constructor-accessibility)
29class LLZKSymbolTableImplTrait
30 : public mlir::OpTrait::TraitBase<TypeClass, LLZKSymbolTableImplTrait> {
31public:
32 static mlir::LogicalResult verifyRegionTrait(mlir::Operation *op) {
33 // Note: the current op will be checked by the normal `SymbolTable` trait that is
34 // included in `LLZKSymbolTable`. Checking it here would cause the same error described
35 // in `LLZKSymbolTable`.
36 while ((op = op->getParentWithTrait<mlir::OpTrait::SymbolTable>())) {
37 if (mlir::failed(mlir::detail::verifySymbolTable(op))) {
38 return mlir::failure();
39 }
40 }
41 return mlir::success();
42 }
43};

77template <int OperandSegmentIndex> struct VerifySizesForMultiAffineOps {
78 template <typename TypeClass> class Impl : public mlir::OpTrait::TraitBase<TypeClass, Impl> {
79 inline static mlir::LogicalResult verifyHelper(mlir::Operation *op, int32_t segmentSize) {
80 TypeClass c = llvm::cast<TypeClass>(op);
81 return affineMapHelpers::verifySizesForMultiAffineOps(
82 op, segmentSize, c.getMapOpGroupSizesAttr(), c.getMapOperands(), c.getNumDimsPerMapAttr()
83 );
84 }
85
86 public:
87 static mlir::LogicalResult verifyTrait(mlir::Operation *op) {
88 if (TypeClass::template hasTrait<mlir::OpTrait::AttrSizedOperandSegments>()) {
89 // If the AttrSizedOperandSegments trait is present, must have `OperandSegmentIndex`.
90 static_assert(
91 OperandSegmentIndex >= 0,
92 "When the `AttrSizedOperandSegments` trait is present, the index of `$mapOperands` "
93 "within the `operandSegmentSizes` attribute must be specified."
94 );
95 mlir::DenseI32ArrayAttr segmentSizes = op->getAttrOfType<mlir::DenseI32ArrayAttr>(
96 mlir::OpTrait::AttrSizedOperandSegments<TypeClass>::getOperandSegmentSizeAttr()
97 );
98 assert(
99 OperandSegmentIndex < segmentSizes.size() &&
100 "Parameter of `VerifySizesForMultiAffineOps` exceeds the number of ODS-declared "
101 "operands"
102 );
103 return verifyHelper(op, segmentSizes[OperandSegmentIndex]);
104 } else {
105 // If the trait is not present, the `OperandSegmentIndex` is ignored. Pass `-1` to indicate
106 // that the checks against `operandSegmentSizes` should be skipped.
107 return verifyHelper(op, -1);
108 }
109 }
110 };
111};
