49static mlir::WalkResult unwrap(MlirWalkResult result) {
50 switch (result) {
51 case MlirWalkResultAdvance:
52 return mlir::WalkResult::advance();
53
54 case MlirWalkResultInterrupt:
55 return mlir::WalkResult::interrupt();
56
57 case MlirWalkResultSkip:
58 return mlir::WalkResult::skip();
59 }
60 llvm_unreachable("unknown result in WalkResult::unwrap");
61}

63void mlirOperationWalkReverse(
64 MlirOperation from, MlirOperationWalkCallback callback, void *userData, MlirWalkOrder walkOrder
65) {
66 switch (walkOrder) {
67 case MlirWalkPreOrder:
68 unwrap(from)->walk<WalkOrder::PreOrder, ReverseIterator>([callback, userData](Operation *op) {
69 return unwrap(callback(wrap(op), userData));
70 });
71 break;
72 case MlirWalkPostOrder:
73 unwrap(from)->walk<WalkOrder::PostOrder, ReverseIterator>([callback, userData](Operation *op) {
74 return unwrap(callback(wrap(op), userData));
75 });
76 }
77}

84template <typename T> void appendElems(T const *src, intptr_t srcSize, T *&dst, intptr_t &dstSize) {
85 assert(srcSize >= 0 && "Negative source size");
86 assert(dstSize >= 0 && "Negative destination size");
87 T *newDst = static_cast<T *>(std::realloc(dst, (srcSize + dstSize) * sizeof(T)));
88 assert(newDst && "Failed to increase the size of buffer");
89 dst = newDst; // this temporary + assert prevents `clang-tidy` warnings
90 std::memcpy(dst + dstSize, src, srcSize * sizeof(T));
91 dstSize += srcSize;
92}

169void llzkAffineMapOperandsBuilderConvertDimsPerMapToArray(LlzkAffineMapOperandsBuilder *builder) {
170 if (builder->nDimsPerMap >= 0) {
171 return;
172 }
173 auto attrData = mlir::unwrap_cast<DenseI32ArrayAttr>(builder->dimsPerMap.attr).asArrayRef();
174 size_t realSize = attrData.size() * sizeof(decltype(attrData)::value_type);
175 builder->dimsPerMap.array = static_cast<int32_t *>(std::malloc(realSize));
176 std::memcpy(builder->dimsPerMap.array, attrData.data(), realSize);
177 builder->nDimsPerMap = llzk::checkedCast<intptr_t>(attrData.size());
178}
