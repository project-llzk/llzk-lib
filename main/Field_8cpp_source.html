49FailureOr<std::reference_wrapper<const Field>> Field::tryGetField(StringRef fieldName) {
50 static std::once_flag fieldsInit;
51 std::call_once(fieldsInit, initKnownFields);
52
53 if (auto it = knownFields.find(fieldName); it != knownFields.end()) {
54 return {it->second};
55 }
56 return failure();
57}

66const Field &Field::getField(StringRef fieldName, EmitErrorFn errFn) {
67 auto res = tryGetField(fieldName);
68 if (succeeded(res)) {
69 return res.value().get();
70 }
71 std::string msg = "field \"" + fieldName.str() + "\" is unsupported";
72 if (errFn) {
73 errFn().append(msg).report();
74 }
75 llvm::report_fatal_error(msg.c_str());
76}

78void Field::addField(Field &&f, EmitErrorFn errFn) {
79 // Use `tryGetField()` to ensure knownFields is initialized before checking for conflicts.
80 auto existing = Field::tryGetField(f.name());
81 if (succeeded(existing)) {
82 // Field exists and conflicts with existing definition.
83 std::string msg;
84 debug::Appender(msg) << "Definition of \"" << f.name()
85 << "\" conflicts with prior definition: prior="
86 << existing.value().get().prime() << ", new=" << f.prime();
87 if (errFn) {
88 errFn().append(msg).report();
89 } else {
90 llvm::report_fatal_error(msg.c_str());
91 }
92 return;
93 }
94 // Field does not exist, add it.
95 knownFields.try_emplace(f.name(), f);
96}

120DynamicAPInt Field::reduce(const DynamicAPInt &i) const {
121 DynamicAPInt m = i % prime();

136// Parses Fields from the given attribute, if able.
137static LogicalResult parseFields(Attribute a) {
138 // clang-format off
139 return TypeSwitch<
140 Attribute, FailureOr<SmallVector<std::reference_wrapper<const Field>>>>(a)
141 .Case<UnitAttr>(
142 [](auto) {
143 return success();
144 })
145 .Case<StringAttr>(
146 [](auto s) -> FailureOr<SmallVector<std::reference_wrapper<const Field>>> {
147 auto fieldRes = Field::tryGetField(s);
148 if (failed(fieldRes)) {
149 return failure();
150 }
151 return SmallVector<std::reference_wrapper<const Field>> {fieldRes.value()};
152 })
153 .Case<ArrayAttr>(
154 [](auto arr) -> FailureOr<SmallVector<std::reference_wrapper<const Field>>> {
155 // An ArrayAttr may only contain inner StringAttr
156 SmallVector<std::reference_wrapper<const Field>> res;
157 for (Attribute elem : arr) {
158 if (auto s = llvm::dyn_cast<StringAttr>(elem)) {
159 auto fieldRes = Field::tryGetField(s);
160 if (failed(fieldRes)) {
161 return failure();
162 }
163 res.push_back(fieldRes.value());
164 } else {
165 return failure();
166 }
167 }
168 return res;
169 })
170 .Default([](auto) { return failure(); });
171 // clang-format on
172}

174LogicalResult addSpecifiedFields(ModuleOp modOp) {
175 if (Attribute a = modOp->getAttr(FIELD_ATTR_NAME)) {
176 return parseFields(a);
177 }
178 // Always recurse.
179 if (ModuleOp parentMod = modOp->getParentOfType<ModuleOp>()) {
180 return addSpecifiedFields(parentMod);
181 }
182 return success();
183}

204static void handleType(mlir::Type type, FieldsCtx &ctx) {
205 TypeSwitch<mlir::Type> ts(type);
206 ts.Case([&ctx](llzk::felt::FeltType felt) {
207 if (felt.hasField()) {
208 ctx.fields.insert(felt.getField());
209 } else {
210 ctx.status = failure();
211 ctx.scope->emitWarning()
212 << "felt type does not declare its field. This may cause some passes to fail";
213 }
214 })
215 .Case([&ctx](llzk::array::ArrayType array) { handleType(array.getElementType(), ctx); })
216 .Case([&ctx](llzk::pod::PodType pod) {
217 for (auto record : pod.getRecords()) {
218 handleAttribute(record, ctx);
219 }
220 }).Case([&ctx](mlir::FunctionType funcType) {
221 for (auto i : funcType.getInputs()) {
222 handleType(i, ctx);
223 }
224 for (auto o : funcType.getResults()) {
225 handleType(o, ctx);
226 }
227 });
228 // Accepted types that are no-ops.
229 doNothing<
230 // clang-format off
231 llzk::component::StructType,
232 llzk::string::StringType,
233 mlir::IntegerType,
234 llzk::polymorphic::TypeVarType
235 // clang-format on
236 >(ts);
237 // Fail on any type we don't recognize.
238 ts.Default([&ctx](auto t) {
239 ctx.scope->emitOpError() << "unhandled type " << t
240 << " while extracting fields from felt types";
241 });
242}

244static void handleAttribute(mlir::Attribute attr, FieldsCtx &ctx) {
245 TypeSwitch<mlir::Attribute> ts(attr);
246 ts.Case([&ctx](mlir::TypeAttr typeAttr) { handleType(typeAttr.getValue(), ctx); })
247 .Case([&ctx](mlir::ArrayAttr arrayAttr) {
248 for (auto a : arrayAttr) {
249 handleAttribute(a, ctx);
250 }
251 })
252 .Case([&ctx](mlir::DictionaryAttr dictAttr) {
253 for (auto a : dictAttr.getValue()) {
254 handleAttribute(a.getValue(), ctx);
255 }
256 }).Case([&ctx](llzk::pod::RecordAttr recordAttr) {
257 handleType(recordAttr.getType(), ctx);
258 }).Default([](auto) {});
259}

261LogicalResult llzk::collectFields(mlir::Operation *root, llzk::FieldSet &fields) {
262 if (!root) {
263 return success(); // Nothing to do
264 }
265 LogicalResult status = success();
266 root->walk([&fields, &status](mlir::Operation *op) {
267 FieldsCtx ctx = {.fields = fields, .status = status, .scope = op};
268 // Crawl for types in the results,
269 for (auto result : op->getOpResults()) {
270 handleType(result.getType(), ctx);
271 }
272 // the attributes,
273 for (auto attr : op->getAttrs()) {
274 handleAttribute(attr.getValue(), ctx);
275 }
276 // block arguments (if any)
277 for (auto &region : op->getRegions()) {
278 for (auto &block : region) {
279 for (auto &arg : block.getArguments()) {
280 handleType(arg.getType(), ctx);
281 }
282 }
283 }
284 });
285
286 return status;
287}
