67template <> LogicalResult SetFuncAllowAttrs<StructDefOp>::verifyTrait(Operation *structOp) {
68 assert(llvm::isa<StructDefOp>(structOp));
69 Region &bodyRegion = llvm::cast<StructDefOp>(structOp).getBodyRegion();
70 if (!bodyRegion.empty()) {
71 bodyRegion.front().walk([](FuncDefOp funcDef) {
72 if (funcDef.nameIsConstrain()) {
73 funcDef.setAllowConstraintAttr();
74 funcDef.setAllowWitnessAttr(false);
75 } else if (funcDef.nameIsCompute()) {
76 funcDef.setAllowConstraintAttr(false);
77 funcDef.setAllowWitnessAttr();
78 } else if (funcDef.nameIsProduct()) {
79 funcDef.setAllowConstraintAttr();
80 funcDef.setAllowWitnessAttr();
81 }
82 });
83 }
84 return success();
85}

101 return origin->emitError() << '\'' << StructDefOp::getOperationName() << "' op "

109LogicalResult checkSelfType(
110 SymbolTableCollection &tables, StructDefOp expectedStruct, Type actualType, Operation *origin,
111 const char *aspect
112) {
113 if (StructType actualStructType = llvm::dyn_cast<StructType>(actualType)) {
114 auto actualStructOpt =
115 lookupTopLevelSymbol<StructDefOp>(tables, actualStructType.getNameRef(), origin);
116 if (failed(actualStructOpt)) {
117 return origin->emitError().append(
118 "could not find '", StructDefOp::getOperationName(), "' named \"",
119 actualStructType.getNameRef(), '"'
120 );
121 }
122 StructDefOp actualStruct = actualStructOpt.value().get();
123 if (actualStruct != expectedStruct) {
124 return genCompareErr(expectedStruct, origin, aspect)
125 .attachNote(actualStruct.getLoc())
126 .append("uses this type instead");
127 }
128 // Check for an EXACT match in the parameter list since it must reference the "self" type.
129 if (expectedStruct.getConstParamsAttr() != actualStructType.getParams()) {
130 // To make error messages more consistent and meaningful, if the parameters don't match
131 // because the actual type uses symbols that are not defined, generate an error about the
132 // undefined symbol(s).
133 if (ArrayAttr tyParams = actualStructType.getParams()) {
134 if (failed(verifyParamsOfType(tables, tyParams.getValue(), actualStructType, origin))) {
135 return failure();
136 }
137 }
138 // Otherwise, generate an error stating the parent struct type must be used.
139 return genCompareErr(expectedStruct, origin, aspect)
140 .attachNote(actualStruct.getLoc())
141 .append("should be type of this '", StructDefOp::getOperationName(), '\'');
142 }
143 } else {
144 return genCompareErr(expectedStruct, origin, aspect);
145 }
146 return success();
147}

193LogicalResult StructDefOp::verifySymbolUses(SymbolTableCollection &tables) {
194 if (ArrayAttr params = this->getConstParamsAttr()) {
195 // Ensure struct parameter names are unique
196 llvm::StringSet<> uniqNames;
197 for (Attribute attr : params) {
198 assert(llvm::isa<FlatSymbolRefAttr>(attr)); // per ODS
199 StringRef name = llvm::cast<FlatSymbolRefAttr>(attr).getValue();
200 if (!uniqNames.insert(name).second) {
201 return this->emitOpError().append("has more than one parameter named \"@", name, '"');
202 }
203 }
204 // Ensure they do not conflict with existing symbols
205 for (Attribute attr : params) {
206 auto res = lookupTopLevelSymbol(tables, llvm::cast<FlatSymbolRefAttr>(attr), *this, false);
207 if (succeeded(res)) {
208 return this->emitOpError()
209 .append("parameter name \"@")
210 .append(llvm::cast<FlatSymbolRefAttr>(attr).getValue())
211 .append("\" conflicts with an existing symbol")
212 .attachNote(res->get()->getLoc())
213 .append("symbol already defined here");
214 }
215 }
216 }
217 return success();
218}

244inline LogicalResult verifyStructComputeConstrain(
245 StructDefOp structDef, FuncDefOp computeFunc, FuncDefOp constrainFunc
246) {
247 // ASSERT: The `SetFuncAllowAttrs` trait on StructDefOp set the attributes correctly.
248 assert(constrainFunc.hasAllowConstraintAttr());
249 assert(!computeFunc.hasAllowConstraintAttr());
250 assert(!constrainFunc.hasAllowWitnessAttr());
251 assert(computeFunc.hasAllowWitnessAttr());
252
253 // Verify parameter types are valid. Skip the first parameter of the "constrain" function; it is
254 // already checked via verifyFuncTypeConstrain() in Function/IR/Ops.cpp.
255 ArrayRef<Type> computeParams = computeFunc.getFunctionType().getInputs();
256 ArrayRef<Type> constrainParams = constrainFunc.getFunctionType().getInputs().drop_front();
257 if (structDef.isMainComponent()) {
258 // Verify the input parameter types are legal. The error message is explicit about what types
259 // are allowed so there is no benefit to report multiple errors if more than one parameter in
260 // the referenced function has an illegal type.
261 for (Type t : computeParams) {
262 if (failed(checkMainFuncParamType(t, computeFunc, std::nullopt))) {
263 return failure(); // checkMainFuncParamType() already emits a sufficient error message
264 }
265 }
266 auto appendSelf = std::make_optional(structDef.getType());
267 for (Type t : constrainParams) {
268 if (failed(checkMainFuncParamType(t, constrainFunc, appendSelf))) {
269 return failure(); // checkMainFuncParamType() already emits a sufficient error message
270 }
271 }
272 }
273
274 if (!typeListsUnify(computeParams, constrainParams)) {
275 return constrainFunc.emitError()
276 .append(
277 "expected \"@", FUNC_NAME_CONSTRAIN,
278 "\" function argument types (sans the first one) to match \"@", FUNC_NAME_COMPUTE,
279 "\" function argument types"
280 )
281 .attachNote(computeFunc.getLoc())
282 .append("\"@", FUNC_NAME_COMPUTE, "\" function defined here");
283 }
284
285 return success();
286}

288inline LogicalResult verifyStructProduct(StructDefOp structDef, FuncDefOp productFunc) {
289 // ASSERT: The `SetFuncAllowAttrs` trait on StructDefOp set the attributes correctly
290 assert(productFunc.hasAllowConstraintAttr());
291 assert(productFunc.hasAllowWitnessAttr());
292
293 // Verify parameter types are valid
294 if (structDef.isMainComponent()) {
295 ArrayRef<Type> productParams = productFunc.getFunctionType().getInputs();
296 // Verify the input parameter types are legal. The error message is explicit about what types
297 // are allowed so there is no benefit to report multiple errors if more than one parameter in
298 // the referenced function has an illegal type.
299 for (Type t : productParams) {
300 if (failed(checkMainFuncParamType(t, productFunc, std::nullopt))) {
301 return failure(); // checkMainFuncParamType() already emits a sufficient error message
302 }
303 }
304 }
305
306 return success();
307}

311LogicalResult StructDefOp::verifyRegions() {
312 std::optional<FuncDefOp> foundCompute = std::nullopt;
313 std::optional<FuncDefOp> foundConstrain = std::nullopt;
314 std::optional<FuncDefOp> foundProduct = std::nullopt;
315 {
316 // Verify the following:
317 // 1. The only ops within the body are member and function definitions
318 // 2. The only functions defined in the struct are `@compute()` and `@constrain()`, or
319 // `@product()`
320 OwningEmitErrorFn emitError = getEmitOpErrFn(this);
321 Region &bodyRegion = getBodyRegion();
322 if (!bodyRegion.empty()) {
323 for (Operation &op : bodyRegion.front()) {
324 if (!llvm::isa<MemberDefOp>(op)) {
325 if (FuncDefOp funcDef = llvm::dyn_cast<FuncDefOp>(op)) {
326 if (funcDef.nameIsCompute()) {
327 if (foundCompute) {
328 return structFuncDefError(funcDef.getOperation())
329 << "found multiple \"@" << FUNC_NAME_COMPUTE << "\" functions";
330 }
331 foundCompute = std::make_optional(funcDef);
332 } else if (funcDef.nameIsConstrain()) {
333 if (foundConstrain) {
334 return structFuncDefError(funcDef.getOperation())
335 << "found multiple \"@" << FUNC_NAME_CONSTRAIN << "\" functions";
336 }
337 foundConstrain = std::make_optional(funcDef);
338 } else if (funcDef.nameIsProduct()) {
339 if (foundProduct) {
340 return structFuncDefError(funcDef.getOperation())
341 << "found multiple \"@" << FUNC_NAME_PRODUCT << "\" functions";
342 }
343 foundProduct = std::make_optional(funcDef);
344 } else {
345 // Must do a little more than a simple call to '?.emitOpError()' to
346 // tag the error with correct location and correct op name.
347 return structFuncDefError(funcDef.getOperation())
348 << "found \"@" << funcDef.getSymName() << '"';
349 }
350 } else {
351 return op.emitOpError()
352 << "invalid operation in '" << StructDefOp::getOperationName() << "'; only '"
353 << MemberDefOp::getOperationName() << '\'' << " and '"
354 << FuncDefOp::getOperationName() << "' operations are permitted";
355 }
356 }
357 }
358 }
359
360 if (!foundCompute.has_value() && foundConstrain.has_value()) {
361 return structFuncDefError(getOperation()) << "found \"@" << FUNC_NAME_CONSTRAIN
362 << "\", missing \"@" << FUNC_NAME_COMPUTE << "\"";
363 }
364 if (!foundConstrain.has_value() && foundCompute.has_value()) {
365 return structFuncDefError(getOperation()) << "found \"@" << FUNC_NAME_COMPUTE
366 << "\", missing \"@" << FUNC_NAME_CONSTRAIN << "\"";
367 }
368 }
369
370 if (!foundCompute.has_value() && !foundConstrain.has_value() && !foundProduct.has_value()) {
371 return structFuncDefError(getOperation())
372 << "could not find \"@" << FUNC_NAME_PRODUCT << "\", \"@" << FUNC_NAME_COMPUTE
373 << "\", or \"@" << FUNC_NAME_CONSTRAIN << "\"";
374 }
375
376 // Check which funcs are present and not marked with {llzk.derived}
377 auto nonderived = [](std::optional<FuncDefOp> op) -> bool {
378 return op && !(*op)->hasAttr(DERIVED_ATTR_NAME);
379 };
380
381 auto attachDerivedNotes = [&foundCompute, &foundConstrain,
382 &foundProduct](InFlightDiagnostic &&error) {
383 if (foundProduct && (*foundProduct)->hasAttr(DERIVED_ATTR_NAME)) {
384 error.attachNote(foundProduct->getLoc()) << "derived \"@" << FUNC_NAME_PRODUCT << "\" here";
385 }
386 if (foundCompute && (*foundCompute)->hasAttr(DERIVED_ATTR_NAME)) {
387 error.attachNote(foundCompute->getLoc()) << "derived \"@" << FUNC_NAME_COMPUTE << "\" here";
388 }
389 if (foundConstrain && (*foundConstrain)->hasAttr(DERIVED_ATTR_NAME)) {
390 error.attachNote(foundConstrain->getLoc())
391 << "derived \"@" << FUNC_NAME_CONSTRAIN << "\" here";
392 }
393 return error;
394 };
395
396 // We know that (@compute+@constrain) is present or @product is present, or both
397
398 // Error cases:
399 // Everything is derived
400 if (!nonderived(foundCompute) && !nonderived(foundConstrain) && !nonderived(foundProduct)) {
401 return attachDerivedNotes(
402 structFuncDefError(getOperation())
403 << "could not find non-derived \"@" << FUNC_NAME_PRODUCT << "\", \"@" << FUNC_NAME_COMPUTE
404 << "\", or \"@" << FUNC_NAME_CONSTRAIN << "\""
405 );
406 }
407
408 // Only one of @compute/@constrain is non-derived
409 if (nonderived(foundCompute) ^ nonderived(foundConstrain)) {
410 return attachDerivedNotes(
411 structFuncDefError(getOperation())
412 << "\"@" << FUNC_NAME_COMPUTE << "\" and \"@" << FUNC_NAME_CONSTRAIN
413 << "\" must both be either derived or non-derived"
414 );
415 }
416
417 // Here, at least one thing is non-derived, and @compute/@constrain are derived or non-derived
418 // together so everything is fine
419 if (nonderived(foundCompute) && nonderived(foundConstrain) && !nonderived(foundProduct)) {
420 return verifyStructComputeConstrain(*this, *foundCompute, *foundConstrain);
421 }
422
423 assert(!nonderived(foundCompute) && !nonderived(foundConstrain) && nonderived(foundProduct));
424 return verifyStructProduct(*this, *foundProduct);
425}

524static LogicalResult
525verifyMemberDefTypeImpl(Type memberType, SymbolTableCollection &tables, Operation *origin) {
526 if (StructType memberStructType = llvm::dyn_cast<StructType>(memberType)) {
527 // Special case for StructType verifies that the member type can resolve and that it is NOT the
528 // parent struct (i.e., struct members cannot create circular references).
529 auto memberTypeRes = verifyStructTypeResolution(tables, memberStructType, origin);
530 if (failed(memberTypeRes)) {
531 return failure(); // above already emits a sufficient error message
532 }
533 StructDefOp parentRes = getParentOfType<StructDefOp>(origin);
534 assert(parentRes && "MemberDefOp parent is always StructDefOp"); // per ODS def
535 if (memberTypeRes.value() == parentRes) {
536 return origin->emitOpError()
537 .append("type is circular")
538 .attachNote(parentRes.getLoc())
539 .append("references parent component defined here");
540 }
541 return success();
542 } else {
543 return verifyTypeResolution(tables, origin, memberType);
544 }
545}

547LogicalResult MemberDefOp::verifySymbolUses(SymbolTableCollection &tables) {
548 Type memberType = this->getType();
549 if (failed(verifyMemberDefTypeImpl(memberType, tables, *this))) {
550 return failure();
551 }
552
553 if (!getColumn()) {
554 return success();
555 }
556 // If the member is marked as a column only a small subset of types are allowed.
557 if (!isValidColumnType(getType(), tables, *this)) {
558 return emitOpError() << "marked as column can only contain felts, arrays of column types, or "
559 "structs with columns, but has type "
560 << getType();
561 }
562 return success();
563}

570FailureOr<SymbolLookupResult<MemberDefOp>>
571getMemberDefOpImpl(MemberRefOpInterface refOp, SymbolTableCollection &tables, StructType tyStruct) {
572 Operation *op = refOp.getOperation();
573 auto structDefRes = tyStruct.getDefinition(tables, op);
574 if (failed(structDefRes)) {
575 return failure(); // getDefinition() already emits a sufficient error message
576 }
577 // Copy namespace because we will need it later.
578 llvm::SmallVector<llvm::StringRef> structDefOpNs(structDefRes->getNamespace());
579 auto res = llzk::lookupSymbolIn<MemberDefOp>(
580 tables, SymbolRefAttr::get(refOp->getContext(), refOp.getMemberName()),
581 std::move(*structDefRes), op
582 );
583 if (failed(res)) {
584 return refOp->emitError() << "could not find '" << MemberDefOp::getOperationName()
585 << "' named \"@" << refOp.getMemberName() << "\" in \""
586 << tyStruct.getNameRef() << '"';
587 }
588 // Prepend the namespace of the struct lookup since the type of the member is meant to be resolved
589 // within that scope.
590 res->prependNamespace(structDefOpNs);
591 return std::move(res.value());
592}

594static FailureOr<SymbolLookupResult<MemberDefOp>>
595findMember(MemberRefOpInterface refOp, SymbolTableCollection &tables) {
596 // Ensure the base component/struct type reference can be resolved.
597 StructType tyStruct = refOp.getStructType();
598 if (failed(tyStruct.verifySymbolRef(tables, refOp.getOperation()))) {
599 return failure();
600 }
601 // Ensure the member name can be resolved in that struct.
602 return getMemberDefOpImpl(refOp, tables, tyStruct);
603}
604
605static LogicalResult verifySymbolUsesImpl(
606 MemberRefOpInterface refOp, SymbolTableCollection &tables,
607 SymbolLookupResult<MemberDefOp> &member
608) {
609 // Ensure the type of the referenced member declaration matches the type used in this op.
610 Type actualType = refOp.getVal().getType();
611 Type memberType = member.get().getType();
612 if (!typesUnify(actualType, memberType, member.getNamespace())) {
613 return refOp->emitOpError() << "has wrong type; expected " << memberType << ", got "
614 << actualType;
615 }
616 // Ensure any SymbolRef used in the type are valid
617 return verifyTypeResolution(tables, refOp.getOperation(), actualType);
618}

620LogicalResult verifySymbolUsesImpl(MemberRefOpInterface refOp, SymbolTableCollection &tables) {
621 // Ensure the member name can be resolved in that struct.
622 auto member = findMember(refOp, tables);
623 if (failed(member)) {
624 return member; // getMemberDefOp() already emits a sufficient error message
625 }
626 return verifySymbolUsesImpl(refOp, tables, *member);
627}

636LogicalResult MemberReadOp::verifySymbolUses(SymbolTableCollection &tables) {
637 auto member = findMember(*this, tables);
638 if (failed(member)) {
639 return failure();
640 }
641 if (failed(verifySymbolUsesImpl(*this, tables, *member))) {
642 return failure();
643 }
644 // If the member is not a column and an offset was specified then fail to validate
645 if (!member->get().getColumn() && getTableOffset().has_value()) {
646 return emitOpError("cannot read with table offset from a member that is not a column")
647 .attachNote(member->get().getLoc())
648 .append("member defined here");
649 }
650 // If the member is private and this read is outside the struct, then fail to validate.
651 // The current op may be inside a struct or a free function, but the
652 // member op (the member definition) is always inside a struct.
653 FailureOr<StructDefOp> memberParentRes = verifyInStruct(member->get());
654 if (failed(memberParentRes)) {
655 return failure(); // verifyInStruct() already emits a sufficient error message
656 }
657 StructDefOp thisParent = getParentOfType<StructDefOp>(*this);
658 StructDefOp memberParentStruct = memberParentRes.value();
659 if (!member->get().hasPublicAttr() && (!thisParent || thisParent != memberParentStruct)) {
660 return emitOpError()
661 .append(
662 "cannot read from private member of struct \"", memberParentStruct.getHeaderString(),
663 "\""
664 )
665 .attachNote(member->get().getLoc())
666 .append("member defined here");
667 }
668 return success();
669}
670
671LogicalResult MemberWriteOp::verifySymbolUses(SymbolTableCollection &tables) {
672 // Ensure the write op only targets members in the current struct.
673 FailureOr<StructDefOp> getParentRes = verifyInStruct(*this);
674 if (failed(getParentRes)) {
675 return failure(); // verifyInStruct() already emits a sufficient error message
676 }
677 if (failed(checkSelfType(tables, *getParentRes, getComponent().getType(), *this, "base value"))) {
678 return failure(); // checkSelfType() already emits a sufficient error message
679 }
680 // Perform the standard member ref checks.
681 return verifySymbolUsesImpl(*this, tables);
682}
