44struct ForbiddenPreconditionInfluenceInfo {

46 llvm::SmallSetVector<mlir::Location, 2> structMemberLocs = {};

96inline ForbiddenPreconditionInfluenceInfo mergeInfluenceInfo(
97 ForbiddenPreconditionInfluenceInfo lhs, const ForbiddenPreconditionInfluenceInfo &rhs
