21static DynamicAPInt po2(const DynamicAPInt &e) {
22 // Ensure parameter is not negative and that it can be safely cast to unsigned.
23 assert(e >= 0);
24 assert(e <= std::numeric_limits<unsigned>::max() /* upcast from unsigned -> int64_t */);
25 unsigned shiftAmt = llzk::toAPSInt(e).getZExtValue();
26 APSInt p = APSInt::get(1) << shiftAmt;
27 return llzk::toDynamicAPInt(p);
28}

30static DynamicAPInt fromBigEndian(const std::vector<bool> &bits) {
31 APSInt rawInt(bits.size(), /* isUnsigned */ false);
32 for (unsigned i = 0; i < bits.size(); ++i) {
33 rawInt.setBitVal(i, bits[i]);
34 }
35 return llzk::toDynamicAPInt(rawInt);
36}

38static DynamicAPInt
39binaryBitOp(const DynamicAPInt &lhs, const DynamicAPInt &rhs, function_ref<bool(bool, bool)> fn) {
40 DynamicAPInt a = lhs, b = rhs;
41 std::vector<bool> bits;
42 while (a != 0 || b != 0) {
43 // bits are sign extended
44 bool abit = a != 0 ? bool(int64_t(a % 2)) : lhs < 0;
45 bool bbit = b != 0 ? bool(int64_t(b % 2)) : rhs < 0;
46 bits.push_back(fn(abit, bbit));
47 a /= 2;
48 b /= 2;
49 }
50 // Insert final sign bit, as the above will ignore 0 sign bits. This is also
51 // acceptable when both numbers are signed, as it acts as a sign extension.
52 bits.push_back(fn(lhs < 0, rhs < 0));
53 return fromBigEndian(bits);
54}

75DynamicAPInt operator>>(const DynamicAPInt &lhs, const DynamicAPInt &rhs) {
76 if (lhs >= 0) {
77 return lhs / po2(rhs);
78 } else {
79 // round towards negative infinity
80 DynamicAPInt divisor = po2(rhs);
81 if (lhs % divisor == 0) {
82 return lhs / divisor;
83 } else {
84 return (lhs - (divisor - 1)) / divisor;
85 }
86 }
87}

94DynamicAPInt toDynamicAPInt(const APSInt &i) {
95 if (i.getBitWidth() <= 64) {
96 // Fast path for smaller values, just use the int64_t conversion
97 return DynamicAPInt(i.isNegative() ? i.getSExtValue() : static_cast<int64_t>(i.getZExtValue()));
98 }
99
100 DynamicAPInt res(0), po2(1);
101 // Since LLVM 20 doesn't have a direct APInt to DynamicAPInt constructor, we
102 // manually construct the DynamicAPInt from bits of the input.
103 // We use the positive representation so our negation works at the end.
104 APSInt raw = i < 0 ? -i : i;
105 for (unsigned b = 0; b < raw.getActiveBits(); b++) {
106 DynamicAPInt bitSet(raw[b]);
107 res += (bitSet * po2);
108 po2 *= 2;
109 }
110 if (i.isNegative() && res > 0) {
111 res = -res;
112 }
113 return res;
114}

116APSInt toAPSInt(const DynamicAPInt &i) {
117 if (numeric_limits<int64_t>::min() <= i && i <= numeric_limits<int64_t>::max()) {
118 // Fast path for smaller values, just use the int64_t conversion
119 return APSInt::get(int64_t(i));
120 }
121
122 // Else, convert to string and parse back as an APSInt.
123 // This may not be the most efficient implementation, but it is the cleanest
124 // due to the lack of direct conversions between DynamicAPInt and APInts.
125 std::string repr;
126 llvm::raw_string_ostream ss(repr);
127 ss << i;
128
129 APSInt res(repr);
130 // For consistency, we add a bit and mark these as signed integers, since
131 // DynamicAPInts are inherently signed.
132 res = res.extend(res.getBitWidth() + 1);
133 res.setIsSigned(true);
134
135 return res;
136}

138DynamicAPInt modExp(const DynamicAPInt &base, const DynamicAPInt &exp, const DynamicAPInt &mod) {
139 DynamicAPInt result(1);
140 DynamicAPInt b = base;
141 DynamicAPInt e = exp;
142 DynamicAPInt one(1);
143
144 while (e != 0) {
145 if (e % 2 != 0) {
146 result = (result * b) % mod;
147 }
148
149 b = (b * b) % mod;
150 e = e >> one;
151 }
152 assert((base * result) % mod == 1 && "inverse is incorrect");
153 return result;
154}

156llvm::DynamicAPInt modInversePrime(const DynamicAPInt &f, const DynamicAPInt &p) {
157 assert(f != 0 && "0 has no inverse");
158 // Fermat: f^(p-2) mod p
159 DynamicAPInt exp = p - 2;
160 return modExp(f, exp, p);
161}
