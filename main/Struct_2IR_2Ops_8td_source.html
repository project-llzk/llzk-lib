41def LLZK_StructDefOp
42 : StructDialectOp<
43 "def", [HasParent<"::mlir::ModuleOp">, Symbol, SymbolTable,
44 IsolatedFromAbove, SetFuncAllowAttrs,
45 DeclareOpInterfaceMethods<SymbolUserOpInterface>,
46 NoRegionArguments]#GraphRegionNoTerminator.traits> {
47 let summary = "circuit component definition";
48 let description = [{
49 This operation describes a component in a circuit. It can contain any number
50 of members that hold inputs, outputs, intermediate values, and subcomponents
51 of the defined component. It also contains a `compute()` function that holds
52 the witness generation code for the component and a `constrain()` function
53 that holds that constraint generation code for the component.
54
55 Example:
56
57 ```llzk
58 struct.def @ComponentA {
59 member @f1 : !array.type<5 x index>
60 member @f2 : !felt.type {llzk.pub}
61
62 function.def @compute(%p: !felt.type) -> !struct.type<@ComponentA> {
63 %self = struct.new : !struct.type<@ComponentA>
64 // initialize all members of `%self` here
65 return %self : !struct.type<@ComponentA>
66 }
67
68 function.def @constrain(%self: !struct.type<@ComponentA>, %p: !felt.type) {
69 // emit constraints here
70 return
71 }
72 }
73 ```
74
75 The optional `llzk.main = !struct.type<...>` attribute on the top-level module in an
76 LLZK IR program expresses the main entry point of the circuit as a concrete instantiation
77 of a specific struct. For example, `llzk.main = !struct.type<@Top<[52,12]>>` or
78 `llzk.main = !struct.type<@Main<[i1, !felt.type, 256]>>`.
79 This component has additional restrictions:
80 1. The parameter types of its functions (besides the required "self" parameter) can
81 only be `!felt.type` or `!array.type<.. x !felt.type>`.
82 2. The members marked as public (i.e., with the `{llzk.pub}` attribute) represent
83 the public outputs of the overall circuit.
84
85 Example of a `Main` component:
86
87 ```llzk
88 struct.def @Main {
89 member @out : !felt.type {llzk.pub} // public output
90 member @intermediate : !felt.type
91 function.def @compute(%p: !felt.type) -> !struct.type<@Main> {
92 // ...
93 }
94 function.def @constrain(%self: !struct.type<@Main>, %p: !felt.type) {
95 // ...
96 }
97 }
98 ```
99 }];
100
101 // Note: `$const_params` contains symbol definitions that do not use the
102 // standard SymbolTable mechanism. Instead hasParamNamed() can be used to
103 // check if a certain FlatSymbolRefAttr is a parameter in the function.
104 let arguments = (ins SymbolNameAttr:$sym_name,
105 OptionalAttr<FlatSymbolRefArrayAttr>:$const_params);
106
107 let regions = (region SizedRegion<1>:$bodyRegion);
108
109 let assemblyFormat = [{
110 $sym_name (`<` $const_params^ `>`)? $bodyRegion attr-dict
111 }];
112
113 let extraClassDeclaration = [{
114 /// Gets the StructType representing this struct. If the `constParams` to use in
115 /// the type are not given, the StructType will use `this->getConstParamsAttr()`.
116 StructType getType(::std::optional<::mlir::ArrayAttr> constParams = {});
117 inline StructType getType(::std::optional<::mlir::ArrayAttr> constParams = {}) const {
118 return const_cast<StructDefOp*>(this)->getType(constParams);
119 }
120
121 /// Gets the MemberDefOp that defines the member in this
122 /// structure with the given name, if present.
123 MemberDefOp getMemberDef(::mlir::StringAttr memberName);
124
125 /// Get all MemberDefOp in this structure.
126 ::std::vector<MemberDefOp> getMemberDefs();
127
128 /// Returns whether the struct defines members marked as columns.
129 ::mlir::LogicalResult hasColumns() {
130 return ::mlir::success(::llvm::any_of(getMemberDefs(), [](MemberDefOp memberOp) {
131 return memberOp.getColumn();
132 }));
133 }
134
135 /// Returns whether the struct defines members marked as signals.
136 ::mlir::LogicalResult hasSignals() {
137 return ::mlir::success(::llvm::any_of(getMemberDefs(), [](MemberDefOp memberOp) {
138 return memberOp.getSignal();
139 }));
140 }
141
142 /// Gets the FuncDefOp that defines the compute function in this structure, if present, or `nullptr` otherwise.
143 ::llzk::function::FuncDefOp getComputeFuncOp();
144
145 /// Gets the FuncDefOp that defines the constrain function in this structure, if present, or `nullptr` otherwise.
146 ::llzk::function::FuncDefOp getConstrainFuncOp();
147
148 /// Gets the FuncDefOp that defines the product function in this structure, if present, or `nullptr` otherwise
149 ::llzk::function::FuncDefOp getProductFuncOp();
150
151 /// Returns `true` iff this structure defines compute and constrain functions.
152 bool hasComputeConstrain() { return lookupSymbol(FUNC_NAME_COMPUTE) != nullptr && lookupSymbol(FUNC_NAME_CONSTRAIN) != nullptr; }
153
154 /// Generate header string, in the same format as the assemblyFormat
155 ::std::string getHeaderString();
156
157 /// Return `false` iff `getConstParamsAttr()` returns `nullptr`
158 bool hasConstParamsAttr() { return getProperties().const_params != nullptr; };
159
160 /// Return `true` iff this StructDefOp has a parameter with the given name
161 bool hasParamNamed(::mlir::StringRef find);
162
163 /// Return `true` iff this StructDefOp has a parameter with the given name
164 inline bool hasParamNamed(::mlir::StringAttr find) {
165 return hasParamNamed(find.strref());
166 }
167
168 /// Return `true` iff this StructDefOp has a parameter with the given name
169 inline bool hasParamNamed(::mlir::FlatSymbolRefAttr find) {
170 return hasParamNamed(find.getRootReference());
171 }
172
173 /// Return the full name for this struct from the root module, including
174 /// any surrounding module scopes.
175 ::mlir::SymbolRefAttr getFullyQualifiedName();
176
177 /// Return `true` iff this StructDefOp is the main struct. See `llzk::MAIN_ATTR_NAME`.
178 bool isMainComponent();
179 }];
180
181 let hasRegionVerifier = 1;
182}

184def LLZK_MemberDefOp
185 : StructDialectOp<
186 "member", [HasParent<"::llzk::component::StructDefOp">,
187 DeclareOpInterfaceMethods<SymbolUserOpInterface>,
188 Symbol]> {
189 let summary = "struct member definition";
190 let description = [{
191 This operation describes a member in a struct/component.
192
193 Example:
194
195 ```llzk
196 struct.member @f1 : !felt.type
197 struct.member @f2 : !felt.type {llzk.pub}
198 struct.member @col1 : !felt.type {column}
199 struct.member @sig1 : !felt.type {signal}
200 struct.member @colsig1 : !felt.type {column, signal}
201 struct.member @pubcol : !felt.type {llzk.pub, column}
202 ```
203
204 - Members marked with the `{llzk.pub}` attribute are considered public
205 and represent outputs of their defining struct/component.
206 - Members marked with the `{column}` attribute can be read/written with
207 table offsets using the `readm` and `writem` operations.
208 - Members marked with the `{signal}` attribute are constraint variables that
209 are stored in the witness; non-signal values are intermediate expressions.
210 }];
211
212 let arguments = (ins SymbolNameAttr:$sym_name, TypeAttrOf<AnyLLZKType>:$type,
213 UnitAttr:$column, UnitAttr:$signal);
214
215 // Define builders manually to avoid the default ones that have extra
216 // TypeRange parameters that must always be empty.
217 let skipDefaultBuilders = 1;
218 let builders =
219 [OpBuilder<(ins "::mlir::StringAttr":$sym_name, "::mlir::TypeAttr":$type,
220 CArg<"bool", "false">:$isSignal, CArg<"bool", "false">:$isColumn)>,
221 OpBuilder<(ins "::llvm::StringRef":$sym_name, "::mlir::Type":$type,
222 CArg<"bool", "false">:$isSignal, CArg<"bool", "false">:$isColumn)>,
223 OpBuilder<(ins "::mlir::TypeRange":$resultTypes,
224 "::mlir::ValueRange":$operands,
225 "::llvm::ArrayRef<::mlir::NamedAttribute>":$attributes,
226 CArg<"bool", "false">:$isSignal, CArg<"bool", "false">:$isColumn)>,
227 // Simpler version since 'resultTypes' and 'operands' must be empty
228 OpBuilder<
229 (ins "::llvm::ArrayRef<::mlir::NamedAttribute>":$attributes,
230 CArg<"bool", "false">:$isSignal,
231 CArg<"bool", "false">:$isColumn),
232 [{ build($_builder, $_state, {}, {}, attributes, isSignal, isColumn); }]>];
233
234 let assemblyFormat = [{ $sym_name `:` $type attr-dict }];
235
236 let extraClassDeclaration = [{
237 inline bool hasPublicAttr() { return getOperation()->hasAttr(llzk::PublicAttr::name); }
238 void setPublicAttr(bool newValue = true);
239 }];
240}

261def LLZK_MemberReadOp
262 : MemberRefOpBase<"readm", [VerifySizesForMultiAffineOps<1>]> {
263 let summary = "read value of a struct member";
264 let description = [{
265 This operation reads the value of a named member in a struct/component.
266
267 A struct can read its own members regardless of whether they are marked as
268 public (i.e., with the `llzk.pub` attribute) or private (members without the
269 `llzk.pub` attribute). However, when reading members of other components, only
270 public members can be accessed. Free functions may also only read public members.
271
272 The value can be read from the signals table, in which case it can be
273 offset by a constant value. A negative value represents reading a value
274 backwards and a positive value represents reading a value forward.
275 Only members marked as columns can be read in this manner.
276 }];
277 let isRead = 1;
278
279 // See `VerifySizesForMultiAffineOps` for more explanation of these arguments.
280 let arguments = (ins LLZK_StructType:$component,
281 FlatSymbolRefAttr:$member_name,
282 OptionalAttr<AnyAttrOf<[SymbolRefAttr, IndexAttr,
283 AffineMapAttr]>>:$tableOffset,
284 // List of AffineMap operand groups where each group provides the
285 // arguments to instantiate the next (left-to-right) AffineMap used in
286 // `tableOffset`.
287 VariadicOfVariadic<Index, "mapOpGroupSizes">:$mapOperands,
288 // Within each group in '$mapOperands', denotes the number of values that
289 // are AffineMap "dimensional" arguments with the remaining values being
290 // AffineMap "symbolic" arguments.
291 DefaultValuedAttr<DenseI32ArrayAttr, "{}">:$numDimsPerMap,
292 // Denotes the size of each variadic group in '$mapOperands'.
293 DenseI32ArrayAttr:$mapOpGroupSizes);
294 let results = (outs AnyLLZKType:$val);
295
296 // Define builders manually so inference of operand layout attributes is not
297 // circumvented.
298 let skipDefaultBuilders = 1;
299 let builders =
300 [OpBuilder<(ins "::mlir::Type":$resultType, "::mlir::Value":$component,
301 "::mlir::StringAttr":$member)>,
302 OpBuilder<(ins "::mlir::Type":$resultType, "::mlir::Value":$component,
303 "::mlir::StringAttr":$member, "::mlir::Attribute":$dist,
304 "::mlir::ValueRange":$mapOperands,
305 "std::optional<int32_t>":$numDims)>,
306 OpBuilder<(ins "::mlir::Type":$resultType, "::mlir::Value":$component,
307 "::mlir::StringAttr":$member,
308 "::mlir::SymbolRefAttr":$dist),
309 [{
310 build($_builder, $_state, resultType, component, member, dist, ::mlir::ValueRange(), std::nullopt);
311 }]>,
312 OpBuilder<(ins "::mlir::Type":$resultType, "::mlir::Value":$component,
313 "::mlir::StringAttr":$member, "::mlir::IntegerAttr":$dist),
314 [{
315 build($_builder, $_state, resultType, component, member, dist, ::mlir::ValueRange(), std::nullopt);
316 }]>,
317 OpBuilder<(ins "::mlir::TypeRange":$resultTypes,
318 "::mlir::ValueRange":$operands,
319 "::mlir::ArrayRef<::mlir::NamedAttribute>":$attrs)>];
320
321 let assemblyFormat = [{
322 $component `[` $member_name `]`
323 ( `{` custom<MultiDimAndSymbolList>($mapOperands, $numDimsPerMap)^ `}` )?
324 `:` type($component) `,` type($val)
325 attr-dict
326 }];
327
328 let hasVerifier = 1;
329}
