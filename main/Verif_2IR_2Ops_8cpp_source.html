53// Check if the op is a valid contract target.
54bool isValidTarget(Operation *op) {
55 if (auto fnOp = dyn_cast<FuncDefOp>(op)) {
56 // Cannot target struct functions directly
57 return fnOp->getParentOfType<StructDefOp>() == nullptr;
58 }
59 // Only other supported target currently is a struct
60 return isa<StructDefOp>(op);
61}

63struct TargetTypeInfo {
64 FunctionType funcType {};
65 ArrayAttr argAttrs {};
66};
67
68FailureOr<TargetTypeInfo> getTargetTypeInfo(Operation *op) {
69 if (auto fnOp = dyn_cast<FuncDefOp>(op)) {
70 // Recreate the function type with return types appended to the arguments.
71 FunctionType fnTy = fnOp.getFunctionType();
72 ArrayRef<Type> curInputs = fnTy.getInputs(), curResults = fnTy.getResults();
73 SmallVector<Type> newInputs;
74 newInputs.reserve(curInputs.size() + curResults.size());
75 newInputs.insert(newInputs.end(), curInputs.begin(), curInputs.end());
76 newInputs.insert(newInputs.end(), curResults.begin(), curResults.end());
77 // And no return types
78 auto newFnTy = fnTy.clone(newInputs, {});
79 // Do the same appending to the return attrs
80 ArrayAttr curArgAttrs = fnOp.getArgAttrsAttr(), curResAttrs = fnOp.getResAttrsAttr();
81 ArrayAttr newArgAttrsAttr {};
82 if (curArgAttrs || curResAttrs) {
83 auto *ctx = op->getContext();
84 SmallVector<Attribute> newArgAttrs;
85 // Since there are some attributes, it must match the length of the input arguments
86 newArgAttrs.reserve(newInputs.size());
87 if (curArgAttrs) {
88 newArgAttrs.insert(newArgAttrs.end(), curArgAttrs.begin(), curArgAttrs.end());
89 } else {
90 // Pad
91 newArgAttrs.insert(newArgAttrs.end(), curInputs.size(), DictionaryAttr::get(ctx));
92 }
93 if (curResAttrs) {
94 newArgAttrs.insert(newArgAttrs.end(), curResAttrs.begin(), curResAttrs.end());
95 } else {
96 // pad
97 newArgAttrs.insert(newArgAttrs.end(), curResults.size(), DictionaryAttr::get(ctx));
98 }
99 newArgAttrsAttr = ArrayAttr::get(ctx, newArgAttrs);
100 }
101
102 return TargetTypeInfo {
103 .funcType = newFnTy,
104 .argAttrs = newArgAttrsAttr,
105 };
106 }
107 if (auto structOp = dyn_cast<StructDefOp>(op)) {
108 if (structOp.hasComputeConstrain()) {
109 auto fnOp = structOp.getConstrainFuncOp();
110 return TargetTypeInfo {
111 .funcType = fnOp.getFunctionType(),
112 .argAttrs = fnOp.getArgAttrsAttr(),
113 };
114 } else {
115 FuncDefOp productFn = structOp.getProductFuncOp();
116 assert(productFn);
117 // Augment the product function signature to accept the self argument.
118 FunctionType fnTy = productFn.getFunctionType();
119 ArrayRef<Type> curInputs = fnTy.getInputs();
120 // Accept the self struct type in addition to existing inputs
121 SmallVector<Type> newInputs;
122 newInputs.reserve(curInputs.size() + 1);
123 newInputs.push_back(structOp.getType());
124 newInputs.insert(newInputs.end(), curInputs.begin(), curInputs.end());
125 // And no return types
126 auto newFnTy = fnTy.clone(newInputs, {});
127 // We also need to expand the arg attributes by one
128 auto *ctx = op->getContext();
129 ArrayAttr curArgAttrs = productFn.getArgAttrsAttr();
130 ArrayAttr newArgAttrsAttr = curArgAttrs;
131 if (curArgAttrs) {
132 SmallVector<Attribute> newArgAttrs;
133 newArgAttrs.reserve(curArgAttrs.size() + 1);
134 newArgAttrs.push_back(DictionaryAttr::get(ctx));
135 newArgAttrs.insert(newArgAttrs.end(), curArgAttrs.begin(), curArgAttrs.end());
136 newArgAttrsAttr = ArrayAttr::get(ctx, newArgAttrs);
137 }
138 return TargetTypeInfo {
139 .funcType = newFnTy,
140 .argAttrs = newArgAttrsAttr,
141 };
142 }
143 }
144
145 return failure();
146}

156void ContractOp::build(
157 ::mlir::OpBuilder &odsBuilder, ::mlir::OperationState &odsState, ::llvm::StringRef name,
158 llvm::StringRef target
159) {
160 // Any errors here in the construction from the target information are not
161 // reported here, but will instead be reported when the verify function fails
162 // to verify this op.
163 SymbolTableCollection tables;
164 MLIRContext *ctx = odsBuilder.getContext();
165 SymbolRefAttr targetAttr = SymbolRefAttr::get(ctx, target);
166 // Find the target of the contract
167 FailureOr<SymbolLookupResultUntyped> targetRes =
168 lookupTopLevelSymbol(tables, targetAttr, odsBuilder.getBlock()->getParentOp());
169 if (failed(targetRes)) {
170 return;
171 }
172 Operation *targetOp = targetRes->get();
173 if (!isValidTarget(targetOp)) {
174 return;
175 }
176 FailureOr<TargetTypeInfo> infoRes = getTargetTypeInfo(targetOp);
177 if (failed(infoRes)) {
178 return;
179 }
180 TargetTypeInfo &info = *infoRes;
181 build(
182 odsBuilder, odsState, name, SymbolRefAttr::get(odsBuilder.getContext(), target),
183 info.funcType, info.argAttrs
184 );
185}

225LogicalResult ContractOp::verifySymbolUses(SymbolTableCollection &tables) {
226 // Verify the target of the contract
227 FailureOr<ModuleOp> rootRes = getRootModule(getOperation());
228 if (failed(rootRes)) {
229 return emitOpError().append("could not lookup root module");
230 }
231 FailureOr<SymbolLookupResultUntyped> targetRes =
232 lookupTopLevelSymbol(tables, getTargetAttr(), rootRes->getOperation());
233 if (failed(targetRes)) {
234 return emitOpError().append("could not find target \"@", getTarget(), "\"");
235 }
236
237 FunctionType contractTy = getFunctionType();
238 // Verify the symbols in the contract argument
239 if (failed(verifyTypeResolution(tables, *this, contractTy))) {
240 // verifyTypeResolution already reports error messages
241 return failure();
242 }
243
244 // Verify the target symbol
245 Operation *targetOp = targetRes->get();
246 if (!isValidTarget(targetOp)) {
247 return emitOpError()
248 .append("target \"", getTargetAttr(), "\" is not a supported contract target")
249 .attachNote(targetOp->getLoc())
250 .append("target defined here");
251 }
252 FailureOr<TargetTypeInfo> targetInfoRes = getTargetTypeInfo(targetOp);
253 if (failed(targetInfoRes)) {
254 return emitOpError()
255 .append("unsupported target type \"", targetOp->getName(), "\"")
256 .attachNote(targetOp->getLoc())
257 .append("target defined here");
258 }
259 TargetTypeInfo &targetInfo = *targetInfoRes;
260 if (targetInfo.funcType != contractTy) {
261 return emitOpError()
262 .append("contract type does not match target type")
263 .attachNote(targetOp->getLoc())
264 .append("target defined here");
265 }
266 if (targetInfo.argAttrs != getArgAttrsAttr()) {
267 return emitOpError()
268 .append(
269 "contract arg attributes ", getArgAttrsAttr(), " does not match target arg attributes ",
270 targetInfo.argAttrs
271 )
272 .attachNote(targetOp->getLoc())
273 .append("target defined here");
274 }
275
276 // Also confirm that the types are in the same template op if they are.
277 TemplateOp contractTmpl = getParentOfType<TemplateOp>(*this);
278 TemplateOp targetTmpl = targetOp->getParentOfType<TemplateOp>();
279 if (contractTmpl != targetTmpl) {
280 if (targetTmpl) {
281 return emitOpError()
282 .append("contract must reside within the template containing the target")
283 .attachNote(targetTmpl.getLoc())
284 .append("target template defined here");
285 } else if (contractTmpl) {
286 return emitOpError()
287 .append("contract cannot be within a template that does not contain the target")
288 .attachNote(contractTmpl.getLoc())
289 .append("contract template defined here");
290 }
291 }
292
293 return success();
294}

296// Parse the ContractOp syntax using the built-in parsing of function-like
297// operations. We'll verify contract-specific restrictions in `verify`.
298ParseResult ContractOp::parse(OpAsmParser &parser, OperationState &result) {
299 StringAttr typeAttrName = getFunctionTypeAttrName(result.name);
300 StringAttr argAttrsName = getArgAttrsAttrName(result.name);
301
302 SmallVector<OpAsmParser::Argument> entryArgs;
303 SmallVector<DictionaryAttr> resultAttrs;
304 SmallVector<Type> resultTypes;
305 auto &builder = parser.getBuilder();
306
307 // Parse the name as a symbol.
308 StringAttr nameAttr;
309 if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(), result.attributes)) {
310 return failure();
311 }
312
313 // Parse the target symbol
314 if (parser.parseKeyword("for")) {
315 return failure();
316 }
317
318 SymbolRefAttr targetAttr;
319 if (parser.parseCustomAttributeWithFallback(
320 targetAttr, parser.getBuilder().getType<::mlir::NoneType>()
321 )) {
322 return failure();
323 }
324 if (!targetAttr) {
325 return failure();
326 }
327 result.getOrAddProperties<ContractOp::Properties>().target = targetAttr;
328
329 // Parse the function signature.
330 SMLoc signatureLocation = parser.getCurrentLocation();
331 bool isVariadic = false;
332
333 if (function_interface_impl::parseFunctionSignature(
334 parser, /*allowVariadic*/ false, entryArgs, isVariadic, resultTypes, resultAttrs
335 )) {
336 return failure();
337 }
338 assert(isVariadic == false);
339 // There should be no return types or attributes.
340 if (!resultTypes.empty() || !resultAttrs.empty()) {
341 return failure();
342 }
343
344 std::string errorMessage;
345 SmallVector<Type> argTypes;
346 argTypes.reserve(entryArgs.size());
347 for (auto &arg : entryArgs) {
348 argTypes.push_back(arg.type);
349 }
350 Type type = builder.getFunctionType(argTypes, resultTypes);
351 if (!type) {
352 return parser.emitError(signatureLocation)
353 << "failed to construct function type" << (errorMessage.empty() ? "" : ": ")
354 << errorMessage;
355 }
356 result.addAttribute(typeAttrName, TypeAttr::get(type));
357
358 // If function attributes are present, parse them.
359 NamedAttrList parsedAttributes;
360 SMLoc attributeDictLocation = parser.getCurrentLocation();
361 if (parser.parseOptionalAttrDictWithKeyword(parsedAttributes)) {
362 return failure();
363 }
364
365 // Disallow attributes that are inferred from elsewhere in the attribute
366 // dictionary.
367 for (StringRef disallowed :
368 {SymbolTable::getVisibilityAttrName(), SymbolTable::getSymbolAttrName(),
369 typeAttrName.getValue()}) {
370 if (parsedAttributes.get(disallowed)) {
371 return parser.emitError(attributeDictLocation, "'")
372 << disallowed
373 << "' is an inferred attribute and should not be specified in the "
374 "explicit attribute dictionary";
375 }
376 }
377 result.attributes.append(parsedAttributes);
378
379 // Add the attributes to the function arguments.
380 function_interface_impl::addArgAndResultAttrs(
381 builder, result, entryArgs, resultAttrs, argAttrsName,
382 /*resAttrsName*/ StringAttr::get(parser.getContext())
383 );
384
385 // Parse the required contract body.
386 auto *body = result.addRegion();
387 SMLoc loc = parser.getCurrentLocation();
388 if (parser.parseRegion(
389 *body, entryArgs,
390 /*enableNameShadowing=*/false
391 )) {
392 return failure();
393 }
394
395 // Contract body was parsed, make sure its not empty.
396 if (body->empty()) {
397 return parser.emitError(loc, "expected non-empty contract body");
398 }
399
400 return success();
401}

403void ContractOp::print(OpAsmPrinter &p) {
404 // Print the operation and the contract name.
405 p << ' ';
406 p.printSymbolName(getSymName());
407
408 // Print the name of the contract's target.
409 p << " for ";
410 p.printAttributeWithoutType(getTarget());
411 p << ' ';
412
413 ArrayRef<Type> argTypes = getArgumentTypes();
414 function_interface_impl::printFunctionSignature(
415 p, *this, argTypes, /*isVariadic*/ false, /*resultTypes*/ ArrayRef<Type>()
416 );
417 function_interface_impl::printFunctionAttributes(
418 p, *this,
419 /*elided*/ {getFunctionTypeAttrName(), getArgAttrsAttrName(), getTargetAttrName()}
420 );
421 // Print the body.
422 Region &body = getRegion();
423 p << ' ';
424 p.printRegion(
425 body, /*printEntryBlockArgs=*/false,
426 /*printBlockTerminators=*/true
427 );
428}

430LogicalResult ContractOp::verify() {
431 OwningEmitErrorFn emitErrorFunc = getEmitOpErrFn(this);
432
433 if ((*this)->hasAttr(ARG_NAME_ATTR_NAME)) {
434 return emitOpError() << '\'' << ARG_NAME_ATTR_NAME << "' is only valid on function arguments";
435 }
436
437 if (ArrayAttr argAttrs = getAllArgAttrs()) {
438 llvm::DenseSet<StringAttr> seenNames;
439 for (auto [i, attr] : llvm::enumerate(argAttrs)) {
440 auto dictAttr = llvm::dyn_cast<DictionaryAttr>(attr);
441 if (!dictAttr) {
442 continue;
443 }
444 Attribute argNameAttr = dictAttr.get(ARG_NAME_ATTR_NAME);
445 if (!argNameAttr) {
446 continue;
447 }
448 auto argName = llvm::dyn_cast<StringAttr>(argNameAttr);
449 if (!argName) {
450 return emitOpError() << '\'' << ARG_NAME_ATTR_NAME << "' on argument " << i
451 << " must be a string attribute";
452 }
453 if (!llvm::isa<NoneType>(argName.getType())) {
454 return emitOpError() << '\'' << ARG_NAME_ATTR_NAME << "' on argument " << i
455 << " must not have an explicit type";
456 }
457 if (argName.getValue().empty()) {
458 return emitOpError() << '\'' << ARG_NAME_ATTR_NAME << "' on argument " << i
459 << " must not be empty";
460 }
461 if (!seenNames.insert(argName).second) {
462 return emitOpError() << "duplicate '" << ARG_NAME_ATTR_NAME << "' value \""
463 << argName.getValue() << "\" on argument " << i;
464 }
465 }
466 }
467
468 // Unlike for FuncDefOps, we don't verify that the inputs are valid LLZK types,
469 // as we will check that the args match the target arguments in `verifySymbolUses()`
470
471 // Ensure that the contract does not contain nested modules, structs, or functions.
472 WalkResult res = this->walk<WalkOrder::PreOrder>([this](Operation *op) {
473 if (isa<ModuleOp, TemplateOp, FuncDefOp, StructDefOp>(op)) {
474 getEmitOpErrFn(op)().append(
475 "cannot be nested within '", getOperation()->getName(), "' operations"
476 );
477 return WalkResult::interrupt();
478 }
479 return WalkResult::advance();
480 });
481 return failure(res.wasInterrupted());
482}

534LogicalResult IncludeOp::verifyTemplateParamCompatibility(
535 Attribute paramFromIncludeOp, TemplateParamOp targetParam
536) {
537 // A wildcard `?` (represented as kDynamic) defers inference to a later pass.
538 // It is only valid for parameters with a `!poly.tvar` type restriction.
539 if (auto intAttr = llvm::dyn_cast<IntegerAttr>(paramFromIncludeOp)) {
540 if (isDynamic(intAttr)) {
541 std::optional<Type> declaredType = targetParam.getTypeOpt();
542 if (!declaredType || !llvm::isa<TypeVarType>(*declaredType)) {
543 auto diag = this->emitOpError().append(
544 "wildcard `?` can only be used for template parameters with `!poly.tvar` "
545 "type restriction, but parameter \"@",
546 targetParam.getName(), "\" has "
547 );
548 if (declaredType) {
549 diag.append("type restriction ", *declaredType);
550 } else {
551 diag.append("no type restriction");
552 }
553 return diag;
554 }
555 return success();
556 }
557 }
558 if (std::optional<Type> declaredType = targetParam.getTypeOpt()) {
559 // Note: `declaredType` is restricted by `isValidConstReadType()`
560 bool compatible = false;
561 if (llvm::isa<TypeVarType>(*declaredType)) {
562 compatible = llvm::isa<TypeAttr>(paramFromIncludeOp);
563 } else if (llvm::isa<FeltType>(*declaredType)) {
564 compatible = llvm::isa<FeltConstAttr, IntegerAttr>(paramFromIncludeOp) &&
565 isValidConstReadType(llvm::cast<TypedAttr>(paramFromIncludeOp).getType());
566 } else if (llvm::isa<IndexType, IntegerType>(*declaredType)) {
567 // Note: Just like struct type instantiation, there is no restriction on passing a
568 // larger value to an `i1`. The flattening pass will treat 0 as false and any other
569 // value as true (but give a warning if it's not 1).
570 compatible = llvm::isa<IntegerAttr>(paramFromIncludeOp) &&
571 isValidConstReadType(llvm::cast<TypedAttr>(paramFromIncludeOp).getType());
572 } else {
573 llvm_unreachable("inconsistent with `isValidConstReadType()`");
574 }
575 if (!compatible) {
576 return this->emitOpError().append(
577 "instantiation value '", paramFromIncludeOp, "' is not compatible with parameter \"@",
578 targetParam.getName(), "\" type restriction ", *declaredType
579 );
580 }
581 }
582 return success();
583}

585LogicalResult IncludeOp::verifyTemplateParamCompatibility(
586 llvm::iterator_range<Region::op_iterator<TemplateParamOp>> targetParamDefs
587) {
588 ArrayAttr callParams = this->getTemplateParamsAttr();
589 assert(!isNullOrEmpty(callParams) && "pre-condition");
590 assert((callParams.size() == llvm::range_size(targetParamDefs)) && "pre-condition");
591
592 for (auto [paramOp, attr] : llvm::zip_equal(targetParamDefs, callParams.getValue())) {
593 if (failed(verifyTemplateParamCompatibility(attr, paramOp))) {
594 return failure();
595 }
596 }
597 return success();
598}
599
600LogicalResult IncludeOp::verifyTemplateParamsMatchInferred(
601 llvm::iterator_range<Region::op_iterator<TemplateParamOp>> targetParamDefs,
602 const UnificationMap &unifications
603) {
604 ArrayAttr callParams = this->getTemplateParamsAttr();
605 assert(!isNullOrEmpty(callParams) && "pre-condition");
606 assert((callParams.size() == llvm::range_size(targetParamDefs)) && "pre-condition");
607
608 for (auto [paramOp, attr] : llvm::zip_equal(targetParamDefs, callParams.getValue())) {
609 // Skip wildcards (`?` / kDynamic) - their value will be resolved by a later inference pass.
610 if (isDynamic(llvm::dyn_cast<IntegerAttr>(attr))) {
611 continue;
612 }
613 auto it = unifications.find({FlatSymbolRefAttr::get(paramOp.getNameAttr()), Side::RHS});
614 if (it != unifications.end() && !typeParamsUnify({attr}, {it->second})) {
615 return this->emitOpError().append(
616 "template instantiation value '", attr, "' for parameter \"@", paramOp.getName(),
617 "\" conflicts with value '", it->second, "' inferred from function type signature"
618 );
619 }
620 }
621 return success();
622}

626struct IncludeOpVerifier {
627 explicit IncludeOpVerifier(IncludeOp *c) : includeOp(c) {}
628 virtual ~IncludeOpVerifier() = default;
629
630 LogicalResult verify() {
631 // Rather than immediately returning on failure, we check all verifier steps and aggregate to
632 // provide as many errors are possible in a single verifier run.
633 LogicalResult aggregateResult = success();
634 if (failed(verifyInputs())) {
635 aggregateResult = failure();
636 }
637 if (failed(verifyTemplateParams())) {
638 aggregateResult = failure();
639 }
640 return aggregateResult;
641 }
642
643protected:
644 IncludeOp *includeOp;
645
646 virtual LogicalResult verifyInputs() = 0;
647 virtual LogicalResult verifyTemplateParams() = 0;
648
649 LogicalResult verifyNoTemplateInstantiations() {
650 if (!isNullOrEmpty(includeOp->getTemplateParamsAttr())) {
651 return includeOp->emitOpError().append(
652 "can only have template instantiations when targeting a templated contract"
653 );
654 }
655 return success();
656 }
657};

659struct KnownTargetVerifier : public IncludeOpVerifier {
660 KnownTargetVerifier(IncludeOp *c, SymbolLookupResult<ContractOp> &&tgtRes)
661 : IncludeOpVerifier(c), tgt(*tgtRes), tgtType(tgt.getFunctionType()),
662 includeSymNames(tgtRes.getNamespace()) {}
663
664 LogicalResult verifyInputs() override {
665 return verifyTypesMatch(includeOp->getArgOperands().getTypes(), tgtType.getInputs(), "operand");
666 }
667
668 LogicalResult verifyTemplateParams() override {
669 auto tgtOp = tgt.getOperation();
670 if (TemplateOp tgtOpParent = getParentOfType<TemplateOp>(tgtOp)) {
671 // When the target function is a free function within a TemplateOp, the IncludeOp may have
672 // template parameter instantiations that must be checked against the template parameters.
673 // - If the function type signature references all template parameters, then the parameter
674 // instantiation list on the IncludeOp is optional, otherwise it's required.
675 // - If present, the instantiation list must provide a value for every template parameter
676 // and the value must be type-compatible with the parameter's declared type (if any).
677 // - If present, the instantiation list must result in a function type signature that can
678 // be unified with the IncludeOp's operand and result types.
679 auto realParams = tgtOpParent.getConstOps<TemplateParamOp>();
680 ArrayAttr callParams = includeOp->getTemplateParamsAttr();
681
682 // When there is no instantiation list, just ensure that it's not required.
683 if (isNullOrEmpty(callParams)) {
684 llvm::SmallDenseSet<SymbolRefAttr> referencedInSignature;
685 llzk::getSymbolsUsedIn(tgtType.getInputs(), referencedInSignature);
686 llzk::getSymbolsUsedIn(tgtType.getResults(), referencedInSignature);
687
688 bool allParamsReferenced = llvm::all_of(realParams, [&](TemplateParamOp p) {
689 return referencedInSignature.contains(FlatSymbolRefAttr::get(p.getNameAttr()));
690 });
691 if (allParamsReferenced) {
692 return success();
693 }
694 return includeOp->emitOpError().append(
695 "must provide template instantiation parameters when calling \"@", tgt.getSymName(),
696 "\" because not all template parameters of \"@", tgtOpParent.getSymName(),
697 "\" appear in the function type signature"
698 );
699 }
700
701 // Ensure `forceIntAttrTypes()` was successful on the IncludeOp's template parameters.
702 if (failed(llzk::forceIntAttrTypes(callParams.getValue(), [this] {
703 return llzk::InFlightDiagnosticWrapper(this->includeOp->emitOpError());
704 }))) {
705 return failure();
706 }
707
708 // The instantiation list is present. Check it has exactly one entry per template param.
709 size_t numTemplateParams = llvm::range_size(realParams);
710 if (callParams.size() != numTemplateParams) {
711 return includeOp->emitOpError().append(
712 "template instantiation has ", callParams.size(), " parameter(s) but \"@",
713 tgtOpParent.getSymName(), "\" expects ", numTemplateParams, " template parameter(s)"
714 );
715 }
716
717 // Check type compatibility of each provided value with the declared parameter type (if any).
718 if (failed(includeOp->verifyTemplateParamCompatibility(realParams))) {
719 return failure();
720 }
721
722 // Check that the provided instantiation values are consistent with what type unification
723 // of the target function types against the call's operand and result types would determine.
724 FailureOr<UnificationMap> unifyResult = includeOp->unifyTypeSignature(tgtType);
725 // This is already checked by `verifyInputs()`, but `verifyTemplateParams()` is called
726 // even if `verifyInputs()` fails for error aggregation, so we still need to return
727 // early here.
728 if (failed(unifyResult)) {
729 return failure();
730 }
731 return includeOp->verifyTemplateParamsMatchInferred(realParams, unifyResult.value());
732 } else {
733 // Non-template functions cannot contain template parameter instantiations.
734 return verifyNoTemplateInstantiations();
735 }
736 }
737
738private:
739 template <typename T>
740 LogicalResult
741 verifyTypesMatch(ValueTypeRange<T> includeOpTypes, ArrayRef<Type> tgtTypes, const char *aspect) {
742 if (tgtTypes.size() != includeOpTypes.size()) {
743 return includeOp->emitOpError()
744 .append("incorrect number of ", aspect, "s for callee, expected ", tgtTypes.size())
745 .attachNote(tgt.getLoc())
746 .append("callee defined here");
747 }
748 for (unsigned i = 0, e = tgtTypes.size(); i != e; ++i) {
749 if (!typesUnify(includeOpTypes[i], tgtTypes[i], includeSymNames)) {
750 return includeOp->emitOpError().append(
751 aspect, " type mismatch: expected type ", tgtTypes[i], ", but found ",
752 includeOpTypes[i], " for ", aspect, " number ", i
753 );
754 }
755 }
756 return success();
757 }
758
759 ContractOp tgt;
760 FunctionType tgtType;
761 std::vector<llvm::StringRef> includeSymNames;
762};

766LogicalResult IncludeOp::verifySymbolUses(SymbolTableCollection &tables) {
767 // First, verify symbol resolution in all input and output types.
768 if (failed(verifyTypeResolution(tables, *this, getTypeSignature()))) {
769 return failure(); // verifyTypeResolution() already emits a sufficient error message
770 }
771
772 // Check that the callee attribute was specified.
773 SymbolRefAttr calleeAttr = getCalleeAttr();
774 if (!calleeAttr) {
775 return emitOpError("requires a 'callee' symbol reference attribute");
776 }
777
778 // If the callee references a parameter of the template where this call appears, perform
779 // the subset of checks that can be done even though the target is unknown.
780 if (calleeAttr.getNestedReferences().size() == 1) {
781 if (TemplateOp parent = getParentOfType<TemplateOp>(*this)) {
782 if (auto constParam = parent.getConstNamed<TemplateParamOp>(calleeAttr.getRootReference())) {
783 return this->emitError("expected parameterized callee to target a struct function")
784 .attachNote(constParam->getLoc())
785 .append(
786 " (i.e. \"@", FUNC_NAME_PRODUCT, "\", \"@", FUNC_NAME_COMPUTE, "\", or \"@",
787 FUNC_NAME_CONSTRAIN, "\")"
788 );
789 }
790 }
791 }
792
793 // Otherwise, callee must be specified via full path from the root module. Perform the full set of
794 // checks against the known target function.
795 auto tgtOpt = lookupTopLevelSymbol<ContractOp>(
796 tables, calleeAttr, getParentOfType<ModuleOp>(getOperation())
797 );
798 if (failed(tgtOpt)) {
799 return this->emitError() << "expected '" << ContractOp::getOperationName() << "' named \""
800 << calleeAttr << '"';
801 }
802 return KnownTargetVerifier(this, std::move(*tgtOpt)).verify();
803}
