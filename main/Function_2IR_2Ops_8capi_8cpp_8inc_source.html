184 MlirOperationState state = mlirOperationStateGet(mlirStringRefCreateFromCString("function.def"), location);
185 MlirContext ctx = mlirOpBuilderGetContext(builder);
186 llvm::SmallVector<MlirNamedAttribute, 4> attributes;
187 if (!mlirAttributeIsNull(reinterpret_cast<MlirAttribute&>(sym_name))) {
188 attributes.push_back(mlirNamedAttributeGet(mlirIdentifierGet(ctx, mlirStringRefCreateFromCString("sym_name")), reinterpret_cast<MlirAttribute&>(sym_name)));
189 }
190 if (!mlirAttributeIsNull(function_type)) {
191 attributes.push_back(mlirNamedAttributeGet(mlirIdentifierGet(ctx, mlirStringRefCreateFromCString("function_type")), function_type));
192 }
193 if (!mlirAttributeIsNull(arg_attrs)) {
194 attributes.push_back(mlirNamedAttributeGet(mlirIdentifierGet(ctx, mlirStringRefCreateFromCString("arg_attrs")), arg_attrs));
195 }
196 if (!mlirAttributeIsNull(res_attrs)) {
197 attributes.push_back(mlirNamedAttributeGet(mlirIdentifierGet(ctx, mlirStringRefCreateFromCString("res_attrs")), res_attrs));
198 }
199 mlirOperationStateAddAttributes(&state, attributes.size(), attributes.data());
200 llvm::SmallVector<MlirRegion, 1> regions;
201 regions.push_back(mlirRegionCreate());
202 mlirOperationStateAddOwnedRegions(&state, regions.size(), regions.data());
203
204 return mlirOpBuilderInsert(builder, mlirOperationCreate(&state));
