156def LLZK_SignedIntDivFeltOp : FeltDialectBinaryOp<"sintdiv", [NotFieldNative]> {
157 let summary = "signed integer division operator for field elements";
158 let description = [{
159 Treats the operands as if they were signed integers with bitwidth
160 equal to that of the prime modulus (no additional sign bit is added)
161 and performs division rounding towards zero.
162
163 The signed integer representation of felt `f` in prime field with modulus
164 `p` follows the following formula:
165
166 signed_int(f) = f if 0 <= f < p/2 + 1
167 "p/2" here is unsigned integer division rounding towards 0
168 signed_int(f) = f-p if p/2 + 1 <= f < p
169 }];
170}
