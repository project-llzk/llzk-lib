83static MlirType
84lookupRecordImpl(PodType type, StringRef name, llvm::function_ref<InFlightDiagnostic()> emitError) {
85 auto attr = type.getRecord(name, emitError);
86 if (failed(attr)) {
87 return MlirType {.ptr = nullptr};
88 }
89 return wrap(*attr);
90}

94MlirType llzkPod_PodTypeLookupRecord(MlirType type, MlirStringRef name) {
95 auto pod = unwrap_cast<PodType>(type);
96 return lookupRecordImpl(pod, unwrap(name), [pod] {
97 auto *ctx = pod.getContext();
98 return ctx->getDiagEngine().emit(Builder(ctx).getUnknownLoc(), DiagnosticSeverity::Error);
99 });
100}
