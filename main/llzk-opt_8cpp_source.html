60int main(int argc, char **argv) {
61 llvm::sys::PrintStackTraceOnErrorSignal(llvm::StringRef());
62 llvm::setBugReportMsg(
63 "PLEASE submit a bug report to " BUG_REPORT_URL
64 " and include the crash backtrace, relevant LLZK files,"
65 " and associated run script(s).\n"
66 );
67 llvm::cl::AddExtraVersionPrinter([](llvm::raw_ostream &os) {
68 os << "\nLLZK (" LLZK_URL "):\n LLZK version " LLZK_VERSION_STRING "\n";
69 });
70
71 // MLIR initialization
72 mlir::DialectRegistry registry;
73 llzk::registerAllDialects(registry);
74 r1cs::registerAllDialects(registry);
75 zklean::registerAllDialects(registry);
76#if LLZK_WITH_PCL
77 pcl::registerAllDialects(registry);
78#endif // LLZK_WITH_PCL
79
80 llzk::registerValidationPasses();
81 llzk::registerAnalysisPasses();
82 llzk::registerTransformationPasses();
83 llzk::array::registerTransformationPasses();
84 llzk::include::registerTransformationPasses();
85 llzk::polymorphic::registerTransformationPasses();
86 r1cs::registerTransformationPasses();
87 zklean::registerConversionPasses();
88 zklean::registerZKLeanPasses();
89#if LLZK_WITH_PCL
90 pcl::registerTransformationPasses();
91 pcl::conversion::registerPCLTransformationPasses();
92#endif // LLZK_WITH_PCL
93
94 llzk::registerTransformationPassPipelines();
95 r1cs::registerTransformationPassPipelines();
96
97 // Register and parse command line options.
98 std::string inputFilename, outputFilename;
99 std::tie(inputFilename, outputFilename) =
100 registerAndParseCLIOptions(argc, argv, "llzk-opt", registry);
101
102 if (PrintAllOps) {
103 mlir::MLIRContext context;
104 context.appendDialectRegistry(registry);
105 context.loadAllAvailableDialects();
106 llvm::outs() << "All ops registered in LLZK IR: {\n";
107 for (const auto &opName : context.getRegisteredOperations()) {
108 llvm::outs().indent(2) << opName.getStringRef() << '\n';
109 }
110 llvm::outs() << "}\n";
111 return EXIT_SUCCESS;
112 }
113
114 // Set the include directories from CL option
115 if (mlir::failed(llzk::GlobalSourceMgr::get().setup(IncludeDirs))) {
116 return EXIT_FAILURE;
117 }
118
119 // Run 'mlir-opt'
120 auto result = mlir::MlirOptMain(argc, argv, inputFilename, outputFilename, registry);
121 return mlir::asMainReturnCode(result);
122}
