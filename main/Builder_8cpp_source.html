28static MlirOpBuilderInsertPoint wrapInsertPoint(OpBuilder::InsertPoint point) {
29 auto *block = point.getBlock();
30 if (!block) {
31 return {.block = {.ptr = nullptr}, .point = {.ptr = nullptr}};
32 }
33
34 if (point.getPoint() == block->end()) {
35 return {.block = wrap(block), .point = {.ptr = nullptr}};
36 }
37
38 return {.block = wrap(block), .point = wrap(&*point.getPoint())};
39}

43class ListenerT : public OpBuilder::Listener {
44public:
45 ListenerT(MlirNotifyOperationInserted op, MlirNotifyBlockInserted block, void *data)
46 : opInsertedCb(op), blockInsertedCb(block), userData(data) {}
47
50 void notifyOperationInserted(Operation *op, OpBuilder::InsertPoint previous) final {
51 opInsertedCb(wrap(op), wrapInsertPoint(previous), userData);
52 }
53
56 void notifyBlockInserted(Block *block, Region *previous, Region::iterator previousIt) final {
57 blockInsertedCb(
58 wrap(block), wrap(previous),
59 (!previous || previous->end() == previousIt) ? MlirBlock {.ptr = nullptr}
60 : wrap(&*previousIt),
61 userData
62 );
63 }
64
65private:
66 MlirNotifyOperationInserted opInsertedCb;
67 MlirNotifyBlockInserted blockInsertedCb;
68 void *userData = nullptr;
69};

131void mlirOpBuilderRestoreInsertionPoint(MlirOpBuilder builder, MlirOpBuilderInsertPoint rawIp) {
132 auto *block = unwrap(rawIp.block);
133 if (!block) {
134 OpBuilderT::InsertPoint ip;
135 unwrap(builder)->restoreInsertionPoint(ip);
136 } else {
137 Block::iterator it = rawIp.point.ptr ? Block::iterator(unwrap(rawIp.point)) : block->end();
138 OpBuilderT::InsertPoint ip(block, it);
139 unwrap(builder)->restoreInsertionPoint(ip);
140 }
141}
