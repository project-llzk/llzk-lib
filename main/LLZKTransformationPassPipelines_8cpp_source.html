41void buildFullStructInliningPipelineImpl(
42 OpPassManager &pm, polymorphic::FlatteningPassOptions flattening, bool arrayToScalar,
43 bool podToScalar, std::unique_ptr<Pass> inliningPass
44) {
45 // default to `main-as-root` if unspecified to avoid leaving parameterized templates
46 // that cause the later struct inlining pass to crash
47 if (flattening.cleanupMode == polymorphic::FlatteningCleanupMode::Unspecified) {
48 flattening.cleanupMode = polymorphic::FlatteningCleanupMode::MainAsRoot;
49 }
50 pm.addPass(polymorphic::createFlatteningPass(flattening));
51
52 // Run array-to-scalar first because it can split arrays within a pod
53 // but pod-to-scalar cannot split pods within an array.
54 if (arrayToScalar) {
55 pm.addPass(array::createArrayToScalarPass());
56 }
57 if (podToScalar) {
58 pm.addPass(pod::createPodToScalarPass());
59 }
60 // Canonicalize to remove known-condition `scf.if` regions so struct inlining
61 // can link "@compute" calls to struct members.
62 pm.addPass(mlir::createCanonicalizerPass());
63 pm.addPass(std::move(inliningPass));
64
65 // Remove struct and member definitions that are no longer used after inlining.
66 pm.addPass(createUnusedDeclarationEliminationPass(
67 UnusedDeclarationEliminationPassOptions {.removeStructs = true}
68 ));
69}
70
71void buildFullPolyLoweringPipelineImpl(
72 OpPassManager &pm, polymorphic::FlatteningPassOptions flattening, bool arrayToScalar,
73 bool podToScalar, std::unique_ptr<Pass> inliningPass, std::unique_ptr<Pass> polyLoweringPass
74) {
75 // 1. Struct flattening and inlining
76 buildFullStructInliningPipelineImpl(
77 pm, flattening, arrayToScalar, podToScalar, std::move(inliningPass)
78 );
79 // 2. Degree lowering
80 pm.addPass(std::move(polyLoweringPass));
81 // 3. Cleanup
82 buildRemoveUnnecessaryOpsAndDefsPipeline(pm);
83}

87void buildRemoveUnnecessaryOpsPipeline(mlir::OpPassManager &pm) {
88 pm.addPass(createRedundantReadAndWriteEliminationPass());
89 pm.addPass(createRedundantOperationEliminationPass());
90}
91
92void buildRemoveUnnecessaryOpsAndDefsPipeline(mlir::OpPassManager &pm) {
93 buildRemoveUnnecessaryOpsPipeline(pm);
94 pm.addPass(createUnusedDeclarationEliminationPass());
95}
96
97void buildProductProgramPipeline(OpPassManager &pm) {
98 pm.addPass(createComputeConstrainToProductPass());
99 pm.addPass(createFuseProductLoopsPass());
100}

102void buildFullStructInliningPipeline(OpPassManager &pm, const FullStructInliningConfig &cfg) {
103 buildFullStructInliningPipelineImpl(
104 pm, cfg.flattening, cfg.arrayToScalar, cfg.podToScalar,
105 component::createInlineStructsPass(cfg.inlining)
106 );
107}

109void buildFullPolyLoweringPipeline(OpPassManager &pm, const FullPolyLoweringConfig &cfg) {
110 buildFullPolyLoweringPipelineImpl(
111 pm, cfg.structInlining.flattening, cfg.structInlining.arrayToScalar,
112 cfg.structInlining.podToScalar,
113 component::createInlineStructsPass(cfg.structInlining.inlining),
114 createPolyLoweringPass(cfg.polyLowering)
115 );
116}

122void registerTransformationPassPipelines() {
123 PassPipelineRegistration<>(
124 "llzk-remove-unnecessary-ops",
125 "Remove unnecessary operations, such as redundant reads or repeated constraints",
126 buildRemoveUnnecessaryOpsPipeline
127 );
128
129 PassPipelineRegistration<>(
130 "llzk-remove-unnecessary-ops-and-defs",
131 "Remove unnecessary operations, member definitions, and struct definitions",
132 buildRemoveUnnecessaryOpsAndDefsPipeline
133 );
134
135 PassPipelineRegistration<>(
136 "llzk-product-program",
137 "Convert @compute/@constrain functions to @product function and perform alignment",
138 buildProductProgramPipeline
139 );
140
141 PassPipelineRegistration<FullStructInliningOptions>(
142 "llzk-full-struct-inlining",
143 "Run flattening and inlining of all struct definitions into the `main` struct. This "
144 "pipeline uses the `main-as-root` cleanup mode in the flattening pass by default. It "
145 "is not recommended to override this cleanup mode because other cleanup modes may "
146 "leave behind parameterized templates that later cause `llzk-inline-structs` to crash.",
147 [](OpPassManager &pm, const FullStructInliningOptions &opts) {
148 auto flattening = opts.flattening.getValue().createOptions();
149 buildFullStructInliningPipelineImpl(
150 pm, flattening->createPassOptions(), opts.arrayToScalar, opts.podToScalar,
151 createConfiguredPass(opts.inlining)
152 );
153 }
154 );
155
156 PassPipelineRegistration<FullPolyLoweringOptions>(
157 "llzk-full-poly-lowering",
158 "Run flattening and inlining of all struct definitions into the `main` struct, then lower "
159 "polynomial constraints to a given max degree, and finally remove unnecessary operations and "
160 "definitions. This pipeline uses the `main-as-root` cleanup mode in the flattening pass by "
161 "default. It is not recommended to override this cleanup mode because other cleanup modes "
162 "may leave behind parameterized templates that later cause `llzk-inline-structs` to crash.",
163 [](OpPassManager &pm, const FullPolyLoweringOptions &opts) {
164 auto structInlining = opts.structInlining.getValue().createOptions();
165 auto flattening = structInlining->flattening.getValue().createOptions();
166 buildFullPolyLoweringPipelineImpl(
167 pm, flattening->createPassOptions(), structInlining->arrayToScalar,
168 structInlining->podToScalar, createConfiguredPass(structInlining->inlining),
169 createConfiguredPass(opts.polyLowering)
170 );
171 }
172 );
173}
