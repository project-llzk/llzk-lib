39bool hasEmptyParamList(StructType t) {
40 if (ArrayAttr paramList = t.getParams()) {
41 if (paramList.empty()) {
42 return true;
43 }
44 }
45 return false;
46}

85class EmptyParamRemovalPass
86 : public llzk::polymorphic::impl::EmptyParamListRemovalPassBase<EmptyParamRemovalPass> {
87
88 void runOnOperation() override {
89 ModuleOp modOp = getOperation();
90 MLIRContext *ctx = modOp.getContext();
91 EmptyParamListStructTypeConverter tyConv;
92 ConversionTarget target = newConverterDefinedTarget<>(tyConv, ctx);
93 // Mark StructDefOp with empty parameter list as illegal
94 target.addDynamicallyLegalOp<StructDefOp>([](StructDefOp op) {
95 return !hasEmptyParamList(op.getType());
96 });
97 RewritePatternSet patterns = newGeneralRewritePatternSet(tyConv, ctx, target);
98 patterns.add<CallOpTypeReplacePattern>(tyConv, ctx);
99 if (failed(applyFullConversion(modOp, target, std::move(patterns)))) {
100 signalPassFailure();
101 }
102 }
103};
