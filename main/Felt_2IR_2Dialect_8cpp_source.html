40mlir::Attribute FieldSpecAttr::parse(mlir::AsmParser &odsParser, mlir::Type) {
41 mlir::Builder odsBuilder(odsParser.getContext());
42 llvm::SMLoc odsLoc = odsParser.getCurrentLocation();
43 mlir::FailureOr<mlir::StringAttr> fieldNameAttrRes;
44 mlir::FailureOr<llvm::APInt> primeRes;
45
46 // Parse literal '<'
47 if (odsParser.parseLess()) {
48 return {};
49 }
50
51 // Parse variable 'fieldName'
52 fieldNameAttrRes = mlir::FieldParser<mlir::StringAttr>::parse(odsParser);
53 if (mlir::failed(fieldNameAttrRes)) {
54 odsParser.emitError(
55 odsParser.getCurrentLocation(), "failed to parse LLZK_FieldSpecAttr parameter 'fieldName' "
56 "which is to be a `mlir::StringAttr`"
57 );
58 return {};
59 }
60 // Parse literal ','
61 if (odsParser.parseComma()) {
62 return {};
63 }
64
65 // Parse variable 'prime'
66 primeRes = mlir::FieldParser<llvm::APInt>::parse(odsParser);
67 if (mlir::failed(primeRes)) {
68 odsParser.emitError(
69 odsParser.getCurrentLocation(),
70 "failed to parse LLZK_FieldSpecAttr parameter 'prime' which is to be a `llvm::APInt`"
71 );
72 return {};
73 }
74 // Parse literal '>'
75 if (odsParser.parseGreater()) {
76 return {};
77 }
78 assert(mlir::succeeded(fieldNameAttrRes));
79 assert(mlir::succeeded(primeRes));
80
81 // Custom logic: cache the field, reporting an error if there's a conflict
82 auto errFn = [&odsParser]() {
83 return InFlightDiagnosticWrapper(odsParser.emitError(odsParser.getCurrentLocation()));
84 };
85 Field::addField(fieldNameAttrRes.value(), primeRes.value(), errFn);
86
87 return odsParser.getChecked<FieldSpecAttr>(
88 odsLoc, odsParser.getContext(), mlir::StringAttr(*fieldNameAttrRes), llvm::APInt(*primeRes)
89 );
90}

92void FieldSpecAttr::print(mlir::AsmPrinter &odsPrinter) const {
93 mlir::Builder odsBuilder(getContext());
94 odsPrinter << "<";
95 odsPrinter.printStrippedAttrOrType(getFieldName());
96 odsPrinter << ", ";
97 odsPrinter.printStrippedAttrOrType(getPrime());
98 odsPrinter << '>';
99}

107llvm::LogicalResult
108FeltType::verify(llvm::function_ref<mlir::InFlightDiagnostic()> errFn, mlir::StringAttr fieldName) {
109 return fieldName ? Field::verifyFieldDefined(
110 fieldName.getValue(), wrapNonNullableInFlightDiagnostic(errFn)
111 )
112 : mlir::success();
113}
