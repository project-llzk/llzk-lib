56LogicalResult AbstractSparseForwardDataFlowAnalysis::initialize(Operation *top) {
57 // Mark the entry block arguments as having reached their pessimistic
58 // fixpoints.
59 for (Region &region : top->getRegions()) {
60 if (region.empty()) {
61 continue;
62 }
63 for (Value argument : region.front().getArguments()) {
64 setToEntryState(getLatticeElement(argument));
65 }
66 }
67
68 return initializeRecursively(top);
69}

71LogicalResult AbstractSparseForwardDataFlowAnalysis::initializeRecursively(Operation *op) {
72 // Initialize the analysis by visiting every owner of an SSA value (all
73 // operations and blocks).
74 if (failed(visitOperation(op))) {
75 return failure();
76 }
77
78 for (Region &region : op->getRegions()) {
79 for (Block &block : region) {
80 getOrCreate<Executable>(getProgramPointBefore(&block))->blockContentSubscribe(this);
81 visitBlock(&block);
82 // LLZK: Renamed "op" -> "containedOp" to avoid shadowing.
83 for (Operation &containedOp : block) {
84 if (failed(initializeRecursively(&containedOp))) {
85 return failure();
86 }
87 }
88 }
89 }
90
91 return success();
92}

94LogicalResult AbstractSparseForwardDataFlowAnalysis::visit(ProgramPoint *point) {
95 if (!point->isBlockStart()) {
96 return visitOperation(point->getPrevOp());
97 }
98 visitBlock(point->getBlock());
99 return success();
100}

102LogicalResult AbstractSparseForwardDataFlowAnalysis::visitOperation(Operation *op) {
106
107 // If the containing block is not executable, bail out.
108 if (op->getBlock() != nullptr &&
109 !getOrCreate<Executable>(getProgramPointBefore(op->getBlock()))->isLive()) {
110 return success();
111 }
112
113 // Get the result lattices.
114 SmallVector<AbstractSparseLattice *> resultLattices;
115 resultLattices.reserve(op->getNumResults());
116 for (Value result : op->getResults()) {
117 AbstractSparseLattice *resultLattice = getLatticeElement(result);
118 resultLattices.push_back(resultLattice);
119 }
120
121 // The results of a region branch operation are determined by control-flow.
122 if (auto branch = dyn_cast<RegionBranchOpInterface>(op)) {
123 visitRegionSuccessors(
124 getProgramPointAfter(branch), branch,
125 /*successor=*/RegionBranchPoint::parent(), resultLattices
126 );
127 return success();
128 }
129
130 // Grab the lattice elements of the operands.
131 SmallVector<const AbstractSparseLattice *> operandLattices;
132 operandLattices.reserve(op->getNumOperands());
133 for (Value operand : op->getOperands()) {
134 AbstractSparseLattice *operandLattice = getLatticeElement(operand);
135 operandLattice->useDefSubscribe(this);
136 operandLattices.push_back(operandLattice);
137 }
138
139 if (auto call = dyn_cast<CallOpInterface>(op)) {
140 // If the call operation is to an external function, attempt to infer the
141 // results from the call arguments.
142 auto callable = dyn_cast_if_present<CallableOpInterface>(call.resolveCallable());
143 if (!getSolverConfig().isInterprocedural() || (callable && !callable.getCallableRegion())) {
144 visitExternalCallImpl(call, operandLattices, resultLattices);
145 return success();
146 }
147
148 // Otherwise, the results of a call operation are determined by the
149 // callgraph.
150 const auto *predecessors =
151 getOrCreateFor<PredecessorState>(getProgramPointAfter(op), getProgramPointAfter(call));
152 // If not all return sites are known, then conservatively assume we can't
153 // reason about the data-flow.
154 if (!predecessors->allPredecessorsKnown()) {
155 setAllToEntryStates(resultLattices);
156 return success();
157 }
158 for (Operation *predecessor : predecessors->getKnownPredecessors()) {
159 for (auto &&[operand, resLattice] : llvm::zip(predecessor->getOperands(), resultLattices)) {
160 join(resLattice, *getLatticeElementFor(getProgramPointAfter(op), operand));
161 }
162 }
163 return success();
164 }
165
166 // Invoke the operation transfer function.
167 return visitOperationImpl(op, operandLattices, resultLattices);
168}

170void AbstractSparseForwardDataFlowAnalysis::visitBlock(Block *block) {
171 // Exit early on blocks with no arguments.
172 if (block->getNumArguments() == 0) {
173 return;
174 }
175
176 // If the block is not executable, bail out.
177 if (!getOrCreate<Executable>(getProgramPointBefore(block))->isLive()) {
178 return;
179 }
180
181 // Get the argument lattices.
182 SmallVector<AbstractSparseLattice *> argLattices;
183 argLattices.reserve(block->getNumArguments());
184 for (BlockArgument argument : block->getArguments()) {
185 AbstractSparseLattice *argLattice = getLatticeElement(argument);
186 argLattices.push_back(argLattice);
187 }
188
189 // The argument lattices of entry blocks are set by region control-flow or the
190 // callgraph.
191 if (block->isEntryBlock()) {
192 // Check if this block is the entry block of a callable region.
193 auto callable = dyn_cast<CallableOpInterface>(block->getParentOp());
194 if (callable && callable.getCallableRegion() == block->getParent()) {
195 const auto *callsites = getOrCreateFor<PredecessorState>(
196 getProgramPointBefore(block), getProgramPointAfter(callable)
197 );
198 // If not all callsites are known, conservatively mark all lattices as
199 // having reached their pessimistic fixpoints.
200 if (!callsites->allPredecessorsKnown() || !getSolverConfig().isInterprocedural()) {
201 return setAllToEntryStates(argLattices);
202 }
203 for (Operation *callsite : callsites->getKnownPredecessors()) {
204 auto call = cast<CallOpInterface>(callsite);
205 for (auto it : llvm::zip(call.getArgOperands(), argLattices)) {
206 join(
207 std::get<1>(it), *getLatticeElementFor(getProgramPointBefore(block), std::get<0>(it))
208 );
209 }
210 }
211 return;
212 }
213
214 // Check if the lattices can be determined from region control flow.
215 if (auto branch = dyn_cast<RegionBranchOpInterface>(block->getParentOp())) {
216 return visitRegionSuccessors(
217 getProgramPointBefore(block), branch, block->getParent(), argLattices
218 );
219 }
220
221 // Otherwise, we can't reason about the data-flow.
222 return visitNonControlFlowArgumentsImpl(
223 block->getParentOp(), RegionSuccessor(block->getParent()), argLattices, /*firstIndex=*/0
224 );
225 }
226
227 // Iterate over the predecessors of the non-entry block.
228 for (Block::pred_iterator it = block->pred_begin(), e = block->pred_end(); it != e; ++it) {
229 Block *predecessor = *it;
230
231 // If the edge from the predecessor block to the current block is not live,
232 // bail out.
233 auto *edgeExecutable = getOrCreate<Executable>(getLatticeAnchor<CFGEdge>(predecessor, block));
234 edgeExecutable->blockContentSubscribe(this);
235 if (!edgeExecutable->isLive()) {
236 continue;
237 }
238
239 // Check if we can reason about the data-flow from the predecessor.
240 if (auto branch = dyn_cast<BranchOpInterface>(predecessor->getTerminator())) {
241 SuccessorOperands operands = branch.getSuccessorOperands(it.getSuccessorIndex());
242 for (auto [idx, lattice] : llvm::enumerate(argLattices)) {
243 if (Value operand = operands[idx]) {
244 join(lattice, *getLatticeElementFor(getProgramPointBefore(block), operand));
245 } else {
246 // Conservatively consider internally produced arguments as entry
247 // points.
248 setAllToEntryStates(lattice);
249 }
250 }
251 } else {
252 return setAllToEntryStates(argLattices);
253 }
254 }

257void AbstractSparseForwardDataFlowAnalysis::visitRegionSuccessors(
258 ProgramPoint *point, RegionBranchOpInterface branch, RegionBranchPoint successor,
259 ArrayRef<AbstractSparseLattice *> lattices
260) {
261 const auto *predecessors = getOrCreateFor<PredecessorState>(point, point);
262 assert(predecessors->allPredecessorsKnown() && "unexpected unresolved region successors");
263
264 for (Operation *op : predecessors->getKnownPredecessors()) {
265 // Get the incoming successor operands.
266 std::optional<OperandRange> operands;
267
268 // Check if the predecessor is the parent op.
269 if (op == branch) {
270 operands = branch.getEntrySuccessorOperands(successor);
271 // Otherwise, try to deduce the operands from a region return-like op.
272 } else if (auto regionTerminator = dyn_cast<RegionBranchTerminatorOpInterface>(op)) {
273 operands = regionTerminator.getSuccessorOperands(successor);
274 }
275
276 if (!operands) {
277 // We can't reason about the data-flow.
278 return setAllToEntryStates(lattices);
279 }
280
281 ValueRange inputs = predecessors->getSuccessorInputs(op);
282 assert(
283 inputs.size() == operands->size() &&
284 "expected the same number of successor inputs as operands"
285 );
286
287 unsigned firstIndex = 0;
288 if (inputs.size() != lattices.size()) {
289 if (!point->isBlockStart()) {
290 if (!inputs.empty()) {
291 firstIndex = cast<OpResult>(inputs.front()).getResultNumber();
292 }
293 visitNonControlFlowArgumentsImpl(
294 branch, RegionSuccessor(branch->getResults().slice(firstIndex, inputs.size())),
295 lattices, firstIndex
296 );
297 } else {
298 if (!inputs.empty()) {
299 firstIndex = cast<BlockArgument>(inputs.front()).getArgNumber();
300 }
301 Region *region = point->getBlock()->getParent();
302 visitNonControlFlowArgumentsImpl(
303 branch,
304 RegionSuccessor(region, region->getArguments().slice(firstIndex, inputs.size())),
305 lattices, firstIndex
306 );
307 }
308 }
309
310 for (auto it : llvm::zip(*operands, lattices.drop_front(firstIndex))) {
311 join(std::get<1>(it), *getLatticeElementFor(point, std::get<0>(it)));
312 }
313 }
