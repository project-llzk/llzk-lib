101class ConversionTracker {
103 bool modified;
106 DenseMap<StructType, StructType> structInstantiations;
108 DenseMap<StructType, StructType> reverseInstantiations;
110 DenseSet<SymbolRefAttr> funcInstantiations;
113 DenseMap<StructType, SmallVector<Diagnostic>> delayedDiagnostics;
114
115public:
116 bool isModified() const { return modified; }
117 void resetModifiedFlag() { modified = false; }
118 void updateModifiedFlag(bool currStepModified) { modified |= currStepModified; }
119
120 void recordInstantiation(StructType oldType, StructType newType) {
121 assert(!isNullOrEmpty(oldType.getParams()) && "cannot instantiate with no params");
122
123 auto forwardResult = structInstantiations.try_emplace(oldType, newType);
124 if (forwardResult.second) {
125 // Insertion was successful
126 // ASSERT: The reverse map does not contain this mapping either
127 assert(!reverseInstantiations.contains(newType));
128 reverseInstantiations[newType] = oldType;
129 // Set the modified flag
130 modified = true;
131 } else {
132 // ASSERT: If a mapping already existed for `oldType` it must be `newType`
133 assert(forwardResult.first->getSecond() == newType);
134 // ASSERT: The reverse mapping is already present as well
135 assert(reverseInstantiations.lookup(newType) == oldType);
136 }
137 assert(structInstantiations.size() == reverseInstantiations.size());
138 }
139
141 std::optional<StructType> getInstantiation(StructType oldType) const {
142 auto cachedResult = structInstantiations.find(oldType);
143 if (cachedResult != structInstantiations.end()) {
144 return cachedResult->second;
145 }
146 return std::nullopt;
147 }
148
149 /// Record that the given free function was instantiated.
150 void recordInstantiation(SymbolRefAttr funcName) {
151 funcInstantiations.insert(funcName);
152 modified = true;
153 }
154
156 DenseSet<SymbolRefAttr> getInstantiatedDefinitionNames() const {
157 DenseSet<SymbolRefAttr> instantiatedNames = funcInstantiations;
158 for (const auto &[origRemoteTy, _] : structInstantiations) {
159 instantiatedNames.insert(origRemoteTy.getNameRef());
160 }
161 return instantiatedNames;
162 }
163
164 void reportDelayedDiagnostics(StructType newType, CallOp caller) {
165 auto res = delayedDiagnostics.find(newType);
166 if (res != delayedDiagnostics.end()) {
167 ::reportDelayedDiagnostics(caller, std::move(res->second));
168
169 // Emitting a Diagnostic consumes it (per DiagnosticEngine::emit) so remove them from the map.
170 // Unfortunately, this means if the key StructType is the result of instantiation at multiple
171 // `compute()` calls it will only be reported at one of those locations, not all.
172 delayedDiagnostics.erase(newType);
173 }
174 }
175
176 SmallVector<Diagnostic> &delayedDiagnosticSet(StructType newType) {
177 return delayedDiagnostics[newType];
178 }

180 /// Check if the type conversion is legal, i.e., the new type unifies with and is more concrete
181 /// than the old type with additional allowance for the results of struct flattening conversions.
182 bool isLegalConversion(Type oldType, Type newType, const char *patName) const {
183 std::function<bool(Type, Type)> checkInstantiations = [&](Type oTy, Type nTy) {
184 // Check if `oTy` is a struct with a known instantiation to `nTy`
185 if (StructType oldStructType = llvm::dyn_cast<StructType>(oTy)) {
186 // Note: The values in `structInstantiations` must be no-parameter struct types
187 // so there is no need for recursive check, simple equality is sufficient.
188 if (this->structInstantiations.lookup(oldStructType) == nTy) {
189 return true;
190 }
191 }
192 // Check if `nTy` is the result of a struct instantiation and if the pre-image of
193 // that instantiation (i.e., the parameterized version of the instantiated struct)
194 // is a more concrete unification of `oTy`.
195 if (StructType newStructType = llvm::dyn_cast<StructType>(nTy)) {
196 if (auto preImage = this->reverseInstantiations.lookup(newStructType)) {
197 if (isMoreConcreteUnification(oTy, preImage, checkInstantiations)) {
198 return true;
199 }
200 }
201 }
202 return false;
203 };
204
205 if (isMoreConcreteUnification(oldType, newType, checkInstantiations)) {
206 return true;
207 }
208 LLVM_DEBUG(
209 llvm::dbgs() << "[" << patName << "] Cannot replace old type " << oldType
210 << " with new type " << newType
211 << " because it does not define a compatible and more concrete type.\n";
212 );
213 return false;
214 }
215
216 template <typename T, typename U>
217 inline bool areLegalConversions(T oldTypes, U newTypes, const char *patName) const {
218 return llvm::all_of(
219 llvm::zip_equal(oldTypes, newTypes), [this, &patName](std::tuple<Type, Type> oldThenNew) {
220 return this->isLegalConversion(std::get<0>(oldThenNew), std::get<1>(oldThenNew), patName);
221 }
222 );
223 }

231 SymbolUserHelper(
232 TypeConverter &converter, MLIRContext *ctx, unsigned patternBenefit,
233 const DenseMap<Attribute, Attribute> &paramNameToInstantiatedValue
234 )
235 : OpConversionPattern<Op>(converter, ctx, patternBenefit),
236 paramNameToValue(paramNameToInstantiatedValue) {}
237
238public:
239 using OpAdaptor = typename mlir::OpConversionPattern<Op>::OpAdaptor;
240
241 virtual Attribute getNameAttr(Op) const = 0;
242
243 virtual LogicalResult handleDefaultRewrite(
244 Attribute, Op op, OpAdaptor, ConversionPatternRewriter &, Attribute a
245 ) const {
246 return op->emitOpError().append("expected value with type ", op.getType(), " but found ", a);

248

250 matchAndRewrite(Op op, OpAdaptor adaptor, ConversionPatternRewriter &rewriter) const override {
251 LLVM_DEBUG(llvm::dbgs() << "[SymbolUserHelper] op: " << op << '\n');
252 auto res = this->paramNameToValue.find(getNameAttr(op));
253 if (res == this->paramNameToValue.end()) {
254 LLVM_DEBUG(llvm::dbgs() << "[SymbolUserHelper] no instantiation for " << op << '\n');
255 return failure();
256 }
257 llvm::TypeSwitch<Attribute, LogicalResult> TS(res->second);
258 llvm::TypeSwitch<Attribute, LogicalResult> *ptr = &TS;
259
260 ((ptr = &(ptr->template Case<HandledAttrs>([&](HandledAttrs a) {
261 return static_cast<const Impl *>(this)->handleRewrite(res->first, op, adaptor, rewriter, a);
262 }))),
263 ...);
264
265 return TS.Default([&](Attribute a) {
266 return handleDefaultRewrite(res->first, op, adaptor, rewriter, a);
267 });
268 }

292 LogicalResult handleRewrite(

295 APInt attrValue = a.getValue();
296 Type origResTy = op.getType();
297 if (FeltType ty = llvm::dyn_cast<FeltType>(origResTy)) {

300 );

310 // Treat 0 as false and any other value as true (but give a warning if it's not 1)
311 if (attrValue.isZero()) {
312 replaceOpWithNewOp<arith::ConstantIntOp>(rewriter, op, false, origResTy);
313 return success();
314 }
315 if (!attrValue.isOne()) {
316 Location opLoc = op.getLoc();
317 Diagnostic diag(opLoc, DiagnosticSeverity::Warning);
318 diag << "Interpreting non-zero value " << stringWithoutType(a) << " as true";
319 if (getContext()->shouldPrintOpOnDiagnostic()) {
320 diag.attachNote(opLoc) << "see current operation: " << *op;
321 }
322 diag.attachNote(UnknownLoc::get(getContext()))
323 << "when instantiating '" << StructDefOp::getOperationName() << "' parameter \"" << sym
324 << "\" for this call";
325 diagnostics.push_back(std::move(diag));
326 }
327 replaceOpWithNewOp<arith::ConstantIntOp>(rewriter, op, true, origResTy);
328 return success();
329 }
330 return op->emitOpError().append("unexpected result type ", origResTy);
331 }

335 ) const {
336 replaceOpWithNewOp<FeltConstantOp>(rewriter, op, a);

379template <bool AllowStructParams = true> AttrConcreteness classifyAttrConcreteness(Attribute a) {

440static std::optional<Attribute>
441evaluateExpr(TemplateExprOp exprOp, const DenseMap<Attribute, Attribute> &paramNameToConcrete) {
442 // Map from SSA value in the expr body to its concrete Attribute.
443 DenseMap<Value, Attribute> valueMap;
444 for (Operation &bodyOp : exprOp.getInitializerRegion().front()) {
445 if (auto yieldOp = llvm::dyn_cast<YieldOp>(bodyOp)) {
446 auto it = valueMap.find(yieldOp.getVal());
447 return it != valueMap.end() ? std::make_optional(it->second) : std::nullopt;
448 }
449
450 if (auto constReadOp = llvm::dyn_cast<ConstReadOp>(bodyOp)) {
451 auto it = paramNameToConcrete.find(constReadOp.getConstNameAttr());
452 if (it == paramNameToConcrete.end()) {
453 return std::nullopt; // a referenced param is not concrete
454 }
455 // If the attribute type is `FeltType` but it's stored as an IntegerAttr, promote to
456 // a `FeltConstAttr`.
457 Attribute val = it->second;
458 if (auto intAttr = llvm::dyn_cast<IntegerAttr>(val)) {
459 if (auto feltTy = llvm::dyn_cast<FeltType>(constReadOp.getResult().getType())) {
460 val = FeltConstAttr::get(bodyOp.getContext(), intAttr.getValue(), feltTy);
461 }
462 }
463 valueMap[constReadOp.getResult()] = val;
464 continue;
465 }
466
467 // Gather constant attributes for all operands.
468 SmallVector<Attribute> operandAttrs;
469 operandAttrs.reserve(bodyOp.getNumOperands());
470 for (Value operand : bodyOp.getOperands()) {
471 auto it = valueMap.find(operand);
472 if (it == valueMap.end()) {
473 return std::nullopt; // operand not known as a constant
474 }
475 operandAttrs.push_back(it->second);
476 }
477
478 // Try constant folding.
479 SmallVector<OpFoldResult> foldResults;
480 if (succeeded(bodyOp.fold(operandAttrs, foldResults)) &&
481 foldResults.size() == bodyOp.getNumResults()) {
482 for (auto [result, fr] : llvm::zip_equal(bodyOp.getResults(), foldResults)) {
483 if (Attribute a = llvm::dyn_cast<Attribute>(fr)) {
484 valueMap[result] = a;
485 } else {
486 return std::nullopt;
487 }
488 }
489 }
490 }
491 return std::nullopt; // no YieldOp found (shouldn't happen in a valid expr)
492}

497static void
498evaluateTemplateExprs(TemplateOp templateOp, DenseMap<Attribute, Attribute> &paramNameToConcrete) {
499 LLVM_DEBUG(
500 llvm::dbgs() << "[evaluateTemplateExprs] before: " << debug::toStringList(paramNameToConcrete)
501 << '\n'

503 for (TemplateExprOp exprOp : templateOp.getConstOps<TemplateExprOp>()) {

517 );
518}

534 class MappedTypeConverter : public TypeConverter {
535 StructType origTy;
536 StructType newTy;
537 const DenseMap<Attribute, Attribute> &paramNameToValue;
538
539 inline Attribute convertIfPossible(Attribute a) const {
540 auto res = this->paramNameToValue.find(a);
541 return (res != this->paramNameToValue.end()) ? res->second : a;
542 }
543
544 public:
545 MappedTypeConverter(
546 StructType originalType, StructType newType,
548 const DenseMap<Attribute, Attribute> &paramNameToInstantiatedValue
549 )
550 : TypeConverter(), origTy(originalType), newTy(newType),
551 paramNameToValue(paramNameToInstantiatedValue) {
552
553 addConversion([](Type inputTy) { return inputTy; });
554
555 addConversion([this](StructType inputTy) {
556 LLVM_DEBUG(llvm::dbgs() << "[MappedTypeConverter] convert " << inputTy << '\n');
557
558 // Check for replacement of the full type
559 if (inputTy == this->origTy) {
560 return this->newTy;
561 }
562 // Check for replacement of parameter symbol names with concrete values
563 if (ArrayAttr inputTyParams = inputTy.getParams()) {
564 SmallVector<Attribute> updated;
565 for (Attribute a : inputTyParams) {
566 if (TypeAttr ta = dyn_cast<TypeAttr>(a)) {
567 updated.push_back(TypeAttr::get(this->convertType(ta.getValue())));
568 } else {
569 updated.push_back(convertIfPossible(a));
570 }
571 }
572 return StructType::get(
573 inputTy.getNameRef(), ArrayAttr::get(inputTy.getContext(), updated)
574 );
575 }
576 // Otherwise, return the type unchanged
577 return inputTy;
578 });
579
580 addConversion([this](ArrayType inputTy) {
581 // Check for replacement of parameter symbol names with concrete values
582 ArrayRef<Attribute> dimSizes = inputTy.getDimensionSizes();
583 if (!dimSizes.empty()) {
584 SmallVector<Attribute> updated;
585 for (Attribute a : dimSizes) {
586 updated.push_back(convertIfPossible(a));
587 }
588 return ArrayType::get(this->convertType(inputTy.getElementType()), updated);
589 }
590 // Otherwise, return the type unchanged
591 return inputTy;
592 });
593
594 addConversion([this](TypeVarType inputTy) -> Type {
595 // Check for replacement of parameter symbol name with a concrete type
596 if (TypeAttr tyAttr = llvm::dyn_cast<TypeAttr>(convertIfPossible(inputTy.getNameRef()))) {
597 Type convertedType = tyAttr.getValue();
598 // Use the new type unless it contains a TypeVarType because a TypeVarType from a
599 // different struct references a parameter name from that other struct, not from the
600 // current struct so the reference would be invalid.
601 if (isConcreteType(convertedType)) {
602 return convertedType;
603 }
604 }
605 return inputTy;
606 });
607 }

653 FailureOr<StructType> genClone(StructType typeAtCaller, ArrayRef<Attribute> typeAtCallerParams) {
654 LLVM_DEBUG(llvm::dbgs() << "[StructCloner] attempting clone of " << typeAtCaller << '\n');

657 typeAtCaller.getDefinition(symTables, rootMod, reportMissing);

662 LLVM_DEBUG(llvm::dbgs() << "[StructCloner] found definition\n";);
663
664 StructDefOp origStruct = r->get();
665 StructType typeAtDef = origStruct.getType();
666 MLIRContext *ctx = origStruct.getContext();
667
668 // Map of StructDefOp parameter name to concrete Attribute at the current instantiation site.
669 DenseMap<Attribute, Attribute> paramNameToConcrete;
670 // List of concrete Attributes from the struct instantiation with `nullptr` at any positions
671 // where the original attribute from the current instantiation site was not concrete. This is
672 // used for generating the new struct name. See `BuildShortTypeString::from()`.
673 SmallVector<Attribute> attrsForInstantiatedNameSuffix;
674 // List of template const param names that must be preserved because they
675 // were not assigned concrete values at the current instantiation site.
676 SmallVector<Attribute> remainingNames;
677 // Reduced from `typeAtCallerParams` to contain only the non-concrete Attributes.
678 ArrayAttr reducedCallerParams = nullptr;
679 {
680 ArrayAttr paramNames = typeAtDef.getParams();
681
682 // pre-conditions
683 assert(!isNullOrEmpty(paramNames));
684 assert(paramNames.size() == typeAtCallerParams.size());
685
686 SmallVector<Attribute> nonConcreteParams;
687 for (size_t i = 0, e = paramNames.size(); i < e; ++i) {
688 Attribute next = typeAtCallerParams[i];
689 if (isConcreteAttr<false>(next)) {
690 paramNameToConcrete[paramNames[i]] = next;
691 attrsForInstantiatedNameSuffix.push_back(next);
692 } else {
693 remainingNames.push_back(paramNames[i]);
694 nonConcreteParams.push_back(next);
695 attrsForInstantiatedNameSuffix.push_back(nullptr);
696 }
697 }
698 // post-conditions
699 assert(remainingNames.size() == nonConcreteParams.size());
700 assert(attrsForInstantiatedNameSuffix.size() == paramNames.size());
701 assert(remainingNames.size() + paramNameToConcrete.size() == paramNames.size());
702
703 if (paramNameToConcrete.empty()) {
704 LLVM_DEBUG(llvm::dbgs() << "[StructCloner] skip: no concrete params \n");
705 return failure();
706 }
707 if (!remainingNames.empty()) {
708 reducedCallerParams = ArrayAttr::get(ctx, nonConcreteParams);
709 }
710 }
711
712 // This list will be used to build the new remote/external type.
713 SmallVector<FlatSymbolRefAttr> typeAtCallerSymPieces = getPieces(typeAtCaller.getNameRef());
714 typeAtCallerSymPieces.pop_back(); // drop struct name
715 // Name of template with instantiated parameter values.
716 std::string templateNameWithAttrs = BuildShortTypeString::from(
717 typeAtCallerSymPieces.back().getValue().str(), attrsForInstantiatedNameSuffix
718 );
719
720 // Get parent refs
721 TemplateOp parentTemplate = getParentOfType<TemplateOp>(origStruct);
722 assert(parentTemplate && "parameterized struct must be nested in a TemplateOp");
723 ModuleOp parentModule = getParentOfType<ModuleOp>(parentTemplate);
724 assert(parentModule && "TemplateOp must be nested in a ModuleOp");
725
726 // Evaluate any poly.expr symbols whose param dependencies are now concrete; add them to the
727 // map so ClonedBodyConstReadOpPattern can replace uses of those symbols too.
728 evaluateTemplateExprs(parentTemplate, paramNameToConcrete);
729
730 // Clone the original struct.
731 StructDefOp newStruct = origStruct.clone();
732 convertCalleesInPlace(newStruct, paramNameToConcrete);
733 if (remainingNames.empty()) { // FULL INSTANTIATION CASE
734 // Set name of the new struct by prepending its name with instantiated template name.
735 newStruct.setSymName(
736 (templateNameWithAttrs + mlir::Twine('_') + newStruct.getSymName()).str()
737 );
738 // Insert 'newStruct' into the parent ModuleOp of the original TemplateOp. Use the
739 // `SymbolTable::insert()` function so that the name will be made unique if necessary.
740 symTables.getSymbolTable(parentModule).insert(newStruct, Block::iterator(parentTemplate));
741 // Drop the old template name from the list.
742 typeAtCallerSymPieces.pop_back();
743 } else { // PARTIAL INSTANTIATION CASE
744 // Clone the template and set instantiated name.
745 TemplateOp newTemplate = parentTemplate.cloneWithoutRegions();
746 newTemplate.setSymName(templateNameWithAttrs);
747 assert(newTemplate->getNumRegions() > 0 && "region exists"); // it just doesn't have a block
748 newTemplate.getBodyRegion().emplaceBlock();
749
750 // Clone preserved const param/expr ops.
751 for (Attribute name : remainingNames) {
752 FlatSymbolRefAttr nameSym = llvm::dyn_cast<FlatSymbolRefAttr>(name);
753 assert(nameSym && "expected FlatSymbolRefAttr");
754
755 Operation *symOp = symTables.getSymbolTable(parentTemplate).lookup(nameSym.getAttr());
756 assert(symOp && "symbol must exist");
757 newTemplate.insert(newTemplate.begin(), symOp->clone());
758 }
759
760 // Insert the struct into the template and the template into the module. Use the
761 // `SymbolTable::insert()` function so that the name will be made unique if necessary.
762 symTables.getSymbolTable(newTemplate).insert(newStruct);
763 symTables.getSymbolTable(parentModule).insert(newTemplate, Block::iterator(parentTemplate));
764
765 // Replace the old template name in the list with the new one (get template name after
766 // symbol table insertion since it may be modified to make it unique).
767 typeAtCallerSymPieces.back() = FlatSymbolRefAttr::get(newTemplate.getSymNameAttr());
768 }
769
770 // Retrieve the new type AFTER inserting since the struct name may be appended to make
771 // it unique and use the remaining non-concrete parameters from the original type.
772 StructType newLocalType = newStruct.getType(reducedCallerParams);
773 typeAtCallerSymPieces.push_back(
774 FlatSymbolRefAttr::get(newLocalType.getNameRef().getLeafReference())
775 );
776 StructType newRemoteType =
777 StructType::get(asSymbolRefAttr(typeAtCallerSymPieces), newLocalType.getParams());
778 LLVM_DEBUG({
779 llvm::dbgs() << "[StructCloner] original def type: " << typeAtDef << '\n';
780 llvm::dbgs() << "[StructCloner] cloned def type: " << newStruct.getType() << '\n';
781 llvm::dbgs() << "[StructCloner] original remote type: " << typeAtCaller << '\n';
782 llvm::dbgs() << "[StructCloner] cloned local type: " << newLocalType << '\n';
783 llvm::dbgs() << "[StructCloner] cloned remote type: " << newRemoteType << '\n';
784 });
785
786 // Within the new struct, replace all references to the original StructType (i.e., the
787 // locally-parameterized version) with the new locally-parameterized StructType,
788 // and replace all uses of the removed struct parameters with the concrete values.
789 MappedTypeConverter tyConv(typeAtDef, newStruct.getType(), paramNameToConcrete);
790 ConversionTarget target =
791 newConverterDefinedTarget<EmitEqualityOp>(tyConv, ctx, tableOffsetIsntSymbol);
792 target.addDynamicallyLegalOp<ConstReadOp>([&paramNameToConcrete](ConstReadOp op) {
793 // Legal if it's not in the map of concrete attribute instantiations
794 return !paramNameToConcrete.contains(op.getConstNameAttr());
795 });
796
797 RewritePatternSet patterns = newGeneralRewritePatternSet<EmitEqualityOp>(tyConv, ctx, target);
798 patterns.add<ClonedBodyConstReadOpPattern>(
799 tyConv, ctx, paramNameToConcrete, tracker_.delayedDiagnosticSet(newLocalType)
800 );
801 patterns.add<ClonedStructMemberReadOpPattern>(tyConv, ctx, paramNameToConcrete);
802 if (failed(applyFullConversion(newStruct, target, std::move(patterns)))) {
803 LLVM_DEBUG(llvm::dbgs() << "[StructCloner] instantiating body of struct failed \n");
804 return failure();
805 }
806 return newRemoteType;
807 }
808
809public:
810 StructCloner(ConversionTracker &tracker, ModuleOp root)
811 : tracker_(tracker), rootMod(root), symTables() {}
812
813 FailureOr<StructType> createInstantiatedClone(StructType orig) {
814 LLVM_DEBUG(llvm::dbgs() << "[StructCloner] orig: " << orig << '\n');
815 if (ArrayAttr params = orig.getParams()) {
816 return genClone(orig, params.getValue());
817 }
818 LLVM_DEBUG(llvm::dbgs() << "[StructCloner] skip: nullptr for params \n");
819 return failure();
820 }

908 calleeAttr = appendLeaf(newStTy.getNameRef(), calleeAttr.getLeafReference());

912 LLVM_DEBUG(llvm::dbgs() << "[CallStructFuncPattern] replaced " << op);
913 CallOp newOp = replaceOpWithNewOp<CallOp>(
914 rewriter, op, newResultTypes, calleeAttr, adapter.getMapOperands(),
915 op.getNumDimsPerMapAttr(), adapter.getArgOperands()
916 );

918 LLVM_DEBUG(llvm::dbgs() << " with " << newOp << '\n');
919 return success();
920 }

1011class FuncInstTypeConverter : public TypeConverter {
1012 DenseMap<Attribute, Attribute> paramNameToValue;
1013
1014 Attribute convertIfPossible(Attribute a) const {
1015 auto res = paramNameToValue.find(a);
1016 return (res != paramNameToValue.end()) ? res->second : a;
1017 }
1018
1019public:
1020 explicit FuncInstTypeConverter(DenseMap<Attribute, Attribute> paramNameToConcrete)
1021 : TypeConverter(), paramNameToValue(std::move(paramNameToConcrete)) {
1022 addConversion([](Type t) { return t; });
1023
1024 addConversion([this](TypeVarType inputTy) -> Type {
1025 if (TypeAttr tyAttr = llvm::dyn_cast<TypeAttr>(convertIfPossible(inputTy.getNameRef()))) {
1026 Type convertedType = tyAttr.getValue();
1027 if (isConcreteType(convertedType)) {
1028 return convertedType;
1029 }
1030 }
1031 return inputTy;
1032 });
1033
1034 addConversion([this](ArrayType inputTy) {
1035 SmallVector<Attribute> updated;
1036 bool changed = false;
1037 for (Attribute a : inputTy.getDimensionSizes()) {
1038 Attribute converted = convertIfPossible(a);
1039 updated.push_back(converted);
1040 if (converted != a) {
1041 changed = true;
1042 }
1043 }
1044 Type newElemTy = this->convertType(inputTy.getElementType());
1045 if (!changed && newElemTy == inputTy.getElementType()) {
1046 return inputTy;
1047 }
1048 return flattenInstantiatedArrayType(
1049 inputTy.cloneWith(inputTy.getElementType(), updated), newElemTy
1050 );
1051 });
1052
1053 addConversion([this](StructType inputTy) -> StructType {
1054 if (ArrayAttr params = inputTy.getParams()) {
1055 SmallVector<Attribute> updated;
1056 bool changed = false;
1057 for (Attribute a : params) {
1058 if (TypeAttr ta = dyn_cast<TypeAttr>(a)) {
1059 Type newTy = this->convertType(ta.getValue());
1060 if (newTy != ta.getValue()) {
1061 updated.push_back(TypeAttr::get(newTy));
1062 changed = true;
1063 continue;
1064 }
1065 } else {
1066 Attribute converted = convertIfPossible(a);
1067 if (converted != a) {
1068 updated.push_back(converted);
1069 changed = true;
1070 continue;
1071 }
1072 }
1073 updated.push_back(a);
1074 }
1075 if (changed) {
1076 return StructType::get(
1077 inputTy.getNameRef(), ArrayAttr::get(inputTy.getContext(), updated)
1078 );
1079 }
1080 }
1081 return inputTy;
1082 });
1083 }
1084

1086 if (TypeAttr tyAttr = llvm::dyn_cast<TypeAttr>(attr)) {
1087 Type convertedTy = convertType(tyAttr.getValue());
1088 if (convertedTy != tyAttr.getValue()) {
1089 return TypeAttr::get(convertedTy);
1090 }
1091 }
1092 return convertIfPossible(attr);
1093 }
1094
1095 bool containsParam(Attribute nameAttr) const { return paramNameToValue.contains(nameAttr); }
1096 const DenseMap<Attribute, Attribute> &getParamMap() const { return paramNameToValue; }

1126class WildcardTypeBodyInferer final {
1127 SymbolTableCollection &symTables_;
1128 const DenseMap<Attribute, Attribute> &paramNameToConcrete_;
1129 SmallVector<std::pair<Operation *, FlatSymbolRefAttr>> activeInferences_;
1130
1131public:
1132 WildcardTypeBodyInferer(
1133 SymbolTableCollection &symTables, const DenseMap<Attribute, Attribute> &paramNameToConcrete
1134 )
1135 : symTables_(symTables), paramNameToConcrete_(paramNameToConcrete) {}
1136
1137 std::optional<Attribute> infer(FuncDefOp func, FlatSymbolRefAttr paramName) {
1138 if (llvm::any_of(activeInferences_, [&](const auto &e) {
1139 return e.first == func.getOperation() && e.second == paramName;
1140 })) {
1141 return std::nullopt;
1142 }
1143 activeInferences_.emplace_back(func.getOperation(), paramName);
1144
1145 FuncInstTypeConverter tyConv((paramNameToConcrete_));
1146 std::optional<Attribute> inferred;
1147 bool ambiguous = false;
1148
1149 // Record a concrete candidate unless it conflicts with an earlier one, in which
1150 // case the wildcard is treated as ambiguous and left unresolved.
1151 auto noteCandidate = [&inferred, &ambiguous](Attribute candidate) {
1152 if (!candidate || !isConcreteAttr(candidate)) {
1153 return WalkResult::advance();
1154 }
1155 if (!inferred.has_value()) {
1156 inferred = candidate;
1157 return WalkResult::advance();
1158 }
1159 if (*inferred != candidate) {
1160 ambiguous = true;
1161 return WalkResult::interrupt();
1162 }
1163 return WalkResult::advance();
1164 };
1165
1166 WalkResult walkResult = func.walk([&](Operation *bodyOp) {
1167 if (auto castOp = llvm::dyn_cast<UnifiableCastOp>(bodyOp)) {
1168 Type inputTy = tyConv.convertType(castOp.getInput().getType());
1169 Type resultTy = tyConv.convertType(castOp.getResult().getType());
1170 if (auto inputTvar = llvm::dyn_cast<TypeVarType>(inputTy);
1171 inputTvar && inputTvar.getNameRef() == paramName && isConcreteType(resultTy)) {
1172 return noteCandidate(TypeAttr::get(resultTy));
1173 }
1174 if (auto resultTvar = llvm::dyn_cast<TypeVarType>(resultTy);
1175 resultTvar && resultTvar.getNameRef() == paramName && isConcreteType(inputTy)) {
1176 return noteCandidate(TypeAttr::get(inputTy));
1177 }
1178 return WalkResult::advance();
1179 }
1180
1181 auto nestedCall = llvm::dyn_cast<CallOp>(bodyOp);
1182 if (!nestedCall) {
1183 return WalkResult::advance();
1184 }
1185
1186 FailureOr<SymbolLookupResult<FuncDefOp>> nestedTgtOpt =
1187 nestedCall.getCalleeTarget(symTables_);
1188 if (failed(nestedTgtOpt)) {
1189 return WalkResult::advance();
1190 }
1191 FuncDefOp nestedTgt = nestedTgtOpt->get();
1192 auto nestedTemplate = llvm::dyn_cast<TemplateOp>(nestedTgt->getParentOp());
1193 if (!nestedTemplate) {
1194 return WalkResult::advance();
1195 }
1196
1197 TypeRange nestedResultTypes = nestedTgt.getFunctionType().getResults();
1198 for (auto [result, nestedResultTy] :
1199 llvm::zip_equal(nestedCall.getResults(), nestedResultTypes)) {
1200 Type convertedResultTy = tyConv.convertType(result.getType());
1201 auto resultTvar = llvm::dyn_cast<TypeVarType>(convertedResultTy);
1202 auto nestedTvar = llvm::dyn_cast<TypeVarType>(nestedResultTy);
1203 if (!resultTvar || !nestedTvar || resultTvar.getNameRef() != paramName) {
1204 continue;
1205 }
1206 if (std::optional<Attribute> candidate = inferFromExplicitNestedCallParams(
1207 nestedCall, nestedTemplate, nestedTvar.getNameRef(), tyConv
1208 )) {
1209 WalkResult candidateResult = noteCandidate(*candidate);
1210 if (candidateResult.wasInterrupted()) {
1211 return candidateResult;
1212 }
1213 continue;
1214 }
1215 if (std::optional<Attribute> candidate = infer(nestedTgt, nestedTvar.getNameRef())) {
1216 WalkResult candidateResult = noteCandidate(*candidate);
1217 if (candidateResult.wasInterrupted()) {
1218 return candidateResult;
1219 }
1220 }
1221 }
1222 return WalkResult::advance();
1223 });
1224
1225 activeInferences_.pop_back();
1226 if (ambiguous || (walkResult.wasInterrupted() && !inferred.has_value())) {
1227 return std::nullopt;
1228 }
1229 return inferred;
1230 }
1231
1232private:
1233 std::optional<Attribute> inferFromExplicitNestedCallParams(
1234 CallOp nestedCall, TemplateOp nestedTemplate, FlatSymbolRefAttr nestedParamName,
1235 const FuncInstTypeConverter &tyConv
1236 ) const {
1237 ArrayAttr nestedCallParams = nestedCall.getTemplateParamsAttr();
1238 if (isNullOrEmpty(nestedCallParams)) {
1239 return std::nullopt;
1240 }
1241
1242 for (auto [paramOp, attr] :
1243 llvm::zip_equal(nestedTemplate.getConstOps<TemplateParamOp>(), nestedCallParams)) {
1244 auto paramName = FlatSymbolRefAttr::get(paramOp.getSymNameAttr());
1245 if (paramName != nestedParamName) {
1246 continue;
1247 }
1248 Attribute convertedAttr = tyConv.convertAttr(attr);
1249 return isConcreteAttr(convertedAttr) ? std::make_optional(convertedAttr) : std::nullopt;
1250 }
1251 return std::nullopt;
1252 }

1266static InstantiationLayout buildInstantiationLayout(
1267 TemplateOp parentTemplate, ArrayAttr callParams,
1268 const DenseMap<Attribute, Attribute> &paramNameToConcrete
1269) {
1270 SmallVector<Attribute> remainingNames;
1271 SmallVector<Attribute> attrsForInstantiatedNameSuffix;
1272 for (Attribute paramName : parentTemplate.getConstNames<TemplateParamOp>()) {
1273 auto it = paramNameToConcrete.find(paramName);
1274 if (it != paramNameToConcrete.end()) {
1275 attrsForInstantiatedNameSuffix.push_back(it->second);
1276 } else {
1277 attrsForInstantiatedNameSuffix.push_back(nullptr);
1278 remainingNames.push_back(paramName);
1279 }
1280 }
1281
1282 ArrayAttr rewrittenCallParams = nullptr;
1283 if (!isNullOrEmpty(callParams) && !remainingNames.empty()) {
1284 SmallVector<Attribute> remainingCallParams;
1285 for (auto [paramOp, attr] :

1289 remainingCallParams.push_back(attr);
1290 }
1291 }
1292 rewrittenCallParams = ArrayAttr::get(parentTemplate.getContext(), remainingCallParams);
1293 }
1294
1295 return {
1296 std::move(remainingNames),
1297 BuildShortTypeString::from(parentTemplate.getSymName().str(), attrsForInstantiatedNameSuffix),
1298 rewrittenCallParams,
1299 };

1373class InstantiateFuncAtCallOp final : public OpRewritePattern<CallOp> {
1374 ConversionTracker &tracker_;
1375
1376public:
1377 InstantiateFuncAtCallOp(MLIRContext *ctx, ConversionTracker &tracker)
1378 : OpRewritePattern<CallOp>(ctx), tracker_(tracker) {}
1379
1380 LogicalResult matchAndRewrite(CallOp op, PatternRewriter &rewriter) const override {
1381 LLVM_DEBUG(llvm::dbgs() << "[InstantiateFuncAtCallOp] op: " << op << '\n');
1382
1383 if (calleeReferencesTemplateParam(op)) {
1384 return failure();
1385 }
1386
1387 // Lookup callee target function
1388 SymbolTableCollection symTables;
1389 FailureOr<SymbolLookupResult<FuncDefOp>> callTgtOpt = op.getCalleeTarget(symTables);
1390 if (failed(callTgtOpt)) {
1391 return rewriter.notifyMatchFailure(op, [](Diagnostic &diag) {
1392 diag << "could not find target function for call";
1393 });
1394 }
1395 FuncDefOp callTgt = callTgtOpt->get();
1396
1397 // Check if callee is within a TemplateOp
1398 TemplateOp parentTemplate = llvm::dyn_cast<TemplateOp>(callTgt->getParentOp());
1399 if (!parentTemplate) {
1400 return failure(); // nothing to do if not parameterized
1401 }
1402 LLVM_DEBUG(
1403 llvm::dbgs() << "[InstantiateFuncAtCallOp] target function in template "
1404 << parentTemplate.getSymName() << '\n'
1405 );
1406
1407 // Perform type unification with tracking to infer the instantiated type(s). Even though
1408 // `CallOp` verification already checked that caller and callee types unify, the progress of
1409 // instantiation so far may have brought together a chain of calls across templates where each
1410 // individual unification check passed due to permissive type variables and/or symbols in the
1411 // middle but the overall chain does not unify. Hence, this unification may fail and should
1412 // produce a meaningful error message if it does.
1413 // See: `test/Transforms/Flattening/instantiate_funcs_fail.llzk`
1414 FailureOr<UnificationMap> unifyResult = unifyTypeSignature(op, callTgt, rewriter);
1415 if (failed(unifyResult)) {
1416 return failure();
1417 }
1418 LLVM_DEBUG(
1419 llvm::dbgs() << "[InstantiateFuncAtCallOp] unifications of types: "
1420 << debug::toStringList(unifyResult.value()) << '\n'
1421 );
1422
1423 // Maps template parameter symbols to the instantiation value at the call site.
1424 DenseMap<Attribute, Attribute> paramNameToConcrete;
1425 if (failed(collectConcreteTemplateParams(
1426 op, rewriter, symTables, callTgt, parentTemplate, unifyResult.value(),
1427 paramNameToConcrete
1428 ))) {
1429 return failure();
1430 }
1431
1432 if (paramNameToConcrete.empty()) {
1433 LLVM_DEBUG(llvm::dbgs() << "[InstantiateFuncAtCallOp] skip: no concrete params\n");
1434 return failure();
1435 }
1436
1437 evaluateTemplateExprs(parentTemplate, paramNameToConcrete);
1438
1439 InstantiationLayout layout =
1440 buildInstantiationLayout(parentTemplate, op.getTemplateParamsAttr(), paramNameToConcrete);
1441 ModuleOp parentModule = getParentOfType<ModuleOp>(parentTemplate);
1442 assert(parentModule && "TemplateOp must be nested in a ModuleOp");
1443
1444 SymbolRefAttr originalCalleeAttr = op.getCalleeAttr();
1445 FailureOr<SymbolRefAttr> newCalleeAttr =
1446 layout.remainingNames.empty()
1447 ? instantiateFully(
1448 op, rewriter, symTables, callTgt, parentTemplate, parentModule,
1449 layout.templateNameWithAttrs, paramNameToConcrete
1450 )
1451 : instantiatePartially(
1452 op, rewriter, symTables, callTgt, parentTemplate, parentModule, layout,
1453 paramNameToConcrete
1454 );
1455 if (failed(newCalleeAttr)) {
1456 return failure();
1457 }
1458
1459 tracker_.recordInstantiation(originalCalleeAttr);
1460
1461 // Update the CallOp to point to the instantiated function and mark the module as modified.
1462 rewriter.modifyOpInPlace(op, [&op, &newCalleeAttr, &layout]() {
1463 LLVM_DEBUG({
1464 llvm::dbgs() << "[InstantiateFuncAtCallOp] updating callee from " << op.getCalleeAttr()
1465 << " to " << *newCalleeAttr << '\n';
1466 });
1467 op.setCalleeAttr(*newCalleeAttr);
1468 op.setTemplateParamsAttr(layout.rewrittenCallParams);
1469 });
1470 tracker_.updateModifiedFlag(true);
1471 return success();
1472 }
1473
1474private:
1477 static FailureOr<UnificationMap>
1478 unifyTypeSignature(CallOp op, FuncDefOp callTgt, PatternRewriter &rewriter) {
1479 FailureOr<UnificationMap> unifyResult = op.unifyTypeSignature(callTgt.getFunctionType());
1480 if (succeeded(unifyResult)) {
1481 return unifyResult;
1482 }
1483 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
1484 diag.append("target function type does not unify with call type ")
1485 .append(op.getTypeSignature())
1486 .attachNote(callTgt.getLoc())
1487 .append("target function declared here");
1488 });
1489 }

1493 static LogicalResult collectConcreteTemplateParams(
1494 CallOp op, PatternRewriter &rewriter, SymbolTableCollection &symTables, FuncDefOp callTgt,
1495 TemplateOp parentTemplate, const UnificationMap &unifyResult,
1496 DenseMap<Attribute, Attribute> &paramNameToConcrete
1497 ) {
1498 auto realParams = parentTemplate.getConstOps<TemplateParamOp>();
1499 ArrayAttr callParams = op.getTemplateParamsAttr();
1500 LLVM_DEBUG(
1501 llvm::dbgs() << "[InstantiateFuncAtCallOp] TemplateParamsAttr: " << callParams << '\n'
1502 );
1503
1504 auto recordConcreteParam = [&](FlatSymbolRefAttr paramName, TemplateParamOp paramOp,
1505 Attribute concreteValue) {
1506 if (failed(op.verifyTemplateParamCompatibility(concreteValue, paramOp))) {
1507 return failIncompatibleInferredParam(op, rewriter, paramName, paramOp);
1508 }
1509 paramNameToConcrete[paramName] = concreteValue;
1510 return success();
1511 };
1512
1513 // If there's no template instantiation list, must infer all template parameters.
1514 if (isNullOrEmpty(callParams)) {
1515 for (auto paramOp : realParams) {
1516 auto paramName = FlatSymbolRefAttr::get(paramOp.getSymNameAttr());
1517 auto inferredValOpt = inferUnifiedParam(unifyResult, paramName);
1518 if (!inferredValOpt.has_value()) {
1519 LLVM_DEBUG(
1520 llvm::dbgs() << "[InstantiateFuncAtCallOp] unification for param '" << paramName
1521 << "': not found\n"
1522 );
1523 continue;
1524 }
1525 Attribute inferredVal = *inferredValOpt;
1526 LLVM_DEBUG(
1527 llvm::dbgs() << "[InstantiateFuncAtCallOp] inferredVal: " << inferredVal << '\n'
1528 );
1529 if (!isConcreteAttr(inferredVal)) {
1530 LLVM_DEBUG(
1531 llvm::dbgs() << "[InstantiateFuncAtCallOp] unification for param '" << paramName
1532 << "': not concrete, " << inferredVal << '\n'
1533 );
1534 continue;
1535 }
1536 if (failed(recordConcreteParam(paramName, paramOp, inferredVal))) {
1537 return failure();
1538 }
1539 }
1540 return success();
1541 }
1542
1543 // As stated earlier, need to run the verification checks again to ensure the
1544 // instantiation is valid, except for the size check because that cannot change.
1545 assert((callParams.size() == llvm::range_size(realParams)) && "per CallOpVerifier");
1546 if (failed(op.verifyTemplateParamCompatibility(realParams))) {
1547 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
1548 diag.append("incompatible with specified param type(s)");
1549 });
1550 }
1551 if (failed(op.verifyTemplateParamsMatchInferred(realParams, unifyResult))) {
1552 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
1553 diag.append("incompatible with inferred param value(s)");
1554 });
1555 }
1556
1557 // When template parameters are specified on the CallOp, use them as the source of truth
1558 // for concrete arguments, then infer wildcard parameters against the full explicit map.
1559 SmallVector<std::pair<TemplateParamOp, FlatSymbolRefAttr>> wildcardParams;
1560 for (auto [paramOp, attr] : llvm::zip_equal(realParams, callParams.getValue())) {
1561 auto paramName = FlatSymbolRefAttr::get(paramOp.getSymNameAttr());
1562 AttrConcreteness classification = classifyAttrConcreteness(attr);
1563 if (classification == AttrConcreteness::Concrete) {
1564 paramNameToConcrete[paramName] = attr;
1565 continue;
1566 }
1567
1568 if (classification == AttrConcreteness::NonConcrete) {
1569 LLVM_DEBUG(
1570 llvm::dbgs() << "[InstantiateFuncAtCallOp] unification for param '" << paramName
1571 << "': not concrete, " << attr << '\n'
1572 );
1573 continue;
1574 }
1575 wildcardParams.emplace_back(paramOp, paramName);
1576 }
1577
1578 WildcardTypeBodyInferer bodyInferer(symTables, paramNameToConcrete);
1579 for (auto [paramOp, paramName] : wildcardParams) {
1580 auto inferredValOpt = inferUnifiedParam(unifyResult, paramName);
1581 if (inferredValOpt.has_value() && isConcreteAttr(*inferredValOpt)) {
1582 LLVM_DEBUG(
1583 llvm::dbgs() << "[InstantiateFuncAtCallOp] inferredVal: " << *inferredValOpt << '\n'
1584 );
1585 if (failed(recordConcreteParam(paramName, paramOp, *inferredValOpt))) {
1586 return failure();
1587 }
1588 continue;
1589 }
1590
1591 inferredValOpt = bodyInferer.infer(callTgt, paramName);
1592 if (inferredValOpt.has_value() && isConcreteAttr(*inferredValOpt)) {
1593 LLVM_DEBUG(
1594 llvm::dbgs() << "[InstantiateFuncAtCallOp] body-inferred value for param '"
1595 << paramName << "': " << *inferredValOpt << '\n'
1596 );
1597 if (failed(recordConcreteParam(paramName, paramOp, *inferredValOpt))) {
1598 return failure();
1599 }
1600 }
1601 }
1602 return success();
1603 }

1607 static FailureOr<SymbolRefAttr> instantiateFully(
1608 CallOp op, PatternRewriter &rewriter, SymbolTableCollection &symTables, FuncDefOp callTgt,
1609 TemplateOp parentTemplate, ModuleOp parentModule, StringRef templateNameWithAttrs,
1610 const DenseMap<Attribute, Attribute> &paramNameToConcrete
1611 ) {
1612 MLIRContext *ctx = op.getContext();
1613 std::string newFuncName =
1614 (mlir::Twine(templateNameWithAttrs) + "_" + callTgt.getSymName()).str();
1615 StringRef actualNewFuncName = newFuncName;
1616 if (!symTables.getSymbolTable(parentModule).lookup(newFuncName)) {
1617 FuncDefOp newFunc = callTgt.clone();
1618 newFunc.setSymName(newFuncName);
1619 convertCalleesInPlace(newFunc, paramNameToConcrete);
1620 // Insert before the TemplateOp; symbol table may adjust the name to ensure uniqueness.
1621 symTables.getSymbolTable(parentModule).insert(newFunc, Block::iterator(parentTemplate));
1622 actualNewFuncName = newFunc.getSymName();
1623 LLVM_DEBUG(
1624 llvm::dbgs() << "[InstantiateFuncAtCallOp] created full instantiation function: "
1625 << actualNewFuncName << '\n'
1626 );
1627 if (failed(applyBodyConversions(op, newFunc, paramNameToConcrete))) {
1628 LLVM_DEBUG(
1629 llvm::dbgs() << "[InstantiateFuncAtCallOp] body conversion failed for "
1630 << actualNewFuncName << '\n'
1631 );
1632 newFunc->erase();
1633 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
1634 diag.append("failure while creating instantiated function '", actualNewFuncName, '\'');
1635 });
1636 }
1637 } else {
1638 LLVM_DEBUG(
1639 llvm::dbgs() << "[InstantiateFuncAtCallOp] reusing full instantiation function: "
1640 << actualNewFuncName << '\n'
1641 );
1642 }
1643
1644 // Callee: drop template & original function names, add the new module-level function name.
1645 // Original: @[prefix...]::@TemplateName::@funcName
1646 // New: @[prefix...]::@newFuncName
1647 SmallVector<FlatSymbolRefAttr> symPieces = getPieces(op.getCalleeAttr());
1648 assert(symPieces.size() >= 2 && "callee must include at least template and function names");
1649 symPieces.pop_back(); // remove original function name
1650 symPieces.pop_back(); // remove template name
1651 symPieces.push_back(FlatSymbolRefAttr::get(StringAttr::get(ctx, actualNewFuncName)));
1652 return asSymbolRefAttr(symPieces);
1653 }

1659 static FailureOr<SymbolRefAttr> instantiatePartially(
1660 CallOp op, PatternRewriter &rewriter, SymbolTableCollection &symTables, FuncDefOp callTgt,
1661 TemplateOp parentTemplate, ModuleOp parentModule, const InstantiationLayout &layout,
1662 const DenseMap<Attribute, Attribute> &paramNameToConcrete
1663 ) {
1664 TemplateOp newTemplate;
1665 if (Operation *existing =
1666 symTables.getSymbolTable(parentModule).lookup(layout.templateNameWithAttrs)) {
1667 newTemplate = llvm::dyn_cast<TemplateOp>(existing);
1668 }
1669 if (!newTemplate) {
1670 newTemplate = parentTemplate.cloneWithoutRegions();
1671 newTemplate.setSymName(layout.templateNameWithAttrs);
1672 assert(newTemplate->getNumRegions() > 0 && "region exists");
1673 newTemplate.getBodyRegion().emplaceBlock();
1674
1675 Block &newTemplateBody = newTemplate.getBodyRegion().front();
1676 for (Attribute name : layout.remainingNames) {
1677 FlatSymbolRefAttr nameSym = llvm::cast<FlatSymbolRefAttr>(name);
1678 Operation *paramOp = symTables.getSymbolTable(parentTemplate).lookup(nameSym.getAttr());
1679 assert(paramOp && "symbol must exist");
1680 newTemplateBody.push_back(paramOp->clone());
1681 }
1682
1683 // Clone and partially convert the function (concretize only the concrete params).
1684 FuncDefOp newFunc = callTgt.clone();
1685 convertCalleesInPlace(newFunc, paramNameToConcrete);
1686
1687 // Insert before body conversion so nested concrete callees verify from the root module. Use
1688 // the `SymbolTable::insert()` function so that the name will be made unique if necessary.
1689 symTables.getSymbolTable(newTemplate).insert(newFunc);
1690 symTables.getSymbolTable(parentModule).insert(newTemplate, Block::iterator(parentTemplate));
1691 if (failed(applyBodyConversions(op, newFunc, paramNameToConcrete))) {
1692 StringRef newFuncName = newFunc.getSymName();
1693 LLVM_DEBUG(
1694 llvm::dbgs() << "[InstantiateFuncAtCallOp] body conversion failed for " << newFuncName
1695 << '\n'
1696 );
1697 newTemplate->erase();
1698 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
1699 diag.append("failure while creating instantiated function '", newFuncName, '\'');
1700 });
1701 }
1702
1703 LLVM_DEBUG(
1704 llvm::dbgs() << "[InstantiateFuncAtCallOp] created partial instantiation template: "
1705 << newTemplate.getSymName() << '\n'
1706 );
1707 } else {
1708 LLVM_DEBUG(
1709 llvm::dbgs() << "[InstantiateFuncAtCallOp] reusing partial instantiation template: "
1710 << newTemplate.getSymName() << '\n'
1711 );
1712 }
1713
1714 // Callee: replace old template name with new template name, keep the function name.
1715 // Original: @[prefix...]::@TemplateName::@funcName
1716 // New: @[prefix...]::@newTemplateName::@funcName
1717 SmallVector<FlatSymbolRefAttr> symPieces = getPieces(op.getCalleeAttr());
1718 assert(symPieces.size() >= 2 && "callee must include at least template and function names");
1719 symPieces.pop_back(); // remove original function name (will be re-appended)
1720 symPieces.pop_back(); // remove original template name
1721 symPieces.push_back(FlatSymbolRefAttr::get(newTemplate.getSymNameAttr()));
1722 symPieces.push_back(FlatSymbolRefAttr::get(callTgt.getSymNameAttr()));
1723 return asSymbolRefAttr(symPieces);
1724 }
1725};

1728 MLIRContext *ctx = modOp.getContext();

1731 MatchFailureListener failureListener;

1800struct AffineMapFolder {
1801 struct Input {
1802 OperandRangeRange mapOpGroups;
1803 DenseI32ArrayAttr dimsPerGroup;
1804 ArrayRef<Attribute> paramsOfStructTy;
1805 };
1806
1807 struct Output {
1808 SmallVector<SmallVector<Value>> mapOpGroups;
1809 SmallVector<int32_t> dimsPerGroup;
1810 SmallVector<Attribute> paramsOfStructTy;
1811 };
1812
1813 static inline SmallVector<ValueRange> getConvertedMapOpGroups(Output out) {
1814 return llvm::map_to_vector(out.mapOpGroups, [](const SmallVector<Value> &grp) {
1815 return ValueRange(grp);
1816 });
1817 }
1818
1819 static LogicalResult
1820 fold(PatternRewriter &rewriter, const Input &in, Output &out, Operation *op, const char *aspect) {
1821 if (in.mapOpGroups.empty()) {
1822 // No affine map operands so nothing to do
1823 return failure();
1824 }
1825
1826 assert(in.mapOpGroups.size() <= in.paramsOfStructTy.size());
1827 assert(std::cmp_equal(in.mapOpGroups.size(), in.dimsPerGroup.size()));
1828
1829 size_t idx = 0; // index in `mapOpGroups`, i.e., the number of AffineMapAttr encountered
1830 for (Attribute sizeAttr : in.paramsOfStructTy) {
1831 if (AffineMapAttr m = dyn_cast<AffineMapAttr>(sizeAttr)) {
1832 ValueRange currMapOps = in.mapOpGroups[idx++];
1833 LLVM_DEBUG(
1834 llvm::dbgs() << "[AffineMapFolder] currMapOps: " << debug::toStringList(currMapOps)
1835 << '\n'
1836 );
1837 SmallVector<OpFoldResult> currMapOpsCast = getAsOpFoldResult(currMapOps);
1838 LLVM_DEBUG(
1839 llvm::dbgs() << "[AffineMapFolder] currMapOps as fold results: "
1840 << debug::toStringList(currMapOpsCast) << '\n'
1841 );
1842 if (auto constOps = Step4_InstantiateAffineMaps::getConstantIntValues(currMapOpsCast)) {
1843 SmallVector<Attribute> result;
1844 bool hasPoison = false; // indicates divide by 0 or mod by <1
1845 auto constAttrs = llvm::map_to_vector(*constOps, [&rewriter](int64_t v) -> Attribute {
1846 return rewriter.getIndexAttr(v);
1847 });
1848 LogicalResult foldResult = m.getAffineMap().constantFold(constAttrs, result, &hasPoison);
1849 if (hasPoison) {
1850 // Diagnostic remark: could be removed for release builds if too noisy
1851 op->emitRemark()
1852 .append(
1853 "Cannot fold affine_map for ", aspect, ' ', out.paramsOfStructTy.size(),
1854 " due to divide by 0 or modulus with negative divisor"
1855 )
1856 .report();
1857 return failure();
1858 }
1859 if (failed(foldResult)) {
1860 // Diagnostic remark: could be removed for release builds if too noisy
1861 op->emitRemark()
1862 .append(
1863 "Folding affine_map for ", aspect, ' ', out.paramsOfStructTy.size(), " failed"
1864 )
1865 .report();
1866 return failure();
1867 }
1868 if (result.size() != 1) {
1869 // Diagnostic remark: could be removed for release builds if too noisy
1870 op->emitRemark()
1871 .append(
1872 "Folding affine_map for ", aspect, ' ', out.paramsOfStructTy.size(),
1873 " produced ", result.size(), " results but expected 1"
1874 )
1875 .report();
1876 return failure();
1877 }
1878 assert(!llvm::isa<AffineMapAttr>(result[0]) && "not converted");
1879 out.paramsOfStructTy.push_back(result[0]);
1880 continue;
1881 }

1939class InstantiateAtCallOpCompute final : public OpRewritePattern<CallOp> {
1940 ConversionTracker &tracker_;
1941
1942public:
1943 InstantiateAtCallOpCompute(MLIRContext *ctx, ConversionTracker &tracker)
1944 : OpRewritePattern(ctx), tracker_(tracker) {}
1945
1946 LogicalResult matchAndRewrite(CallOp op, PatternRewriter &rewriter) const override {
1947 if (!op.calleeIsStructCompute()) {
1948 // this pattern only applies when the callee is "compute()" within a struct
1949 return failure();
1950 }
1951 LLVM_DEBUG(llvm::dbgs() << "[InstantiateAtCallOpCompute] target: " << op.getCallee() << '\n');
1952 StructType oldRetTy = op.getSingleResultTypeOfCompute();
1953 LLVM_DEBUG(llvm::dbgs() << "[InstantiateAtCallOpCompute] oldRetTy: " << oldRetTy << '\n');
1954 ArrayAttr params = oldRetTy.getParams();
1955 if (isNullOrEmpty(params)) {
1956 // nothing to do if the StructType is not parameterized
1957 return failure();
1958 }
1959
1960 AffineMapFolder::Output out;
1961 AffineMapFolder::Input in = {
1962 op.getMapOperands(),
1963 op.getNumDimsPerMapAttr(),
1964 params.getValue(),
1965 };
1966 if (!in.mapOpGroups.empty()) {
1967 // If there are affine map operands, attempt to fold them to a constant.
1968 if (failed(AffineMapFolder::fold(rewriter, in, out, op, "struct parameter"))) {
1969 return failure();
1970 }
1971 LLVM_DEBUG({
1972 llvm::dbgs() << "[InstantiateAtCallOpCompute] folded affine_map in result type params\n";
1973 });
1974 } else {
1975 // If there are no affine map operands, attempt to refine the result type of the CallOp using
1976 // the function argument types and the type of the target function.
1977 auto callArgTypes = op.getArgOperands().getTypes();
1978 if (callArgTypes.empty()) {
1979 // no refinement possible if no function arguments
1980 return failure();
1981 }
1982 if (calleeReferencesTemplateParam(op)) {
1983 return failure();
1984 }
1985 SymbolTableCollection tables;
1986 auto lookupRes = lookupTopLevelSymbol<FuncDefOp>(tables, op.getCalleeAttr(), op);
1987 if (failed(lookupRes)) {
1988 return failure();
1989 }
1990 if (failed(instantiateViaTargetType(in, out, callArgTypes, lookupRes->get()))) {
1991 return failure();
1992 }
1993 LLVM_DEBUG({
1994 llvm::dbgs() << "[InstantiateAtCallOpCompute] propagated instantiations via symrefs in "
1995 "result type params: "
1996 << debug::toStringList(out.paramsOfStructTy) << '\n';
1997 });
1998 }
1999
2000 StructType newRetTy = StructType::get(oldRetTy.getNameRef(), out.paramsOfStructTy);
2001 LLVM_DEBUG(llvm::dbgs() << "[InstantiateAtCallOpCompute] newRetTy: " << newRetTy << '\n');
2002 if (newRetTy == oldRetTy) {
2003 return failure(); // nothing changed
2004 }
2005 // The `newRetTy` is computed via instantiateViaTargetType() which can only preserve the
2006 // original Attribute or convert to a concrete attribute via the unification process. Thus, if
2007 // the conversion here is illegal it means there is a type conflict within the LLZK code that
2008 // prevents instantiation of the struct with the requested type.
2009 if (!tracker_.isLegalConversion(oldRetTy, newRetTy, "InstantiateAtCallOpCompute")) {
2010 return rewriter.notifyMatchFailure(op, [&](Diagnostic &diag) {
2011 diag.append(
2012 "result type mismatch: due to struct instantiation, expected type ", newRetTy,
2013 ", but found ", oldRetTy
2014 );
2015 });
2016 }
2017 LLVM_DEBUG(llvm::dbgs() << "[InstantiateAtCallOpCompute] replaced " << op);
2018 CallOp newOp = replaceOpWithNewOp<CallOp>(
2019 rewriter, op, TypeRange {newRetTy}, op.getCallee(),
2020 AffineMapFolder::getConvertedMapOpGroups(out), out.dimsPerGroup, op.getArgOperands()
2021 );
2022 (void)newOp; // tell compiler it's intentionally unused in release builds
2023 LLVM_DEBUG(llvm::dbgs() << " with " << newOp << '\n');
2024 return success();
2025 }
2026
2027private:
2030 inline LogicalResult instantiateViaTargetType(
2031 const AffineMapFolder::Input &in, AffineMapFolder::Output &out,
2032 OperandRange::type_range callArgTypes, FuncDefOp targetFunc
2033 ) const {
2034 assert(targetFunc.isStructCompute()); // since `op.calleeIsStructCompute()`
2035 ArrayAttr targetResTyParams = targetFunc.getSingleResultTypeOfCompute().getParams();
2036 assert(!isNullOrEmpty(targetResTyParams)); // same cardinality as `in.paramsOfStructTy`
2037 assert(in.paramsOfStructTy.size() == targetResTyParams.size()); // verifier ensures this
2038
2039 if (llvm::all_of(in.paramsOfStructTy, isConcreteAttr<>)) {
2040 // Nothing can change if everything is already concrete
2041 return failure();
2042 }
2043
2044 LLVM_DEBUG({
2045 llvm::dbgs() << '[' << __FUNCTION__ << ']'
2046 << " call arg types: " << debug::toStringList(callArgTypes) << '\n';
2047 llvm::dbgs() << '[' << __FUNCTION__ << ']' << " target func arg types: "
2048 << debug::toStringList(targetFunc.getArgumentTypes()) << '\n';
2049 llvm::dbgs() << '[' << __FUNCTION__ << ']'
2050 << " struct params @ call: " << debug::toStringList(in.paramsOfStructTy) << '\n';
2051 llvm::dbgs() << '[' << __FUNCTION__ << ']'
2052 << " target struct params: " << debug::toStringList(targetResTyParams) << '\n';
2053 });
2054
2055 UnificationMap unifications;
2056 bool unifies = typeListsUnify(targetFunc.getArgumentTypes(), callArgTypes, {}, &unifications);
2057 (void)unifies; // tell compiler it's intentionally unused in builds without assertions
2058 assert(unifies && "should have been checked by verifiers");
2059
2060 LLVM_DEBUG({
2061 llvm::dbgs() << '[' << __FUNCTION__ << ']'
2062 << " unifications of arg types: " << debug::toStringList(unifications) << '\n';
2063 });
2064
2065 // Check for LHS SymRef (i.e., from the target function) that have RHS concrete Attributes (i.e.
2066 // from the call argument types) without any struct parameters (because the type with concrete
2067 // struct parameters will be used to instantiate the target struct rather than the fully
2068 // flattened struct type resulting in type mismatch of the callee to target) and perform those
2069 // replacements in the `targetFunc` return type to produce the new result type for the CallOp.
2070 SmallVector<Attribute> newReturnStructParams = llvm::map_to_vector(
2071 llvm::zip_equal(targetResTyParams.getValue(), in.paramsOfStructTy),
2072 [&unifications](std::tuple<Attribute, Attribute> p) {
2073 Attribute fromCall = std::get<1>(p);
2074 // Preserve attributes that are already concrete at the call site. Otherwise attempt to lookup
2075 // non-parameterized concrete unification for the target struct parameter symbol.
2076 if (!isConcreteAttr(fromCall)) {
2077 Attribute fromTgt = std::get<0>(p);
2078 LLVM_DEBUG({
2079 llvm::dbgs() << "[instantiateViaTargetType] fromCall = " << fromCall << '\n';
2080 llvm::dbgs() << "[instantiateViaTargetType] fromTgt = " << fromTgt << '\n';
2081 });
2082 assert(llvm::isa<SymbolRefAttr>(fromTgt));
2083 auto it = unifications.find(std::make_pair(llvm::cast<SymbolRefAttr>(fromTgt), Side::LHS));
2084 if (it != unifications.end()) {
2085 Attribute unifiedAttr = it->second;
2086 LLVM_DEBUG({
2087 llvm::dbgs() << "[instantiateViaTargetType] unifiedAttr = " << unifiedAttr << '\n';
2088 });
2089 if (unifiedAttr && isConcreteAttr<false>(unifiedAttr)) {
2090 return unifiedAttr;
2091 }
2092 }
2093 }
2094 return fromCall;
2095 }
2096 );
2097
2098 out.paramsOfStructTy = newReturnStructParams;
2099 assert(out.paramsOfStructTy.size() == in.paramsOfStructTy.size() && "post-condition");
2100 assert(out.mapOpGroups.empty() && "post-condition");
2101 assert(out.dimsPerGroup.empty() && "post-condition");
2102 return success();
2103 }
2104};

2109 patterns.add<

2111 InstantiateAtCallOpCompute // CallOp, targeting struct "compute()"

2114 return applyAndFoldGreedily(modOp, tracker, std::move(patterns));
2115}

2551 // Initialize roots from the given symbol definitions.
2552 SetVector<SymbolOpInterface> roots(keep.begin(), keep.end());
2553 // Add GlobalDefOp to the set of roots.
2554 rootMod.walk([&roots](global::GlobalDefOp gdef) { roots.insert(gdef); });
2555
2556 // Use a SymbolDefTree to find all Symbol defs reachable from one of the root nodes. Then
2557 // collect all Symbol uses reachable from those def nodes. These are the symbols that should
2558 // be preserved. All other symbol defs should be removed.
2559 DenseSet<Operation *> defsToKeep;
2560 llvm::df_iterator_default_set<const SymbolUseGraphNode *> symbolsToKeep;
2561 for (size_t i = 0; i < roots.size(); ++i) { // iterate for safe insertion
2562 SymbolOpInterface keepRoot = roots[i];
2563 LLVM_DEBUG({ llvm::dbgs() << "[EraseUnreachable] root: " << keepRoot << '\n'; });
2564 const SymbolDefTreeNode *keepRootNode = defTree.lookupNode(keepRoot);
2565 assert(keepRootNode && "every symbol def must be in the def tree");
2566 for (const SymbolDefTreeNode *reachableDefNode : llvm::depth_first(keepRootNode)) {
2567 LLVM_DEBUG({
2568 llvm::dbgs() << "[EraseUnreachable] can reach: " << reachableDefNode->getOp() << '\n';
2569 });
2570 if (SymbolOpInterface reachableDef = reachableDefNode->getOp()) {
2571 if (isErasableDefinition(reachableDef.getOperation())) {
2572 defsToKeep.insert(reachableDef.getOperation());
2573 }
2574 // Use 'depth_first_ext()' to get all symbol uses reachable from the current Symbol def
2575 // node. There are no uses if the node is not in the graph. Within the loop that populates
2576 // 'depth_first_ext()', also check if the symbol is an erasable definition and ensure it
2577 // is in 'roots' so the outer loop preserves all symbols reachable from it.
2578 if (const SymbolUseGraphNode *useGraphNodeForDef = useGraph.lookupNode(reachableDef)) {
2579 for (const SymbolUseGraphNode *usedSymbolNode :
2580 depth_first_ext(useGraphNodeForDef, symbolsToKeep)) {
2581 LLVM_DEBUG({
2582 llvm::dbgs() << "[EraseUnreachable] uses symbol: "
2583 << usedSymbolNode->getSymbolPath() << '\n';
2584 });
2585 // Ignore struct/template parameter symbols (before doing the lookup below because it
2586 // would fail anyway and then cause the "failed" case to be triggered unnecessarily).
2587 if (usedSymbolNode->isTemplateSymbolBinding()) {
2588 continue;
2589 }
2590 // If `usedSymbolNode` references an erasable definition, ensure it's considered in
2591 // the roots so symbols reachable from its body are preserved too.
2592 auto lookupRes = usedSymbolNode->lookupSymbol(tables);
2593 if (failed(lookupRes)) {
2594 LLVM_DEBUG(useGraph.dumpToDotFile());
2595 return failure();
2596 }
2597 // If loaded via an IncludeOp it's not in the current AST anyway so ignore.
2598 if (lookupRes->viaInclude()) {
2599 continue;
2600 }
2601 Operation *usedOp = lookupRes->get();
2602 if (isErasableDefinition(usedOp)) {
2603 SymbolOpInterface asSymbol = llvm::cast<SymbolOpInterface>(usedOp);
2604 bool insertRes = roots.insert(asSymbol);
2605 (void)insertRes; // tell compiler it's intentionally unused in release builds
2606 LLVM_DEBUG({
2607 if (insertRes) {
2608 llvm::dbgs() << "[EraseUnreachable] found another root: " << asSymbol << '\n';
2609 }
2610 });
2611 }
2612 }
2613 }
2614 }
2615 }
2616 }
2617
2618 SmallVector<SymbolOpInterface> toErase;
2619 rootMod.walk([this, &defsToKeep, &symbolsToKeep, &toErase](Operation *op) {
2620 if (!isErasableDefinition(op) || defsToKeep.contains(op)) {
2621 return;
2622 }
2623 SymbolOpInterface symOp = llvm::cast<SymbolOpInterface>(op);
2624 const SymbolUseGraphNode *n = this->useGraph.lookupNode(symOp);
2625 if (!n || !symbolsToKeep.contains(n)) {
2626 LLVM_DEBUG(llvm::dbgs() << "[EraseUnreachable] removing: " << symOp.getNameAttr() << '\n');
2627 toErase.push_back(symOp);
2628 }
2629 });
2630 for (SymbolOpInterface symOp : toErase) {
2631 symOp.erase();
2632 }
2633
2634 return success();
2635 }

2699 bool collectSafeToErase(SymbolOpInterface check) {
2700 assert(check); // pre-condition
2701
2702 // If previously visited, return the safety result.
2703 auto visited = visitedPlusSafetyResult.find(check);
2704 if (visited != visitedPlusSafetyResult.end()) {
2705 return visited->second;
2706 }
2707
2708 // If it's an erasable definition that is not in `tryToErase` then it cannot be erased.
2709 if (FromKeepSet::isErasableDefinition(check.getOperation()) && !tryToErase.contains(check)) {
2710 visitedPlusSafetyResult[check] = false;
2711 return false;
2712 }
2713
2714 // Otherwise, temporarily mark as safe b/c a node cannot keep itself live (and this prevents
2715 // the recursion from getting stuck in an infinite loop).
2716 visitedPlusSafetyResult[check] = true;
2717
2718 // Check if it's safe according to both the def tree and use graph.
2719 // Note: Every symbol must have a def node but ModuleOp and TemplateOp symbols may not have a
2720 // use node since they are not "terminal" symbols (i.e. they are not referred to directly).
2721 if (collectSafeToErase(defTree.lookupNode(check))) {
2722 const auto *useNode = useGraph.lookupNode(check);
2723 assert(useNode || (llvm::isa<ModuleOp, TemplateOp>(check.getOperation())));
2724 if (!useNode || collectSafeToErase(useNode)) {
2725 return true;
2726 }
2727 }
2728
2729 // Otherwise, revert the safety decision and return it.
2730 visitedPlusSafetyResult[check] = false;
2731 return false;
2732 }

2735 bool collectSafeToErase(const SymbolDefTreeNode *check) {
2736 assert(check); // pre-condition
2737 if (const SymbolDefTreeNode *p = check->getParent()) {
2738 if (SymbolOpInterface checkOp = p->getOp()) { // safe if parent is root
2739 return collectSafeToErase(checkOp);
2740 }
2741 }
2742 return true;
2743 }

2746 bool collectSafeToErase(const SymbolUseGraphNode *check) {
2747 assert(check); // pre-condition
2748 for (const SymbolUseGraphNode *p : check->predecessorIter()) {
2749 if (SymbolOpInterface checkOp = cachedLookup(p)) { // safe if via IncludeOp
2750 if (!collectSafeToErase(checkOp)) {
2751 return false;
2752 }
2753 }
2754 }
2755 return true;
2756 }

2762 SymbolOpInterface cachedLookup(const SymbolUseGraphNode *node) {
2763 assert(node && "must provide a node"); // pre-condition
2764 // Check for cached result
2765 auto fromCache = lookupCache.find(node);
2766 if (fromCache != lookupCache.end()) {
2767 return fromCache->second;
2768 }
2769 // Otherwise, perform lookup and cache
2770 auto lookupRes = node->lookupSymbol(tables);
2771 assert(succeeded(lookupRes) && "graph contains node with invalid path");
2772 assert(lookupRes->get() != nullptr && "lookup must return an Operation");
2773 // If loaded via an IncludeOp it's not in the current AST anyway so ignore.
2774 // NOTE: The SymbolUseGraph does contain nodes for struct parameters which cannot cast to
2775 // SymbolOpInterface. However, those will always be leaf nodes in the SymbolUseGraph and
2776 // therefore will not be traversed by this analysis so directly casting is fine.
2777 SymbolOpInterface actualRes =
2778 lookupRes->viaInclude() ? nullptr : llvm::cast<SymbolOpInterface>(lookupRes->get());
2779 // Cache and return
2780 lookupCache[node] = actualRes;
2781 assert((!actualRes == lookupRes->viaInclude()) && "not found iff included"); // post-condition
2782 return actualRes;
2783 }
