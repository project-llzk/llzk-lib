#mapSum = affine_map<()[s0, s1] -> (s0 + s1)>
#dynLen = affine_map<(d0) -> (d0 * 5 + 1)>

module attributes {
    llzk.lang = "unknown",
    llzk.main  = !struct.type<@Top>,
    // Multiple fields: single form is llzk.fields = #felt.field<"name", prime>.
    llzk.fields = [
        #felt.field<"foo", 101>,
        #felt.field<"bar",  11>
    ]} {

  // ── Sub-modules (namespace / package) ──────────────────────────────────────
  module @util {
    function.def private @log(!string.type) -> i1
    // Private decl with {llzk.pub} argument attributes.
    function.def private @pub_args(%a: !felt.type {llzk.pub}, %b: index {llzk.pub})

  // This path doesn't exist but that's okay since `@stdlib` is not referenced.
  include.from "lib/stdlib.llzk" as @stdlib

  module @gstore {
    // global.def: const/mutable, initialized/uninitialized, all basic types.
    global.def const @BOUND : !felt.type  = 16
    global.def       @tally : index       = 0
    global.def       @flag  : i1          = false
    global.def       @label : !string.type                         // uninitialized
    global.def const @BITS  : !array.type<2,2 x i1>               = [0, 1, 1, 0]
    global.def       @score : !array.type<2,1,2 x !felt.type>     = [1, 2, 3, 4]
  }

  // ── @Box ──────────────────────────────────────────────────────────────────
  // Polymorphic struct: type-variable @T, quoted member name, {llzk.pub}.
  poly.template @Box {
    poly.param @T : !poly.tvar<@T>
    struct.def @Box {
      struct.member @val    : !poly.tvar<@T> {llzk.pub}
      struct.member @"$tag" : i1                         // quoted symbol name

      function.def @compute(%arg0: !poly.tvar<@T>) -> !struct.type<@Box::@Box<[@T]>> {
        %self = struct.new : !struct.type<@Box::@Box<[@T]>>
        struct.writem %self[@val] = %arg0 : <@Box::@Box<[@T]>>, !poly.tvar<@T>
        %true = arith.constant true
        struct.writem %self[@"$tag"] = %true : <@Box::@Box<[@T]>>, i1
        function.return %self : !struct.type<@Box::@Box<[@T]>>
      }

      function.def @constrain(%arg0: !struct.type<@Box::@Box<[@T]>>, %arg1: !poly.tvar<@T>) {
        %0 = struct.readm %arg0[@val] : <@Box::@Box<[@T]>>, !poly.tvar<@T>
        constrain.eq %0, %arg1 : !poly.tvar<@T>, !poly.tvar<@T>
        function.return
      }
    }
  }

  // ── @Reg ──────────────────────────────────────────────────────────────────
  // Numeric param @N, {column} member, struct.readm with {tableOffset}.
  poly.template @Reg {
    poly.param @N : index
    struct.def @Reg {
      struct.member @data : !array.type<@N x !felt.type> {column}

      function.def @compute(%arg0: !array.type<@N x !felt.type>) -> !struct.type<@Reg::@Reg<[@N]>> {
        %self = struct.new : !struct.type<@Reg::@Reg<[@N]>>
        struct.writem %self[@data] = %arg0 : <@Reg::@Reg<[@N]>>, !array.type<@N x !felt.type>
        %0 = struct.readm %self[@data] : <@Reg::@Reg<[@N]>>, !array.type<@N x !felt.type> {tableOffset = -1 : index}
        function.return %self : !struct.type<@Reg::@Reg<[@N]>>
      }

      function.def @constrain(%arg0: !struct.type<@Reg::@Reg<[@N]>>, %arg1: !array.type<@N x !felt.type>) {
        %0 = struct.readm %arg0[@data] : <@Reg::@Reg<[@N]>>, !array.type<@N x !felt.type>
        constrain.eq %0, %arg1 : !array.type<@N x !felt.type>, !array.type<@N x !felt.type>
        function.return
      }
    }
  }

  // ── @ArrComp ──────────────────────────────────────────────────────────────
  // poly.expr, poly.read_const, poly.unifiable_cast, function.call with sym bindings.
  poly.template @ArrComp {
    poly.param @X
    poly.param @Y
    poly.expr @N {
      // This is similar to what the circom frontend would do: treat all as felt and cast final result.
      %a = poly.read_const @X : !felt.type
      %b = poly.read_const @Y : !felt.type
      %c = felt.add %a, %b
      %d = cast.toindex %c : !felt.type
      poly.yield %d : index
    }

    struct.def @ArrComp {
      struct.member @reg : !struct.type<@Reg::@Reg<[#mapSum]>>

      function.def @compute(%arg0: !array.type<@N x !felt.type>) -> !struct.type<@ArrComp::@ArrComp<[@X, @Y]>> {
        %self = struct.new : !struct.type<@ArrComp::@ArrComp<[@X, @Y]>>
        %0 = poly.read_const @X : index
        %1 = poly.read_const @Y : index
        // poly.unifiable_cast: assert @N is compatible with the two-symbol map #mapSum.
        %2 = poly.unifiable_cast %arg0 : (!array.type<@N x !felt.type>) -> !array.type<#mapSum x !felt.type>
        // function.call with explicit affine-symbol bindings {()[%x, %y]}.
        %3 = function.call @Reg::@Reg::@compute(%2) {()[%0, %1]} : (!array.type<#mapSum x !felt.type>) -> !struct.type<@Reg::@Reg<[#mapSum]>>
        struct.writem %self[@reg] = %3 : <@ArrComp::@ArrComp<[@X, @Y]>>, !struct.type<@Reg::@Reg<[#mapSum]>>
        function.return %self : !struct.type<@ArrComp::@ArrComp<[@X, @Y]>>
      }

      function.def @constrain(%arg0: !struct.type<@ArrComp::@ArrComp<[@X, @Y]>>, %arg1: !array.type<@N x !felt.type>) {
        %0 = poly.unifiable_cast %arg1 : (!array.type<@N x !felt.type>) -> !array.type<#mapSum x !felt.type>
        %1 = struct.readm %arg0[@reg] : <@ArrComp::@ArrComp<[@X, @Y]>>, !struct.type<@Reg::@Reg<[#mapSum]>>
        function.call @Reg::@Reg::@constrain(%1, %0) : (!struct.type<@Reg::@Reg<[#mapSum]>>, !array.type<#mapSum x !felt.type>) -> ()
        function.return
      }
    }
  }

  // ── @Entry ────────────────────────────────────────────────────────────────
  // Struct with empty explicit parameter list (the <[]> syntax).
  poly.template @Entry {
    struct.def @Entry {
      function.def @compute() -> !struct.type<@Entry::@Entry<[]>> {
        %self = struct.new : !struct.type<@Entry::@Entry<[]>>
        function.return %self : !struct.type<@Entry::@Entry<[]>>
      }
      function.def @constrain(%arg0: !struct.type<@Entry::@Entry<[]>>) {
        function.return
      }
    }
  }

  // ── @Top (llzk.main) ──────────────────────────────────────────────────────
  struct.def @Top {
    struct.member @bit  : !felt.type {signal}
    struct.member @reg4 : !struct.type<@Reg::@Reg<[4]>>
    struct.member @box  : !struct.type<@Box::@Box<[!felt.type]>>

    function.def @compute(%a: !felt.type, %pub: !felt.type {llzk.pub})
        -> !struct.type<@Top> attributes {function.allow_witness} {
      %self = struct.new : !struct.type<@Top>

      // felt: const, add, sub, mul, div, neg
      %c0  = felt.const 0
      %c1  = felt.const 1
      %c7  = felt.const 7
      %add = felt.add %a, %c7
      %sub = felt.sub %a, %c1
      %mul = felt.mul %a, %c7
      %div = felt.div %a, %c7
      %neg = felt.neg %div

      // bool: all six comparison predicates on felt
      %beq = bool.cmp eq(%a, %pub)
      %bne = bool.cmp ne(%a, %pub)
      %blt = bool.cmp lt(%a, %pub)
      %ble = bool.cmp le(%a, %pub)
      %bgt = bool.cmp gt(%a, %pub)
      %bge = bool.cmp ge(%a, %pub)

      // bool: assert (plain) and assert with message
      bool.assert %bge
      bool.assert %beq, "a must equal pub"

      // string.new
      %str = string.new "hello LLZK"

      // index: addi, subi, muli, divui, cmpi (eq, ne, slt, sle)
      %i0    = arith.constant 0 : index
      %ub    = arith.constant 4 : index
      %step  = arith.constant 1 : index
      %ia    = arith.addi  %i0, %ub    : index
      %is    = arith.subi  %ub, %step  : index
      %im    = arith.muli  %ub, %step  : index
      %id    = arith.divui %ub, %step  : index
      %cieq  = arith.cmpi eq,  %i0, %ub : index
      %cine  = arith.cmpi ne,  %i0, %ub : index
      %cislt = arith.cmpi slt, %i0, %ub : index
      %cisle = arith.cmpi sle, %i0, %ub : index

      // cast: i1 -> felt,  index -> felt  (cast.toindex shown in @non_native)
      %flt_a = cast.tofelt %beq : i1
      %flt_b = cast.tofelt %i0  : index
      // also demonstrate cast.tofelt with explicit result-type form
      %flt_c = cast.tofelt %beq : i1, !felt.type<"goldilocks">
      %flt_d = cast.tofelt %i0  : index, !felt.type<"bn128">
      // also demonstrate cast.tofelt with explicit overflow semantics
      %flt_e = cast.tofelt assert %i0 : index
      %flt_f = cast.tofelt sat %i0 : index, !felt.type<"mersenne31">
      %flt_g = cast.tofelt trunc %i0 : index
      %flt_h = cast.tofelt wrap %i0 : index, !felt.type<"koalabear">

      // arith ops on i1 width
      %tf  = arith.constant true
      %ff  = arith.constant false
      %ai  = arith.andi %tf, %ff : i1
      %ao  = arith.ori  %tf, %ff : i1
      %ax  = arith.xori %tf, %ff : i1

      // poly.applymap: no-args constant, dims-only, syms-only, mixed
      %pa0 = poly.applymap ()             affine_map<() -> (42)>
      %pa1 = poly.applymap (%ub)[]        affine_map<(d0) -> (d0 + 1)>
      %pa2 = poly.applymap ()[%ub, %step] affine_map<()[s0, s1] -> (s0 + s1)>
      %pa3 = poly.applymap (%ub)[%step]   affine_map<(d0)[s0] -> (d0 * s0)>

      // 1-D array: new (with values), read, write
      %arr4 = array.new %c0, %c1, %add, %mul : !array.type<4 x !felt.type>
      %e0   = array.read  %arr4[%i0]         : !array.type<4 x !felt.type>, !felt.type
      array.write %arr4[%i0] = %neg          : !array.type<4 x !felt.type>, !felt.type

      // 2-D array: new, extract row, insert row, multi-index read, len
      %i1   = arith.constant 1 : index
      %arr8 = array.new %c0, %c1, %add, %mul, %c0, %c1, %add, %mul
                : !array.type<2,4 x !felt.type>
      %row  = array.extract %arr8[%i0]   : !array.type<2,4 x !felt.type>
      array.insert %arr8[%i0] = %arr4    : !array.type<2,4 x !felt.type>, !array.type<4 x !felt.type>
      %e11  = array.read %arr8[%i1, %i1] : !array.type<2,4 x !felt.type>, !felt.type
      %len  = array.len  %arr8, %i0      : !array.type<2,4 x !felt.type>

      // scf.if with result value
      %bit = scf.if %beq -> !felt.type {
        scf.yield %c0 : !felt.type
      } else {
        scf.yield %c1 : !felt.type
      }
      struct.writem %self[@bit] = %bit : !struct.type<@Top>, !felt.type

      // scf.if without result (side-effect only)
      scf.if %bne {
        array.write %arr4[%i0] = %c1 : !array.type<4 x !felt.type>, !felt.type
      }

      // scf.for with iter_args
      %sum = scf.for %iv = %i0 to %ub step %step
                 iter_args(%acc = %c0) -> !felt.type {
        %v = felt.add %a, %acc
        scf.yield %v : !felt.type
      }

      // scf.while with #llzk.loopbounds attribute
      %wres = scf.while (%w = %c0) : (!felt.type) -> !felt.type {
        %cond = bool.cmp lt(%w, %a)
        scf.condition(%cond) %w : !felt.type
      } do {
        ^bb0(%w1: !felt.type):
          %nxt = felt.add %w1, %c1
          scf.yield %nxt : !felt.type
      } attributes { llzk.loopbounds = #llzk.loopbounds<0 to 10 step 1> }

      // sub-struct compute calls
      %boxed = function.call @Box::@Box::@compute(%a)
                 : (!felt.type) -> !struct.type<@Box::@Box<[!felt.type]>>
      struct.writem %self[@box] = %boxed : !struct.type<@Top>, !struct.type<@Box::@Box<[!felt.type]>>

      %inp4 = array.new : !array.type<4 x !felt.type>
      %reg4 = function.call @Reg::@Reg::@compute(%inp4)
                 : (!array.type<4 x !felt.type>) -> !struct.type<@Reg::@Reg<[4]>>
      struct.writem %self[@reg4] = %reg4 : !struct.type<@Top>, !struct.type<@Reg::@Reg<[4]>>

      // global.read / global.write  (only valid in allow_witness / @compute)
      %prime = global.read  @gstore::@BOUND      : !felt.type
      global.write          @gstore::@tally = %i0 : index

      function.return %self : !struct.type<@Top>
    }

    function.def @constrain(
        %self: !struct.type<@Top>,
        %a:   !felt.type,
        %pub: !felt.type {llzk.pub}) attributes {function.allow_constraint} {
      // struct.readm
      %b   = struct.readm %self[@bit]  : !struct.type<@Top>, !felt.type
      %bxm = struct.readm %self[@box]  : !struct.type<@Top>, !struct.type<@Box::@Box<[!felt.type]>>
      %rm  = struct.readm %self[@reg4] : !struct.type<@Top>, !struct.type<@Reg::@Reg<[4]>>

      // llzk.nondet: non-deterministic witness  (only valid in @constrain)
      %nd = llzk.nondet : !felt.type

      // constrain.eq: verify bit*(bit-1) == 0  (boolean soundness)
      %c0  = felt.const 0
      %c1  = felt.const 1
      %bm1 = felt.sub %b, %c1
      %chk = felt.mul %b, %bm1
      constrain.eq %chk, %c0 : !felt.type

      // constrain.in: bit must be in {0, 1}
      %vals = array.new %c0, %c1 : !array.type<2 x !felt.type>
      constrain.in %vals, %b : !array.type<2 x !felt.type>, !felt.type

      // delegate constrain to sub-structs
      function.call @Box::@Box::@constrain(%bxm, %a)
          : (!struct.type<@Box::@Box<[!felt.type]>>, !felt.type) -> ()
      %inp4 = array.new : !array.type<4 x !felt.type>
      function.call @Reg::@Reg::@constrain(%rm, %inp4)
          : (!struct.type<@Reg::@Reg<[4]>>, !array.type<4 x !felt.type>) -> ()

      function.return
    }
  }

  // ── Standalone functions ──────────────────────────────────────────────────

  // bool.and/or/xor/not require function.allow_non_native_field_ops.
  function.def @bool_logic(%a: i1, %b: i1) -> (i1, i1, i1, i1)
      attributes {function.allow_non_native_field_ops} {
    %and = bool.and %a, %b
    %or  = bool.or  %a, %b
    %xor = bool.xor %a, %b
    %not = bool.not %a
    function.return %and, %or, %xor, %not : i1, i1, i1, i1

  // Multi-return function + caller with destructured result.
  function.def @pair(%x: index {function.arg_name = "x"})
      -> (index {function.res_name = "first"},
          index {function.res_name = "second"}) {
    %one = arith.constant 1 : index
    function.return %x, %one : index, index
  }
  function.def @use_pair(%x: index) -> (index, index) {
    %a, %b = function.call @pair(%x) : (index) -> (index, index)
    function.return %a, %b : index, index
  }

  // cast.toindex requires function.allow_non_native_field_ops.
  function.def @non_native(%f1: !felt.type, %f2: !felt.type<"bn254">)
      -> (index, index, index, index, index, index)
      attributes {function.allow_non_native_field_ops} {
    %1  = cast.toindex %f1
    %2  = cast.toindex %f2 : !felt.type<"bn254">

    // also demonstrate cast.toindex with explicit overflow semantics
    %3  = cast.toindex assert %f1
    %4  = cast.toindex sat %f2 : !felt.type<"bn254">
    %5  = cast.toindex trunc %f1
    %6  = cast.toindex wrap %f2 : !felt.type<"bn254">

    function.return %1, %2, %3, %4, %5, %6 : index, index, index, index, index, index
  }

  // Dynamic-length array: array.new{(<dim>)} with affine-map element count.
  function.def @dyn_array(%n: index) -> !array.type<#dynLen x !felt.type> {
    %arr = array.new{(%n)} : !array.type<#dynLen x !felt.type>
    function.return %arr : !array.type<#dynLen x !felt.type>
  }

  // POD type: empty, uninitialized, initialized; pod.read, pod.write.
  function.def @pod_ops(%n: index) -> !pod.type<[@n: index]> {
    %empty  = pod.new               : <[]>
    %uninit = pod.new               : <[@n: index]>
    %init   = pod.new { @n = %n }  : <[@n: index]>
    %v      = pod.read  %init[@n]  : !pod.type<[@n: index]>, index
    pod.write %init[@n] = %v       : !pod.type<[@n: index]>, index
    function.return %init : !pod.type<[@n: index]>
  }

  // RAM dialect ops
  function.def @ram_ops() -> !felt.type attributes {function.allow_witness} {
    %c0 = arith.constant 0 : index
    %v = felt.const 42
    ram.store %c0, %v : !felt.type
    %loaded = ram.load %c0 : !felt.type
    function.return %loaded : !felt.type
  }
